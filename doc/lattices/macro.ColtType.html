macro_rules! ColtType {
    ($a:ty => ()) => { ... };
    ($a:ty => $c:ty) => { ... };
    ($a:ty => $c:ty, $( $d:ty ),*) => { ... };
    ($a:ty, $( $b:ty ),* => ()) => { ... };
    ($a:ty, $( $b:ty ),* => $c:ty) => { ... };
    ($a:ty, $( $b:ty ),* => $c:ty, $( $d:ty ),*) => { ... };
    ($a:ty) => { ... };
    ($a:ty, $( $b:ty ),*) => { ... };
}
Expand description

Construct a forest of Ghts (i.e. a ColtForest) with the given schema and storage type.

use lattices::ColtType;

type MyColt = ColtType!(u16, u32, u64);
