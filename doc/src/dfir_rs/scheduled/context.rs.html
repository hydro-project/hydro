27/// Internal state storage for operator accumulators.
28struct StateData {
29    state: Box<dyn Any>,
30    lifespan_hook_fn: Option<LifespanResetFn>,
31    /// `None` for static.
32    lifespan: Option<StateLifespan>,
33}
34type LifespanResetFn = Box<dyn FnMut(&mut dyn Any)>;

36/// Coordinates waking between [`Context`] (inside the tick closure) and [`Dfir`]
37/// (the external runner). Shared via `Arc` between both.
38///
39/// When external data arrives (e.g., a tokio stream receives a message), the [`Context::waker`]
40/// fires, which sets `can_start_tick` and wakes the [`Dfir::run`](Dfir::run) task so it starts a new tick.
41/// Implements [`Wake`] directly so it can be used as a `Waker` without an extra wrapper.
42#[doc(hidden)]
43pub struct WakeState {
44    /// Set to `true` when external data arrives, signaling that a new tick should run.
45    /// Checked by [`Dfir::run_tick`](Dfir::run_tick) and [`Dfir::run_available`](Dfir::run_available).
46    can_start_tick: std::sync::atomic::AtomicBool,
47    /// Wakes the [`Dfir::run`](Dfir::run) task from its idle `poll_fn` sleep.
48    task_waker: futures::task::AtomicWaker,
49}
50
51impl Default for WakeState {
52    fn default() -> Self {
53        Self {
54            can_start_tick: std::sync::atomic::AtomicBool::new(false),
55            task_waker: futures::task::AtomicWaker::new(),
56        }
57    }
58}
59
60impl Wake for WakeState {
61    fn wake(self: Arc<Self>) {
62        self.wake_by_ref();
63    }
64
65    fn wake_by_ref(self: &Arc<Self>) {
66        self.can_start_tick.store(true, Ordering::Relaxed);
67        self.task_waker.wake();
68    }
69}

71/// A lightweight context for inline codegen that avoids the overhead of the full
72/// [`Context`] (no tokio channels, no scheduler queues, no loop machinery).
73///
74/// Exposes the same method names that operator-generated code calls on both
75/// `df` (for prologues: `add_state`, `set_state_lifespan_hook`) and
76/// `context` (for iterators: `state_ref_unchecked`, `is_first_run_this_tick`, etc.).
77#[doc(hidden)]
78#[derive(Default)]
79pub struct Context {
80    /// Storage for the operator-facing State API.
81    states: SlotVec<StateTag, StateData>,
82    /// Counter for number of ticks run.
83    current_tick: TickInstant,
84    /// Coordinates waking between [`Context`] (inside the tick closure) and [`Dfir`]
85    /// (the external runner). Shared via `Arc` between both. Implements [`Wake`].
86    wake_state: Arc<WakeState>,
87    /// Live-updating DFIR runtime metrics via interior mutability.
88    metrics: Rc<DfirMetrics>,
89    /// Tasks buffered via [`Self::request_task`], spawned by [`Dfir::spawn_tasks`]
90    /// once the runtime is running inside a tokio `LocalSet`.
91    tasks_to_spawn: Vec<Pin<Box<dyn Future<Output = ()> + 'static>>>,
92}
93
94impl Context {
95    /// Create a new inline context with shared wake state and metrics.
96    pub fn new(wake_state: Arc<WakeState>, metrics: Rc<DfirMetrics>) -> Self {
97        Self {
98            states: SlotVec::new(),
99            current_tick: TickInstant::default(),
100            wake_state,
101            metrics,
102            tasks_to_spawn: Vec::new(),
103        }
104    }
105
106    // --- Methods called as `df.xxx()` in operator prologues ---
107
108    /// Adds state and returns a handle.
109    pub fn add_state<T>(&mut self, state: T) -> StateHandle<T>
110    where
111        T: Any,
112    {
113        let state_data = StateData {
114            state: Box::new(state),
115            lifespan_hook_fn: None,
116            lifespan: None,
117        };
118        let state_id = self.states.insert(state_data);
119        StateHandle {
120            state_id,
121            _phantom: PhantomData,
122        }
123    }
124
125    /// Sets a hook to modify state at the end of each tick.
126    pub fn set_state_lifespan_hook<T>(
127        &mut self,
128        handle: StateHandle<T>,
129        _lifespan: StateLifespan,
130        mut hook_fn: impl 'static + FnMut(&mut T),
131    ) where
132        T: Any,
133    {
134        let state_data = self
135            .states
136            .get_mut(handle.state_id)
137            .expect("Failed to find state with given handle.");
138        state_data.lifespan_hook_fn = Some(Box::new(move |state| {
139            (hook_fn)(state.downcast_mut::<T>().unwrap());
140        }));
141        state_data.lifespan = Some(_lifespan);
142    }
143
144    /// Buffers an async task to be spawned later by [`Dfir::spawn_tasks`].
145    ///
146    /// Tasks are deferred because `write_prologue` runs during graph construction,
147    /// which may occur before a tokio `LocalSet` is entered. Buffered tasks are
148    /// drained and spawned via `tokio::task::spawn_local` at the start of
149    /// [`Dfir::run_tick`]. Tasks requested after that point remain buffered until
150    /// the next call to [`Dfir::run_tick`].
151    pub fn request_task<Fut>(&mut self, future: Fut)
152    where
153        Fut: Future<Output = ()> + 'static,
154    {
155        self.tasks_to_spawn.push(Box::pin(future));
156    }
157
158    // --- Methods called as `context.xxx()` in operator iterators ---
159
160    /// Returns a shared reference to the state.
161    ///
162    /// # Safety
163    /// `StateHandle<T>` must be from _this_ instance.
164    pub unsafe fn state_ref_unchecked<T>(&self, handle: StateHandle<T>) -> &'_ T
165    where
166        T: Any,
167    {
168        let state = self
169            .states
170            .get(handle.state_id)
171            .expect("Failed to find state with given handle.")
172            .state
173            .as_ref();
174        debug_assert!(state.is::<T>());
175        unsafe { &*(state as *const dyn Any as *const T) }
176    }
177
178    /// Always returns `true` in inline mode. The inline codegen runs the entire DAG
179    /// once per tick with no re-execution, so every subgraph is always on its first
180    /// (and only) run within each tick.
181    pub fn is_first_run_this_tick(&self) -> bool {
182        true
183    }
184
185    /// Gets the current tick count.
186    pub fn current_tick(&self) -> TickInstant {
187        self.current_tick
188    }
189
190    /// Returns a reference to the runtime metrics.
191    pub fn metrics(&self) -> &Rc<DfirMetrics> {
192        &self.metrics
193    }
194
195    /// No-op: inline mode has no subgraph scheduling.
196    pub fn current_subgraph(&self) -> SubgraphId {
197        SubgraphId::from_raw(0)
198    }
199
200    /// In inline mode, every subgraph runs unconditionally each tick, so the `sg_id`
201    /// parameter is ignored. Only `is_external` matters: when `true`, it signals that
202    /// external data has arrived and a new tick should be started.
203    pub fn schedule_subgraph(&self, _sg_id: SubgraphId, is_external: bool) {
204        if is_external {
205            self.wake_state.wake_by_ref();
206        }
207    }
208
209    /// Returns a waker that signals external data has arrived.
210    pub fn waker(&self) -> std::task::Waker {
211        std::task::Waker::from(self.wake_state.clone())
212    }
213
214    /// Runs end-of-tick state hooks and increments the tick counter.
215    /// Called by the generated tick closure at the end of each tick.
216    #[doc(hidden)]
217    pub fn __end_tick(&mut self) {
218        for state_data in self.states.values_mut() {
219            let StateData {
220                state,
221                lifespan_hook_fn: Some(lifespan_hook_fn),
222                lifespan: Some(StateLifespan::Tick),
223            } = state_data
224            else {
225                continue;
226            };
227            (lifespan_hook_fn)(Box::deref_mut(state));
228        }
229        self.current_tick += crate::scheduled::ticks::TickDuration::SINGLE_TICK;
230    }
231}

233/// A wrapper around an inline-codegen tick closure that provides [`Self::run`],
234/// [`Self::run_available`], and [`Self::run_tick`] methods — mirroring the [`Dfir`](super::context::Dfir)
235/// API.
236///
237/// # Design
238///
239/// The inline codegen generates an `async move |df: &mut Context|` closure that captures
240/// dataflow-specific state (handoff buffers, source iterators) and receives the [`Context`]
241/// (operator accumulators, tick counter) by reference each tick. `Dfir` owns both the
242/// closure and the context, and coordinates tick lifecycle and idle/wake behavior.
243///
244/// We use a single opaque closure rather than generating a bespoke struct per dataflow because:
245/// - The closure naturally captures exactly the state it needs with correct lifetimes
246/// - No codegen needed for struct definitions, field accessors, or initialization
247/// - Rust's async closure machinery handles the complex state machine (suspend/resume across
248///   `.await` points) that would be very difficult to replicate in a generated struct
249///
250/// The `Tick` type parameter is bounded by [`TickClosure`] (not `AsyncFnMut` directly) to
251/// support type erasure via [`TickClosureErased`] / [`DfirErased`] for heterogeneous
252/// collections (e.g., the sim runtime storing multiple locations in a `Vec`). The concrete
253/// (non-erased) path used by trybuild and embedded has zero overhead.
254#[doc(hidden)]
255pub struct Dfir<Tick> {
256    /// Async closure which runs a single tick when called.
257    tick_closure: Tick,
258    /// Coordinates waking between [`Context`] (inside the tick closure) and [`Dfir`]
259    /// (the external runner). Shared via `Arc` between both. Implements [`Wake`].
260    wake_state: Arc<WakeState>,
261    /// The inline context, owned by `Dfir` and passed to the tick closure by reference.
262    context: Context,
263    /// See [`Self::meta_graph()`].
264    #[cfg(feature = "meta")]
265    meta_graph: Option<DfirGraph>,
266    /// See [`Self::diagnostics()`].
267    #[cfg(feature = "meta")]
268    diagnostics: Option<Vec<Diagnostic<SerdeSpan>>>,
269}
270
271/// Trait for tick closures — abstracts over both concrete async closures
272/// and type-erased boxed versions ([`TickClosureErased`]).
273///
274/// The `&mut Context` parameter is owned by [`Dfir`] and lent to the
275/// closure each tick, avoiding shared-ownership overhead for the context.
276#[doc(hidden)]
277pub trait TickClosure {
278    /// Call the tick closure. Returns `true` if any subgraph received input data.
279    fn call_tick<'a>(&'a mut self, ctx: &'a mut Context) -> impl Future<Output = bool> + 'a;
280}
281
282impl<F: for<'a> AsyncFnMut(&'a mut Context) -> bool> TickClosure for F {
283    fn call_tick<'a>(&'a mut self, ctx: &'a mut Context) -> impl Future<Output = bool> + 'a {
284        self(ctx)
285    }
286}

298/// Type-erased tick function for use in heterogeneous collections (e.g., the sim runtime).
299#[doc(hidden)]
300pub struct TickClosureErased(Box<dyn TickClosureErasedInner>);
301
302/// Object-safe inner trait for [`TickClosureErased`]. Needed because `AsyncFnMut` is not
303/// object-safe (GAT return type), but a trait with `&mut self -> Pin<Box<dyn Future + '_>>`
304/// is — the returned future borrows from the trait object which owns the closure.
305trait TickClosureErasedInner {
306    fn call_tick<'a>(
307        &'a mut self,
308        ctx: &'a mut Context,
309    ) -> Pin<Box<dyn Future<Output = bool> + 'a>>;
310}
311
312impl<F: for<'a> AsyncFnMut(&'a mut Context) -> bool> TickClosureErasedInner for F {
313    fn call_tick<'a>(
314        &'a mut self,
315        ctx: &'a mut Context,
316    ) -> Pin<Box<dyn Future<Output = bool> + 'a>> {
317        Box::pin(self(ctx))
318    }
319}
320
321impl TickClosure for TickClosureErased {
322    fn call_tick<'a>(&'a mut self, ctx: &'a mut Context) -> impl Future<Output = bool> + 'a {
323        self.0.call_tick(ctx)
324    }
325}

331impl<Tick: TickClosure> Dfir<Tick> {
332    /// Create a new `Dfir` from a tick closure, inline context,
333    /// and meta graph / diagnostics JSON strings.
334    #[doc(hidden)]
335    pub fn new(
336        tick_closure: Tick,
337        context: Context,
338        meta_graph_json: Option<&str>,
339        diagnostics_json: Option<&str>,
340    ) -> Self {
341        #[cfg(not(feature = "meta"))]
342        let _ = (meta_graph_json, diagnostics_json);
343        Self {
344            tick_closure,
345            wake_state: context.wake_state.clone(),
346            context,
347            #[cfg(feature = "meta")]
348            meta_graph: meta_graph_json.map(|json| {
349                let mut meta_graph: DfirGraph =
350                    serde_json::from_str(json).expect("Failed to deserialize graph.");
351                let mut op_inst_diagnostics = Diagnostics::new();
352                meta_graph.insert_node_op_insts_all(&mut op_inst_diagnostics);
353                assert!(
354                    op_inst_diagnostics.is_empty(),
355                    "Expected no diagnostics, got: {:#?}",
356                    op_inst_diagnostics
357                );
358                meta_graph
359            }),
360            #[cfg(feature = "meta")]
361            diagnostics: diagnostics_json.map(|json| {
362                serde_json::from_str(json).expect("Failed to deserialize diagnostics.")
363            }),
364        }
365    }
366
367    /// Return a handle to the meta graph, if set.
368    #[cfg(feature = "meta")]
369    #[cfg_attr(docsrs, doc(cfg(feature = "meta")))]
370    pub fn meta_graph(&self) -> Option<&DfirGraph> {
371        self.meta_graph.as_ref()
372    }
373
374    /// Returns any diagnostics generated by the surface syntax macro.
375    #[cfg(feature = "meta")]
376    #[cfg_attr(docsrs, doc(cfg(feature = "meta")))]
377    pub fn diagnostics(&self) -> Option<&[Diagnostic<SerdeSpan>]> {
378        self.diagnostics.as_deref()
379    }
380
381    /// Returns a reference-counted handle to the continually-updated runtime metrics for this DFIR instance.
382    pub fn metrics(&self) -> Rc<DfirMetrics> {
383        Rc::clone(self.context.metrics())
384    }
385
386    /// Gets the current tick (local time) count.
387    pub fn current_tick(&self) -> TickInstant {
388        self.context.current_tick()
389    }
390
391    /// Returns a [`DfirMetricsIntervals`] handle where each call to
392    /// [`DfirMetricsIntervals::take_interval`] ends the current interval and returns its metrics.
393    ///
394    /// The first call to `take_interval` returns metrics since this DFIR instance was created. Each subsequent call to
395    /// `take_interval` returns metrics since the previous call.
396    ///
397    /// Cloning the handle "forks" it from the original, as afterwards each interval may return different metrics
398    /// depending on when exactly `take_interval` is called.
399    pub fn metrics_intervals(&self) -> DfirMetricsIntervals {
400        DfirMetricsIntervals {
401            curr: self.metrics(),
402            prev: None,
403        }
404    }

407impl<Tick: TickClosure> Dfir<Tick> {
408    /// Spawns all tasks buffered via [`Context::request_task`].
409    ///
410    /// This drains the buffer, so subsequent calls are no-ops until new tasks are requested.
411    fn spawn_tasks(&mut self) {
412        for task in self.context.tasks_to_spawn.drain(..) {
413            tokio::task::spawn_local(task);
414        }
415    }
416
417    /// Run a single tick. Returns `true` if any subgraph received input data.
418    ///
419    /// Checks both handoff buffers (via `work_done` flag set in generated recv port code)
420    /// and external events (via `can_start_tick` set by wakers/schedule_subgraph).
421    pub async fn run_tick(&mut self) -> bool {
422        self.spawn_tasks();
423        let had_external = self
424            .wake_state
425            .can_start_tick
426            .swap(false, Ordering::Relaxed);
427        let tick_had_work = self.tick_closure.call_tick(&mut self.context).await;
428        had_external || tick_had_work || self.wake_state.can_start_tick.load(Ordering::Relaxed)
429    }
430
431    /// Run a single tick synchronously. Panics if the tick yields (async suspension).
432    /// Returns `true` if work was done (see [`Self::run_tick`]).
433    pub fn run_tick_sync(&mut self) -> bool {
434        let mut fut = std::pin::pin!(self.run_tick());
435        let mut ctx = std::task::Context::from_waker(std::task::Waker::noop());
436        match fut.as_mut().poll(&mut ctx) {
437            std::task::Poll::Ready(result) => result,
438            std::task::Poll::Pending => {
439                panic!("Dfir::run_tick_sync: tick yielded asynchronously.")
440            }
441        }
442    }
443
444    /// Run ticks as long as work is available, then return.
445    pub async fn run_available(&mut self) {
446        // Always run at least one tick.
447        self.wake_state
448            .can_start_tick
449            .store(false, Ordering::Relaxed);
450        loop {
451            self.run_tick().await;
452            let can_start_tick = self
453                .wake_state
454                .can_start_tick
455                .swap(false, Ordering::Relaxed);
456            if !can_start_tick {
457                break;
458            }
459            // Yield between each tick to receive more events.
460            tokio::task::yield_now().await;
461        }
462    }
463
464    /// [`Self::run_available`] but panics if any tick yields asynchronously.
465    pub fn run_available_sync(&mut self) {
466        self.wake_state
467            .can_start_tick
468            .store(false, Ordering::Relaxed);
469        loop {
470            self.run_tick_sync();
471            let can_start_tick = self
472                .wake_state
473                .can_start_tick
474                .swap(false, Ordering::Relaxed);
475            if !can_start_tick {
476                break;
477            }
478        }
479    }
480
481    /// Run forever, processing ticks when work is available and yielding when idle.
482    pub async fn run(&mut self) -> crate::Never {
483        loop {
484            self.run_available().await;
485            // Wait for an external event to wake us.
486            std::future::poll_fn(|cx| {
487                // Register waker first to avoid race: if an event fires between
488                // the check and the register, the waker is already in place.
489                self.wake_state.task_waker.register(cx.waker());
490                if self.wake_state.can_start_tick.load(Ordering::Relaxed) {
491                    std::task::Poll::Ready(())
492                } else {
493                    std::task::Poll::Pending
494                }
495            })
496            .await;
497        }
498    }
499}

501impl<Tick: 'static + for<'a> AsyncFnMut(&'a mut Context) -> bool> Dfir<Tick> {
502    /// Type-erase the tick closure for use in heterogeneous collections.
503    ///
504    /// Wraps the concrete async closure in [`TickClosureErased`], which boxes the future
505    /// returned by each tick call. This adds one heap allocation per tick, but enables
506    /// storing multiple `Dfir`s with different closure types in a single `Vec`.
507    ///
508    /// Only needed for the sim runtime path. The trybuild and embedded paths keep the
509    /// concrete type and pay no erasure cost.
510    pub fn into_erased(self) -> DfirErased {
511        Dfir {
512            tick_closure: TickClosureErased(Box::new(self.tick_closure)),
513            wake_state: self.wake_state,
514            context: self.context,
515            #[cfg(feature = "meta")]
516            meta_graph: self.meta_graph,
517            #[cfg(feature = "meta")]
518            diagnostics: self.diagnostics,
519        }
520    }
