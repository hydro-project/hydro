32/// The main state and scheduler of the runtime instance. Provided as the `context` API to each
33/// subgraph/operator as it is run.
34///
35/// Each instance stores eactly one Context inline. Before the `Context` is provided to
36/// a running operator, the `subgraph_id` field must be updated.
37pub struct Context {
38    /// Storage for the user-facing State API.
39    states: SlotVec<StateTag, StateData>,
40
41    /// Priority stack for handling strata within loops. Prioritized by loop depth.
42    pub(super) stratum_stack: PriorityStack<usize>,
43
44    /// Stack of loop nonces. Used to identify when a new loop iteration begins.
45    pub(super) loop_nonce_stack: Vec<usize>,
46
47    /// TODO(mingwei):
48    /// used for loop iteration scheduling.
49    pub(super) schedule_deferred: Vec<SubgraphId>,
50
51    /// TODO(mingwei): separate scheduler into its own struct/trait?
52    /// Index is stratum, value is FIFO queue for that stratum.
53    pub(super) stratum_queues: Vec<VecDeque<SubgraphId>>,
54
55    /// Receive events, if second arg indicates if it is an external "important" event (true).
56    pub(super) event_queue_recv: UnboundedReceiver<(SubgraphId, bool)>,
57    /// If external events or data can justify starting the next tick.
58    pub(super) can_start_tick: bool,
59    /// If the events have been received for this tick.
60    pub(super) events_received_tick: bool,
61
62    // TODO(mingwei): as long as this is here, it's impossible to know when all work is done.
63    // Second field (bool) is for if the event is an external "important" event (true).
64    pub(super) event_queue_send: UnboundedSender<(SubgraphId, bool)>,
65
66    /// If the current subgraph wants to reschedule the current loop block (in the current tick).
67    pub(super) reschedule_loop_block: Cell<bool>,
68    pub(super) allow_another_iteration: Cell<bool>,
69
70    pub(super) current_tick: TickInstant,
71    pub(super) current_stratum: usize,
72
73    pub(super) current_tick_start: SystemTime,
74    pub(super) is_first_run_this_tick: bool,
75    pub(super) loop_iter_count: usize,
76
77    /// Depth of loop (zero for top-level).
78    pub(super) loop_depth: SlotVec<LoopTag, usize>,
79    /// For each loop, state which needs to be reset between loop executions.
80    loop_states: SecondarySlotVec<LoopTag, Vec<StateId>>,
81    /// Used to differentiate between loop executions. Incremented at the start of each loop execution.
82    pub(super) loop_nonce: usize,
83
84    /// For each subgraph, state which needs to be reset between executions.
85    subgraph_states: SecondarySlotVec<SubgraphTag, Vec<StateId>>,
86
87    /// The SubgraphId of the currently running operator. When this context is
88    /// not being forwarded to a running operator, this field is meaningless.
89    pub(super) subgraph_id: SubgraphId,
90
91    tasks_to_spawn: Vec<Pin<Box<dyn Future<Output = ()> + 'static>>>,
92    /// Join handles for spawned tasks.
93    task_join_handles: Vec<JoinHandle<()>>,
94}

146    /// Returns a `Waker` for interacting with async Rust.
147    /// Waker events are considered to be extenral.
148    pub fn waker(&self) -> std::task::Waker {
149        use std::sync::Arc;
150        use std::task::Wake;
151
152        struct ContextWaker {
153            subgraph_id: SubgraphId,
154            event_queue_send: UnboundedSender<(SubgraphId, bool)>,
155        }
156        impl Wake for ContextWaker {
157            fn wake(self: Arc<Self>) {
158                self.wake_by_ref();
159            }
160
161            fn wake_by_ref(self: &Arc<Self>) {
162                let _recv_closed_error = self.event_queue_send.send((self.subgraph_id, true));
163            }
164        }
165
166        let context_waker = ContextWaker {
167            subgraph_id: self.subgraph_id,
168            event_queue_send: self.event_queue_send.clone(),
169        };
170        std::task::Waker::from(Arc::new(context_waker))
171    }

173    /// Returns a shared reference to the state.
174    ///
175    /// # Safety
176    /// `StateHandle<T>` must be from _this_ instance, created via [`Self::add_state`].
177    pub unsafe fn state_ref_unchecked<T>(&self, handle: StateHandle<T>) -> &'_ T
178    where
179        T: Any,
180    {
181        let state = self

241    /// Sets a hook to modify the state at the end of each tick, using the supplied closure.
242    pub fn set_state_lifespan_hook<T>(
243        &mut self,
244        handle: StateHandle<T>,
245        lifespan: StateLifespan,
246        mut hook_fn: impl 'static + FnMut(&mut T),
247    ) where
248        T: Any,
249    {
250        let state_data = self
251            .states
252            .get_mut(handle.state_id)
253            .expect("Failed to find state with given handle.");
254        state_data.lifespan_hook_fn = Some(Box::new(move |state| {
255            (hook_fn)(state.downcast_mut::<T>().unwrap());
256        }));
257        state_data.lifespan = Some(lifespan);
258
259        match lifespan {
260            StateLifespan::Subgraph(key) => {
261                self.subgraph_states
262                    .get_or_insert_with(key, Vec::new)
263                    .push(handle.state_id);
264            }
265            StateLifespan::Loop(loop_id) => {
266                self.loop_states
267                    .get_or_insert_with(loop_id, Vec::new)
268                    .push(handle.state_id);
269            }
270            StateLifespan::Tick => {
271                // Already included in `run_state_hooks_tick`.
272            }
273            StateLifespan::Static => {
274                // Never resets.
275            }
276        }
277    }

279    /// Prepares an async task to be launched by [`Self::spawn_tasks`].
280    pub fn request_task<Fut>(&mut self, future: Fut)
281    where
282        Fut: Future<Output = ()> + 'static,
283    {
284        self.tasks_to_spawn.push(Box::pin(future));

287    /// Launches all tasks requested with [`Self::request_task`] on the internal Tokio executor.
288    pub fn spawn_tasks(&mut self) {
289        for task in self.tasks_to_spawn.drain(..) {
290            self.task_join_handles.push(tokio::task::spawn_local(task));
291        }
292    }
293
294    /// Aborts all tasks spawned with [`Self::spawn_tasks`].
295    pub fn abort_tasks(&mut self) {
296        for task in self.task_join_handles.drain(..) {
297            task.abort();
298        }
299    }
300
301    /// Waits for all tasks spawned with [`Self::spawn_tasks`] to complete.
302    ///
303    /// Will probably just hang.
304    pub async fn join_tasks(&mut self) {
305        futures::future::join_all(self.task_join_handles.drain(..)).await;
306    }

353/// Internal APIs.
354impl Context {
355    /// Makes sure stratum STRATUM is initialized.
356    pub(super) fn init_stratum(&mut self, stratum: usize) {
357        if self.stratum_queues.len() <= stratum {
358            self.stratum_queues
359                .resize_with(stratum + 1, Default::default);
360        }
361    }
362
363    /// Call this at the end of a tick,
364    pub(super) fn run_state_hooks_tick(&mut self) {
365        tracing::trace!("Running state hooks for tick.");
366        for state_data in self.states.values_mut() {
367            let StateData {
368                state,
369                lifespan_hook_fn: Some(lifespan_hook_fn),
370                lifespan: Some(StateLifespan::Tick),
371            } = state_data
372            else {
373                continue;
374            };
375            (lifespan_hook_fn)(Box::deref_mut(state));
376        }
377    }
378
379    pub(super) fn run_state_hooks_subgraph(&mut self, subgraph_id: SubgraphId) {
380        tracing::trace!("Running state hooks for subgraph.");
381        for state_id in self.subgraph_states.get(subgraph_id).into_iter().flatten() {
382            let StateData {
383                state,
384                lifespan_hook_fn,
385                lifespan: _,
386            } = self
387                .states
388                .get_mut(*state_id)
389                .expect("Failed to find state with given ID.");
390
391            if let Some(lifespan_hook_fn) = lifespan_hook_fn {
392                (lifespan_hook_fn)(Box::deref_mut(state));
393            }
394        }
395    }
396
397    // Run the state hooks for each state in the loop.
398    // Call at the end of each loop execution.
399    pub(super) fn run_state_hooks_loop(&mut self, loop_id: LoopId) {
400        tracing::trace!(
401            loop_id = loop_id.to_string(),
402            "Running state hooks for loop."
403        );
404        for state_id in self.loop_states.get(loop_id).into_iter().flatten() {
405            let StateData {
406                state,
407                lifespan_hook_fn,
408                lifespan: _,
409            } = self
410                .states
411                .get_mut(*state_id)
412                .expect("Failed to find state with given ID.");
413
414            if let Some(lifespan_hook_fn) = lifespan_hook_fn {
415                (lifespan_hook_fn)(Box::deref_mut(state));
416            }
417        }
418    }

430/// Coordinates waking between [`InlineContext`] (inside the tick closure) and [`InlineDfir`]
431/// (the external runner). Shared via `Arc` between both.
432///
433/// When external data arrives (e.g., a tokio stream receives a message), the [`InlineContext::waker`]
434/// fires, which sets `can_start_tick` and wakes the [`InlineDfir::run`] task so it starts a new tick.
435/// Implements [`Wake`] directly so it can be used as a `Waker` without an extra wrapper.
436#[doc(hidden)]
437pub struct InlineWakeState {
438    /// Set to `true` when external data arrives, signaling that a new tick should run.
439    /// Checked by [`InlineDfir::run_tick`] and [`InlineDfir::run_available`].
440    can_start_tick: std::sync::atomic::AtomicBool,
441    /// Wakes the [`InlineDfir::run`] task from its idle `poll_fn` sleep.
442    task_waker: futures::task::AtomicWaker,
443}
444
445impl Default for InlineWakeState {
446    fn default() -> Self {
447        Self {
448            can_start_tick: std::sync::atomic::AtomicBool::new(false),
449            task_waker: futures::task::AtomicWaker::new(),
450        }
451    }
452}
453
454impl Wake for InlineWakeState {
455    fn wake(self: std::sync::Arc<Self>) {
456        self.wake_by_ref();
457    }
458
459    fn wake_by_ref(self: &std::sync::Arc<Self>) {
460        self.can_start_tick.store(true, Ordering::Relaxed);
461        self.task_waker.wake();
462    }
463}

588    /// Runs end-of-tick state hooks and increments the tick counter.
589    /// Called by the generated tick closure at the end of each tick.
590    #[doc(hidden)]
591    pub fn __end_tick(&mut self) {
592        for state_data in self.states.values_mut() {
593            let StateData {
594                state,
595                lifespan_hook_fn: Some(lifespan_hook_fn),
596                lifespan: Some(StateLifespan::Tick),

608/// A wrapper around an inline-codegen tick closure that provides [`Self::run`],
609/// [`Self::run_available`], and [`Self::run_tick`] methods — mirroring the [`super::graph::Dfir`]
610/// API.
611///
612/// # Design
613///
614/// The inline codegen generates an `async move ||` closure that captures all dataflow state
615/// (operator accumulators, handoff buffers, source iterators) and runs one tick per call.
616/// `InlineDfir` wraps this closure and adds tick lifecycle and idle/wake coordination.
617///
618/// We use a single opaque closure rather than generating a bespoke struct per dataflow because:
619/// - The closure naturally captures exactly the state it needs with correct lifetimes
620/// - No codegen needed for struct definitions, field accessors, or initialization
621/// - Rust's async closure machinery handles the complex state machine (suspend/resume across
622///   `.await` points) that would be very difficult to replicate in a generated struct
623///
624/// The `Tick` type parameter is bounded by [`TickClosure`] (not `AsyncFnMut` directly) to
625/// support type erasure via [`TickClosureErased`] / [`InlineDfirErased`] for heterogeneous
626/// collections (e.g., the sim runtime storing multiple locations in a `Vec`). The concrete
627/// (non-erased) path used by trybuild and embedded has zero overhead.
628#[doc(hidden)]
629pub struct InlineDfir<Tick> {
630    tick_closure: Tick,
631    wake_state: std::sync::Arc<InlineWakeState>,
632
633    /// Shared tick counter, updated by [`InlineContext::__end_tick`] inside the closure.
634    current_tick: Rc<Cell<TickInstant>>,
635
636    /// Live-updating DFIR runtime metrics via interior mutability.
637    metrics: Rc<DfirMetrics>,
638
639    #[cfg(feature = "meta")]
640    /// See [`Self::meta_graph()`].
641    meta_graph: Option<DfirGraph>,
642
643    #[cfg(feature = "meta")]
644    /// See [`Self::diagnostics()`].
645    diagnostics: Option<Vec<Diagnostic<SerdeSpan>>>,
646}

648/// Trait for tick closures — abstracts over both concrete async closures
649/// and type-erased boxed versions ([`TickClosureErased`]).
650#[doc(hidden)]
651pub trait TickClosure {
652    /// Call the tick closure. Returns `true` if any subgraph received input data.
653    fn call_tick(&mut self) -> impl Future<Output = bool>;
654}

656impl<F: AsyncFnMut() -> bool> TickClosure for F {
657    fn call_tick(&mut self) -> impl Future<Output = bool> {
658        self()
659    }
660}

662/// Type-erased tick function for use in heterogeneous collections (e.g., the sim runtime).
663#[doc(hidden)]
664pub struct TickClosureErased(Box<dyn TickClosureErasedInner>);
665
666/// Object-safe inner trait for [`TickClosureErased`]. Needed because `AsyncFnMut` is not
667/// object-safe (GAT return type), but a trait with `&mut self -> Pin<Box<dyn Future + '_>>`
668/// is — the returned future borrows from the trait object which owns the closure.
669trait TickClosureErasedInner {
670    fn call_tick(&mut self) -> Pin<Box<dyn Future<Output = bool> + '_>>;
671}
672
673impl<F: AsyncFnMut() -> bool> TickClosureErasedInner for F {
674    fn call_tick(&mut self) -> Pin<Box<dyn Future<Output = bool> + '_>> {
675        Box::pin(self())
676    }
677}
678
679impl TickClosure for TickClosureErased {
680    fn call_tick(&mut self) -> impl Future<Output = bool> {
681        self.0.call_tick()
682    }
683}

768impl<Tick: TickClosure> InlineDfir<Tick> {
769    /// Run a single tick. Returns `true` if any subgraph received input data.
770    ///
771    /// Checks both handoff buffers (via `work_done` flag set in generated recv port code)
772    /// and external events (via `can_start_tick` set by wakers/schedule_subgraph).
773    pub async fn run_tick(&mut self) -> bool {
774        let had_external = self
775            .wake_state
776            .can_start_tick
777            .swap(false, Ordering::Relaxed);
778        let tick_had_work = self.tick_closure.call_tick().await;
779        had_external || tick_had_work || self.wake_state.can_start_tick.load(Ordering::Relaxed)
780    }
781
782    /// Run a single tick synchronously. Panics if the tick yields (async suspension).
783    /// Returns `true` if work was done (see [`Self::run_tick`]).
784    pub fn run_tick_sync(&mut self) -> bool {
785        let mut fut = std::pin::pin!(self.run_tick());
786        let mut ctx = std::task::Context::from_waker(std::task::Waker::noop());
787        match fut.as_mut().poll(&mut ctx) {
788            std::task::Poll::Ready(result) => result,
789            std::task::Poll::Pending => {
790                panic!("InlineDfir::run_tick_sync: tick yielded asynchronously.")
791            }
792        }
793    }
794
795    /// Run ticks as long as work is available, then return.
796    pub async fn run_available(&mut self) {
797        // Always run at least one tick.
798        self.wake_state
799            .can_start_tick
800            .store(false, Ordering::Relaxed);
801        loop {
802            self.run_tick().await;
803            let can_start_tick = self
804                .wake_state
805                .can_start_tick
806                .swap(false, Ordering::Relaxed);
807            if !can_start_tick {
808                break;
809            }
810            // Yield between each tick to receive more events.
811            tokio::task::yield_now().await;
812        }
813    }
814
815    /// [`Self::run_available`] but panics if any tick yields asynchronously.
816    pub fn run_available_sync(&mut self) {
817        self.wake_state
818            .can_start_tick
819            .store(false, Ordering::Relaxed);
820        loop {
821            self.run_tick_sync();
822            let can_start_tick = self
823                .wake_state
824                .can_start_tick
825                .swap(false, Ordering::Relaxed);
826            if !can_start_tick {
827                break;
828            }
829        }
830    }
831
832    /// Run forever, processing ticks when work is available and yielding when idle.
833    pub async fn run(&mut self) -> crate::Never {
834        loop {
835            self.run_available().await;
836            // Wait for an external event to wake us.
837            std::future::poll_fn(|cx| {
838                // Register waker first to avoid race: if an event fires between
839                // the check and the register, the waker is already in place.
840                self.wake_state.task_waker.register(cx.waker());
841                if self.wake_state.can_start_tick.load(Ordering::Relaxed) {
842                    std::task::Poll::Ready(())
843                } else {
844                    std::task::Poll::Pending
845                }
846            })
847            .await;
848        }
849    }
850}

852impl<Tick: AsyncFnMut() -> bool + 'static> InlineDfir<Tick> {
853    /// Type-erase the tick closure for use in heterogeneous collections.
854    ///
855    /// Wraps the concrete async closure in [`TickClosureErased`], which boxes the future
856    /// returned by each tick call. This adds one heap allocation per tick, but enables
857    /// storing multiple `InlineDfir`s with different closure types in a single `Vec`.
858    ///
859    /// Only needed for the sim runtime path. The trybuild and embedded paths keep the
860    /// concrete type and pay no erasure cost.
861    pub fn into_erased(self) -> InlineDfirErased {
862        InlineDfir {
863            tick_closure: TickClosureErased(Box::new(self.tick_closure)),
864            wake_state: self.wake_state,
865            current_tick: self.current_tick,
866            metrics: self.metrics,
867            #[cfg(feature = "meta")]
868            meta_graph: self.meta_graph,
869            #[cfg(feature = "meta")]
870            diagnostics: self.diagnostics,
871        }
872    }
873}
