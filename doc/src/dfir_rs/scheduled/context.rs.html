27/// Internal state storage for operator accumulators.
28struct StateData {
29    state: Box<dyn Any>,
30    lifespan_hook_fn: Option<LifespanResetFn>,
31    /// `None` for static.
32    lifespan: Option<StateLifespan>,
33}
34type LifespanResetFn = Box<dyn FnMut(&mut dyn Any)>;

36/// Coordinates waking between [`Context`] (inside the tick closure) and [`Dfir`]
37/// (the external runner). Shared via `Arc` between both.
38///
39/// When external data arrives (e.g., a tokio stream receives a message), the [`Context::waker`]
40/// fires, which sets `can_start_tick` and wakes the [`Dfir::run`](Dfir::run) task so it starts a new tick.
41/// Implements [`Wake`] directly so it can be used as a `Waker` without an extra wrapper.
42#[doc(hidden)]
43pub struct WakeState {
44    /// Set to `true` when external data arrives, signaling that a new tick should run.
45    /// Checked by [`Dfir::run_tick`](Dfir::run_tick) and [`Dfir::run_available`](Dfir::run_available).
46    can_start_tick: std::sync::atomic::AtomicBool,
47    /// Wakes the [`Dfir::run`](Dfir::run) task from its idle `poll_fn` sleep.
48    task_waker: futures::task::AtomicWaker,
49}
50
51impl Default for WakeState {
52    fn default() -> Self {
53        Self {
54            can_start_tick: std::sync::atomic::AtomicBool::new(false),
55            task_waker: futures::task::AtomicWaker::new(),
56        }
57    }
58}
59
60impl Wake for WakeState {
61    fn wake(self: Arc<Self>) {
62        self.wake_by_ref();
63    }
64
65    fn wake_by_ref(self: &Arc<Self>) {
66        self.can_start_tick.store(true, Ordering::Relaxed);
67        self.task_waker.wake();
68    }
69}

71/// A lightweight context for inline codegen that avoids the overhead of the full
72/// [`Context`] (no tokio channels, no scheduler queues, no loop machinery).
73///
74/// Exposes the same method names that operator-generated code calls on both
75/// `df` (for prologues: `add_state`, `set_state_lifespan_hook`) and
76/// `context` (for iterators: `state_ref_unchecked`, `is_first_run_this_tick`, etc.).
77#[doc(hidden)]
78#[derive(Default)]
79pub struct Context {
80    /// Storage for the operator-facing State API.
81    states: SlotVec<StateTag, StateData>,
82    /// Counter for number of ticks run.
83    current_tick: TickInstant,
84    /// Coordinates waking between [`Context`] (inside the tick closure) and [`Dfir`]
85    /// (the external runner). Shared via `Arc` between both. Implements [`Wake`].
86    wake_state: Arc<WakeState>,
87    /// Live-updating DFIR runtime metrics via interior mutability.
88    metrics: Rc<DfirMetrics>,
89}
90
91impl Context {
92    /// Create a new inline context with shared wake state and metrics.
93    pub fn new(wake_state: Arc<WakeState>, metrics: Rc<DfirMetrics>) -> Self {
94        Self {
95            states: SlotVec::new(),
96            current_tick: TickInstant::default(),
97            wake_state,
98            metrics,
99        }
100    }
101
102    // --- Methods called as `df.xxx()` in operator prologues ---
103
104    /// Adds state and returns a handle.
105    pub fn add_state<T>(&mut self, state: T) -> StateHandle<T>
106    where
107        T: Any,
108    {
109        let state_data = StateData {
110            state: Box::new(state),
111            lifespan_hook_fn: None,
112            lifespan: None,
113        };
114        let state_id = self.states.insert(state_data);
115        StateHandle {
116            state_id,
117            _phantom: PhantomData,
118        }
119    }
120
121    /// Sets a hook to modify state at the end of each tick.
122    pub fn set_state_lifespan_hook<T>(
123        &mut self,
124        handle: StateHandle<T>,
125        _lifespan: StateLifespan,
126        mut hook_fn: impl 'static + FnMut(&mut T),
127    ) where
128        T: Any,
129    {
130        let state_data = self
131            .states
132            .get_mut(handle.state_id)
133            .expect("Failed to find state with given handle.");
134        state_data.lifespan_hook_fn = Some(Box::new(move |state| {
135            (hook_fn)(state.downcast_mut::<T>().unwrap());
136        }));
137        state_data.lifespan = Some(_lifespan);
138    }
139
140    // --- Methods called as `context.xxx()` in operator iterators ---
141
142    /// Returns a shared reference to the state.
143    ///
144    /// # Safety
145    /// `StateHandle<T>` must be from _this_ instance.
146    pub unsafe fn state_ref_unchecked<T>(&self, handle: StateHandle<T>) -> &'_ T
147    where
148        T: Any,
149    {
150        let state = self
151            .states
152            .get(handle.state_id)
153            .expect("Failed to find state with given handle.")
154            .state
155            .as_ref();
156        debug_assert!(state.is::<T>());
157        unsafe { &*(state as *const dyn Any as *const T) }
158    }
159
160    /// Always returns `true` in inline mode. The inline codegen runs the entire DAG
161    /// once per tick with no re-execution, so every subgraph is always on its first
162    /// (and only) run within each tick.
163    pub fn is_first_run_this_tick(&self) -> bool {
164        true
165    }
166
167    /// Gets the current tick count.
168    pub fn current_tick(&self) -> TickInstant {
169        self.current_tick
170    }
171
172    /// Returns a reference to the runtime metrics.
173    pub fn metrics(&self) -> &Rc<DfirMetrics> {
174        &self.metrics
175    }
176
177    /// No-op: inline mode has no subgraph scheduling.
178    pub fn current_subgraph(&self) -> SubgraphId {
179        SubgraphId::from_raw(0)
180    }
181
182    /// In inline mode, every subgraph runs unconditionally each tick, so the `sg_id`
183    /// parameter is ignored. Only `is_external` matters: when `true`, it signals that
184    /// external data has arrived and a new tick should be started.
185    pub fn schedule_subgraph(&self, _sg_id: SubgraphId, is_external: bool) {
186        if is_external {
187            self.wake_state.wake_by_ref();
188        }
189    }
190
191    /// Returns a waker that signals external data has arrived.
192    pub fn waker(&self) -> std::task::Waker {
193        std::task::Waker::from(self.wake_state.clone())
194    }
195
196    /// Runs end-of-tick state hooks and increments the tick counter.
197    /// Called by the generated tick closure at the end of each tick.
198    #[doc(hidden)]
199    pub fn __end_tick(&mut self) {
200        for state_data in self.states.values_mut() {
201            let StateData {
202                state,
203                lifespan_hook_fn: Some(lifespan_hook_fn),
204                lifespan: Some(StateLifespan::Tick),
205            } = state_data
206            else {
207                continue;
208            };
209            (lifespan_hook_fn)(Box::deref_mut(state));
210        }
211        self.current_tick += crate::scheduled::ticks::TickDuration::SINGLE_TICK;
212    }
213}

215/// A wrapper around an inline-codegen tick closure that provides [`Self::run`],
216/// [`Self::run_available`], and [`Self::run_tick`] methods — mirroring the [`Dfir`](super::context::Dfir)
217/// API.
218///
219/// # Design
220///
221/// The inline codegen generates an `async move |df: &mut Context|` closure that captures
222/// dataflow-specific state (handoff buffers, source iterators) and receives the [`Context`]
223/// (operator accumulators, tick counter) by reference each tick. `Dfir` owns both the
224/// closure and the context, and coordinates tick lifecycle and idle/wake behavior.
225///
226/// We use a single opaque closure rather than generating a bespoke struct per dataflow because:
227/// - The closure naturally captures exactly the state it needs with correct lifetimes
228/// - No codegen needed for struct definitions, field accessors, or initialization
229/// - Rust's async closure machinery handles the complex state machine (suspend/resume across
230///   `.await` points) that would be very difficult to replicate in a generated struct
231///
232/// The `Tick` type parameter is bounded by [`TickClosure`] (not `AsyncFnMut` directly) to
233/// support type erasure via [`TickClosureErased`] / [`DfirErased`] for heterogeneous
234/// collections (e.g., the sim runtime storing multiple locations in a `Vec`). The concrete
235/// (non-erased) path used by trybuild and embedded has zero overhead.
236#[doc(hidden)]
237pub struct Dfir<Tick> {
238    /// Async closure which runs a single tick when called.
239    tick_closure: Tick,
240    /// Coordinates waking between [`Context`] (inside the tick closure) and [`Dfir`]
241    /// (the external runner). Shared via `Arc` between both. Implements [`Wake`].
242    wake_state: Arc<WakeState>,
243    /// The inline context, owned by `Dfir` and passed to the tick closure by reference.
244    context: Context,
245    /// See [`Self::meta_graph()`].
246    #[cfg(feature = "meta")]
247    meta_graph: Option<DfirGraph>,
248    /// See [`Self::diagnostics()`].
249    #[cfg(feature = "meta")]
250    diagnostics: Option<Vec<Diagnostic<SerdeSpan>>>,
251}
252
253/// Trait for tick closures — abstracts over both concrete async closures
254/// and type-erased boxed versions ([`TickClosureErased`]).
255///
256/// The `&mut Context` parameter is owned by [`Dfir`] and lent to the
257/// closure each tick, avoiding shared-ownership overhead for the context.
258#[doc(hidden)]
259pub trait TickClosure {
260    /// Call the tick closure. Returns `true` if any subgraph received input data.
261    fn call_tick<'a>(&'a mut self, ctx: &'a mut Context) -> impl Future<Output = bool> + 'a;
262}
263
264impl<F: for<'a> AsyncFnMut(&'a mut Context) -> bool> TickClosure for F {
265    fn call_tick<'a>(&'a mut self, ctx: &'a mut Context) -> impl Future<Output = bool> + 'a {
266        self(ctx)
267    }
268}

280/// Type-erased tick function for use in heterogeneous collections (e.g., the sim runtime).
281#[doc(hidden)]
282pub struct TickClosureErased(Box<dyn TickClosureErasedInner>);
283
284/// Object-safe inner trait for [`TickClosureErased`]. Needed because `AsyncFnMut` is not
285/// object-safe (GAT return type), but a trait with `&mut self -> Pin<Box<dyn Future + '_>>`
286/// is — the returned future borrows from the trait object which owns the closure.
287trait TickClosureErasedInner {
288    fn call_tick<'a>(
289        &'a mut self,
290        ctx: &'a mut Context,
291    ) -> Pin<Box<dyn Future<Output = bool> + 'a>>;
292}
293
294impl<F: for<'a> AsyncFnMut(&'a mut Context) -> bool> TickClosureErasedInner for F {
295    fn call_tick<'a>(
296        &'a mut self,
297        ctx: &'a mut Context,
298    ) -> Pin<Box<dyn Future<Output = bool> + 'a>> {
299        Box::pin(self(ctx))
300    }
301}
302
303impl TickClosure for TickClosureErased {
304    fn call_tick<'a>(&'a mut self, ctx: &'a mut Context) -> impl Future<Output = bool> + 'a {
305        self.0.call_tick(ctx)
306    }
307}

313impl<Tick: TickClosure> Dfir<Tick> {
314    /// Create a new `Dfir` from a tick closure, inline context,
315    /// and meta graph / diagnostics JSON strings.
316    #[doc(hidden)]
317    pub fn new(
318        tick_closure: Tick,
319        context: Context,
320        meta_graph_json: Option<&str>,
321        diagnostics_json: Option<&str>,
322    ) -> Self {
323        #[cfg(not(feature = "meta"))]
324        let _ = (meta_graph_json, diagnostics_json);
325        Self {
326            tick_closure,
327            wake_state: context.wake_state.clone(),
328            context,
329            #[cfg(feature = "meta")]
330            meta_graph: meta_graph_json.map(|json| {
331                let mut meta_graph: DfirGraph =
332                    serde_json::from_str(json).expect("Failed to deserialize graph.");
333                let mut op_inst_diagnostics = Diagnostics::new();
334                meta_graph.insert_node_op_insts_all(&mut op_inst_diagnostics);
335                assert!(
336                    op_inst_diagnostics.is_empty(),
337                    "Expected no diagnostics, got: {:#?}",
338                    op_inst_diagnostics
339                );
340                meta_graph
341            }),
342            #[cfg(feature = "meta")]
343            diagnostics: diagnostics_json.map(|json| {
344                serde_json::from_str(json).expect("Failed to deserialize diagnostics.")
345            }),
346        }
347    }
348
349    /// Return a handle to the meta graph, if set.
350    #[cfg(feature = "meta")]
351    #[cfg_attr(docsrs, doc(cfg(feature = "meta")))]
352    pub fn meta_graph(&self) -> Option<&DfirGraph> {
353        self.meta_graph.as_ref()
354    }
355
356    /// Returns any diagnostics generated by the surface syntax macro.
357    #[cfg(feature = "meta")]
358    #[cfg_attr(docsrs, doc(cfg(feature = "meta")))]
359    pub fn diagnostics(&self) -> Option<&[Diagnostic<SerdeSpan>]> {
360        self.diagnostics.as_deref()
361    }
362
363    /// Returns a reference-counted handle to the continually-updated runtime metrics for this DFIR instance.
364    pub fn metrics(&self) -> Rc<DfirMetrics> {
365        Rc::clone(self.context.metrics())
366    }
367
368    /// Gets the current tick (local time) count.
369    pub fn current_tick(&self) -> TickInstant {
370        self.context.current_tick()
371    }
372
373    /// Returns a [`DfirMetricsIntervals`] handle where each call to
374    /// [`DfirMetricsIntervals::take_interval`] ends the current interval and returns its metrics.
375    ///
376    /// The first call to `take_interval` returns metrics since this DFIR instance was created. Each subsequent call to
377    /// `take_interval` returns metrics since the previous call.
378    ///
379    /// Cloning the handle "forks" it from the original, as afterwards each interval may return different metrics
380    /// depending on when exactly `take_interval` is called.
381    pub fn metrics_intervals(&self) -> DfirMetricsIntervals {
382        DfirMetricsIntervals {
383            curr: self.metrics(),
384            prev: None,
385        }
386    }

389impl<Tick: TickClosure> Dfir<Tick> {
390    /// Run a single tick. Returns `true` if any subgraph received input data.
391    ///
392    /// Checks both handoff buffers (via `work_done` flag set in generated recv port code)
393    /// and external events (via `can_start_tick` set by wakers/schedule_subgraph).
394    pub async fn run_tick(&mut self) -> bool {
395        let had_external = self
396            .wake_state
397            .can_start_tick
398            .swap(false, Ordering::Relaxed);
399        let tick_had_work = self.tick_closure.call_tick(&mut self.context).await;
400        had_external || tick_had_work || self.wake_state.can_start_tick.load(Ordering::Relaxed)
401    }
402
403    /// Run a single tick synchronously. Panics if the tick yields (async suspension).
404    /// Returns `true` if work was done (see [`Self::run_tick`]).
405    pub fn run_tick_sync(&mut self) -> bool {
406        let mut fut = std::pin::pin!(self.run_tick());
407        let mut ctx = std::task::Context::from_waker(std::task::Waker::noop());
408        match fut.as_mut().poll(&mut ctx) {
409            std::task::Poll::Ready(result) => result,
410            std::task::Poll::Pending => {
411                panic!("Dfir::run_tick_sync: tick yielded asynchronously.")
412            }
413        }
414    }
415
416    /// Run ticks as long as work is available, then return.
417    pub async fn run_available(&mut self) {
418        // Always run at least one tick.
419        self.wake_state
420            .can_start_tick
421            .store(false, Ordering::Relaxed);
422        loop {
423            self.run_tick().await;
424            let can_start_tick = self
425                .wake_state
426                .can_start_tick
427                .swap(false, Ordering::Relaxed);
428            if !can_start_tick {
429                break;
430            }
431            // Yield between each tick to receive more events.
432            tokio::task::yield_now().await;
433        }
434    }
435
436    /// [`Self::run_available`] but panics if any tick yields asynchronously.
437    pub fn run_available_sync(&mut self) {
438        self.wake_state
439            .can_start_tick
440            .store(false, Ordering::Relaxed);
441        loop {
442            self.run_tick_sync();
443            let can_start_tick = self
444                .wake_state
445                .can_start_tick
446                .swap(false, Ordering::Relaxed);
447            if !can_start_tick {
448                break;
449            }
450        }
451    }
452
453    /// Run forever, processing ticks when work is available and yielding when idle.
454    pub async fn run(&mut self) -> crate::Never {
455        loop {
456            self.run_available().await;
457            // Wait for an external event to wake us.
458            std::future::poll_fn(|cx| {
459                // Register waker first to avoid race: if an event fires between
460                // the check and the register, the waker is already in place.
461                self.wake_state.task_waker.register(cx.waker());
462                if self.wake_state.can_start_tick.load(Ordering::Relaxed) {
463                    std::task::Poll::Ready(())
464                } else {
465                    std::task::Poll::Pending
466                }
467            })
468            .await;
469        }
470    }
471}

473impl<Tick: 'static + for<'a> AsyncFnMut(&'a mut Context) -> bool> Dfir<Tick> {
474    /// Type-erase the tick closure for use in heterogeneous collections.
475    ///
476    /// Wraps the concrete async closure in [`TickClosureErased`], which boxes the future
477    /// returned by each tick call. This adds one heap allocation per tick, but enables
478    /// storing multiple `Dfir`s with different closure types in a single `Vec`.
479    ///
480    /// Only needed for the sim runtime path. The trybuild and embedded paths keep the
481    /// concrete type and pay no erasure cost.
482    pub fn into_erased(self) -> DfirErased {
483        Dfir {
484            tick_closure: TickClosureErased(Box::new(self.tick_closure)),
485            wake_state: self.wake_state,
486            context: self.context,
487            #[cfg(feature = "meta")]
488            meta_graph: self.meta_graph,
489            #[cfg(feature = "meta")]
490            diagnostics: self.diagnostics,
491        }
492    }
