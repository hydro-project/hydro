7pub fn partition<'a, F: Fn((MemberId<()>, String)) -> (MemberId<()>, String) + 'a>(
8    cluster1: Cluster<'a, ()>,
9    cluster2: Cluster<'a, ()>,
10    dist_policy: impl IntoQuotedMut<'a, F, Cluster<'a, ()>>,
11) -> (Cluster<'a, ()>, Cluster<'a, ()>) {
12    cluster1
13        .source_iter(q!(vec!(CLUSTER_SELF_ID)))
14        .map(q!(move |id| (
15            MemberId::<()>::from_raw(id.raw_id),
16            format!("Hello from {}", id.raw_id)
17        )))
18        .send_partitioned(&cluster2, dist_policy)
19        .assume_ordering(nondet!(/** testing, order does not matter */))
20        .for_each(q!(move |message| println!(
21            "My self id is {}, my message is {:?}",
22            CLUSTER_SELF_ID.raw_id, message
23        )));
24    (cluster1, cluster2)
25}

27pub fn decouple_cluster<'a>(flow: &FlowBuilder<'a>) -> (Cluster<'a, ()>, Cluster<'a, ()>) {
28    let cluster1 = flow.cluster();
29    let cluster2 = flow.cluster();
30    cluster1
31        .source_iter(q!(vec!(CLUSTER_SELF_ID)))
32        // .for_each(q!(|message| println!("hey, {}", message)))
33        .inspect(q!(|message| println!("Cluster1 node sending message: {}", message)))
34        .decouple_cluster(&cluster2)
35        .for_each(q!(move |message| println!(
36            "My self id is {}, my message is {}",
37            CLUSTER_SELF_ID, message
38        )));
39    (cluster1, cluster2)
40}

52pub fn simple_cluster<'a>(flow: &FlowBuilder<'a>) -> (Process<'a, ()>, Cluster<'a, ()>) {
53    let process = flow.process();
54    let cluster = flow.cluster();
55
56    let numbers = process.source_iter(q!(0..5));
57    let ids = process
58        .source_cluster_members(&cluster)
59        .entries()
60        .filter_map(q!(|(i, e)| match e {
61            MembershipEvent::Joined => Some(i),
62            MembershipEvent::Left => None,
63        }));
64
65    ids.cross_product(numbers)
66        .map(q!(|(id, n)| (id, (id, n))))
67        .demux_bincode(&cluster)
68        .inspect(q!(move |n| println!(
69            "cluster received: {:?} (self cluster id: {})",
70            n, CLUSTER_SELF_ID
71        )))
72        .send_bincode(&process)
73        .entries()
74        .assume_ordering(nondet!(/** testing, order does not matter */))
75        .for_each(q!(|(id, d)| println!("node received: ({}, {:?})", id, d)));
76
77    (process, cluster)
78}

101        let (node, cluster) = super::simple_cluster(&builder);
102
103        let nodes = builder
104            .with_default_optimize()
105            .with_process(&node, deployment.Localhost())
106            .with_cluster(&cluster, (0..2).map(|_| deployment.Localhost()))
107            .deploy(&mut deployment);
108
109        deployment.deploy().await.unwrap();
110
111        let mut node_stdout = nodes.get_process(&node).stdout().await;
112        let cluster_stdouts = futures::future::join_all(
113            nodes
114                .get_cluster(&cluster)
115                .members()
116                .iter()
117                .map(|node| node.stdout()),

119        .await;

121        deployment.start().await.unwrap();
122
123        for (i, mut stdout) in cluster_stdouts.into_iter().enumerate() {
124            for j in 0..5 {
125                assert_eq!(
126                    stdout.recv().await.unwrap(),
127                    format!(
128                        "cluster received: (MemberId::<()>({}), {}) (self cluster id: MemberId::<()>({}))",
129                        i, j, i
130                    )
131                );
132            }
133        }
134
135        let mut node_outs = vec![];
136        for _i in 0..10 {
137            node_outs.push(node_stdout.recv().await.unwrap());
138        }
139        node_outs.sort();
140
141        for (i, n) in node_outs.into_iter().enumerate() {
142            assert_eq!(
143                n,
144                format!(
145                    "node received: (MemberId::<()>({}), (MemberId::<()>({}), {}))",
