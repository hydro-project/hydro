8#[expect(clippy::type_complexity, reason = "Paxos internals")]
9pub fn sequence_payloads<'a, K: KvKey, V: KvValue, L: Location<'a> + NoTick>(
10    replica_tick: &Tick<L>,
11    p_to_replicas: Stream<SequencedKv<K, V>, L, Unbounded, NoOrder>,
12) -> (
13    Stream<SequencedKv<K, V>, Tick<L>, Bounded>,
14    TickCycleHandle<'a, Singleton<usize, Tick<L>, Bounded>>,
15) {
16    let (r_buffered_payloads_complete_cycle, r_buffered_payloads) =
17        replica_tick.cycle::<Stream<SequencedKv<K, V>, Tick<L>, Bounded>>();
18    // p_to_replicas.inspect(q!(|payload: ReplicaPayload| println!("Replica received payload: {:?}", payload)));
19    let r_sorted_payloads = p_to_replicas.batch(replica_tick, nondet!(
20            /// because we fill slots one-by-one, we can safely batch
21            /// because non-determinism is resolved when we sort by slots
22        ))
23        .chain(r_buffered_payloads) // Combine with all payloads that we've received and not processed yet
24        .sort();
25    // Create a cycle since we'll use this seq before we define it
26    let (r_next_slot_complete_cycle, r_next_slot) =
27        replica_tick.cycle_with_initial(replica_tick.singleton(q!(0)));
28    // Find highest the sequence number of any payload that can be processed in this tick. This is the payload right before a hole.
29    let r_next_slot_after_processing_payloads =
30        r_sorted_payloads.clone().cross_singleton(r_next_slot).fold(
31            q!(|| 0),
32            q!(|new_next_slot, (sorted_payload, next_slot)| {
33                if sorted_payload.seq == std::cmp::max(*new_next_slot, next_slot) {
34                    *new_next_slot = sorted_payload.seq + 1;
35                }
36            }),
37        );
38    // Find all payloads that can and cannot be processed in this tick.
39    let r_processable_payloads = r_sorted_payloads
40        .clone()
41        .cross_singleton(r_next_slot_after_processing_payloads.clone())
42        .filter(q!(
43            |(sorted_payload, highest_seq)| sorted_payload.seq < *highest_seq
44        ))
45        .map(q!(|(sorted_payload, _)| { sorted_payload }));
46    let r_new_non_processable_payloads = r_sorted_payloads
47        .cross_singleton(r_next_slot_after_processing_payloads)
48        .filter(q!(
49            |(sorted_payload, highest_seq)| sorted_payload.seq > *highest_seq
50        ))
51        .map(q!(|(sorted_payload, _)| { sorted_payload }));
52    // Save these, we can process them once the hole has been filled
53    r_buffered_payloads_complete_cycle.complete_next_tick(r_new_non_processable_payloads);
54
55    (r_processable_payloads, r_next_slot_complete_cycle)
56}

58#[cfg(test)]
59mod tests {
60    use hydro_lang::prelude::*;
61
62    use super::super::SequencedKv;
63    use super::*;
64
65    #[test]
66    fn sequence_payloads_sequences_all() {
67        let mut flow = FlowBuilder::new();
68        let node = flow.process::<()>();
69        let tick = node.tick();
70
71        let (in_send, input_payloads) = node.sim_input();
72        let (sequenced, complete_next_slot) = sequence_payloads(&tick, input_payloads);
73
74        complete_next_slot.complete_next_tick(sequenced.clone().across_ticks(|s| {
75            s.fold(
76                q!(|| 0),
77                q!(|next_slot, payload: SequencedKv<(), ()>| {
78                    *next_slot = payload.seq + 1;
79                }),
80            )
81        }));
82
83        let out_recv = sequenced.all_ticks().sim_output();
84
85        flow.sim().exhaustive(async || {
86            in_send.send_many_unordered([
87                SequencedKv { seq: 0, kv: None },
88                SequencedKv { seq: 1, kv: None },
89                SequencedKv { seq: 2, kv: None },
90                SequencedKv { seq: 3, kv: None },
91            ]);
92
93            out_recv
94                .assert_yields_only([
95                    SequencedKv { seq: 0, kv: None },
96                    SequencedKv { seq: 1, kv: None },
97                    SequencedKv { seq: 2, kv: None },
98                    SequencedKv { seq: 3, kv: None },
99                ])
100                .await;
101        });
102    }
