39impl<'a> PaxosLike<'a> for CoreCompartmentalizedPaxos<'a> {
40    type PaxosIn = Proposer;
41    type PaxosLog = Acceptor;
42    type PaxosOut = ProxyLeader;
43    type Ballot = Ballot;
44
45    fn payload_recipients(&self) -> &Cluster<'a, Self::PaxosIn> {
46        &self.proposers
47    }
48
49    fn log_stores(&self) -> &Cluster<'a, Self::PaxosLog> {
50        &self.acceptors
51    }
52
53    fn get_recipient_from_ballot<L: Location<'a>>(
54        ballot: Optional<Self::Ballot, L, Unbounded>,
55    ) -> Optional<MemberId<Self::PaxosIn>, L, Unbounded> {
56        ballot.map(q!(|ballot| ballot.proposer_id))
57    }
58
59    fn build<P: PaxosPayload>(
60        self,
61        with_ballot: impl FnOnce(
62            Stream<Ballot, Cluster<'a, Self::PaxosIn>, Unbounded>,
63        ) -> Stream<P, Cluster<'a, Self::PaxosIn>, Unbounded>,
64        a_checkpoint: Optional<usize, Cluster<'a, Acceptor>, Unbounded>,
65        nondet_leader: NonDet,
66        nondet_commit: NonDet,
67    ) -> Stream<(usize, Option<P>), Cluster<'a, Self::PaxosOut>, Unbounded, NoOrder> {
68        compartmentalized_paxos_core(
69            &self.proposers,
70            &self.proxy_leaders,
71            &self.acceptors,
72            a_checkpoint,
73            with_ballot,
74            self.config,
75            nondet_leader,
76            nondet_commit,
77        )
78        .1
79    }
80}

82/// Implements the Compartmentalized Paxos algorithm as described in "Scaling Replicated State Machines with Compartmentalization",
83/// which augments regular Paxos with a cluster of Proxy Leaders.
84///
85/// Proposers that wish to broadcast p2as to acceptors or collect p2bs from acceptors instead
86/// go through the Proxy Leaders, which offload networking. The slot is used to determine which Proxy Leader to offload to.
87/// Acceptors are arranged into a grid, where each row and column must have at least f+1 members.
88/// Rows represent "write quorums"; an entire row of acceptors must confirm a payload before it is committed.
89/// Columns represent "read quorums"; an entire column of acceptors must respond to a p1b before a proposer is elected the leader.
90/// Read and write quorums were introduced in "Flexible Paxos: Quorum Intersection Revisited".
91///
92/// Returns a stream of ballots, where new values are emitted when a new leader is elected,
93/// and a stream of sequenced payloads with an index and optional payload (in the case of
94/// holes in the log).
95///
96/// # Non-Determinism
97/// When the leader is stable, the algorithm will commit incoming payloads to the leader
98/// in deterministic order. However, when the leader is changing, payloads may be
99/// non-deterministically dropped. The stream of ballots is also non-deterministic because
100/// leaders are elected in a non-deterministic process.
101#[expect(
102    clippy::type_complexity,
103    clippy::too_many_arguments,
104    reason = "internal paxos code // TODO"
105)]
106pub fn compartmentalized_paxos_core<'a, P: PaxosPayload>(
107    proposers: &Cluster<'a, Proposer>,
108    proxy_leaders: &Cluster<'a, ProxyLeader>,
109    acceptors: &Cluster<'a, Acceptor>,
110    a_checkpoint: Optional<usize, Cluster<'a, Acceptor>, Unbounded>,
111    c_to_proposers: impl FnOnce(
112        Stream<Ballot, Cluster<'a, Proposer>, Unbounded>,
113    ) -> Stream<P, Cluster<'a, Proposer>, Unbounded>,
114    config: CompartmentalizedPaxosConfig,
115    nondet_leader: NonDet,
116    nondet_commit_leader_change: NonDet,
117) -> (
118    Stream<Ballot, Cluster<'a, Proposer>, Unbounded>,
119    Stream<(usize, Option<P>), Cluster<'a, ProxyLeader>, Unbounded, NoOrder>,
120) {
121    proposers
122        .source_iter(q!(["Proposers say hello"]))
123        .for_each(q!(|s| println!("{}", s)));
124
125    proxy_leaders
126        .source_iter(q!(["Proxy leaders say hello"]))
127        .for_each(q!(|s| println!("{}", s)));
128
129    acceptors
130        .source_iter(q!(["Acceptors say hello"]))
131        .for_each(q!(|s| println!("{}", s)));
132
133    let proposer_tick = proposers.tick();
134    let proxy_leader_tick = proxy_leaders.tick();
135    let acceptor_tick = acceptors.tick();
136
137    let (sequencing_max_ballot_complete_cycle, sequencing_max_ballot_forward_reference) =
138        proposers.forward_ref::<Stream<Ballot, _, _, NoOrder>>();
139    let (a_log_complete_cycle, a_log_forward_reference) =
140        acceptor_tick.forward_ref::<Singleton<_, _, _>>();
141
142    let (p_ballot, p_is_leader, p_relevant_p1bs, a_max_ballot) = leader_election(
143        proposers,
144        acceptors,
145        &proposer_tick,
146        &acceptor_tick,
147        config.acceptor_grid_rows,
148        config.acceptor_grid_rows * config.acceptor_grid_cols,
149        config.paxos_config,
150        sequencing_max_ballot_forward_reference,
151        a_log_forward_reference,
152        nondet!(
153            /// The primary non-determinism exposed by leader election algorithm lies in which leader
154            /// is elected, which affects both the ballot at each proposer and the leader flag. But using a stale ballot
155            /// or leader flag will only lead to failure in sequencing rather than commiting the wrong value.
156            nondet_leader
157        ),
158        nondet!(
159            /// Because ballots are non-deterministic, the acceptor max ballot is also non-deterministic, although we are
160            /// guaranteed that the max ballot will match the current ballot of a proposer who believes they are the leader.
161        ),
162    );
163
164    let just_became_leader = p_is_leader
165        .clone()
166        .filter_if_none(p_is_leader.clone().defer_tick());
167
168    let c_to_proposers = c_to_proposers(
169        just_became_leader
170            .clone()
171            .if_some_then(p_ballot.clone())
172            .all_ticks(),
173    );
174
175    let (p_to_replicas, a_log, sequencing_max_ballots) = sequence_payload(
176        proposers,
177        proxy_leaders,
178        acceptors,
179        &proposer_tick,
180        &proxy_leader_tick,
181        &acceptor_tick,
182        c_to_proposers,
183        a_checkpoint,
184        p_ballot.clone(),
185        p_is_leader,
186        p_relevant_p1bs,
187        config,
188        a_max_ballot,
189        nondet!(
190            /// The relevant p1bs are non-deterministic because they come from a arbitrary quorum, but because
191            /// we use a quorum, if we remain the leader there are no missing committed values when we combine the logs.
192            /// The remaining non-determinism is in when incoming payloads are batched versus the leader flag and state
193            /// of acceptors, which in the worst case will lead to dropped payloads as documented.
194            nondet_commit_leader_change
195        ),
196    );
197
198    a_log_complete_cycle.complete(a_log.snapshot_atomic(nondet!(
199        /// We will always write payloads to the log before acknowledging them to the proposers,
200        /// which guarantees that if the leader changes the quorum overlap between sequencing and leader
201        /// election will include the committed value.
202    )));
203    sequencing_max_ballot_complete_cycle.complete(sequencing_max_ballots);
204
205    (
206        // Only tell the clients once when leader election concludes
207        just_became_leader.if_some_then(p_ballot).all_ticks(),
208        p_to_replicas,
209    )
210}

213    clippy::type_complexity,

217fn sequence_payload<'a, P: PaxosPayload>(
218    proposers: &Cluster<'a, Proposer>,
219    proxy_leaders: &Cluster<'a, ProxyLeader>,
220    acceptors: &Cluster<'a, Acceptor>,
221    proposer_tick: &Tick<Cluster<'a, Proposer>>,
222    proxy_leader_tick: &Tick<Cluster<'a, ProxyLeader>>,
223    acceptor_tick: &Tick<Cluster<'a, Acceptor>>,
224    c_to_proposers: Stream<P, Cluster<'a, Proposer>, Unbounded>,
225    a_checkpoint: Optional<usize, Cluster<'a, Acceptor>, Unbounded>,
226
227    p_ballot: Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
228    p_is_leader: Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
229
230    p_relevant_p1bs: Stream<
231        (Option<usize>, HashMap<usize, LogValue<P>>),
232        Tick<Cluster<'a, Proposer>>,
233        Bounded,
234        NoOrder,
235    >,
236    config: CompartmentalizedPaxosConfig,
237    a_max_ballot: Singleton<Ballot, Tick<Cluster<'a, Acceptor>>, Bounded>,
238    nondet_commit_leader_change: NonDet,
239) -> (
240    Stream<(usize, Option<P>), Cluster<'a, ProxyLeader>, Unbounded, NoOrder>,
241    Singleton<
242        (Option<usize>, HashMap<usize, LogValue<P>>),
243        Atomic<Cluster<'a, Acceptor>>,
244        Unbounded,
245    >,
246    Stream<Ballot, Cluster<'a, Proposer>, Unbounded, NoOrder>,
247) {
248    let (p_log_to_recommit, p_max_slot) =
249        recommit_after_leader_election(p_relevant_p1bs, p_ballot.clone(), config.paxos_config.f);
250
251    let p_indexed_payloads = index_payloads(
252        p_max_slot,
253        c_to_proposers
254            .batch(
255                proposer_tick,
256                nondet!(
257                    /// We batch payloads so that we can compute the correct slot based on
258                    /// base slot. In the case of a leader re-election, the base slot is updated which
259                    /// affects the computed payload slots. This non-determinism can lead to non-determinism
260                    /// in which payloads are committed when the leader is changing, which is documented at
261                    /// the function level.
262                    nondet_commit_leader_change
263                ),
264            )
265            .filter_if_some(p_is_leader.clone()),
266    );
267
268    let num_proxy_leaders = config.num_proxy_leaders;
269    let p_to_proxy_leaders_p2a = p_indexed_payloads
270        .cross_singleton(p_ballot.clone())
271        .map(q!(move |((slot, payload), ballot)| (
272            MemberId::<ProxyLeader>::from_raw_id((slot % num_proxy_leaders) as u32),
273            ((slot, ballot), Some(payload))
274        )))
275        .chain(p_log_to_recommit.map(q!(move |((slot, ballot), payload)| (
276            MemberId::<ProxyLeader>::from_raw_id((slot % num_proxy_leaders) as u32),
277            ((slot, ballot), payload)
278        ))))
279        .all_ticks()
280        .demux(proxy_leaders, TCP.bincode())
281        .values()
282        .atomic(proxy_leader_tick);
283
284    // Send to a specific acceptor row
285    let num_acceptor_rows = config.acceptor_grid_rows;
286    let num_acceptor_cols = config.acceptor_grid_cols;
287    let pl_to_acceptors_p2a_thrifty = p_to_proxy_leaders_p2a
288        .clone()
289        .flat_map_unordered(q!(move |((slot, ballot), payload)| {
290            let row = slot % num_acceptor_rows;
291            let mut p2as = Vec::new();
292            for i in 0..num_acceptor_cols {
293                p2as.push((
294                    MemberId::<Acceptor>::from_raw_id((row * num_acceptor_cols + i) as u32),
295                    P2a {
296                        sender: MemberId::<ProxyLeader>::from_raw_id(
297                            (slot % num_proxy_leaders) as u32,
298                        ),
299                        slot,
300                        ballot: ballot.clone(),
301                        value: payload.clone(),
302                    },
303                ));
304            }
305            p2as
306        }))
307        .end_atomic()
308        .demux(acceptors, TCP.bincode())
309        .values();
310
311    let (a_log, a_to_proxy_leaders_p2b) = acceptor_p2(
312        acceptor_tick,
313        a_max_ballot.clone(),
314        pl_to_acceptors_p2a_thrifty,
315        a_checkpoint,
316        proxy_leaders,
317    );
318
319    // TODO: This is a liveness problem if any node in the thrifty quorum fails
320    // Need special operator for per-value timeout detection
321    let (quorums, fails) = collect_quorum(
322        a_to_proxy_leaders_p2b,
323        config.acceptor_grid_cols,
324        config.acceptor_grid_cols,
325    );
326
327    let pl_to_replicas = join_responses(
328        quorums.map(q!(|k| (k, ()))),
329        p_to_proxy_leaders_p2a.batch_atomic(nondet!(
330            /// The metadata will always be generated before we get a quorum
331            /// because our batch of `p_to_proxy_leaders_p2a` is at least after
332            /// what we sent to the acceptors.
333        )),
334    );
335
336    let pl_failed_p2b_to_proposer = fails
337        .map(q!(|(_, ballot)| (ballot.proposer_id.clone(), ballot)))
338        .inspect(q!(|(_, ballot)| println!("Failed P2b: {:?}", ballot)))
339        .demux(proposers, TCP.bincode())
340        .values();
341
342    (
343        pl_to_replicas.map(q!(|((slot, _ballot), (value, _))| (slot, value))),
344        a_log,
345        pl_failed_p2b_to_proposer,
346    )
