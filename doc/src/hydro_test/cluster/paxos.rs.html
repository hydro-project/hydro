119/// Implements the core Paxos algorithm, which uses a cluster of propsers and acceptors
120/// to sequence payloads being sent to the proposers.
121///
122/// Proposers that currently are the leader will work with acceptors to sequence incoming
123/// payloads, but may drop payloads if they are not the lader or lose leadership during
124/// the commit process.
125///
126/// Returns a stream of ballots, where new values are emitted when a new leader is elected,
127/// and a stream of sequenced payloads with an index and optional payload (in the case of
128/// holes in the log).
129///
130/// # Non-Determinism
131/// When the leader is stable, the algorithm will commit incoming payloads to the leader
132/// in deterministic order. However, when the leader is changing, payloads may be
133/// non-deterministically dropped. The stream of ballots is also non-deterministic because
134/// leaders are elected in a non-deterministic process.
135#[expect(clippy::type_complexity, reason = "internal paxos code // TODO")]
136pub fn paxos_core<'a, P: PaxosPayload>(
137    proposers: &Cluster<'a, Proposer>,
138    acceptors: &Cluster<'a, Acceptor>,
139    a_checkpoint: Optional<usize, Cluster<'a, Acceptor>, Unbounded>,
140    c_to_proposers: impl FnOnce(
141        Stream<Ballot, Cluster<'a, Proposer>, Unbounded>,
142    ) -> Stream<P, Cluster<'a, Proposer>, Unbounded>,
143    config: PaxosConfig,
144    nondet_leader: NonDet,
145    nondet_commit: NonDet,
146) -> (
147    Stream<Ballot, Cluster<'a, Proposer>, Unbounded>,
148    Stream<(usize, Option<P>), Cluster<'a, Proposer>, Unbounded, NoOrder>,
149) {
150    let f = config.f;
151
152    proposers
153        .source_iter(q!(["Proposers say hello"]))
154        .for_each(q!(|s| println!("{}", s)));
155
156    acceptors
157        .source_iter(q!(["Acceptors say hello"]))
158        .for_each(q!(|s| println!("{}", s)));
159
160    let proposer_tick = proposers.tick();
161    let acceptor_tick = acceptors.tick();
162
163    let (sequencing_max_ballot_complete_cycle, sequencing_max_ballot_forward_reference) =
164        proposers.forward_ref::<Stream<Ballot, _, _, NoOrder>>();
165    let (a_log_complete_cycle, a_log_forward_reference) =
166        acceptor_tick.forward_ref::<Singleton<_, _, _>>();
167
168    let (p_ballot, p_is_leader, p_relevant_p1bs, a_max_ballot) = leader_election(
169        proposers,
170        acceptors,
171        &proposer_tick,
172        &acceptor_tick,
173        f + 1,
174        2 * f + 1,
175        config,
176        sequencing_max_ballot_forward_reference,
177        a_log_forward_reference,
178        nondet!(
179            /// The primary non-determinism exposed by leader election algorithm lies in which leader
180            /// is elected, which affects both the ballot at each proposer and the leader flag. But using a stale ballot
181            /// or leader flag will only lead to failure in sequencing rather than commiting the wrong value.
182            nondet_leader
183        ),
184        nondet!(
185            /// Because ballots are non-deterministic, the acceptor max ballot is also non-deterministic, although we are
186            /// guaranteed that the max ballot will match the current ballot of a proposer who believes they are the leader.
187            nondet_commit
188        ),
189    );
190
191    let just_became_leader = p_is_leader
192        .clone()
193        .filter_if_none(p_is_leader.clone().defer_tick());
194
195    let c_to_proposers = c_to_proposers(
196        just_became_leader
197            .clone()
198            .if_some_then(p_ballot.clone())
199            .all_ticks(),
200    );
201
202    let (p_to_replicas, a_log, sequencing_max_ballots) = sequence_payload(
203        proposers,
204        acceptors,
205        &proposer_tick,
206        &acceptor_tick,
207        c_to_proposers,
208        a_checkpoint,
209        p_ballot.clone(),
210        p_is_leader,
211        p_relevant_p1bs,
212        f,
213        a_max_ballot,
214        nondet!(
215            /// The relevant p1bs are non-deterministic because they come from a arbitrary quorum, but because
216            /// we use a quorum, if we remain the leader there are no missing committed values when we combine the logs.
217            /// The remaining non-determinism is in when incoming payloads are batched versus the leader flag and state
218            /// of acceptors, which in the worst case will lead to dropped payloads as documented.
219            nondet_commit
220        ),
221    );
222
223    a_log_complete_cycle.complete(a_log.snapshot_atomic(nondet!(
224        /// We will always write payloads to the log before acknowledging them to the proposers,
225        /// which guarantees that if the leader changes the quorum overlap between sequencing and leader
226        /// election will include the committed value.
227    )));
228    sequencing_max_ballot_complete_cycle.complete(sequencing_max_ballots);
229
230    (
231        // Only tell the clients once when leader election concludes
232        just_became_leader.if_some_then(p_ballot).all_ticks(),
233        p_to_replicas,
234    )
235}

242pub fn leader_election<'a, L: Clone + Debug + Serialize + DeserializeOwned>(
243    proposers: &Cluster<'a, Proposer>,
244    acceptors: &Cluster<'a, Acceptor>,
245    proposer_tick: &Tick<Cluster<'a, Proposer>>,
246    acceptor_tick: &Tick<Cluster<'a, Acceptor>>,
247    quorum_size: usize,
248    num_quorum_participants: usize,
249    paxos_config: PaxosConfig,
250    p_received_p2b_ballots: Stream<Ballot, Cluster<'a, Proposer>, Unbounded, NoOrder>,
251    a_log: Singleton<(Option<usize>, L), Tick<Cluster<'a, Acceptor>>, Bounded>,
252    nondet_leader: NonDet,
253    nondet_acceptor_ballot: NonDet,
254) -> (
255    Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
256    Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
257    Stream<(Option<usize>, L), Tick<Cluster<'a, Proposer>>, Bounded, NoOrder>,
258    Singleton<Ballot, Tick<Cluster<'a, Acceptor>>, Bounded>,
259) {
260    let (p1b_fail_complete, p1b_fail) =
261        proposers.forward_ref::<Stream<Ballot, _, Unbounded, NoOrder>>();
262    let (p_to_proposers_i_am_leader_complete_cycle, p_to_proposers_i_am_leader_forward_ref) =
263        proposers.forward_ref::<Stream<_, _, _, NoOrder, AtLeastOnce>>();
264    let (p_is_leader_complete_cycle, p_is_leader_forward_ref) =
265        proposer_tick.forward_ref::<Optional<(), _, _>>();
266    // a_to_proposers_p2b.clone().for_each(q!(|(_, p2b): (u32, P2b)| println!("Proposer received P2b: {:?}", p2b)));
267    // p_to_proposers_i_am_leader.clone().for_each(q!(|ballot: Ballot| println!("Proposer received I am leader: {:?}", ballot)));
268    // c_to_proposers.clone().for_each(q!(|payload: ClientPayload| println!("Client sent proposer payload: {:?}", payload)));
269
270    let p_received_max_ballot = p1b_fail
271        .interleave(p_received_p2b_ballots)
272        .interleave(p_to_proposers_i_am_leader_forward_ref)
273        .max()
274        .unwrap_or(
275            proposers
276                .singleton(q!(Ballot {
277                    num: 0,
278                    proposer_id: MemberId::from_raw_id(0)
279                }))
280                .into(),
281        );
282
283    let (p_ballot, p_has_largest_ballot) = p_ballot_calc(p_received_max_ballot.snapshot(
284        proposer_tick,
285        nondet!(
286            /// A stale max ballot might result in us failing to become the leader, but which proposer
287            /// becomes the leader is non-deterministic anyway.
288            nondet_leader
289        ),
290    ));
291
292    let (p_to_proposers_i_am_leader, p_trigger_election) = p_leader_heartbeat(
293        proposers,
294        proposer_tick,
295        p_is_leader_forward_ref,
296        p_ballot.clone(),
297        paxos_config,
298        nondet!(
299            /// Non-determinism in heartbeats may lead to additional leader election attempts, which
300            /// is propagated to the non-determinism of which leader is elected.
301            nondet_leader
302        ),
303    );
304
305    p_to_proposers_i_am_leader_complete_cycle.complete(p_to_proposers_i_am_leader);
306
307    let p_to_acceptors_p1a = p_trigger_election
308        .if_some_then(p_ballot.clone())
309        .all_ticks()
310        .inspect(q!(|_| println!("Proposer leader expired, sending P1a")))
311        .broadcast(acceptors, TCP.fail_stop().bincode(), nondet!(/** TODO */))
312        .values();
313
314    let (a_max_ballot, a_to_proposers_p1b) = acceptor_p1(
315        acceptor_tick,
316        p_to_acceptors_p1a.batch(
317            acceptor_tick,
318            nondet!(
319                /// Non-deterministic batching may result in different payloads being rejected
320                /// by an acceptor if the payload is batched with another payload with larger ballot.
321                /// But as documented, payloads may be non-deterministically dropped during leader election.
322                nondet_acceptor_ballot
323            ),
324        ),
325        a_log,
326        proposers,
327    );
328
329    let (p_is_leader, p_accepted_values, fail_ballots) = p_p1b(
330        proposer_tick,
331        a_to_proposers_p1b.inspect(q!(|p1b| println!("Proposer received P1b: {:?}", p1b))),
332        p_ballot.clone(),
333        p_has_largest_ballot,
334        quorum_size,
335        num_quorum_participants,
336    );
337    p_is_leader_complete_cycle.complete(p_is_leader.clone());
338    p1b_fail_complete.complete(fail_ballots);
339
340    (p_ballot, p_is_leader, p_accepted_values, a_max_ballot)
341}

343// Proposer logic to calculate the next ballot number. Expects p_received_max_ballot, the largest ballot received so far. Outputs streams: ballot_num, and has_largest_ballot, which only contains a value if we have the largest ballot.
344#[expect(clippy::type_complexity, reason = "internal paxos code // TODO")]
345fn p_ballot_calc<'a>(
346    p_received_max_ballot: Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
347) -> (
348    Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
349    Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
350) {
351    let (p_ballot, p_has_largest_ballot) = sliced! {
352        let p_received_max_ballot = use::atomic(p_received_max_ballot.latest_atomic(), nondet!(/** up to date with tick input */));
353        let mut p_ballot_num = use::state(|l| l.singleton(q!(0)));
354
355        p_ballot_num = p_received_max_ballot
356            .clone()
357            .zip(p_ballot_num)
358            .map(q!(move |(received_max_ballot, ballot_num)| {
359                if received_max_ballot
360                    > (Ballot {
361                        num: ballot_num,
362                        proposer_id: CLUSTER_SELF_ID.clone(),
363                    })
364                {
365                    received_max_ballot.num + 1
366                } else {
367                    ballot_num
368                }
369            }));
370
371        let p_ballot = p_ballot_num.clone().map(q!(move |num| Ballot {
372            num,
373            proposer_id: CLUSTER_SELF_ID.clone()
374        }));
375
376        let p_has_largest_ballot = p_received_max_ballot
377            .zip(p_ballot.clone())
378            .filter(q!(
379                |(received_max_ballot, cur_ballot)| *received_max_ballot <= *cur_ballot
380            ))
381            .map(q!(|_| ()));
382
383        (yield_atomic(p_ballot), yield_atomic(p_has_largest_ballot))
384    };
385
386    (
387        p_ballot.snapshot_atomic(nondet!(/** always up to date with received ballots */)),
388        p_has_largest_ballot
389            .snapshot_atomic(nondet!(/** always up to date with received ballots */)),
390    )
391}

393#[expect(clippy::type_complexity, reason = "internal paxos code // TODO")]
394fn p_leader_heartbeat<'a>(
395    proposers: &Cluster<'a, Proposer>,
396    proposer_tick: &Tick<Cluster<'a, Proposer>>,
397    p_is_leader: Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
398    p_ballot: Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
399    paxos_config: PaxosConfig,
400    nondet_reelection: NonDet,
401) -> (
402    Stream<Ballot, Cluster<'a, Proposer>, Unbounded, NoOrder, AtLeastOnce>,
403    Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
404) {
405    let i_am_leader_send_timeout = paxos_config.i_am_leader_send_timeout;
406    let i_am_leader_check_timeout = paxos_config.i_am_leader_check_timeout;
407    let i_am_leader_check_timeout_delay_multiplier =
408        paxos_config.i_am_leader_check_timeout_delay_multiplier;
409
410    let p_to_proposers_i_am_leader = p_is_leader
411        .clone()
412        .if_some_then(p_ballot)
413        .latest()
414        .sample_every(
415            q!(Duration::from_secs(i_am_leader_send_timeout)),
416            nondet!(
417                /// Delays in heartbeats may lead to leader election attempts even
418                /// if the leader is alive. This will result in the previous leader receiving
419                /// larger ballots from its peers and it will drop its leadership.
420                nondet_reelection
421            ),
422        )
423        .broadcast(proposers, TCP.fail_stop().bincode(), nondet!(/** TODO */))
424        .values();
425
426    let p_leader_expired = p_to_proposers_i_am_leader
427        .clone()
428        .timeout(
429            q!(Duration::from_secs(i_am_leader_check_timeout)),
430            nondet!(
431                /// Delayed timeouts only affect which leader wins re-election. If the leadership flag
432                /// is gained after timeout correctly ignore the timeout. If the flag is lost after
433                /// timeout we correctly attempt to become the leader.
434                nondet_reelection
435            ),
436        )
437        .snapshot(proposer_tick, nondet!(/** absorbed into timeout */))
438        .filter_if_none(p_is_leader);
439
440    // Add random delay depending on node ID so not everyone sends p1a at the same time
441    let p_trigger_election = p_leader_expired.filter_if_some(
442        proposers
443            .source_interval_delayed(
444                q!(Duration::from_secs(
445                    (CLUSTER_SELF_ID.get_raw_id()
446                        * i_am_leader_check_timeout_delay_multiplier as u32)
447                        .into()
448                )),
449                q!(Duration::from_secs(i_am_leader_check_timeout)),
450                nondet!(
451                    /// If the leader 'un-expires' due to non-deterministic delay, we return
452                    /// to a stable leader state. If the leader remains expired, non-deterministic
453                    /// delay is propagated to the non-determinism of which leader is elected.
454                    nondet_reelection
455                ),
456            )
457            .batch(proposer_tick, nondet!(/** absorbed into interval */))
458            .first(),
459    );
460    (p_to_proposers_i_am_leader, p_trigger_election)
461}

463#[expect(clippy::type_complexity, reason = "internal paxos code // TODO")]
464fn acceptor_p1<'a, L: Serialize + DeserializeOwned + Clone>(
465    acceptor_tick: &Tick<Cluster<'a, Acceptor>>,
466    p_to_acceptors_p1a: Stream<Ballot, Tick<Cluster<'a, Acceptor>>, Bounded, NoOrder>,
467    a_log: Singleton<(Option<usize>, L), Tick<Cluster<'a, Acceptor>>, Bounded>,
468    proposers: &Cluster<'a, Proposer>,
469) -> (
470    Singleton<Ballot, Tick<Cluster<'a, Acceptor>>, Bounded>,
471    Stream<(Ballot, Result<(Option<usize>, L), Ballot>), Cluster<'a, Proposer>, Unbounded, NoOrder>,
472) {
473    let a_max_ballot = p_to_acceptors_p1a
474        .clone()
475        .inspect(q!(|p1a| println!("Acceptor received P1a: {:?}", p1a)))
476        .across_ticks(|s| s.max())
477        .unwrap_or(acceptor_tick.singleton(q!(Ballot {
478            num: 0,
479            proposer_id: MemberId::from_raw_id(0)
480        })));
481
482    (
483        a_max_ballot.clone(),
484        p_to_acceptors_p1a
485            .cross_singleton(a_max_ballot)
486            .cross_singleton(a_log)
487            .map(q!(|((ballot, max_ballot), log)| (
488                ballot.proposer_id.clone(),
489                (
490                    ballot.clone(),
491                    if ballot == max_ballot {
492                        Ok(log)
493                    } else {
494                        Err(max_ballot)
495                    }
496                )
497            )))
498            .all_ticks()
499            .demux(proposers, TCP.fail_stop().bincode())
500            .values(),
501    )
502}

504// Proposer logic for processing p1bs, determining if the proposer is now the leader, which uncommitted messages to commit, what the maximum slot is in the p1bs, and which no-ops to commit to fill log holes.
505#[expect(clippy::type_complexity, reason = "internal paxos code // TODO")]
506fn p_p1b<'a, P: Clone + Serialize + DeserializeOwned>(
507    proposer_tick: &Tick<Cluster<'a, Proposer>>,
508    a_to_proposers_p1b: Stream<
509        (Ballot, Result<(Option<usize>, P), Ballot>),
510        Cluster<'a, Proposer>,
511        Unbounded,
512        NoOrder,
513    >,
514    p_ballot: Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
515    p_has_largest_ballot: Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
516    quorum_size: usize,
517    num_quorum_participants: usize,
518) -> (
519    Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
520    Stream<(Option<usize>, P), Tick<Cluster<'a, Proposer>>, Bounded, NoOrder>,
521    Stream<Ballot, Cluster<'a, Proposer>, Unbounded, NoOrder>,
522) {
523    let (quorums, fails) =
524        collect_quorum_with_response(a_to_proposers_p1b, quorum_size, num_quorum_participants);
525
526    let p_received_quorum_of_p1bs = quorums
527        .into_keyed()
528        .assume_ordering::<TotalOrder>(nondet!(
529            /// We use `flatten_unordered` later
530        ))
531        .fold_early_stop(
532            q!(|| vec![]),
533            q!(move |logs, log| {
534                logs.push(log);
535                logs.len() >= quorum_size
536            }),
537        )
538        .get_max_key()
539        .snapshot(
540            proposer_tick,
541            nondet!(
542                /// If the max_by_key result is stale in this snapshot, we might be delayed in
543                /// realizing that we are the leader. This does not result in any safety issues.
544                /// In the meantime, ballots that acceptors rejected might be fed back into the max
545                /// ballot calculation, and change `p_ballot`. By using an async snapshot, we might
546                /// miss a chance to become the leader for a short period of time (until the rejected
547                /// ballots are processed). This is fine, if there is a higher ballot somewhere out
548                /// there, we should not be the leader anyways.
549            ),
550        )
551        .zip(p_ballot.clone())
552        .filter_map(q!(
553            move |((quorum_ballot, quorum_accepted), my_ballot)| if quorum_ballot == my_ballot {
554                Some(quorum_accepted)
555            } else {
556                None
557            }
558        ));
559
560    let p_is_leader = p_received_quorum_of_p1bs
561        .clone()
562        .map(q!(|_| ()))
563        .filter_if_some(p_has_largest_ballot.clone());
564
565    (
566        p_is_leader,
567        // we used an unordered accumulator, so flattened has no order
568        p_received_quorum_of_p1bs.flatten_unordered(),
569        fails.map(q!(|(_, ballot)| ballot)),
570    )
571}

573#[expect(clippy::type_complexity, reason = "internal paxos code // TODO")]
574pub fn recommit_after_leader_election<'a, P: PaxosPayload>(
575    accepted_logs: Stream<
576        (Option<usize>, HashMap<usize, LogValue<P>>),
577        Tick<Cluster<'a, Proposer>>,
578        Bounded,
579        NoOrder,
580    >,
581    p_ballot: Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
582    f: usize,
583) -> (
584    Stream<((usize, Ballot), Option<P>), Tick<Cluster<'a, Proposer>>, Bounded, NoOrder>,
585    Optional<usize, Tick<Cluster<'a, Proposer>>, Bounded>,
586) {
587    let p_p1b_max_checkpoint = accepted_logs
588        .clone()
589        .filter_map(q!(|(checkpoint, _log)| checkpoint))
590        .max()
591        .into_singleton();
592    let p_p1b_highest_entries_and_count = accepted_logs
593        .map(q!(|(_checkpoint, log)| log))
594        .flatten_unordered() // Convert HashMap log back to stream
595        .into_keyed()
596        .fold::<(usize, Option<LogValue<P>>), _, _, _, _>(q!(|| (0, None)), q!(|curr_entry, new_entry| {
597            if let Some(curr_entry_payload) = &mut curr_entry.1 {
598                let same_values = new_entry.value == curr_entry_payload.value;
599                let higher_ballot = new_entry.ballot > curr_entry_payload.ballot;
600                // Increment count if the values are the same
601                if same_values {
602                    curr_entry.0 += 1;
603                }
604                // Replace the ballot with the largest one
605                if higher_ballot {
606                    curr_entry_payload.ballot = new_entry.ballot;
607                    // Replace the value with the one from the largest ballot, if necessary
608                    if !same_values {
609                        curr_entry.0 = 1;
610                        curr_entry_payload.value = new_entry.value;
611                    }
612                }
613            } else {
614                *curr_entry = (1, Some(new_entry));
615            }
616        }, commutative = ManualProof(/* TODO */)))
617        .map(q!(|(count, entry)| (count, entry.unwrap())));
618    let p_log_to_try_commit = p_p1b_highest_entries_and_count
619        .clone()
620        .entries()
621        .cross_singleton(p_ballot.clone())
622        .cross_singleton(p_p1b_max_checkpoint.clone())
623        .filter_map(q!(move |(((slot, (count, entry)), ballot), checkpoint)| {
624            if count > f {
625                return None;
626            } else if let Some(checkpoint) = checkpoint
627                && slot <= checkpoint
628            {
629                return None;
630            }
631            Some(((slot, ballot), entry.value))
632        }));
633    let p_max_slot = p_p1b_highest_entries_and_count.clone().keys().max();
634    let p_proposed_slots = p_p1b_highest_entries_and_count.clone().keys();
635    let p_log_holes = p_max_slot
636        .clone()
637        .zip(p_p1b_max_checkpoint)
638        .flat_map_ordered(q!(|(max_slot, checkpoint)| {
639            if let Some(checkpoint) = checkpoint {
640                (checkpoint + 1)..max_slot
641            } else {
642                0..max_slot
643            }
644        }))
645        .filter_not_in(p_proposed_slots)
646        .cross_singleton(p_ballot.clone())
647        .map(q!(move |(slot, ballot)| ((slot, ballot), None)));
648
649    (p_log_to_try_commit.chain(p_log_holes), p_max_slot)
650}

652#[expect(
653    clippy::type_complexity,
654    clippy::too_many_arguments,
655    reason = "internal paxos code // TODO"
656)]
657fn sequence_payload<'a, P: PaxosPayload>(
658    proposers: &Cluster<'a, Proposer>,
659    acceptors: &Cluster<'a, Acceptor>,
660    proposer_tick: &Tick<Cluster<'a, Proposer>>,
661    acceptor_tick: &Tick<Cluster<'a, Acceptor>>,
662    c_to_proposers: Stream<P, Cluster<'a, Proposer>, Unbounded>,
663    a_checkpoint: Optional<usize, Cluster<'a, Acceptor>, Unbounded>,
664
665    p_ballot: Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
666    p_is_leader: Optional<(), Tick<Cluster<'a, Proposer>>, Bounded>,
667
668    p_relevant_p1bs: Stream<
669        (Option<usize>, HashMap<usize, LogValue<P>>),
670        Tick<Cluster<'a, Proposer>>,
671        Bounded,
672        NoOrder,
673    >,
674    f: usize,
675
676    a_max_ballot: Singleton<Ballot, Tick<Cluster<'a, Acceptor>>, Bounded>,
677
678    nondet_commit: NonDet,
679) -> (
680    Stream<(usize, Option<P>), Cluster<'a, Proposer>, Unbounded, NoOrder>,
681    Singleton<
682        (Option<usize>, HashMap<usize, LogValue<P>>),
683        Atomic<Cluster<'a, Acceptor>>,
684        Unbounded,
685    >,
686    Stream<Ballot, Cluster<'a, Proposer>, Unbounded, NoOrder>,
687) {
688    let (p_log_to_recommit, p_max_slot) =
689        recommit_after_leader_election(p_relevant_p1bs, p_ballot.clone(), f);
690
691    let indexed_payloads = index_payloads(
692        p_max_slot,
693        c_to_proposers
694            .batch(
695                proposer_tick,
696                nondet!(
697                    /// We batch payloads so that we can compute the correct slot based on
698                    /// base slot. In the case of a leader re-election, the base slot is updated which
699                    /// affects the computed payload slots. This non-determinism can lead to non-determinism
700                    /// in which payloads are committed when the leader is changing, which is documented at
701                    /// the function level
702                    nondet_commit
703                ),
704            )
705            .filter_if_some(p_is_leader.clone()),
706    );
707
708    let payloads_to_send = indexed_payloads
709        .cross_singleton(p_ballot.clone())
710        .map(q!(|((slot, payload), ballot)| (
711            (slot, ballot),
712            Some(payload)
713        )))
714        .chain(p_log_to_recommit)
715        .filter_if_some(p_is_leader)
716        .all_ticks_atomic();
717
718    let (a_log, a_to_proposers_p2b) = acceptor_p2(
719        acceptor_tick,
720        a_max_ballot.clone(),
721        payloads_to_send
722            .clone()
723            .end_atomic()
724            .map(q!(move |((slot, ballot), value)| P2a {
725                sender: CLUSTER_SELF_ID.clone(),
726                ballot,
727                slot,
728                value
729            }))
730            .broadcast(acceptors, TCP.fail_stop().bincode(), nondet!(/** TODO */))
731            .values(),
732        a_checkpoint,
733        proposers,
734    );
735
736    // TOOD: only persist if we are the leader
737    let (quorums, fails) = collect_quorum(a_to_proposers_p2b, f + 1, 2 * f + 1);
738
739    let p_to_replicas = join_responses(
740        quorums.map(q!(|k| (k, ()))),
741        payloads_to_send.batch_atomic(nondet!(
742            /// The metadata will always be generated before we get a quorum
743            /// because our batch of `payloads_to_send` is at least after
744            /// what we sent to the acceptors.
745        )),
746    );
747
748    (
749        p_to_replicas.map(q!(|((slot, _ballot), (value, _))| (slot, value))),
750        a_log,
751        fails.map(q!(|(_, ballot)| ballot)),
752    )
753}

755// Proposer logic to send p2as, outputting the next slot and the p2as to send to acceptors.
756pub fn index_payloads<'a, L: Location<'a> + NoTick, P: PaxosPayload>(
757    p_max_slot: Optional<usize, Tick<L>, Bounded>,
758    c_to_proposers: Stream<P, Tick<L>, Bounded>,
759) -> Stream<(usize, P), Tick<L>, Bounded> {
760    sliced! {
761        let mut next_slot = use::state(|l| l.singleton(q!(0)));
762        let updated_max_slot = use::atomic(p_max_slot.latest_atomic(), nondet!(/** up to date with tick input */));
763        let payload_batch = use::atomic(c_to_proposers.all_ticks_atomic(), nondet!(/** up to date with tick input */));
764
765        let next_slot_after_reconciling_p1bs = updated_max_slot.map(q!(|s| s + 1));
766        let base_slot = next_slot_after_reconciling_p1bs.unwrap_or(next_slot);
767
768        let indexed_payloads = payload_batch
769            .enumerate()
770            .cross_singleton(base_slot.clone())
771            .map(q!(|((index, payload), base_slot)| (
772                base_slot + index,
773                payload
774            )));
775
776        let num_payloads = indexed_payloads.clone().count();
777        next_slot = num_payloads
778            .zip(base_slot)
779            .map(q!(|(num_payloads, base_slot)| base_slot + num_payloads));
780
781        yield_atomic(indexed_payloads)
782    }.batch_atomic(nondet!(/** up to date with tick input */))
783}

785#[expect(clippy::type_complexity, reason = "internal paxos code // TODO")]
786pub fn acceptor_p2<'a, P: PaxosPayload, S: Clone>(
787    acceptor_tick: &Tick<Cluster<'a, Acceptor>>,
788    a_max_ballot: Singleton<Ballot, Tick<Cluster<'a, Acceptor>>, Bounded>,
789    p_to_acceptors_p2a: Stream<P2a<P, S>, Cluster<'a, Acceptor>, Unbounded, NoOrder>,
790    a_checkpoint: Optional<usize, Cluster<'a, Acceptor>, Unbounded>,
791    proposers: &Cluster<'a, S>,
792) -> (
793    Singleton<
794        (Option<usize>, HashMap<usize, LogValue<P>>),
795        Atomic<Cluster<'a, Acceptor>>,
796        Unbounded,
797    >,
798    Stream<((usize, Ballot), Result<(), Ballot>), Cluster<'a, S>, Unbounded, NoOrder>,
799) {
800    let p_to_acceptors_p2a_batch = p_to_acceptors_p2a.batch(
801        acceptor_tick,
802        nondet!(
803            /// We use batches to ensure that the log is updated before sending
804            /// a confirmation to the proposer. Because we use `persist()` on these
805            /// messages before folding into the log, non-deterministic batch boundaries
806            /// will not affect the eventual log state.
807        ),
808    );
809
810    let a_checkpoint = a_checkpoint.snapshot(
811        acceptor_tick,
812        nondet!(
813            /// We can arbitrarily snapshot the checkpoint sequence number,
814            /// since a delayed garbage collection does not affect correctness.
815        ),
816    );
817    // .inspect(q!(|min_seq| println!("Acceptor new checkpoint: {:?}", min_seq)));
818
819    let a_p2as_to_place_in_log = p_to_acceptors_p2a_batch
820        .clone()
821        .cross_singleton(a_max_ballot.clone()) // Don't consider p2as if the current ballot is higher
822        .filter_map(q!(|(p2a, max_ballot)|
823            if p2a.ballot >= max_ballot {
824                Some((p2a.slot, LogValue {
825                    ballot: p2a.ballot,
826                    value: p2a.value,
827                }))
828            } else {
829                None
830            }
831        ));
832    let a_log = a_p2as_to_place_in_log.across_ticks(|s| {
833        s.into_keyed().reduce_watermark(
834            a_checkpoint.clone(),
835            q!(|prev_entry, entry| {
836                // Insert p2a into the log if it has a higher ballot than what was there before
837                if entry.ballot > prev_entry.ballot {
838                    *prev_entry = LogValue {
839                        ballot: entry.ballot,
840                        value: entry.value,
841                    };
842                }
843            }, commutative = ManualProof(/* max by ballot (TODO: not if two entries with same ballot, need assume) */)),
844        )
845    });
846    let a_log_snapshot = a_log.entries().fold(
847        q!(|| HashMap::new()),
848        q!(
849            |map, (slot, entry)| {
850                map.insert(slot, entry);
851            },
852            commutative = ManualProof(
853                /* inserting elements into map is commutative because no keys will overlap */
854            )
855        ),
856    );
857
858    let a_to_proposers_p2b = p_to_acceptors_p2a_batch
859        .cross_singleton(a_max_ballot)
860        .map(q!(|(p2a, max_ballot)| (
861            p2a.sender,
862            (
863                (p2a.slot, p2a.ballot.clone()),
864                if p2a.ballot == max_ballot {
865                    Ok(())
866                } else {
867                    Err(max_ballot)
868                }
869            )
870        )))
871        .all_ticks()
872        .demux(proposers, TCP.fail_stop().bincode())
873        .values();
874
875    (
876        a_checkpoint
877            .into_singleton()
878            .zip(a_log_snapshot)
879            .latest_atomic(),
880        a_to_proposers_p2b,
881    )
882}
