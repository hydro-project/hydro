2mod tests {
3    use futures::StreamExt;
4    use hydro_deploy::Deployment;
5    use hydro_lang::compile::builder::FlowBuilder;

7    use stageleft::q;

9    #[tokio::test]
10    async fn test_singleton_mut() {
11        let mut deployment = Deployment::new();
12
13        let mut builder = FlowBuilder::new();
14        let external = builder.external::<()>();
15        let p1 = builder.process::<()>();
16
17        // Create a singleton: fold 0..5 => 10
18        let my_count = p1
19            .source_iter(q!(0..5i32))
20            .fold(q!(|| 0i32), q!(|acc: &mut i32, x| *acc += x));
21
22        let count_mut = my_count.by_mut();
23
24        // Use the singleton mut in a map on another stream
25        let out_port = p1
26            .source_iter(q!(1..=3i32))
27            .weaken_ordering::<hydro_lang::live_collections::stream::NoOrder>()
28            .map(q!(
29                |x| {
30                    *count_mut += x;
31                    *count_mut
32                },
33                commutative = hydro_lang::__manual_proof__!(/** addition is commutative */)
34            ))
35            .send_bincode_external(&external);
36
37        let nodes = builder
38            .with_default_optimize()
39            .with_process(&p1, deployment.Localhost())
40            .with_external(&external, deployment.Localhost())
41            .deploy(&mut deployment);
42
43        deployment.deploy().await.unwrap();
44
45        let mut out_recv = nodes.connect(out_port).await;
46
47        deployment.start().await.unwrap();
48
49        let mut results = Vec::new();
50        for _ in 0..3 {
51            results.push(out_recv.next().await.unwrap());
52        }
53        results.sort();
54        assert_eq!(results, vec![11, 13, 16]);
55    }
