1use hydro_lang::live_collections::stream::NoOrder;
2use hydro_lang::prelude::*;
3
4pub fn graph_reachability<'a>(
5    process: &Process<'a>,
6    roots: Stream<u32, Process<'a>, Unbounded>,
7    edges: Stream<(u32, u32), Process<'a>, Unbounded>,
8) -> Stream<u32, Process<'a>, Unbounded, NoOrder> {
9    let reachability_tick = process.tick();
10    let (set_reached_cycle, reached_cycle) = reachability_tick.cycle::<Stream<_, _, _, NoOrder>>();
11
12    let reached = roots
13        .batch(
14            &reachability_tick,
15            nondet!(/** roots can be inserted on any tick because we are fixpointing */),
16        )
17        .chain(reached_cycle);
18    let reachable = reached
19        .clone()
20        .map(q!(|r| (r, ())))
21        .join(
22            edges
23                .batch(
24                    &reachability_tick,
25                    nondet!(/** edges can be inserted on any tick because we are fixpointing */),
26                )
27                .persist(),
28        )
29        .map(q!(|(_from, (_, to))| to));
30    set_reached_cycle.complete_next_tick(reached.clone().chain(reachable));
31
32    reached.all_ticks().unique()
33}
