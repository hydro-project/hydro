1use hydro_lang::live_collections::stream::{NoOrder, TotalOrder};
2use hydro_lang::prelude::*;
3
4#[derive(Debug, Clone)]
5pub enum RequestType {
6    Increment { key: i32 },
7    Get { key: i32 },
8    Invalid,
9}
10
11#[derive(Debug, Clone)]
12pub struct ParsedRequest {
13    pub connection_id: u64,
14    pub request_type: RequestType,
15    pub raw_request: String,
16}
17
18pub fn http_counter_server<'a, P>(
19    in_stream: KeyedStream<u64, String, Process<'a, P>, Unbounded, TotalOrder>,
20    process: &Process<'a, P>,
21) -> KeyedStream<u64, String, Process<'a, P>, Unbounded, NoOrder> {
22    let parsed_requests = in_stream
23        .fold_early_stop(
24            q!(|| String::new()),
25            q!(|buffer, line| {
26                buffer.push_str(&line);
27                buffer.push_str("\r\n");
28                // Check if this is an empty line (end of HTTP headers)
29                line.trim().is_empty()
30            }),
31        )
32        .map_with_key(q!(|(connection_id, raw_request)| {
33            let lines: Vec<&str> = raw_request.lines().collect();
34            let request_line = lines.first().unwrap_or(&"");
35            let parts: Vec<&str> = request_line.split_whitespace().collect();
36            let method = parts.first().unwrap_or(&"GET");
37            let path = parts.get(1).unwrap_or(&"/");
38
39            let request_type = if method == &"POST" && path.starts_with("/increment/") {
40                if let Ok(key) = path[11..].parse::<i32>() {
41                    RequestType::Increment { key }
42                } else {
43                    RequestType::Invalid
44                }
45            } else if method == &"GET" && path.starts_with("/get/") {
46                if let Ok(key) = path[5..].parse::<i32>() {
47                    RequestType::Get { key }
48                } else {
49                    RequestType::Invalid
50                }
51            } else {
52                RequestType::Invalid
53            };
54
55            ParsedRequest {
56                connection_id,
57                request_type,
58                raw_request,
59            }
60        }));
61
62    let increment_lookup_tick = process.tick();
63    let increment_stream = parsed_requests
64        .clone()
65        .filter_map(q!(|req| match req.request_type {
66            RequestType::Increment { key } => Some(key),
67            _ => None,
68        }))
69        .atomic(&increment_lookup_tick);
70
71    let get_stream = parsed_requests
72        .clone()
73        .filter_map(q!(|req| match req.request_type {
74            RequestType::Get { key } => Some(key),
75            _ => None,
76        }));
77
78    let invalid_requests = parsed_requests.filter_map(q!(|req| match req.request_type {
79        RequestType::Invalid => Some(req.raw_request),
80        _ => None,
81    }));
82
83    let counters = increment_stream
84        .clone()
85        .values()
86        .map(q!(|key| (key, ())))
87        .into_keyed()
88        .value_counts();
89
90    let lookup_result = sliced! {
91        let batch_get_requests = use(get_stream, nondet!(/** batch get requests */));
92        let cur_counters = use::atomic(counters, nondet!(/** intentional non-determinism for get timing */));
93
94        batch_get_requests.lookup_keyed_singleton(cur_counters).into_keyed_stream()
95    };
96    let get_responses =
97        lookup_result.map(q!(|(key, maybe_count)| if let Some(count) = maybe_count {
98            format!(
99                "HTTP/1.1 200 OK\r\n\
100                    Content-Type: application/json\r\n\
101                    Content-Length: {}\r\n\
102                    Connection: close\r\n\
103                    \r\n\
104                    {{\"key\": {}, \"count\": {}}}",
105                format!("{{\"key\": {}, \"count\": {}}}", key, count).len(),
106                key,
107                count
108            )
109        } else {
110            format!(
111                "HTTP/1.1 200 OK\r\n\
112                        Content-Type: application/json\r\n\
113                        Content-Length: {}\r\n\
114                        Connection: close\r\n\
115                        \r\n\
116                        {{\"key\": {}, \"count\": 0}}",
117                format!("{{\"key\": {}, \"count\": 0}}", key).len(),
118                key
119            )
120        }));
121
122    // Handle increment responses (just acknowledge)
123    let increment_responses = increment_stream
124        .map(q!(|key| {
125            format!(
126                "HTTP/1.1 200 OK\r\n\
127                 Content-Type: application/json\r\n\
128                 Content-Length: {}\r\n\
129                 Connection: close\r\n\
130                 \r\n\
131                 {{\"key\": {}, \"status\": \"incremented\"}}",
132                format!("{{\"key\": {}, \"status\": \"incremented\"}}", key).len(),
133                key
134            )
135        }))
136        .end_atomic();
137
138    let invalid_responses = invalid_requests.map(q!(|_raw_request| {
139        let error_body =
140            "{\"error\": \"Invalid request. Use POST /increment/{key} or GET /get/{key}\"}";
141        format!(
142            "HTTP/1.1 400 Bad Request\r\n\
143                 Content-Type: application/json\r\n\
144                 Content-Length: {}\r\n\
145                 Connection: close\r\n\
146                 \r\n\
147                 {}",
148            error_body.len(),
149            error_body
150        )
151    }));
152
153    get_responses
154        .interleave(increment_responses.into_keyed_stream())
155        .interleave(invalid_responses.into_keyed_stream())
156}
