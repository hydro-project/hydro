186                replica_tick: &Tick<L>,

190                TickCycleHandle<'a, Singleton<usize, Tick<L::DropConsistency>, Bounded>>,
191            ) {
192                let (r_buffered_payloads_complete_cycle, r_buffered_payloads) = replica_tick
193                    .cycle::<
194                        Stream<SequencedKv<K, V>, Tick<L::DropConsistency>, Bounded>,
195                        _,
196                    >();
197                let r_sorted_payloads = p_to_replicas
198                    .batch(
199                        replica_tick,
200                        nondet!(
201                            #[doc =
202                            " because we fill slots one-by-one, we can safely batch"]

204                            " because non-determinism is resolved when we sort by slots"]

206                    )
207                    .chain(r_buffered_payloads)

210                    .cycle_with_initial(replica_tick.singleton(q!(0)));
211                let r_next_slot_after_processing_payloads = r_sorted_payloads
212                    .clone()
213                    .cross_singleton(r_next_slot)
214                    .fold(
215                        q!(|| 0),
216                        q!(
217                            | new_next_slot, (sorted_payload, next_slot) | { if
218                            sorted_payload.seq == std::cmp::max(* new_next_slot,
219                            next_slot) { * new_next_slot = sorted_payload.seq + 1; } }
220                        ),
221                    );
222                let r_processable_payloads = r_sorted_payloads
223                    .clone()
224                    .cross_singleton(r_next_slot_after_processing_payloads.clone())
225                    .filter(
226                        q!(
227                            | (sorted_payload, highest_seq) | sorted_payload.seq < *
228                            highest_seq
229                        ),
230                    )
231                    .map(q!(| (sorted_payload, _) | { sorted_payload }));
232                let r_new_non_processable_payloads = r_sorted_payloads
233                    .cross_singleton(r_next_slot_after_processing_payloads)

235                        q!(
236                            | (sorted_payload, highest_seq) | sorted_payload.seq > *
237                            highest_seq
238                        ),
239                    )
240                    .map(q!(| (sorted_payload, _) | { sorted_payload }));
241                r_buffered_payloads_complete_cycle
242                    .complete_next_tick(r_new_non_processable_payloads);
243                (r_processable_payloads, r_next_slot_complete_cycle)
244            }

302        pub fn p_ballot_calc<'a>(
303            p_received_max_ballot: Singleton<
304                Ballot,
305                Tick<Cluster<'a, Proposer>>,
306                Bounded,
307            >,
308        ) -> (
309            Singleton<Ballot, Tick<Cluster<'a, Proposer>>, Bounded>,
310            Singleton<bool, Tick<Cluster<'a, Proposer>>, Bounded>,
311        ) {
312            let tick = p_received_max_ballot.location().clone();
313            let (p_ballot, p_has_largest_ballot) = sliced! {
314                let p_received_max_ballot = use ::atomic(p_received_max_ballot
315                .latest_atomic(), nondet!(#[doc = " up to date with tick input "])); let
316                mut p_ballot_num = use ::state(| l | l.singleton(q!(0))); p_ballot_num =
317                p_received_max_ballot.clone().zip(p_ballot_num).map(q!(move |
318                (received_max_ballot, ballot_num) | { if received_max_ballot > (Ballot {
319                num : ballot_num, proposer_id : CLUSTER_SELF_ID.clone(), }) {
320                received_max_ballot.num + 1 } else { ballot_num } })); let p_ballot =
321                p_ballot_num.clone().map(q!(move | num | Ballot { num, proposer_id :
322                CLUSTER_SELF_ID.clone() })); let p_has_largest_ballot =
323                p_received_max_ballot.zip(p_ballot.clone()).map(q!(|
324                (received_max_ballot, cur_ballot) | received_max_ballot <= cur_ballot));
325                (yield_atomic(p_ballot), yield_atomic(p_has_largest_ballot))
326            };
327            (
328                p_ballot
329                    .snapshot_atomic(
330                        &tick,
331                        nondet!(#[doc = " always up to date with received ballots "]),

550            p_is_leader: Singleton<bool, Tick<Cluster<'a, Proposer>>, Bounded>,
551            p_relevant_p1bs: Stream<
552                (Option<usize>, HashMap<usize, LogValue<P>>),
553                Tick<Cluster<'a, Proposer>>,
554                Bounded,
555                NoOrder,
556            >,
557            f: usize,
558            a_max_ballot: Singleton<Ballot, Tick<Cluster<'a, Acceptor>>, Bounded>,
559            nondet_commit: NonDet,
560        ) -> (
561            Stream<(usize, Option<P>), Cluster<'a, Proposer>, Unbounded, NoOrder>,
562            Singleton<
563                (Option<usize>, HashMap<usize, LogValue<P>>),
564                Atomic<Cluster<'a, Acceptor>>,
565                Unbounded,
566            >,
567            Stream<Ballot, Cluster<'a, Proposer>, Unbounded, NoOrder>,
568        ) {
569            let (p_log_to_recommit, p_max_slot) = recommit_after_leader_election(
570                p_relevant_p1bs,
571                p_ballot.clone(),
572                f,
573            );
574            let indexed_payloads = index_payloads(
575                p_max_slot,
576                c_to_proposers
577                    .batch(
578                        proposer_tick,
579                        nondet!(
580                            #[doc =
581                            " We batch payloads so that we can compute the correct slot based on"]
582                            #[doc =
583                            " base slot. In the case of a leader re-election, the base slot is updated which"]
584                            #[doc =
585                            " affects the computed payload slots. This non-determinism can lead to non-determinism"]
586                            #[doc =
587                            " in which payloads are committed when the leader is changing, which is documented at"]
588                            #[doc = " the function level"] nondet_commit
589                        ),
590                    )
591                    .filter_if(p_is_leader.clone()),
592            );
593            let payloads_to_send = indexed_payloads
594                .cross_singleton(p_ballot.clone())
595                .map(q!(| ((slot, payload), ballot) | ((slot, ballot), Some(payload))))
596                .chain(p_log_to_recommit)
597                .filter_if(p_is_leader)
598                .all_ticks_atomic();
599            let (a_log, a_to_proposers_p2b) = acceptor_p2(
600                acceptor_tick,
601                a_max_ballot.clone(),
602                payloads_to_send
603                    .clone()
604                    .end_atomic()
605                    .map(
606                        q!(
607                            move | ((slot, ballot), value) | P2a { sender :
608                            CLUSTER_SELF_ID.clone(), ballot, slot, value }
609                        ),
610                    )
611                    .broadcast(
612                        acceptors,
613                        TCP.fail_stop().bincode(),
614                        nondet!(#[doc = " TODO "]),
615                    )
616                    .values(),
617                a_checkpoint,
618                proposers,
619            );
620            let (quorums, fails) = collect_quorum(a_to_proposers_p2b, f + 1, 2 * f + 1);
621            let p_to_replicas = join_responses(
622                quorums.map(q!(| k | (k, ()))),
623                payloads_to_send
624                    .batch_atomic(
625                        proposer_tick,
626                        nondet!(
627                            #[doc =
628                            " The metadata will always be generated before we get a quorum"]
629                            #[doc =
630                            " because our batch of `payloads_to_send` is at least after"]
631                            #[doc = " what we sent to the acceptors."]
632                        ),
633                    ),
634            );
635            (
636                p_to_replicas.map(q!(| ((slot, _ballot), (value, _)) | (slot, value))),
637                a_log,
638                fails.map(q!(| (_, ballot) | ballot)),
639            )
640        }

642        pub use crate::cluster::paxos::acceptor_p2;

778        pub use crate::distributed::versioning::GossipServer;
779        pub use crate::distributed::versioning::Logger;
780        /// Routes each request to a single `GossipServer` member chosen by `route`, based on a
781        /// nondeterministic snapshot of the cluster's current membership. Returns the demuxed
782        /// [`KeyedStream`] received at the target (keyed by the sending member) plus the membership
783        /// snapshots used for routing.
784        #[expect(
785            clippy::type_complexity,
786            reason = "returns tuple with membership stream"
787        )]
788        pub fn hash_demux<'a, F, N: NetworkFor<Request>>(
789            requests: Stream<Request, Cluster<'a, GossipServer>, Unbounded, TotalOrder>,
790            to: &Cluster<'a, GossipServer>,
791            route: impl IntoQuotedMut<
792                'a,
793                F,
794                Tick<Cluster<'a, GossipServer>>,
795                StreamMapFuncAlgebra,
796            >,
797            via: N,
798            nondet_membership: NonDet,
799        ) -> (
800            KeyedStream<
801                MemberId<GossipServer>,
802                Request,
803                Cluster<'a, GossipServer>,
804                Unbounded,
805                N::OrderingGuarantee,
806                ExactlyOnce,
807            >,
808            Stream<Vec<MemberId<GossipServer>>, Cluster<'a, GossipServer>, Unbounded>,
809        )
810        where
811            F: Fn(
812                    (Request, Vec<MemberId<GossipServer>>),
813                ) -> Option<(MemberId<GossipServer>, Request)> + 'a,
814        {
815            let ids = track_membership(
816                requests
817                    .location()
818                    .source_cluster_membership_stream(to, nondet_membership),
819            );
820            let (filtered, members_out) = sliced! {
821                let members_snapshot = use (ids, nondet_membership); let elements = use
822                (requests, nondet_membership); let current_members = members_snapshot
823                .filter(q!(| b | * b)).keys().assume_ordering::< TotalOrder >
824                (nondet_membership).collect_vec(); let filtered = elements
825                .cross_singleton(current_members.clone()).filter_map::< _, _, _, _, false
826                > (route); (filtered, current_members.into_stream())
827            };
828            (filtered.demux(to, via), members_out)
829        }
830        /// A server that forwards all requests to one other member (chosen by `route`) via gossip; the
831        /// target member folds writes into its state and replies.
832        pub fn gossip_server<'a, F>(
833            requests: Stream<Request, Cluster<'a, GossipServer>, Unbounded, TotalOrder>,
834            servers: &Cluster<'a, GossipServer>,
835            route: impl IntoQuotedMut<
836                'a,
837                F,
838                Tick<Cluster<'a, GossipServer>>,
839                StreamMapFuncAlgebra,
840            >,
841        ) -> (
842            Stream<Response, Cluster<'a, GossipServer>, Unbounded, NoOrder>,
843            Stream<Vec<MemberId<GossipServer>>, Cluster<'a, GossipServer>, Unbounded>,
844        )
845        where
846            F: Fn(
847                    (Request, Vec<MemberId<GossipServer>>),
848                ) -> Option<(MemberId<GossipServer>, Request)> + 'a,
849        {
850            let (on_other, membership) = hash_demux(
851                requests,
852                servers,
853                route,
854                TCP.fail_stop().bincode().name("gossip"),
855                nondet!(#[doc = " gossip membership "]),
856            );
857            let responses = on_other
858                .entries()
859                .assume_ordering::<TotalOrder>(nondet!(#[doc = " ordering on target "]))
860                .scan(
861                    q!(|| 0usize),
862                    q!(
863                        | acc, (sender, req) | { match req { Request::Write { value } =>
864                        { * acc = usize::max(* acc, value); } Request::Read => {} }
865                        Some((sender, Response { value : * acc })) }
866                    ),
867                )
868                .map(q!(| (sender, response) | (sender, response)))
869                .into_keyed()
870                .demux(servers, TCP.fail_stop().bincode().name("gossip_response"))
871                .values()
872                .weaken_ordering();
873            (responses, membership)
874        }
875        pub use crate::distributed::versioning::GossipVersion;
876        pub use crate::distributed::versioning::gossip_server_versioned;
