22fn link_cycles_leaf(leaf: &mut HydroLeaf, sink_inputs: &mut HashMap<Ident, usize>) {
23    if let HydroLeaf::CycleSink { ident, input, .. } = leaf {
24        sink_inputs.insert(ident.clone(), input.metadata().id.unwrap());
25    }
26}

28fn link_cycles_node(node: &mut HydroNode, sources: &mut HashMap<Ident, usize>) {

37// Returns map from CycleSource id to the input IDs of the corresponding CycleSink's input
38// Assumes that metadtata.id is set for all nodes
39pub fn cycle_source_to_sink_input(ir: &mut [HydroLeaf]) -> HashMap<usize, usize> {
40    let mut sources = HashMap::new();
41    let mut sink_inputs = HashMap::new();
42
43    // Can't use traverse_dfir since that skips CycleSink
44    transform_bottom_up(
45        ir,
46        &mut |leaf| {
47            link_cycles_leaf(leaf, &mut sink_inputs);
48        },
49        &mut |node| {
50            link_cycles_node(node, &mut sources);
51        },
52    );
53
54    let mut source_to_sink_input = HashMap::new();
55    for (sink_ident, sink_input_id) in sink_inputs {
56        if let Some(source_id) = sources.get(&sink_ident) {
57            source_to_sink_input.insert(*source_id, sink_input_id);
58        } else {
59            std::panic!(
60                "No source found for CycleSink {}, Input Id {}",
61                sink_ident,
62                sink_input_id
63            );
64        }
65    }
66    println!("Source to sink input: {:?}", source_to_sink_input);
67    source_to_sink_input
68}

70fn inject_location_leaf(
71    leaf: &mut HydroLeaf,
72    id_to_location: &RefCell<HashMap<usize, LocationId>>,
73    missing_location: &RefCell<bool>,
74) {
75    let inputs = leaf.input_metadata();
76    let input_metadata = inputs.first().unwrap();
77    let input_id = input_metadata.id.unwrap();
78
79    if let Some(location) = id_to_location.borrow().get(&input_metadata.id.unwrap()) {
80        let metadata = leaf.metadata_mut();
81        metadata.location_kind.swap_root(location.clone());
82
83        if let HydroLeaf::CycleSink { location_kind, .. } = leaf {
84            *location_kind = location.clone();
85            println!(
86                "Cycle sink with input {} has location {:?}",
87                input_id,
88                location.clone()
89            );
90        }
91    } else {
92        println!("Missing location for leaf: {:?}", leaf.print_root());
93        *missing_location.borrow_mut() = true;
94    }
95}

97fn inject_location_input_persist(input: &mut Box<HydroNode>, new_location: LocationId) {
98    if let HydroNode::Persist {
99        metadata: persist_metadata,
100        ..
101    } = input.as_mut()
102    {
103        persist_metadata.location_kind.swap_root(new_location);
104    }

107fn inject_location_node(
108    node: &mut HydroNode,
109    id_to_location: &RefCell<HashMap<usize, LocationId>>,
110    missing_location: &RefCell<bool>,
111    cycle_source_to_sink_input: &HashMap<usize, usize>,
112) {
113    if let Some(op_id) = node.metadata().id {
114        let inputs = match node {
115            HydroNode::Source { location_kind, .. }
116            | HydroNode::Network {
117                to_location: location_kind,
118                ..
119            } => {
120                // Get location sources from the nodes must have it be correct: Source and Network
121                id_to_location
122                    .borrow_mut()
123                    .insert(op_id, location_kind.clone());
124                return;
125            }
126            HydroNode::Tee { inner, .. } => {
127                vec![inner.0.borrow().metadata().id.unwrap()]
128            }
129            HydroNode::CycleSource { .. } => {
130                vec![*cycle_source_to_sink_input.get(&op_id).unwrap()]
131            }
132            _ => node
133                .input_metadata()
134                .iter()
135                .map(|input_metadata| input_metadata.id.unwrap())
136                .collect(),
137        };
138
139        // Otherwise, get it from (either) input
140        let metadata = node.metadata_mut();
141        for input in inputs {
142            let location = id_to_location.borrow().get(&input).cloned();
143            if let Some(location) = location {
144                metadata.location_kind.swap_root(location.clone());
145                id_to_location.borrow_mut().insert(op_id, location.clone());
146
147                match node {
148                    // Update Persist's location as well (we won't see it during traversal)
149                    HydroNode::CrossProduct { left, right, .. }
150                    | HydroNode::Join { left, right, .. } => {
151                        inject_location_input_persist(left, location.clone());
152                        inject_location_input_persist(right, location);
153                    }
154                    HydroNode::Difference { pos, neg, .. }
155                    | HydroNode::AntiJoin { pos, neg, .. } => {
156                        inject_location_input_persist(pos, location.clone());
157                        inject_location_input_persist(neg, location);
158                    }
159                    HydroNode::Fold { input, .. }
160                    | HydroNode::FoldKeyed { input, .. }
161                    | HydroNode::Reduce { input, .. }
162                    | HydroNode::ReduceKeyed { input, .. }
163                    | HydroNode::Scan { input, .. } => {
164                        inject_location_input_persist(input, location);
165                    }
166                    // CycleSource also stores the location outside of its metadata, so update it as well
167                    HydroNode::CycleSource { location_kind, .. } => {
168                        location_kind.swap_root(location);
169                    }
170                    _ => {}
171                }
172                return;
173            }
174        }
175
176        // If the location was not set, let the recursive function know
177        println!("Missing location for node: {:?}", node.print_root());
178        *missing_location.borrow_mut() = true;
179    }
180}

182pub fn inject_location(ir: &mut [HydroLeaf], cycle_source_to_sink_input: &HashMap<usize, usize>) {
183    let id_to_location = RefCell::new(HashMap::new());
184
185    loop {
186        println!("Attempting to inject location, looping until fixpoint...");
187        let missing_location = RefCell::new(false);
188
189        transform_bottom_up(
190            ir,
191            &mut |leaf| {
192                inject_location_leaf(leaf, &id_to_location, &missing_location);
193            },
194            &mut |node| {
195                inject_location_node(
196                    node,
197                    &id_to_location,
198                    &missing_location,
199                    cycle_source_to_sink_input,
200                );
201            },
202        );
203
204        if !*missing_location.borrow() {
205            println!("Locations injected!");
206            break;
207        }
208    }
209}

211fn remove_counter_node(node: &mut HydroNode, _next_stmt_id: &mut usize) {
212    if let HydroNode::Counter { input, .. } = node {
213        *node = std::mem::replace(input, HydroNode::Placeholder);
214    }
215}
