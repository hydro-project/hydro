22fn link_cycles_root(root: &mut HydroRoot, sink_inputs: &mut HashMap<Ident, usize>) {
23    if let HydroRoot::CycleSink { ident, input, .. } = root {
24        sink_inputs.insert(ident.clone(), input.op_metadata().id.unwrap());
25    }
26}

28fn link_cycles_node(node: &mut HydroNode, sources: &mut HashMap<Ident, usize>) {
29    if let HydroNode::CycleSource {
30        ident, metadata, ..
31    } = node
32    {
33        sources.insert(ident.clone(), metadata.op.id.unwrap());
34    }
35}

37// Returns map from CycleSource id to the input IDs of the corresponding CycleSink's input
38// Assumes that metadtata.id is set for all nodes
39pub fn cycle_source_to_sink_input(ir: &mut [HydroRoot]) -> HashMap<usize, usize> {
40    let mut sources = HashMap::new();
41    let mut sink_inputs = HashMap::new();
42
43    // Can't use traverse_dfir since that skips CycleSink
44    transform_bottom_up(
45        ir,
46        &mut |leaf| {
47            link_cycles_root(leaf, &mut sink_inputs);
48        },
49        &mut |node| {
50            link_cycles_node(node, &mut sources);
51        },
52        false,
53    );
54
55    let mut source_to_sink_input = HashMap::new();
56    for (sink_ident, sink_input_id) in sink_inputs {
57        if let Some(source_id) = sources.get(&sink_ident) {
58            source_to_sink_input.insert(*source_id, sink_input_id);
59        } else {
60            std::panic!(
61                "No source found for CycleSink {}, Input Id {}",
62                sink_ident,
63                sink_input_id
64            );
65        }
66    }
67    println!("Source to sink input: {:?}", source_to_sink_input);
68    source_to_sink_input
69}

71fn inject_location_root(
72    root: &mut HydroRoot,
73    id_to_location: &RefCell<HashMap<usize, LocationId>>,
74    missing_location: &RefCell<bool>,
75) {
76    let inputs = root.input_metadata();
77    let input_metadata = inputs.first().unwrap();
78
79    if let Some(location) = id_to_location.borrow().get(&input_metadata.op.id.unwrap()) {
80        if let HydroRoot::CycleSink { out_location, .. } = root {
81            out_location.swap_root(location.root().clone());
82        }
83    } else {
84        println!("Missing location for root: {:?}", root.print_root());
85        *missing_location.borrow_mut() = true;
86    }
87}

89fn inject_location_input_persist(input: &mut Box<HydroNode>, new_location: LocationId) {
90    if let HydroNode::Persist {
91        metadata: persist_metadata,
92        ..
93    } = input.as_mut()
94    {
95        persist_metadata.location_kind.swap_root(new_location);
96    }

99fn inject_location_node(
100    node: &mut HydroNode,
101    id_to_location: &RefCell<HashMap<usize, LocationId>>,
102    missing_location: &RefCell<bool>,
103    cycle_source_to_sink_input: &HashMap<usize, usize>,
104) {
105    if let Some(op_id) = node.op_metadata().id {
106        let inputs = match node {
107            HydroNode::Source { metadata, .. }
108            | HydroNode::ExternalInput { metadata, .. }
109            | HydroNode::Network { metadata, .. } => {
110                // Get location sources from the nodes must have it be correct: Source and Network
111                id_to_location
112                    .borrow_mut()
113                    .insert(op_id, metadata.location_kind.clone());
114                return;
115            }
116            HydroNode::Tee { inner, .. } => {
117                vec![inner.0.borrow().op_metadata().id.unwrap()]
118            }
119            HydroNode::CycleSource { .. } => {
120                vec![*cycle_source_to_sink_input.get(&op_id).unwrap()]
121            }
122            _ => node
123                .input_metadata()
124                .iter()
125                .map(|input_metadata| input_metadata.op.id.unwrap())
126                .collect(),
127        };
128
129        // Otherwise, get it from (either) input
130        let metadata = node.metadata_mut();
131        for input in inputs {
132            let location = id_to_location.borrow().get(&input).cloned();
133            if let Some(location) = location {
134                metadata.location_kind.swap_root(location.root().clone());
135                id_to_location
136                    .borrow_mut()
137                    .insert(op_id, metadata.location_kind.clone());
138
139                match node {
140                    // Update Persist's location as well (we won't see it during traversal)
141                    HydroNode::CrossProduct { left, right, .. }
142                    | HydroNode::Join { left, right, .. } => {
143                        inject_location_input_persist(left, location.root().clone());
144                        inject_location_input_persist(right, location.root().clone());
145                    }
146                    HydroNode::Difference { pos, neg, .. }
147                    | HydroNode::AntiJoin { pos, neg, .. } => {
148                        inject_location_input_persist(pos, location.root().clone());
149                        inject_location_input_persist(neg, location.root().clone());
150                    }
151                    HydroNode::Fold { input, .. }
152                    | HydroNode::FoldKeyed { input, .. }
153                    | HydroNode::Reduce { input, .. }
154                    | HydroNode::ReduceKeyed { input, .. }
155                    | HydroNode::Scan { input, .. } => {
156                        inject_location_input_persist(input, location.root().clone());
157                    }
158                    _ => {}
159                }
160                return;
161            }
162        }
163
164        // If the location was not set, let the recursive function know
165        println!("Missing location for node: {:?}", node.print_root());
166        *missing_location.borrow_mut() = true;
167    }
168}

170pub fn inject_location(ir: &mut [HydroRoot], cycle_source_to_sink_input: &HashMap<usize, usize>) {
171    let id_to_location = RefCell::new(HashMap::new());
172
173    loop {
174        println!("Attempting to inject location, looping until fixpoint...");
175        let missing_location = RefCell::new(false);
176
177        transform_bottom_up(
178            ir,
179            &mut |leaf| {
180                inject_location_root(leaf, &id_to_location, &missing_location);
181            },
182            &mut |node| {
183                inject_location_node(
184                    node,
185                    &id_to_location,
186                    &missing_location,
187                    cycle_source_to_sink_input,
188                );
189            },
190            false,
191        );
192
193        if !*missing_location.borrow() {
194            println!("Locations injected!");
195
196            // Check well-formedness here
197            transform_bottom_up(ir, &mut |_| {}, &mut |_| {}, true);
198            break;
199        }
200    }
201}

203fn remove_counter_node(node: &mut HydroNode, _next_stmt_id: &mut usize) {
204    if let HydroNode::Counter { input, .. } = node {
205        *node = std::mem::replace(input, HydroNode::Placeholder);
206    }
207}
