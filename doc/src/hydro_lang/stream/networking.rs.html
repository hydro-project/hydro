15pub fn serialize_bincode_with_type(is_demux: bool, t_type: &syn::Type) -> syn::Expr {
16    let root = get_this_crate();
17
18    if is_demux {
19        parse_quote! {
20            ::#root::runtime_support::stageleft::runtime_support::fn1_type_hint::<(#root::ClusterId<_>, #t_type), _>(
21                |(id, data)| {
22                    (id.raw_id, #root::runtime_support::bincode::serialize(&data).unwrap().into())
23                }
24            )
25        }
26    } else {
27        parse_quote! {
28            ::#root::runtime_support::stageleft::runtime_support::fn1_type_hint::<#t_type, _>(
29                |data| {
30                    #root::runtime_support::bincode::serialize(&data).unwrap().into()
31                }
32            )
33        }
34    }
35}

37fn serialize_bincode<T: Serialize>(is_demux: bool) -> syn::Expr {
38    serialize_bincode_with_type(is_demux, &quote_type::<T>())
39}

41pub fn deserialize_bincode_with_type(tagged: Option<&syn::Type>, t_type: &syn::Type) -> syn::Expr {
42    let root = get_this_crate();
43
44    if let Some(c_type) = tagged {
45        parse_quote! {
46            |res| {
47                let (id, b) = res.unwrap();
48                (#root::ClusterId::<#c_type>::from_raw(id), #root::runtime_support::bincode::deserialize::<#t_type>(&b).unwrap())
49            }
50        }
51    } else {
52        parse_quote! {
53            |res| {
54                #root::runtime_support::bincode::deserialize::<#t_type>(&res.unwrap()).unwrap()
55            }
56        }
57    }
58}
59
60pub(crate) fn deserialize_bincode<T: DeserializeOwned>(tagged: Option<&syn::Type>) -> syn::Expr {
61    deserialize_bincode_with_type(tagged, &quote_type::<T>())
62}

64impl<'a, T, L, B, O, R> Stream<T, Cluster<'a, L>, B, O, R> {
65    pub fn send_bincode<L2>(
66        self,
67        other: &Process<'a, L2>,
68    ) -> KeyedStream<ClusterId<L>, T, Process<'a, L2>, Unbounded, O, R>
69    where
70        T: Serialize + DeserializeOwned,
71    {
72        let serialize_pipeline = Some(serialize_bincode::<T>(false));
73
74        let deserialize_pipeline = Some(deserialize_bincode::<T>(Some(&quote_type::<L>())));
75
76        let raw_stream: Stream<(ClusterId<L>, T), Process<'a, L2>, Unbounded, O, R> = Stream::new(
77            other.clone(),
78            HydroNode::Network {
79                serialize_fn: serialize_pipeline.map(|e| e.into()),
80                instantiate_fn: DebugInstantiate::Building,
81                deserialize_fn: deserialize_pipeline.map(|e| e.into()),
82                input: Box::new(self.ir_node.into_inner()),
83                metadata: other.new_node_metadata::<(ClusterId<L>, T)>(),
84            },
85        );
86
87        raw_stream.into_keyed()
88    }
89
90    pub fn broadcast_bincode<L2: 'a>(
91        self,
92        other: &Cluster<'a, L2>,
93    ) -> KeyedStream<ClusterId<L>, T, Cluster<'a, L2>, Unbounded, O, R>
94    where
95        T: Clone + Serialize + DeserializeOwned,
96    {
97        let ids = other.members();
98        self.flat_map_ordered(q!(|v| { ids.iter().map(move |id| (*id, v.clone())) }))
99            .demux_bincode(other)
100    }

140impl<'a, T, L, B> Stream<T, Process<'a, L>, B, TotalOrder, ExactlyOnce> {
141    pub fn round_robin_bincode<L2: 'a>(
142        self,
143        other: &Cluster<'a, L2>,
144    ) -> Stream<T, Cluster<'a, L2>, Unbounded, TotalOrder, ExactlyOnce>
145    where
146        T: Serialize + DeserializeOwned,
147    {
148        let ids = other.members();
149
150        self.enumerate()
151            .map(q!(|(i, w)| (ids[i % ids.len()], w)))
152            .demux_bincode(other)
153    }
154}

231    pub fn send_bincode_external<L2>(self, other: &External<L2>) -> ExternalBincodeStream<T>
232    where
233        T: Serialize + DeserializeOwned,
234    {
235        let serialize_pipeline = Some(serialize_bincode::<T>(false));
236
237        let mut flow_state_borrow = self.location.flow_state().borrow_mut();
238
239        let external_key = flow_state_borrow.next_external_out;
240        flow_state_borrow.next_external_out += 1;
241
242        let leaves = flow_state_borrow.leaves.as_mut().expect("Attempted to add a leaf to a flow that has already been finalized. No leaves can be added after the flow has been compiled()");
243
244        leaves.push(HydroLeaf::SendExternal {
245            to_external_id: other.id,
246            to_key: external_key,
247            to_many: false,
248            serialize_fn: serialize_pipeline.map(|e| e.into()),
249            instantiate_fn: DebugInstantiate::Building,
250            input: Box::new(HydroNode::Unpersist {
251                inner: Box::new(self.ir_node.into_inner()),
252                metadata: self.location.new_node_metadata::<T>(),
253            }),
254        });
255
256        ExternalBincodeStream {
257            process_id: other.id,
258            port_id: external_key,
259            _phantom: PhantomData,
260        }
261    }
