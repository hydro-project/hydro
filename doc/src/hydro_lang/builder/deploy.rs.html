145impl<'a, D: Deploy<'a>> DeployFlow<'a, D> {
146    pub fn compile(mut self, env: &D::CompileEnv) -> CompiledFlow<'a, D::GraphId> {
147        let mut seen_tees: HashMap<_, _> = HashMap::new();
148        let mut seen_tee_locations: HashMap<_, _> = HashMap::new();
149        self.ir.get_mut().iter_mut().for_each(|leaf| {
150            leaf.compile_network::<D>(
151                env,
152                &mut seen_tees,
153                &mut seen_tee_locations,
154                &self.processes,
155                &self.clusters,
156                &self.externals,
157            );
158        });
159
160        CompiledFlow {
161            dfir: build_inner(self.ir.get_mut()),
162            _phantom: PhantomData,
163        }
164    }
165
166    fn extra_stmts(&self, env: &<D as Deploy<'a>>::CompileEnv) -> BTreeMap<usize, Vec<syn::Stmt>> {
167        let mut extra_stmts: BTreeMap<usize, Vec<syn::Stmt>> = BTreeMap::new();
168
169        let mut all_clusters_sorted = self.clusters.keys().collect::<Vec<_>>();
170        all_clusters_sorted.sort();
171
172        for &c_id in all_clusters_sorted {
173            let self_id_ident = syn::Ident::new(
174                &format!("__hydro_lang_cluster_self_id_{}", c_id),
175                Span::call_site(),
176            );
177            let self_id_expr = D::cluster_self_id(env).splice_untyped();
178            extra_stmts
179                .entry(c_id)
180                .or_default()
181                .push(syn::parse_quote! {
182                    let #self_id_ident = #self_id_expr;
183                });
184
185            for other_location in self.processes.keys().chain(self.clusters.keys()) {
186                let other_id_ident = syn::Ident::new(
187                    &format!("__hydro_lang_cluster_ids_{}", c_id),
188                    Span::call_site(),
189                );
190                let other_id_expr = D::cluster_ids(env, c_id).splice_untyped();
191                extra_stmts
192                    .entry(*other_location)
193                    .or_default()
194                    .push(syn::parse_quote! {
195                        let #other_id_ident = #other_id_expr;
196                    });
197            }
198        }
199        extra_stmts
200    }
201}

203impl<'a, D: Deploy<'a, CompileEnv = ()>> DeployFlow<'a, D> {
204    #[must_use]
205    pub fn deploy(mut self, env: &mut D::InstantiateEnv) -> DeployResult<'a, D> {
206        let mut seen_tees_instantiate: HashMap<_, _> = HashMap::new();
207        let mut seen_tee_locations: HashMap<_, _> = HashMap::new();
208        self.ir.get_mut().iter_mut().for_each(|leaf| {
209            leaf.compile_network::<D>(
210                &(),
211                &mut seen_tees_instantiate,
212                &mut seen_tee_locations,
213                &self.processes,
214                &self.clusters,
215                &self.externals,
216            );
217        });
218
219        let mut compiled = build_inner(self.ir.get_mut());
220        let mut extra_stmts = self.extra_stmts(&());
221        let mut meta = D::Meta::default();
222
223        let (mut processes, mut clusters, mut externals) = (
224            std::mem::take(&mut self.processes)
225                .into_iter()
226                .filter_map(|(node_id, node)| {
227                    if let Some(ir) = compiled.remove(&node_id) {
228                        node.instantiate(
229                            env,
230                            &mut meta,
231                            ir,
232                            extra_stmts.remove(&node_id).unwrap_or_default(),
233                        );
234                        Some((node_id, node))
235                    } else {
236                        None
237                    }
238                })
239                .collect::<HashMap<_, _>>(),
240            std::mem::take(&mut self.clusters)
241                .into_iter()
242                .filter_map(|(cluster_id, cluster)| {
243                    if let Some(ir) = compiled.remove(&cluster_id) {
244                        cluster.instantiate(
245                            env,
246                            &mut meta,
247                            ir,
248                            extra_stmts.remove(&cluster_id).unwrap_or_default(),
249                        );
250                        Some((cluster_id, cluster))
251                    } else {
252                        None
253                    }
254                })
255                .collect::<HashMap<_, _>>(),
256            std::mem::take(&mut self.externals)
257                .into_iter()
258                .map(|(external_id, external)| {
259                    external.instantiate(
260                        env,
261                        &mut meta,
262                        compiled.remove(&external_id).unwrap(),
263                        extra_stmts.remove(&external_id).unwrap_or_default(),
264                    );
265                    (external_id, external)
266                })
267                .collect::<HashMap<_, _>>(),
268        );
269
270        for node in processes.values_mut() {
271            node.update_meta(&meta);
272        }
273
274        for cluster in clusters.values_mut() {
275            cluster.update_meta(&meta);
276        }
277
278        for external in externals.values_mut() {
279            external.update_meta(&meta);
280        }
281
282        let mut seen_tees_connect = HashMap::new();
283        self.ir.get_mut().iter_mut().for_each(|leaf| {
284            leaf.connect_network(&mut seen_tees_connect);
285        });
286
287        DeployResult {
288            processes,
289            clusters,
290            externals,
291            cluster_id_name: std::mem::take(&mut self.cluster_id_name)
292                .into_iter()
293                .collect(),
294        }
295    }
296}

305impl<'a, D: Deploy<'a>> DeployResult<'a, D> {
306    pub fn get_process<P>(&self, p: &Process<P>) -> &D::Process {
307        let id = match p.id() {
308            LocationId::Process(id) => id,
309            _ => panic!("Process ID expected"),
310        };
311
312        self.processes.get(&id).unwrap()
313    }
314
315    pub fn get_cluster<C>(&self, c: &Cluster<'a, C>) -> &D::Cluster {
316        let id = match c.id() {
317            LocationId::Cluster(id) => id,
318            _ => panic!("Cluster ID expected"),
319        };
320
321        self.clusters.get(&id).unwrap()
322    }
323
324    pub fn get_all_clusters(&self) -> impl Iterator<Item = (LocationId, String, &D::Cluster)> {
325        self.clusters.iter().map(|(&id, c)| {
326            (
327                LocationId::Cluster(id),
328                self.cluster_id_name.get(&id).unwrap().clone(),
329                c,
330            )
331        })
332    }
333
334    pub fn get_external<P>(&self, p: &ExternalProcess<P>) -> &D::ExternalProcess {
335        self.externals.get(&p.id).unwrap()
336    }
