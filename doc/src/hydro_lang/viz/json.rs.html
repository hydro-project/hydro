246    /// Optimize backtrace data for size efficiency
247    /// 1. Remove redundant/non-essential frames
248    /// 2. Truncate paths
249    /// 3. Remove memory addresses (not useful for visualization)
250    fn optimize_backtrace(&self, backtrace: &Backtrace) -> serde_json::Value {
251        #[cfg(feature = "build")]
252        {
253            let elements = backtrace.elements();
254
255            // filter out obviously internal frames
256            let relevant_frames: Vec<BacktraceFrame> = elements
257                .iter()
258                .map(|elem| {
259                    // Truncate paths and function names for size
260                    let short_filename = elem
261                        .filename
262                        .as_deref()
263                        .map(|f| Self::truncate_path(f))
264                        .unwrap_or_else(|| "unknown".to_string());
265
266                    let short_fn_name = Self::truncate_function_name(&elem.fn_name);
267
268                    BacktraceFrame {
269                        fn_name: short_fn_name.clone(),
270                        function: short_fn_name,
271                        file: short_filename.clone(),
272                        filename: short_filename,
273                        line: elem.lineno,
274                        line_number: elem.lineno,
275                    }
276                })
277                .collect();
278
279            serde_json::to_value(relevant_frames).unwrap_or_else(|_| serde_json::json!([]))
280        }
281        #[cfg(not(feature = "build"))]
282        {
283            serde_json::json!([])
284        }
285    }
286
287    /// Truncate file paths to keep only the relevant parts
288    fn truncate_path(path: &str) -> String {
289        let parts: Vec<&str> = path.split('/').collect();
290
291        // For paths like "/Users/foo/project/src/main.rs", keep "src/main.rs"
292        if let Some(src_idx) = parts.iter().rposition(|&p| p == "src") {
293            parts[src_idx..].join("/")
294        } else if parts.len() > 2 {
295            // Keep last 2 components
296            parts[parts.len().saturating_sub(2)..].join("/")
297        } else {
298            path.to_string()
299        }
300    }
301
302    /// Truncate function names to remove module paths
303    fn truncate_function_name(fn_name: &str) -> String {
304        // Remove everything before the last "::" to get just the function name
305        fn_name.split("::").last().unwrap_or(fn_name).to_string()
306    }

326    fn write_node_definition(
327        &mut self,
328        node_id: usize,
329        node_label: &super::render::NodeLabel,
330        node_type: HydroNodeType,
331        location_id: Option<usize>,
332        location_kind: Option<&LocationId>,
333        backtrace: Option<&Backtrace>,
334    ) -> Result<(), Self::Err> {
335        // Extract tick information and type from location_kind if present
336        let tick_id = location_kind.and_then(|loc_kind| match loc_kind {
337            LocationId::Tick(tick, _) => Some(*tick),
338            _ => None,
339        });
340        // serialize the location type to JSON-friendly string
341        let derived_location_type: Option<String> =
342            location_kind.and_then(|loc_kind| match loc_kind.root() {
343                LocationId::Process(_) => Some("Process".to_string()),
344                LocationId::Cluster(_) => Some("Cluster".to_string()),
345                _ => None,
346            });
347
348        // Create the full label string using DebugExpr::Display for expressions
349        let full_label = match node_label {
350            super::render::NodeLabel::Static(s) => s.clone(),
351            super::render::NodeLabel::WithExprs { op_name, exprs } => {
352                if exprs.is_empty() {
353                    format!("{}()", op_name)
354                } else {
355                    // This is where DebugExpr::Display gets called with q! macro cleanup
356                    let expr_strs: Vec<String> = exprs.iter().map(|e| e.to_string()).collect();
357                    format!("{}({})", op_name, expr_strs.join(", "))
358                }
359            }
360        };
361
362        // Always extract short label for UI toggle functionality
363        let short_label = super::render::extract_short_label(&full_label);
364
365        // If short and full labels are the same or very similar, enhance the full label
366        // Use saturating comparison to avoid underflow when full_label is very short
367        let full_len = full_label.len();
368        let enhanced_full_label = if short_label.len() >= full_len.saturating_sub(2) {
369            // If they're nearly the same length, add more context to full label
370            match short_label.as_str() {
371                "inspect" => "inspect [debug output]".to_string(),
372                "persist" => "persist [state storage]".to_string(),
373                "tee" => "tee [branch dataflow]".to_string(),
374                "delta" => "delta [change detection]".to_string(),
375                "spin" => "spin [delay/buffer]".to_string(),
376                "send_bincode" => "send_bincode [send data to process/cluster]".to_string(),
377                "broadcast_bincode" => {
378                    "broadcast_bincode [send data to all cluster members]".to_string()
379                }
380                "source_iter" => "source_iter [iterate over collection]".to_string(),
381                "source_stream" => "source_stream [receive external data stream]".to_string(),
382                "network(recv)" => "network(recv) [receive from network]".to_string(),
383                "network(send)" => "network(send) [send to network]".to_string(),
384                "dest_sink" => "dest_sink [output destination]".to_string(),
385                _ => {
386                    if full_label.len() < 15 {
387                        format!("{} [{}]", node_label, "hydro operator")
388                    } else {
389                        node_label.to_string()
390                    }
391                }
392            }
393        } else {
394            node_label.to_string()
395        };
396
397        // Convert backtrace to JSON if available (optimized for size)
398        let backtrace_json = if let Some(bt) = backtrace {
399            // Store backtrace for hierarchy generation
400            self.node_backtraces.insert(node_id, bt.clone());
401            self.optimize_backtrace(bt)
402        } else {
403            serde_json::json!([])
404        };
405
406        // Node type string for styling/legend
407        let node_type_str = Self::node_type_to_string(node_type);
408
409        let node = Node {
410            id: node_id.to_string(),
411            node_type: node_type_str.to_string(),
412            full_label: enhanced_full_label,
413            short_label: short_label.clone(),
414            // Primary display label follows configuration (defaults to short)
415            label: if self.use_short_labels {
416                short_label
417            } else {
418                full_label
419            },
420            data: NodeData {
421                location_id,
422                location_type: derived_location_type,
423                tick_id,
424                backtrace: backtrace_json,
425            },
426        };
427        self.nodes
428            .push(serde_json::to_value(node).expect("Node serialization should not fail"));
429
430        // Track node location for cross-location edge detection
431        if let Some(loc_id) = location_id {
432            self.node_locations.insert(node_id, loc_id);
433        }
434
435        // Track node tick for hierarchy generation
436        if let Some(tid) = tick_id {
437            self.node_ticks.insert(node_id, tid);
438        }
439
440        Ok(())
441    }

443    fn write_edge(
444        &mut self,
445        src_id: usize,
446        dst_id: usize,
447        edge_properties: &HashSet<HydroEdgeProp>,
448        label: Option<&str>,
449    ) -> Result<(), Self::Err> {
450        let edge_id = format!("e{}", self.edge_count);
451        self.edge_count = self.edge_count.saturating_add(1);
452
453        // Convert edge properties to semantic tags (string array)
454        let mut semantic_tags: Vec<String> = edge_properties
455            .iter()
456            .map(|p| Self::edge_type_to_string(*p))
457            .collect();
458
459        // Get location information for styling
460        let src_loc = self.node_locations.get(&src_id).copied();
461        let dst_loc = self.node_locations.get(&dst_id).copied();
462
463        // Add Network tag if edge crosses locations; otherwise add Local for completeness
464        if let (Some(src), Some(dst)) = (src_loc, dst_loc)
465            && src != dst
466            && !semantic_tags.iter().any(|t| t == "Network")
467        {
468            semantic_tags.push("Network".to_string());
469        } else if semantic_tags.iter().all(|t| t != "Network") {
470            // Only add Local if Network not present (complement for styling)
471            semantic_tags.push("Local".to_string());
472        }
473
474        // Ensure deterministic ordering of semantic tags
475        semantic_tags.sort();
476
477        let edge = Edge {
478            id: edge_id,
479            source: src_id.to_string(),
480            target: dst_id.to_string(),
481            semantic_tags,
482            label: label.map(|s| s.to_string()),
483        };
484
485        self.edges
486            .push(serde_json::to_value(edge).expect("Edge serialization should not fail"));
487        Ok(())
488    }

543    fn write_epilogue(&mut self) -> Result<(), Self::Err> {
544        // Create multiple hierarchy options
545        let mut hierarchy_choices = Vec::new();
546        let mut node_assignments_choices = serde_json::Map::new();
547
548        // Always add location-based hierarchy
549        let (location_hierarchy, location_assignments) = self.create_location_hierarchy();
550        hierarchy_choices.push(serde_json::json!({
551            "id": "location",
552            "name": "Location",
553            "children": location_hierarchy
554        }));
555        node_assignments_choices.insert(
556            "location".to_string(),
557            serde_json::Value::Object(location_assignments),
558        );
559
560        // Add backtrace-based hierarchy if available
561        if self.has_backtrace_data() {
562            let (backtrace_hierarchy, backtrace_assignments) = self.create_backtrace_hierarchy();
563            hierarchy_choices.push(serde_json::json!({
564                "id": "backtrace",
565                "name": "Backtrace",
566                "children": backtrace_hierarchy
567            }));
568            node_assignments_choices.insert(
569                "backtrace".to_string(),
570                serde_json::Value::Object(backtrace_assignments),
571            );
572        }
573
574        // Before serialization, enforce deterministic ordering for nodes and edges
575        let mut nodes_sorted = self.nodes.clone();
576        nodes_sorted.sort_by(|a, b| a["id"].as_str().cmp(&b["id"].as_str()));
577        let mut edges_sorted = self.edges.clone();
578        edges_sorted.sort_by(|a, b| {
579            let a_src = a["source"].as_str();
580            let b_src = b["source"].as_str();
581            match a_src.cmp(&b_src) {
582                std::cmp::Ordering::Equal => {
583                    let a_dst = a["target"].as_str();
584                    let b_dst = b["target"].as_str();
585                    match a_dst.cmp(&b_dst) {
586                        std::cmp::Ordering::Equal => a["id"].as_str().cmp(&b["id"].as_str()),
587                        other => other,
588                    }
589                }
590                other => other,
591            }
592        });
593
594        // Create the final JSON structure in the format expected by the visualizer
595        let node_type_definitions = Self::get_node_type_definitions();
596        let legend_items = Self::get_legend_items();
597
598        let node_type_config = serde_json::json!({
599            "types": node_type_definitions,
600            "defaultType": "Transform"
601        });
602        let legend = serde_json::json!({
603            "title": "Node Types",
604            "items": legend_items
605        });
606
607        // Determine the selected hierarchy (first one is default)
608        let selected_hierarchy = if !hierarchy_choices.is_empty() {
609            hierarchy_choices[0]["id"].as_str().map(|s| s.to_string())
610        } else {
611            None
612        };
613
614        #[derive(serde::Serialize)]
615        struct GraphPayload {
616            nodes: Vec<serde_json::Value>,
617            edges: Vec<serde_json::Value>,
618            #[serde(rename = "hierarchyChoices")]
619            hierarchy_choices: Vec<serde_json::Value>,
620            #[serde(rename = "nodeAssignments")]
621            node_assignments: serde_json::Map<String, serde_json::Value>,
622            #[serde(rename = "selectedHierarchy", skip_serializing_if = "Option::is_none")]
623            selected_hierarchy: Option<String>,
624            #[serde(rename = "edgeStyleConfig")]
625            edge_style_config: serde_json::Value,
626            #[serde(rename = "nodeTypeConfig")]
627            node_type_config: serde_json::Value,
628            legend: serde_json::Value,
629        }
630
631        let payload = GraphPayload {
632            nodes: nodes_sorted,
633            edges: edges_sorted,
634            hierarchy_choices,
635            node_assignments: node_assignments_choices,
636            selected_hierarchy,
637            edge_style_config: Self::get_edge_style_config(),
638            node_type_config,
639            legend,
640        };
641
642        let final_json = serde_json::to_string_pretty(&payload).unwrap();
643
644        write!(self.write, "{}", final_json)
645    }
646}

649    /// Check if any nodes have meaningful backtrace data
650    fn has_backtrace_data(&self) -> bool {
651        self.nodes.iter().any(|node| {
652            if let Some(backtrace_array) = node["data"]["backtrace"].as_array() {
653                // Check if any frame has meaningful filename or fn_name data
654                backtrace_array.iter().any(|frame| {
655                    let filename = frame["file"].as_str().unwrap_or("");
656                    let fn_name = frame["fn"].as_str().unwrap_or("");
657                    !filename.is_empty() || !fn_name.is_empty()
658                })
659            } else {
660                false
661            }
662        })
663    }

665    /// Create location-based hierarchy with nested Tick support
666    fn create_location_hierarchy(
667        &self,
668    ) -> (
669        Vec<serde_json::Value>,
670        serde_json::Map<String, serde_json::Value>,
671    ) {
672        use std::collections::{HashMap, HashSet};
673
674        // Define the key type once for UnionFind operations
675        slotmap::new_key_type! { struct NodeKey; }
676
677        // Group nodes by (location_id, optional tick_id)
678        let mut location_tick_nodes: HashMap<usize, HashMap<Option<usize>, Vec<String>>> =
679            HashMap::new();
680
681        for node in &self.nodes {
682            if let Some(node_id_str) = node["id"].as_str() {
683                let location_id = node["data"]["locationId"].as_u64().map(|v| v as usize);
684                let tick_id = node["data"]["tickId"].as_u64().map(|v| v as usize);
685
686                if let Some(loc_id) = location_id {
687                    location_tick_nodes
688                        .entry(loc_id)
689                        .or_default()
690                        .entry(tick_id)
691                        .or_default()
692                        .push(node_id_str.to_string());
693                }
694            }
695        }
696
697        // Build hierarchy: locations as top level, with Tick sub-containers when present
698        let mut locs: Vec<(&usize, &(String, Vec<usize>))> = self.locations.iter().collect();
699        locs.sort_by(|a, b| a.0.cmp(b.0));
700
701        let mut hierarchy = Vec::new();
702        let mut node_assignments = serde_json::Map::new();
703
704        for (location_id, (label, _)) in locs {
705            let location_key = format!("loc_{}", location_id);
706
707            if let Some(tick_groups) = location_tick_nodes.get(location_id) {
708                // Collect all tick IDs present for this location
709                let tick_ids: HashSet<Option<usize>> = tick_groups.keys().cloned().collect();
710
711                // If there are nodes in ticks, create sub-containers for each tick
712                if tick_ids.len() > 1 || (tick_ids.len() == 1 && !tick_ids.contains(&None)) {
713                    // Multiple ticks or at least one tick with ID - create nested structure
714                    let mut tick_children = Vec::new();
715                    let mut sorted_ticks: Vec<Option<usize>> = tick_ids.into_iter().collect();
716                    sorted_ticks.sort_by(|a, b| {
717                        match (a, b) {
718                            (None, None) => std::cmp::Ordering::Equal,
719                            (None, Some(_)) => std::cmp::Ordering::Less, // None comes first
720                            (Some(_), None) => std::cmp::Ordering::Greater,
721                            (Some(a_id), Some(b_id)) => a_id.cmp(b_id),
722                        }
723                    });
724
725                    // disconnected components within each tick group belong to independent containers
726                    // need to find connected components per tick group using UnionFind
727                    for tick_id_opt in sorted_ticks {
728                        if let Some(tick_id) = tick_id_opt {
729                            let base_tick_key = format!("{}_tick_{}", location_key, tick_id);
730
731                            // Build connected components for this (location, tick) using UnionFind
732                            let nodes_str: Vec<String> =
733                                tick_groups.get(&tick_id_opt).cloned().unwrap_or_default();
734                            let mut node_nums: Vec<usize> = nodes_str
735                                .iter()
736                                .filter_map(|s| s.parse::<usize>().ok())
737                                .collect();
738                            node_nums.sort_unstable();
739                            let nodes_set: std::collections::HashSet<usize> =
740                                node_nums.iter().cloned().collect();
741
742                            // Create a SlotMap to map node IDs to keys for UnionFind
743                            let mut node_map: SlotMap<NodeKey, usize> = SlotMap::with_key();
744                            let mut node_to_key: HashMap<usize, NodeKey> = HashMap::new();
745                            for &node_id in &node_nums {
746                                let key = node_map.insert(node_id);
747                                node_to_key.insert(node_id, key);
748                            }
749
750                            // Use UnionFind from dfir_lang to find connected components
751                            let mut uf =
752                                dfir_lang::union_find::UnionFind::with_capacity(node_map.len());
753
754                            // Union nodes that are connected by edges
755                            for edge in &self.edges {
756                                let s = edge["source"]
757                                    .as_str()
758                                    .and_then(|v| v.parse::<usize>().ok());
759                                let t = edge["target"]
760                                    .as_str()
761                                    .and_then(|v| v.parse::<usize>().ok());
762                                if let (Some(sid), Some(tid)) = (s, t)
763                                    && nodes_set.contains(&sid)
764                                    && nodes_set.contains(&tid)
765                                    && let (Some(&s_key), Some(&t_key)) =
766                                        (node_to_key.get(&sid), node_to_key.get(&tid))
767                                {
768                                    uf.union(s_key, t_key);
769                                }
770                            }
771
772                            // Group nodes by their component representative
773                            let mut component_map: HashMap<NodeKey, Vec<usize>> = HashMap::new();
774                            for &node_id in &node_nums {
775                                if let Some(&key) = node_to_key.get(&node_id) {
776                                    let root = uf.find(key);
777                                    component_map.entry(root).or_default().push(node_id);
778                                }
779                            }
780
781                            // Convert to vector and sort for determinism
782                            let mut components: Vec<Vec<usize>> = component_map
783                                .into_values()
784                                .map(|mut nodes| {
785                                    nodes.sort_unstable();
786                                    nodes
787                                })
788                                .collect();
789                            // Sort components by smallest node id for determinism
790                            components.sort_by_key(|c| c.first().cloned().unwrap_or(usize::MAX));
791
792                            if components.is_empty() {
793                                // No edges and possibly no nodes (shouldn't happen), create empty container for consistency
794                                tick_children.push(serde_json::json!({
795                                    "id": base_tick_key.clone(),
796                                    "name": format!("Tick({})", tick_id),
797                                    "children": []
798                                }));
799                            } else if components.len() == 1 {
800                                // Single component: single Tick container
801                                let tick_key = base_tick_key.clone();
802                                tick_children.push(serde_json::json!({
803                                    "id": tick_key,
804                                    "name": format!("Tick({})", tick_id),
805                                    "children": []
806                                }));
807                                for n in &components[0] {
808                                    node_assignments.insert(
809                                        n.to_string(),
810                                        serde_json::Value::String(base_tick_key.clone()),
811                                    );
812                                }
813                            } else {
814                                // Multiple components: split into multiple Tick containers
815                                for (idx, comp) in components.iter().enumerate() {
816                                    let tick_key = format!("{}_c{}", base_tick_key, idx + 1);
817                                    tick_children.push(serde_json::json!({
818                                        "id": tick_key.clone(),
819                                        "name": format!("Tick({})", tick_id),
820                                        "children": []
821                                    }));
822                                    for n in comp {
823                                        node_assignments.insert(
824                                            n.to_string(),
825                                            serde_json::Value::String(tick_key.clone()),
826                                        );
827                                    }
828                                }
829                            }
830                        } else {
831                            // Nodes without tick go directly under location
832                            if let Some(nodes) = tick_groups.get(&None) {
833                                for node_id in nodes {
834                                    node_assignments.insert(
835                                        node_id.clone(),
836                                        serde_json::Value::String(location_key.clone()),
837                                    );
838                                }
839                            }
840                        }
841                    }
842
843                    hierarchy.push(serde_json::json!({
844                        "id": location_key,
845                        "name": label,
846                        "children": tick_children
847                    }));
848                } else {
849                    // Only nodes without tick IDs - flat structure
850                    hierarchy.push(serde_json::json!({
851                        "id": location_key.clone(),
852                        "name": label,
853                        "children": []
854                    }));
855
856                    // Assign all nodes directly to location
857                    if let Some(nodes) = tick_groups.get(&None) {
858                        for node_id in nodes {
859                            node_assignments.insert(
860                                node_id.clone(),
861                                serde_json::Value::String(location_key.clone()),
862                            );
863                        }
864                    }
865                }
866            } else {
867                // No nodes for this location - create empty container
868                hierarchy.push(serde_json::json!({
869                    "id": location_key,
870                    "name": label,
871                    "children": []
872                }));
873            }
874        }
875
876        (hierarchy, node_assignments)
877    }

879    /// Create backtrace-based hierarchy using structured backtrace data
880    fn create_backtrace_hierarchy(
881        &self,
882    ) -> (
883        Vec<serde_json::Value>,
884        serde_json::Map<String, serde_json::Value>,
885    ) {
886        use std::collections::HashMap;
887
888        let mut hierarchy_map: HashMap<String, (String, usize, Option<String>)> = HashMap::new(); // path -> (name, depth, parent_path)
889        let mut path_to_node_assignments: HashMap<String, Vec<String>> = HashMap::new(); // path -> [node_ids]
890
891        // Process each node's backtrace using the stored backtraces
892        for node in &self.nodes {
893            if let Some(node_id_str) = node["id"].as_str()
894                && let Ok(node_id) = node_id_str.parse::<usize>()
895                && let Some(backtrace) = self.node_backtraces.get(&node_id)
896            {
897                let elements = backtrace.elements();
898                if elements.is_empty() {
899                    continue;
900                }
901
902                // Do not filter frames for now
903                let user_frames = elements;
904                if user_frames.is_empty() {
905                    continue;
906                }
907
908                // Build hierarchy path from backtrace frames (reverse order for call stack)
909                let mut hierarchy_path = Vec::new();
910                for (i, elem) in user_frames.iter().rev().enumerate() {
911                    let label = if i == 0 {
912                        if let Some(filename) = &elem.filename {
913                            Self::extract_file_path(filename)
914                        } else {
915                            format!("fn_{}", Self::truncate_function_name(&elem.fn_name))
916                        }
917                    } else {
918                        Self::truncate_function_name(&elem.fn_name)
919                    };
920                    hierarchy_path.push(label);
921                }
922
923                // Create hierarchy nodes for this path
924                let mut current_path = String::new();
925                let mut parent_path: Option<String> = None;
926                let mut deepest_path = String::new();
927                // Deduplicate consecutive identical labels for cleanliness
928                let mut deduped: Vec<String> = Vec::new();
929                for seg in hierarchy_path {
930                    if deduped.last().map(|s| s == &seg).unwrap_or(false) {
931                        continue;
932                    }
933                    deduped.push(seg);
934                }
935                for (depth, label) in deduped.iter().enumerate() {
936                    current_path = if current_path.is_empty() {
937                        label.clone()
938                    } else {
939                        format!("{}/{}", current_path, label)
940                    };
941                    if !hierarchy_map.contains_key(&current_path) {
942                        hierarchy_map.insert(
943                            current_path.clone(),
944                            (label.clone(), depth, parent_path.clone()),
945                        );
946                    }
947                    deepest_path = current_path.clone();
948                    parent_path = Some(current_path.clone());
949                }
950
951                if !deepest_path.is_empty() {
952                    path_to_node_assignments
953                        .entry(deepest_path)
954                        .or_default()
955                        .push(node_id_str.to_string());
956                }
957            }
958        }
959        // Build hierarchy tree and create proper ID mapping (deterministic)
960        let (mut hierarchy, mut path_to_id_map, id_remapping) =
961            self.build_hierarchy_tree_with_ids(&hierarchy_map);
962
963        // Create a root node for nodes without backtraces
964        let root_id = "bt_root".to_string();
965        let mut nodes_without_backtrace = Vec::new();
966
967        // Collect all node IDs
968        for node in &self.nodes {
969            if let Some(node_id_str) = node["id"].as_str() {
970                nodes_without_backtrace.push(node_id_str.to_string());
971            }
972        }
973
974        // Remove nodes that already have backtrace assignments
975        for node_ids in path_to_node_assignments.values() {
976            for node_id in node_ids {
977                nodes_without_backtrace.retain(|id| id != node_id);
978            }
979        }
980
981        // If there are nodes without backtraces, create a root container for them
982        if !nodes_without_backtrace.is_empty() {
983            hierarchy.push(serde_json::json!({
984                "id": root_id.clone(),
985                "name": "(no backtrace)",
986                "children": []
987            }));
988            path_to_id_map.insert("__root__".to_string(), root_id.clone());
989        }
990
991        // Create node assignments using the actual hierarchy IDs
992        let mut node_assignments = serde_json::Map::new();
993        let mut pairs: Vec<(String, Vec<String>)> = path_to_node_assignments.into_iter().collect();
994        pairs.sort_by(|a, b| a.0.cmp(&b.0));
995        for (path, mut node_ids) in pairs {
996            node_ids.sort();
997            if let Some(hierarchy_id) = path_to_id_map.get(&path) {
998                for node_id in node_ids {
999                    node_assignments
1000                        .insert(node_id, serde_json::Value::String(hierarchy_id.clone()));
1001                }
1002            }
1003        }
1004
1005        // Assign nodes without backtraces to the root
1006        for node_id in nodes_without_backtrace {
1007            node_assignments.insert(node_id, serde_json::Value::String(root_id.clone()));
1008        }
1009
1010        // CRITICAL FIX: Apply ID remapping to node assignments
1011        // When containers are collapsed, their IDs change, but nodeAssignments still reference old IDs
1012        // We need to update all assignments to use the new (collapsed) container IDs
1013        let mut remapped_assignments = serde_json::Map::new();
1014        for (node_id, container_id_value) in node_assignments.iter() {
1015            if let Some(container_id) = container_id_value.as_str() {
1016                // Check if this container ID was remapped during collapsing
1017                let final_container_id = id_remapping
1018                    .get(container_id)
1019                    .map(|s| s.as_str())
1020                    .unwrap_or(container_id);
1021                remapped_assignments.insert(
1022                    node_id.clone(),
1023                    serde_json::Value::String(final_container_id.to_string()),
1024                );
1025            }
1026        }
1027
1028        (hierarchy, remapped_assignments)
1029    }

1031    /// Build a tree structure and return both the tree and path-to-ID mapping
1032    fn build_hierarchy_tree_with_ids(
1033        &self,
1034        hierarchy_map: &HashMap<String, (String, usize, Option<String>)>,
1035    ) -> (
1036        Vec<serde_json::Value>,
1037        HashMap<String, String>,
1038        HashMap<String, String>,
1039    ) {
1040        // Assign IDs deterministically based on sorted path names
1041        let mut keys: Vec<&String> = hierarchy_map.keys().collect();
1042        keys.sort();
1043        let mut path_to_id: HashMap<String, String> = HashMap::new();
1044        for (i, path) in keys.iter().enumerate() {
1045            path_to_id.insert((*path).clone(), format!("bt_{}", i.saturating_add(1)));
1046        }
1047
1048        // Find root items (depth 0) and sort by name
1049        let mut roots: Vec<(String, String)> = hierarchy_map
1050            .iter()
1051            .filter_map(|(path, (name, depth, _))| {
1052                if *depth == 0 {
1053                    Some((path.clone(), name.clone()))
1054                } else {
1055                    None
1056                }
1057            })
1058            .collect();
1059        roots.sort_by(|a, b| a.1.cmp(&b.1));
1060        let mut root_nodes = Vec::new();
1061        for (path, name) in roots {
1062            let tree_node = Self::build_tree_node(&path, &name, hierarchy_map, &path_to_id);
1063            root_nodes.push(tree_node);
1064        }
1065
1066        // Apply top-down collapsing of single-child container chains
1067        // and build a mapping of old IDs to new IDs
1068        let mut id_remapping: HashMap<String, String> = HashMap::new();
1069        root_nodes = root_nodes
1070            .into_iter()
1071            .map(|node| Self::collapse_single_child_containers(node, None, &mut id_remapping))
1072            .collect();
1073
1074        // Update path_to_id with remappings
1075        let mut updated_path_to_id = path_to_id.clone();
1076        for (path, old_id) in path_to_id.iter() {
1077            if let Some(new_id) = id_remapping.get(old_id) {
1078                updated_path_to_id.insert(path.clone(), new_id.clone());
1079            }
1080        }
1081
1082        (root_nodes, updated_path_to_id, id_remapping)
1083    }

1085    /// Build a single tree node recursively
1086    fn build_tree_node(
1087        current_path: &str,
1088        name: &str,
1089        hierarchy_map: &HashMap<String, (String, usize, Option<String>)>,
1090        path_to_id: &HashMap<String, String>,
1091    ) -> serde_json::Value {
1092        let current_id = path_to_id.get(current_path).unwrap().clone();
1093
1094        // Find children (paths that have this path as parent)
1095        let mut child_specs: Vec<(&String, &String)> = hierarchy_map
1096            .iter()
1097            .filter_map(|(child_path, (child_name, _, parent_path))| {
1098                if let Some(parent) = parent_path {
1099                    if parent == current_path {
1100                        Some((child_path, child_name))
1101                    } else {
1102                        None
1103                    }
1104                } else {
1105                    None
1106                }
1107            })
1108            .collect();
1109        child_specs.sort_by(|a, b| a.1.cmp(b.1));
1110        let mut children = Vec::new();
1111        for (child_path, child_name) in child_specs {
1112            let child_node =
1113                Self::build_tree_node(child_path, child_name, hierarchy_map, path_to_id);
1114            children.push(child_node);
1115        }
1116
1117        if children.is_empty() {
1118            serde_json::json!({
1119                "id": current_id,
1120                "name": name
1121            })
1122        } else {
1123            serde_json::json!({
1124                "id": current_id,
1125                "name": name,
1126                "children": children
1127            })
1128        }
1129    }
1130
1131    /// Collapse single-child container chains (top-down)
1132    /// When a container has exactly one child AND that child is also a container,
1133    /// we collapse them by keeping the child's ID and combining names.
1134    /// parent_name is used to accumulate names during recursion (None for roots)
1135    /// id_remapping tracks which old IDs map to which new IDs after collapsing
1136    fn collapse_single_child_containers(
1137        node: serde_json::Value,
1138        parent_name: Option<String>,
1139        id_remapping: &mut HashMap<String, String>,
1140    ) -> serde_json::Value {
1141        let mut node_obj = match node {
1142            serde_json::Value::Object(obj) => obj,
1143            _ => return node,
1144        };
1145
1146        let current_name = node_obj
1147            .get("name")
1148            .and_then(|v| v.as_str())
1149            .unwrap_or("")
1150            .to_string();
1151
1152        let current_id = node_obj
1153            .get("id")
1154            .and_then(|v| v.as_str())
1155            .unwrap_or("")
1156            .to_string();
1157
1158        // Determine the effective name (combined with parent if collapsing)
1159        // Use → to show call chain (parent called child)
1160        let effective_name = if let Some(parent) = parent_name.clone() {
1161            format!("{} → {}", parent, current_name)
1162        } else {
1163            current_name.clone()
1164        };
1165
1166        // Check if this node has children (is a container)
1167        if let Some(serde_json::Value::Array(children)) = node_obj.get("children") {
1168            // If exactly one child AND that child is also a container
1169            if children.len() == 1
1170                && let Some(child) = children.first()
1171            {
1172                let child_is_container = child
1173                    .get("children")
1174                    .and_then(|v| v.as_array())
1175                    .map(|arr| !arr.is_empty())
1176                    .unwrap_or(false);
1177
1178                if child_is_container {
1179                    let child_id = child
1180                        .get("id")
1181                        .and_then(|v| v.as_str())
1182                        .unwrap_or("")
1183                        .to_string();
1184
1185                    // Record that this parent's ID should map to the child's ID
1186                    if !current_id.is_empty() && !child_id.is_empty() {
1187                        id_remapping.insert(current_id.clone(), child_id.clone());
1188                    }
1189
1190                    // Collapse: recursively process the child with accumulated name
1191                    return Self::collapse_single_child_containers(
1192                        child.clone(),
1193                        Some(effective_name),
1194                        id_remapping,
1195                    );
1196                }
1197            }
1198
1199            // Not collapsing: process children normally and update name if accumulated
1200            let processed_children: Vec<serde_json::Value> = children
1201                .iter()
1202                .map(|child| {
1203                    Self::collapse_single_child_containers(child.clone(), None, id_remapping)
1204                })
1205                .collect();
1206
1207            node_obj.insert(
1208                "name".to_string(),
1209                serde_json::Value::String(effective_name),
1210            );
1211            node_obj.insert(
1212                "children".to_string(),
1213                serde_json::Value::Array(processed_children),
1214            );
1215        } else {
1216            // Leaf node: just update name if accumulated
1217            node_obj.insert(
1218                "name".to_string(),
1219                serde_json::Value::String(effective_name),
1220            );
1221        }
1222
1223        serde_json::Value::Object(node_obj)
1224    }

1226    /// Extract meaningful file path
1227    fn extract_file_path(filename: &str) -> String {
1228        if filename.is_empty() {
1229            return "unknown".to_string();
1230        }
1231
1232        // Extract the most relevant part of the file path
1233        let parts: Vec<&str> = filename.split('/').collect();
1234        let file_name = parts.last().unwrap_or(&"unknown");
1235
1236        // If it's a source file, include the parent directory for context
1237        if file_name.ends_with(".rs") && parts.len() > 1 {
1238            let parent_dir = parts[parts.len() - 2];
1239            format!("{}/{}", parent_dir, file_name)
1240        } else {
1241            file_name.to_string()
1242        }
1243    }
1244}
