24/// Keyed Streams capture streaming elements of type `V` grouped by a key of type `K`,
25/// where the order of keys is non-deterministic but the order *within* each group may
26/// be deterministic.
27///
28/// Type Parameters:
29/// - `K`: the type of the key for each group
30/// - `V`: the type of the elements inside each group
31/// - `Loc`: the [`Location`] where the keyed stream is materialized
32/// - `Bound`: tracks whether the entries are [`Bounded`] (local and finite) or [`Unbounded`] (asynchronous and possibly infinite)
33/// - `Order`: tracks whether the elements within each group have deterministic order
34///   ([`TotalOrder`]) or not ([`NoOrder`])
35/// - `Retries`: tracks whether the elements within each group have deterministic cardinality
36///   ([`ExactlyOnce`]) or may have non-deterministic retries ([`crate::live_collections::stream::AtLeastOnce`])
37pub struct KeyedStream<K, V, Loc, Bound: Boundedness, Order = TotalOrder, Retries = ExactlyOnce> {
38    pub(crate) underlying: Stream<(K, V), Loc, Bound, NoOrder, Retries>,
39    pub(crate) _phantom_order: PhantomData<Order>,
40}
41
42impl<'a, K, V, L, B: Boundedness, R> From<KeyedStream<K, V, L, B, TotalOrder, R>>
43    for KeyedStream<K, V, L, B, NoOrder, R>
44where
45    L: Location<'a>,
46{
47    fn from(stream: KeyedStream<K, V, L, B, TotalOrder, R>) -> KeyedStream<K, V, L, B, NoOrder, R> {
48        KeyedStream {
49            underlying: stream.underlying,
50            _phantom_order: Default::default(),
51        }
52    }
53}

99            underlying: self.underlying,

119    /// Flattens the keyed stream into a single stream of key-value pairs, with non-deterministic
120    /// element ordering.

124    /// # use hydro_lang::prelude::*;

143    /// element ordering.

165    /// Transforms each value by invoking `f` on each element, with keys staying the same
166    /// after transformation. If you need access to the key, see [`KeyedStream::map_with_key`].
167    ///
168    /// If you do not want to modify the stream and instead only want to view
169    /// each item use [`KeyedStream::inspect`] instead.
170    ///
171    /// # Example
172    /// ```rust
173    /// # use hydro_lang::prelude::*;
174    /// # use futures::StreamExt;
175    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
176    /// process
177    ///     .source_iter(q!(vec![(1, 2), (1, 3), (2, 4)]))
178    ///     .into_keyed()
179    ///     .map(q!(|v| v + 1))
180    /// #   .entries()
181    /// # }, |mut stream| async move {
182    /// // { 1: [3, 4], 2: [5] }
183    /// # for w in vec![(1, 3), (1, 4), (2, 5)] {
184    /// #     assert_eq!(stream.next().await.unwrap(), w);
185    /// # }
186    /// # }));
187    /// ```
188    pub fn map<U, F>(self, f: impl IntoQuotedMut<'a, F, L> + Copy) -> KeyedStream<K, U, L, B, O, R>
189    where
190        F: Fn(V) -> U + 'a,
191    {
192        let f: ManualExpr<F, _> = ManualExpr::new(move |ctx: &L| f.splice_fn1_ctx(ctx));
193        KeyedStream {
194            underlying: self.underlying.map(q!({
195                let orig = f;
196                move |(k, v)| (k, orig(v))
197            })),
198            _phantom_order: Default::default(),
199        }
200    }

202    /// Transforms each value by invoking `f` on each key-value pair. The resulting values are **not**
203    /// re-grouped even they are tuples; instead they will be grouped under the original key.
204    ///
205    /// If you do not want to modify the stream and instead only want to view
206    /// each item use [`KeyedStream::inspect_with_key`] instead.
207    ///
208    /// # Example
209    /// ```rust
210    /// # use hydro_lang::prelude::*;
211    /// # use futures::StreamExt;
212    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
213    /// process
214    ///     .source_iter(q!(vec![(1, 2), (1, 3), (2, 4)]))
215    ///     .into_keyed()
216    ///     .map_with_key(q!(|(k, v)| k + v))
217    /// #   .entries()
218    /// # }, |mut stream| async move {
219    /// // { 1: [3, 4], 2: [6] }
220    /// # for w in vec![(1, 3), (1, 4), (2, 6)] {
221    /// #     assert_eq!(stream.next().await.unwrap(), w);
222    /// # }
223    /// # }));
224    /// ```
225    pub fn map_with_key<U, F>(
226        self,
227        f: impl IntoQuotedMut<'a, F, L> + Copy,
228    ) -> KeyedStream<K, U, L, B, O, R>
229    where
230        F: Fn((K, V)) -> U + 'a,
231        K: Clone,
232    {
233        let f: ManualExpr<F, _> = ManualExpr::new(move |ctx: &L| f.splice_fn1_ctx(ctx));
234        KeyedStream {
235            underlying: self.underlying.map(q!({
236                let orig = f;
237                move |(k, v)| {
238                    let out = orig((k.clone(), v));
239                    (k, out)
240                }
241            })),
242            _phantom_order: Default::default(),
243        }
244    }

246    /// Creates a stream containing only the elements of each group stream that satisfy a predicate
247    /// `f`, preserving the order of the elements within the group.
248    ///
249    /// The closure `f` receives a reference `&V` rather than an owned value `v` because filtering does
250    /// not modify or take ownership of the values. If you need to modify the values while filtering
251    /// use [`KeyedStream::filter_map`] instead.
252    ///
253    /// # Example
254    /// ```rust
255    /// # use hydro_lang::prelude::*;
256    /// # use futures::StreamExt;
257    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
258    /// process
259    ///     .source_iter(q!(vec![(1, 2), (1, 3), (2, 4)]))
260    ///     .into_keyed()
261    ///     .filter(q!(|&x| x > 2))
262    /// #   .entries()
263    /// # }, |mut stream| async move {
264    /// // { 1: [3], 2: [4] }
265    /// # for w in vec![(1, 3), (2, 4)] {
266    /// #     assert_eq!(stream.next().await.unwrap(), w);
267    /// # }
268    /// # }));
269    /// ```
270    pub fn filter<F>(self, f: impl IntoQuotedMut<'a, F, L> + Copy) -> KeyedStream<K, V, L, B, O, R>
271    where
272        F: Fn(&V) -> bool + 'a,
273    {
274        let f: ManualExpr<F, _> = ManualExpr::new(move |ctx: &L| f.splice_fn1_borrow_ctx(ctx));
275        KeyedStream {
276            underlying: self.underlying.filter(q!({
277                let orig = f;
278                move |(_k, v)| orig(v)
279            })),
280            _phantom_order: Default::default(),
281        }
282    }

355            _phantom_order: Default::default(),

480impl<'a, K, V, L: Location<'a> + NoTick + NoAtomic, O, R> KeyedStream<K, V, L, Unbounded, O, R> {
481    /// Produces a new keyed stream that "merges" the inputs by interleaving the elements
482    /// of any overlapping groups. The result has [`NoOrder`] on each group because the
483    /// order of interleaving is not guaranteed. If the keys across both inputs do not overlap,
484    /// the ordering will be deterministic and you can safely use [`Self::assume_ordering`].
485    ///
486    /// Currently, both input streams must be [`Unbounded`].
487    ///
488    /// # Example
489    /// ```rust
490    /// # use hydro_lang::prelude::*;
491    /// # use futures::StreamExt;
492    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
493    /// let numbers1 = process.source_iter(q!(vec![(1, 2), (3, 4)])).into_keyed();
494    /// let numbers2 = process.source_iter(q!(vec![(1, 3), (3, 5)])).into_keyed();
495    /// numbers1.interleave(numbers2)
496    /// #   .entries()
497    /// # }, |mut stream| async move {
498    /// // { 1: [2, 3], 3: [4, 5] } with each group in unknown order
499    /// # for w in vec![(1, 2), (3, 4), (1, 3), (3, 5)] {
500    /// #     assert_eq!(stream.next().await.unwrap(), w);
501    /// # }
502    /// # }));
503    /// ```
504    pub fn interleave<O2, R2: MinRetries<R>>(
505        self,
506        other: KeyedStream<K, V, L, Unbounded, O2, R2>,
507    ) -> KeyedStream<K, V, L, Unbounded, NoOrder, R::Min>
508    where
509        R: MinRetries<R2, Min = R2::Min>,
510    {
511        self.entries().interleave(other.entries()).into_keyed()
512    }
513}

515impl<'a, K, V, L, B: Boundedness> KeyedStream<K, V, L, B, TotalOrder, ExactlyOnce>
516where
517    K: Eq + Hash,
518    L: Location<'a>,
519{
520    /// A special case of [`Stream::scan`] for keyd streams. For each key group the values are transformed via the `f` combinator.
521    ///
522    /// Unlike [`Stream::fold_keyed`] which only returns the final accumulated value, `scan` produces a new stream
523    /// containing all intermediate accumulated values paired with the key. The scan operation can also terminate
524    /// early by returning `None`.
525    ///
526    /// The function takes a mutable reference to the accumulator and the current element, and returns
527    /// an `Option<U>`. If the function returns `Some(value)`, `value` is emitted to the output stream.
528    /// If the function returns `None`, the stream is terminated and no more elements are processed.
529    ///
530    /// # Example
531    /// ```rust
532    /// # use hydro_lang::prelude::*;
533    /// # use futures::StreamExt;
534    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
535    /// process
536    ///     .source_iter(q!(vec![(0, 1), (0, 2), (1, 3), (1, 4)]))
537    ///     .into_keyed()
538    ///     .scan(
539    ///         q!(|| 0),
540    ///         q!(|acc, x| {
541    ///             *acc += x;
542    ///             Some(*acc)
543    ///         }),
544    ///     )
545    /// #   .entries()
546    /// # }, |mut stream| async move {
547    /// // Output: { 0: [1, 3], 1: [3, 7] }
548    /// # for w in vec![(0, 1), (0, 3), (1, 3), (1, 7)] {
549    /// #     assert_eq!(stream.next().await.unwrap(), w);
550    /// # }
551    /// # }));
552    /// ```
553    pub fn scan<A, U, I, F>(
554        self,
555        init: impl IntoQuotedMut<'a, I, L> + Copy,
556        f: impl IntoQuotedMut<'a, F, L> + Copy,
557    ) -> KeyedStream<K, U, L, B, TotalOrder, ExactlyOnce>
558    where

560        I: Fn() -> A + 'a,
561        F: Fn(&mut A, V) -> Option<U> + 'a,
562    {
563        KeyedStream {
564            underlying: self
565                .underlying
566                .assume_ordering::<TotalOrder>(
567                    nondet!(/** keyed scan does not rely on order of keys */),
568                )
569                .scan_keyed(init, f)
570                .into(),
571            _phantom_order: Default::default(),
572        }
573    }

575    /// A variant of [`Stream::fold`], intended for keyed streams. The aggregation is executed in-order across the values
576    /// in each group. But the aggregation function returns a boolean, which when true indicates that the aggregated
577    /// result is complete and can be released to downstream computation. Unlike [`Stream::fold_keyed`], this means that
578    /// even if the input stream is [`super::boundedness::Unbounded`], the outputs of the fold can be processed like normal stream elements.
579    ///
580    /// # Example
581    /// ```rust
582    /// # use hydro_lang::prelude::*;
583    /// # use futures::StreamExt;
584    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
585    /// process
586    ///     .source_iter(q!(vec![(0, 2), (0, 3), (1, 3), (1, 6)]))
587    ///     .into_keyed()
588    ///     .fold_early_stop(
589    ///         q!(|| 0),
590    ///         q!(|acc, x| {
591    ///             *acc += x;
592    ///             x % 2 == 0
593    ///         }),
594    ///     )
595    /// #   .entries()
596    /// # }, |mut stream| async move {
597    /// // Output: { 0: 2, 1: 9 }
598    /// # for w in vec![(0, 2), (1, 9)] {
599    /// #     assert_eq!(stream.next().await.unwrap(), w);
600    /// # }
601    /// # }));
602    /// ```
603    pub fn fold_early_stop<A, I, F>(
604        self,
605        init: impl IntoQuotedMut<'a, I, L> + Copy,
606        f: impl IntoQuotedMut<'a, F, L> + Copy,
607    ) -> KeyedSingleton<K, A, L, B::WhenValueBounded>
608    where
609        K: Clone,
610        I: Fn() -> A + 'a,
611        F: Fn(&mut A, V) -> bool + 'a,
612    {
613        KeyedSingleton {
614            underlying: {
615                self.underlying
616                    .assume_ordering::<TotalOrder>(
617                        nondet!(/** keyed fold does not rely on order of keys */),
618                    )
619                    .fold_keyed_early_stop(init, f)
620                    .into()
621            },
622        }
623    }

640    /// Like [`Stream::fold`], aggregates the values in each group via the `comb` closure.
641    ///
642    /// Each group must have a [`TotalOrder`] guarantee, which means that the `comb` closure is allowed
643    /// to depend on the order of elements in the group.
644    ///
645    /// If the input and output value types are the same and do not require initialization then use
646    /// [`KeyedStream::reduce`].
647    ///
648    /// # Example
649    /// ```rust
650    /// # use hydro_lang::prelude::*;
651    /// # use futures::StreamExt;
652    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
653    /// let tick = process.tick();
654    /// let numbers = process
655    ///     .source_iter(q!(vec![(1, 2), (2, 3), (1, 3), (2, 4)]))
656    ///     .into_keyed();
657    /// let batch = numbers.batch(&tick, nondet!(/** test */));
658    /// batch
659    ///     .fold(q!(|| 0), q!(|acc, x| *acc += x))
660    ///     .entries()
661    ///     .all_ticks()
662    /// # }, |mut stream| async move {
663    /// // (1, 5), (2, 7)
664    /// # assert_eq!(stream.next().await.unwrap(), (1, 5));
665    /// # assert_eq!(stream.next().await.unwrap(), (2, 7));
666    /// # }));
667    /// ```
668    pub fn fold<A, I: Fn() -> A + 'a, F: Fn(&mut A, V)>(
669        self,
670        init: impl IntoQuotedMut<'a, I, L>,
671        comb: impl IntoQuotedMut<'a, F, L>,
672    ) -> KeyedSingleton<K, A, L, B::WhenValueUnbounded> {
673        let init = init.splice_fn0_ctx(&self.underlying.location).into();
674        let comb = comb
675            .splice_fn2_borrow_mut_ctx(&self.underlying.location)
676            .into();
677
678        let out_ir = HydroNode::FoldKeyed {
679            init,
680            acc: comb,
681            input: Box::new(self.underlying.ir_node.into_inner()),
682            metadata: self.underlying.location.new_node_metadata::<(K, A)>(),
683        };
684
685        KeyedSingleton {
686            underlying: Stream::new(self.underlying.location, out_ir),
687        }
688    }

690    /// Like [`Stream::reduce`], aggregates the values in each group via the `comb` closure.
691    ///
692    /// Each group must have a [`TotalOrder`] guarantee, which means that the `comb` closure is allowed
693    /// to depend on the order of elements in the stream.
694    ///
695    /// If you need the accumulated value to have a different type than the input, use [`KeyedStream::fold`].
696    ///
697    /// # Example
698    /// ```rust
699    /// # use hydro_lang::prelude::*;
700    /// # use futures::StreamExt;
701    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
702    /// let tick = process.tick();
703    /// let numbers = process
704    ///     .source_iter(q!(vec![(1, 2), (2, 3), (1, 3), (2, 4)]))
705    ///     .into_keyed();
706    /// let batch = numbers.batch(&tick, nondet!(/** test */));
707    /// batch.reduce(q!(|acc, x| *acc += x)).entries().all_ticks()
708    /// # }, |mut stream| async move {
709    /// // (1, 5), (2, 7)
710    /// # assert_eq!(stream.next().await.unwrap(), (1, 5));
711    /// # assert_eq!(stream.next().await.unwrap(), (2, 7));
712    /// # }));
713    /// ```
714    pub fn reduce<F: Fn(&mut V, V) + 'a>(
715        self,
716        comb: impl IntoQuotedMut<'a, F, L>,
717    ) -> KeyedSingleton<K, V, L, B::WhenValueUnbounded> {
718        let f = comb
719            .splice_fn2_borrow_mut_ctx(&self.underlying.location)
720            .into();
721
722        let out_ir = HydroNode::ReduceKeyed {
723            f,
724            input: Box::new(self.underlying.ir_node.into_inner()),
725            metadata: self.underlying.location.new_node_metadata::<(K, V)>(),
726        };
727
728        KeyedSingleton {
729            underlying: Stream::new(self.underlying.location, out_ir),
730        }
731    }

733    /// A special case of [`KeyedStream::reduce`] where tuples with keys less than the watermark are automatically deleted.
734    ///
735    /// Each group must have a [`TotalOrder`] guarantee, which means that the `comb` closure is allowed
736    /// to depend on the order of elements in the stream.
737    ///
738    /// # Example
739    /// ```rust
740    /// # use hydro_lang::prelude::*;
741    /// # use futures::StreamExt;
742    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
743    /// let tick = process.tick();
744    /// let watermark = tick.singleton(q!(1));
745    /// let numbers = process
746    ///     .source_iter(q!([(0, 100), (1, 101), (2, 102), (2, 102)]))
747    ///     .into_keyed();
748    /// let batch = numbers.batch(&tick, nondet!(/** test */));
749    /// batch
750    ///     .reduce_watermark(watermark, q!(|acc, x| *acc += x))
751    ///     .entries()
752    ///     .all_ticks()
753    /// # }, |mut stream| async move {
754    /// // (2, 204)
755    /// # assert_eq!(stream.next().await.unwrap(), (2, 204));
756    /// # }));
757    /// ```
758    pub fn reduce_watermark<O, F>(
759        self,
760        other: impl Into<Optional<O, Tick<L::Root>, Bounded>>,
761        comb: impl IntoQuotedMut<'a, F, L>,
762    ) -> KeyedSingleton<K, V, L, B::WhenValueUnbounded>
763    where
764        O: Clone,
765        F: Fn(&mut V, V) + 'a,
766    {
767        let other: Optional<O, Tick<L::Root>, Bounded> = other.into();
768        check_matching_location(&self.underlying.location.root(), other.location.outer());
769        let f = comb
770            .splice_fn2_borrow_mut_ctx(&self.underlying.location)
771            .into();
772
773        let out_ir = Stream::new(
774            self.underlying.location.clone(),
775            HydroNode::ReduceKeyedWatermark {
776                f,
777                input: Box::new(self.underlying.ir_node.into_inner()),
778                watermark: Box::new(other.ir_node.into_inner()),
779                metadata: self.underlying.location.new_node_metadata::<(K, V)>(),
780            },
781        );
782
783        KeyedSingleton { underlying: out_ir }
784    }

1092    /// A special case of [`Stream::reduce_keyed_commutative_idempotent`] where tuples with keys less than the watermark are automatically deleted.
1093    ///
1094    /// The `comb` closure must be **commutative**, as the order of input items is not guaranteed, and **idempotent**,
1095    /// as there may be non-deterministic duplicates.
1096    ///
1097    /// # Example
1098    /// ```rust
1099    /// # use hydro_lang::prelude::*;
1100    /// # use futures::StreamExt;
1101    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
1102    /// let tick = process.tick();
1103    /// let watermark = tick.singleton(q!(1));
1104    /// let numbers = process
1105    ///     .source_iter(q!([(0, false), (1, false), (2, false), (2, true)]))
1106    ///     .into_keyed();
1107    /// let batch = numbers.batch(&tick, nondet!(/** test */));
1108    /// batch
1109    ///     .reduce_watermark_commutative_idempotent(watermark, q!(|acc, x| *acc |= x))
1110    ///     .entries()
1111    ///     .all_ticks()
1112    /// # }, |mut stream| async move {
1113    /// // (2, true)
1114    /// # assert_eq!(stream.next().await.unwrap(), (2, true));
1115    /// # }));
1116    /// ```
1117    pub fn reduce_watermark_commutative_idempotent<O2, F>(
1118        self,
1119        other: impl Into<Optional<O2, Tick<L::Root>, Bounded>>,
1120        comb: impl IntoQuotedMut<'a, F, L>,
1121    ) -> KeyedSingleton<K, V, L, B::WhenValueUnbounded>
1122    where
1123        O2: Clone,
1124        F: Fn(&mut V, V) + 'a,
1125    {
1126        self.assume_ordering::<TotalOrder>(nondet!(/** the combinator function is commutative */))
1127            .assume_retries::<ExactlyOnce>(nondet!(/** the combinator function is idempotent */))
1128            .reduce_watermark(other, comb)
1129    }

1131    /// Given a bounded stream of keys `K`, returns a new keyed stream containing only the groups
1132    /// whose keys are not in the bounded stream.
1133    ///
1134    /// # Example
1135    /// ```rust
1136    /// # use hydro_lang::prelude::*;
1137    /// # use futures::StreamExt;
1138    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
1139    /// let tick = process.tick();
1140    /// let keyed_stream = process
1141    ///     .source_iter(q!(vec![ (1, 'a'), (2, 'b'), (3, 'c'), (4, 'd') ]))
1142    ///     .batch(&tick, nondet!(/** test */))
1143    ///     .into_keyed();
1144    /// let keys_to_remove = process
1145    ///     .source_iter(q!(vec![1, 2]))
1146    ///     .batch(&tick, nondet!(/** test */));
1147    /// keyed_stream.filter_key_not_in(keys_to_remove).all_ticks()
1148    /// #   .entries()
1149    /// # }, |mut stream| async move {
1150    /// // { 3: ['c'], 4: ['d'] }
1151    /// # for w in vec![(3, 'c'), (4, 'd')] {
1152    /// #     assert_eq!(stream.next().await.unwrap(), w);
1153    /// # }
1154    /// # }));
1155    pub fn filter_key_not_in<O2, R2>(self, other: Stream<K, L, Bounded, O2, R2>) -> Self {
1156        KeyedStream {
1157            underlying: self.entries().anti_join(other),
1158            _phantom_order: Default::default(),
1159        }
1160    }
