22/// A marker trait indicating which components of a [`KeyedSingleton`] may change.
23///
24/// In addition to [`Bounded`] (all entries are fixed) and [`Unbounded`] (entries may be added /
25/// removed / changed), this also includes an additional variant [`BoundedValue`], which indicates
26/// that entries may be added over time, but once an entry is added it will never be removed and
27/// its value will never change.
28pub trait KeyedSingletonBound {
29    /// The [`Boundedness`] of the [`Stream`] underlying the keyed singleton.
30    type UnderlyingBound: Boundedness;
31    /// The [`Boundedness`] of each entry's value; [`Bounded`] means it is immutable.
32    type ValueBound: Boundedness;
33}
34
35impl KeyedSingletonBound for Unbounded {
36    type UnderlyingBound = Unbounded;
37    type ValueBound = Unbounded;
38}
39
40impl KeyedSingletonBound for Bounded {
41    type UnderlyingBound = Bounded;
42    type ValueBound = Bounded;
43}
44
45/// A variation of boundedness specific to [`KeyedSingleton`], which indicates that once a key appears,
46/// its value is bounded and will never change. If the `KeyBound` is [`Bounded`], then the entire set of entries
47/// is bounded, but if it is [`Unbounded`], then new entries may appear asynchronously.
48pub struct BoundedValue;
49
50impl KeyedSingletonBound for BoundedValue {
51    type UnderlyingBound = Unbounded;
52    type ValueBound = Bounded;
53}

55/// Mapping from keys of type `K` to values of type `V`.
56///
57/// Keyed Singletons capture an asynchronously updated mapping from keys of the `K` to values of
58/// type `V`, where the order of keys is non-deterministic. In addition to the standard boundedness
59/// variants ([`Bounded`] for finite and immutable, [`Unbounded`] for asynchronously changing),
60/// keyed singletons can use [`BoundedValue`] to declare that new keys may be added over time, but
61/// keys cannot be removed and the value for each key is immutable.
62///
63/// Type Parameters:
64/// - `K`: the type of the key for each entry
65/// - `V`: the type of the value for each entry
66/// - `Loc`: the [`Location`] where the keyed singleton is materialized
67/// - `Bound`: tracks whether the entries are:
68///     - [`Bounded`] (local and finite)
69///     - [`Unbounded`] (asynchronous with entries added / removed / changed over time)
70///     - [`BoundedValue`] (asynchronous with immutable values for each key and no removals)
71pub struct KeyedSingleton<K, V, Loc, Bound: KeyedSingletonBound> {
72    pub(crate) underlying: Stream<(K, V), Loc, Bound::UnderlyingBound, NoOrder, ExactlyOnce>,
73}

79        KeyedSingleton {

81        }

112    /// Flattens the keyed singleton into an unordered stream of key-value pairs.
113    ///
114    /// The value for each key must be bounded, otherwise the resulting stream elements would be
115    /// non-determinstic. As new entries are added to the keyed singleton, they will be streamed
116    /// into the output.
117    ///
118    /// # Example
119    /// ```rust
120    /// # use hydro_lang::prelude::*;
121    /// # use futures::StreamExt;
122    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
123    /// let keyed_singleton = // { 1: 2, 2: 4 }
124    /// # process
125    /// #     .source_iter(q!(vec![(1, 2), (2, 4)]))
126    /// #     .into_keyed()
127    /// #     .first();
128    /// keyed_singleton.entries()
129    /// # }, |mut stream| async move {
130    /// // (1, 2), (2, 4) in any order
131    /// # let mut results = Vec::new();
132    /// # for _ in 0..2 {
133    /// #     results.push(stream.next().await.unwrap());
134    /// # }
135    /// # results.sort();
136    /// # assert_eq!(results, vec![(1, 2), (2, 4)]);
137    /// # }));
138    /// ```
139    pub fn entries(self) -> Stream<(K, V), L, B::UnderlyingBound, NoOrder, ExactlyOnce> {
140        self.underlying
141    }

147    /// into the output.
148    ///
149    /// # Example
150    /// ```rust
151    /// # use hydro_lang::prelude::*;
152    /// # use futures::StreamExt;
153    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
154    /// let keyed_singleton = // { 1: 2, 2: 4 }

164    /// #     results.push(stream.next().await.unwrap());
165    /// # }
166    /// # results.sort();
167    /// # assert_eq!(results, vec![2, 4]);
168    /// # }));

174    /// Flattens the keyed singleton into an unordered stream of just the keys.
175    ///
176    /// The value for each key must be bounded, otherwise the removal of keys would result in
177    /// non-determinism. As new entries are added to the keyed singleton, they will be streamed
178    /// into the output.
179    ///
180    /// # Example
181    /// ```rust
182    /// # use hydro_lang::prelude::*;

184    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {

196    /// # }
197    /// # results.sort();
198    /// # assert_eq!(results, vec![1, 2]);
199    /// # }));
200    /// ```
201    pub fn keys(self) -> Stream<K, L, B::UnderlyingBound, NoOrder, ExactlyOnce> {
202        self.entries().map(q!(|(k, _)| k))
203    }

252impl<'a, K, V, L: Location<'a>, B: KeyedSingletonBound> KeyedSingleton<K, V, L, B> {
253    /// Transforms each value by invoking `f` on each element, with keys staying the same
254    /// after transformation. If you need access to the key, see [`KeyedStream::map_with_key`].
255    ///
256    /// If you do not want to modify the stream and instead only want to view
257    /// each item use [`KeyedStream::inspect`] instead.
258    ///
259    /// # Example
260    /// ```rust
261    /// # use hydro_lang::prelude::*;
262    /// # use futures::StreamExt;
263    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
264    /// let keyed_singleton = // { 1: 2, 2: 4 }
265    /// # process
266    /// #     .source_iter(q!(vec![(1, 2), (2, 4)]))
267    /// #     .into_keyed()
268    /// #     .first();
269    /// keyed_singleton.map(q!(|v| v + 1))
270    /// #   .entries()
271    /// # }, |mut stream| async move {
272    /// // { 1: 3, 2: 5 }
273    /// # let mut results = Vec::new();
274    /// # for _ in 0..2 {
275    /// #     results.push(stream.next().await.unwrap());
276    /// # }
277    /// # results.sort();
278    /// # assert_eq!(results, vec![(1, 3), (2, 5)]);
279    /// # }));
280    /// ```
281    pub fn map<U, F>(self, f: impl IntoQuotedMut<'a, F, L> + Copy) -> KeyedSingleton<K, U, L, B>
282    where
283        F: Fn(V) -> U + 'a,
284    {
285        let f: ManualExpr<F, _> = ManualExpr::new(move |ctx: &L| f.splice_fn1_ctx(ctx));
286        KeyedSingleton {
287            underlying: self.underlying.map(q!({
288                let orig = f;
289                move |(k, v)| (k, orig(v))
290            })),
291        }
292    }
293
294    /// Transforms each value by invoking `f` on each key-value pair, with keys staying the same
295    /// after transformation. Unlike [`KeyedSingleton::map`], this gives access to both the key and value.
296    ///
297    /// The closure `f` receives a tuple `(K, V)` containing both the key and value, and returns
298    /// the new value `U`. The key remains unchanged in the output.
299    ///
300    /// # Example
301    /// ```rust
302    /// # use hydro_lang::prelude::*;
303    /// # use futures::StreamExt;
304    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
305    /// let keyed_singleton = // { 1: 2, 2: 4 }
306    /// # process
307    /// #     .source_iter(q!(vec![(1, 2), (2, 4)]))
308    /// #     .into_keyed()
309    /// #     .first();
310    /// keyed_singleton.map_with_key(q!(|(k, v)| k + v))
311    /// #   .entries()
312    /// # }, |mut stream| async move {
313    /// // { 1: 3, 2: 6 }
314    /// # let mut results = Vec::new();
315    /// # for _ in 0..2 {
316    /// #     results.push(stream.next().await.unwrap());
317    /// # }
318    /// # results.sort();
319    /// # assert_eq!(results, vec![(1, 3), (2, 6)]);
320    /// # }));
321    /// ```
322    pub fn map_with_key<U, F>(
323        self,
324        f: impl IntoQuotedMut<'a, F, L> + Copy,
325    ) -> KeyedSingleton<K, U, L, B>
326    where
327        F: Fn((K, V)) -> U + 'a,
328        K: Clone,
329    {
330        let f: ManualExpr<F, _> = ManualExpr::new(move |ctx: &L| f.splice_fn1_ctx(ctx));
331        KeyedSingleton {
332            underlying: self.underlying.map(q!({
333                let orig = f;
334                move |(k, v)| {
335                    let out = orig((k.clone(), v));
336                    (k, out)
337                }
338            })),
339        }
340    }

342    /// Creates a keyed singleton containing only the key-value pairs where the value satisfies a predicate `f`.
343    ///
344    /// The closure `f` receives a reference `&V` to each value and returns a boolean. If the predicate
345    /// returns `true`, the key-value pair is included in the output. If it returns `false`, the pair
346    /// is filtered out.
347    ///
348    /// The closure `f` receives a reference `&V` rather than an owned value `V` because filtering does
349    /// not modify or take ownership of the values. If you need to modify the values while filtering
350    /// use [`KeyedSingleton::filter_map`] instead.
351    ///
352    /// # Example
353    /// ```rust
354    /// # use hydro_lang::prelude::*;
355    /// # use futures::StreamExt;
356    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
357    /// let keyed_singleton = // { 1: 2, 2: 4, 3: 1 }
358    /// # process
359    /// #     .source_iter(q!(vec![(1, 2), (2, 4), (3, 1)]))
360    /// #     .into_keyed()
361    /// #     .first();
362    /// keyed_singleton.filter(q!(|&v| v > 1))
363    /// #   .entries()
364    /// # }, |mut stream| async move {
365    /// // { 1: 2, 2: 4 }
366    /// # let mut results = Vec::new();
367    /// # for _ in 0..2 {
368    /// #     results.push(stream.next().await.unwrap());
369    /// # }
370    /// # results.sort();
371    /// # assert_eq!(results, vec![(1, 2), (2, 4)]);
372    /// # }));
373    /// ```
374    pub fn filter<F>(self, f: impl IntoQuotedMut<'a, F, L> + Copy) -> KeyedSingleton<K, V, L, B>
375    where
376        F: Fn(&V) -> bool + 'a,
377    {
378        let f: ManualExpr<F, _> = ManualExpr::new(move |ctx: &L| f.splice_fn1_borrow_ctx(ctx));
379        KeyedSingleton {
380            underlying: self.underlying.filter(q!({
381                let orig = f;
382                move |(_k, v)| orig(v)
383            })),
384        }
385    }
386
387    /// An operator that both filters and maps values. It yields only the key-value pairs where
388    /// the supplied closure `f` returns `Some(value)`.
389    ///
390    /// The closure `f` receives each value `V` and returns `Option<U>`. If the closure returns
391    /// `Some(new_value)`, the key-value pair `(key, new_value)` is included in the output.
392    /// If it returns `None`, the key-value pair is filtered out.
393    ///
394    /// # Example
395    /// ```rust
396    /// # use hydro_lang::prelude::*;
397    /// # use futures::StreamExt;
398    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
399    /// let keyed_singleton = // { 1: "42", 2: "hello", 3: "100" }
400    /// # process
401    /// #     .source_iter(q!(vec![(1, "42"), (2, "hello"), (3, "100")]))
402    /// #     .into_keyed()
403    /// #     .first();
404    /// keyed_singleton.filter_map(q!(|s| s.parse::<i32>().ok()))
405    /// #   .entries()
406    /// # }, |mut stream| async move {
407    /// // { 1: 42, 3: 100 }
408    /// # let mut results = Vec::new();
409    /// # for _ in 0..2 {
410    /// #     results.push(stream.next().await.unwrap());
411    /// # }
412    /// # results.sort();
413    /// # assert_eq!(results, vec![(1, 42), (3, 100)]);
414    /// # }));
415    /// ```
416    pub fn filter_map<F, U>(
417        self,
418        f: impl IntoQuotedMut<'a, F, L> + Copy,
419    ) -> KeyedSingleton<K, U, L, B>
420    where
421        F: Fn(V) -> Option<U> + 'a,
422    {
423        let f: ManualExpr<F, _> = ManualExpr::new(move |ctx: &L| f.splice_fn1_ctx(ctx));
424        KeyedSingleton {
425            underlying: self.underlying.filter_map(q!({
426                let orig = f;
427                move |(k, v)| orig(v).map(|v| (k, v))
428            })),
429        }
430    }

449impl<'a, K: Hash + Eq, V, L: Location<'a>> KeyedSingleton<K, V, Tick<L>, Bounded> {
450    /// Gets the value associated with a specific key from the keyed singleton.
451    ///
452    /// # Example
453    /// ```rust
454    /// # use hydro_lang::prelude::*;
455    /// # use futures::StreamExt;
456    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
457    /// let tick = process.tick();
458    /// let keyed_data = process
459    ///     .source_iter(q!(vec![(1, 2), (2, 3)]))
460    ///     .into_keyed()
461    ///     .batch(&tick, nondet!(/** test */))
462    ///     .fold(q!(|| 0), q!(|acc, x| *acc = x));
463    /// let key = tick.singleton(q!(1));
464    /// keyed_data.get(key).all_ticks()
465    /// # }, |mut stream| async move {
466    /// // 2
467    /// # assert_eq!(stream.next().await.unwrap(), 2);
468    /// # }));
469    /// ```
470    pub fn get(self, key: Singleton<K, Tick<L>, Bounded>) -> Optional<V, Tick<L>, Bounded> {
471        self.entries()
472            .join(key.into_stream().map(q!(|k| (k, ()))))
473            .map(q!(|(_, (v, _))| v))
474            .assume_ordering::<TotalOrder>(nondet!(/** only a single key, so totally ordered */))
475            .first()
476    }
477
478    /// Given a keyed stream of lookup requests, where the key is the lookup and the value
479    /// is some additional metadata, emits a keyed stream of lookup results where the key
480    /// is the same as before, but the value is a tuple of the lookup result and the metadata
481    /// of the request. If the key is not found, no output will be produced.
482    ///
483    /// # Example
484    /// ```rust
485    /// # use hydro_lang::prelude::*;
486    /// # use futures::StreamExt;
487    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
488    /// let tick = process.tick();
489    /// let keyed_data = process
490    ///     .source_iter(q!(vec![(1, 10), (2, 20)]))
491    ///     .into_keyed()
492    ///     .batch(&tick, nondet!(/** test */))
493    ///     .fold(q!(|| 0), q!(|acc, x| *acc = x));
494    /// let other_data = process
495    ///     .source_iter(q!(vec![(1, 100), (2, 200), (1, 101)]))
496    ///     .into_keyed()
497    ///     .batch(&tick, nondet!(/** test */));
498    /// keyed_data.get_many_if_present(other_data).entries().all_ticks()
499    /// # }, |mut stream| async move {
500    /// // { 1: [(10, 100), (10, 101)], 2: [(20, 200)] } in any order
501    /// # let mut results = vec![];
502    /// # for _ in 0..3 {
503    /// #     results.push(stream.next().await.unwrap());
504    /// # }
505    /// # results.sort();
506    /// # assert_eq!(results, vec![(1, (10, 100)), (1, (10, 101)), (2, (20, 200))]);
507    /// # }));
508    /// ```
509    pub fn get_many_if_present<O2: Ordering, R2: Retries, V2>(
510        self,
511        requests: KeyedStream<K, V2, Tick<L>, Bounded, O2, R2>,
512    ) -> KeyedStream<K, (V, V2), Tick<L>, Bounded, NoOrder, R2> {
513        self.entries()
514            .weaker_retries::<R2>()
515            .join(requests.entries())
516            .into_keyed()
517    }

519    #[expect(missing_docs, reason = "TODO")]
520    pub fn get_from<V2: Clone>(
521        self,
522        from: KeyedSingleton<V, V2, Tick<L>, Bounded>,
523    ) -> KeyedSingleton<K, (V, Option<V2>), Tick<L>, Bounded>
524    where
525        K: Clone,
526        V: Hash + Eq + Clone,
527    {
528        let to_lookup = self.entries().map(q!(|(k, v)| (v, k))).into_keyed();
529        let lookup_result = from.get_many_if_present(to_lookup.clone());
530        let missing_values =
531            to_lookup.filter_key_not_in(lookup_result.clone().entries().map(q!(|t| t.0)));
532        KeyedSingleton {
533            underlying: lookup_result
534                .entries()
535                .map(q!(|(v, (v2, k))| (k, (v, Some(v2)))))
536                .chain(missing_values.entries().map(q!(|(v, k)| (k, (v, None))))),
537        }
538    }

541impl<'a, K, V, L, B: KeyedSingletonBound> KeyedSingleton<K, V, L, B>
542where
543    L: Location<'a> + NoTick + NoAtomic,
544{
545    /// Shifts this keyed singleton into an atomic context, which guarantees that any downstream logic
546    /// will all be executed synchronously before any outputs are yielded (in [`KeyedSingleton::end_atomic`]).
547    ///
548    /// This is useful to enforce local consistency constraints, such as ensuring that a write is
549    /// processed before an acknowledgement is emitted. Entering an atomic section requires a [`Tick`]
550    /// argument that declares where the keyed singleton will be atomically processed. Batching a
551    /// keyed singleton into the _same_ [`Tick`] will preserve the synchronous execution, while
552    /// batching into a different [`Tick`] will introduce asynchrony.
553    pub fn atomic(self, tick: &Tick<L>) -> KeyedSingleton<K, V, Atomic<L>, B> {
554        KeyedSingleton {
555            underlying: self.underlying.atomic(tick),
556        }
557    }
558
559    /// Returns a keyed singleton with a snapshot of each key-value entry at a non-deterministic
560    /// point in time.
561    ///
562    /// # Non-Determinism
563    /// Because this picks a snapshot of each entry, which is continuously changing, each output has a
564    /// non-deterministic set of entries since each snapshot can be at an arbitrary point in time.
565    pub fn snapshot(
566        self,
567        tick: &Tick<L>,
568        nondet: NonDet,
569    ) -> KeyedSingleton<K, V, Tick<L>, Bounded> {
570        self.atomic(tick).snapshot(nondet)
571    }
572}

603impl<'a, K, V, L: Location<'a>> KeyedSingleton<K, V, Tick<L>, Bounded> {
604    /// Asynchronously yields this keyed singleton outside the tick, which will
605    /// be asynchronously updated with the latest set of entries inside the tick.
606    ///
607    /// This converts a bounded value _inside_ a tick into an asynchronous value outside the
608    /// tick that tracks the inner value. This is useful for getting the value as of the
609    /// "most recent" tick, but note that updates are propagated asynchronously outside the tick.
610    ///
611    /// The entire set of entries are propagated on each tick, which means that if a tick
612    /// does not have a key "XYZ" that was present in the previous tick, the entry for "XYZ" will
613    /// also be removed from the output.
614    ///
615    /// # Example
616    /// ```rust
617    /// # use hydro_lang::prelude::*;
618    /// # use futures::StreamExt;
619    /// # tokio_test::block_on(hydro_lang::test_util::stream_transform_test(|process| {
620    /// let tick = process.tick();
621    /// # // ticks are lazy by default, forces the second tick to run
622    /// # tick.spin_batch(q!(1)).all_ticks().for_each(q!(|_| {}));
623    /// # let batch_first_tick = process
624    /// #   .source_iter(q!(vec![(1, 2), (2, 3)]))
625    /// #   .batch(&tick, nondet!(/** test */))
626    /// #   .into_keyed();
627    /// # let batch_second_tick = process
628    /// #   .source_iter(q!(vec![(2, 4), (3, 5)]))
629    /// #   .batch(&tick, nondet!(/** test */))
630    /// #   .into_keyed()
631    /// #   .defer_tick(); // appears on the second tick
632    /// # let input_batch = batch_first_tick.chain(batch_second_tick).first();
633    /// input_batch // first tick: { 1: 2, 2: 3 }, second tick: { 2: 4, 3: 5 }
634    ///     .latest()
635    /// # .snapshot(&tick, nondet!(/** test */))
636    /// # .entries()
637    /// # .all_ticks()
638    /// # }, |mut stream| async move {
639    /// // asynchronously changes from { 1: 2, 2: 3 } ~> { 2: 4, 3: 5 }
640    /// # for w in vec![(1, 2), (2, 3), (2, 4), (3, 5)] {
641    /// #     assert_eq!(stream.next().await.unwrap(), w);
642    /// # }
643    /// # }));
644    /// ```
645    pub fn latest(self) -> KeyedSingleton<K, V, L, Unbounded> {
646        KeyedSingleton {
647            underlying: Stream::new(
648                self.underlying.location.outer().clone(),
649                // no need to persist due to top-level replay
650                self.underlying.ir_node.into_inner(),
651            ),
652        }
653    }
654
655    /// Synchronously yields this keyed singleton outside the tick as an unbounded keyed singleton,
656    /// which will be updated with the latest set of entries inside the tick.
657    ///
658    /// Unlike [`KeyedSingleton::latest`], this preserves synchronous execution, as the output
659    /// keyed singleton is emitted in an [`Atomic`] context that will process elements synchronously
660    /// with the input keyed singleton's [`Tick`] context.
661    pub fn latest_atomic(self) -> KeyedSingleton<K, V, Atomic<L>, Unbounded> {
662        KeyedSingleton {
663            underlying: Stream::new(
664                Atomic {
665                    tick: self.underlying.location,
666                },
667                // no need to persist due to top-level replay
668                self.underlying.ir_node.into_inner(),
669            ),
670        }
671    }
672
673    #[expect(missing_docs, reason = "TODO")]
674    pub fn defer_tick(self) -> KeyedSingleton<K, V, Tick<L>, Bounded> {
675        KeyedSingleton {
676            underlying: self.underlying.defer_tick(),
677        }
678    }
679}
