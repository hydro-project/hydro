126    fn write_node_definition(
127        &mut self,
128        node_id: usize,
129        node_label: &super::render::NodeLabel,
130        node_type: HydroNodeType,
131        location_id: Option<usize>,
132        location_type: Option<&str>,
133        backtrace: Option<&crate::backtrace::Backtrace>,
134    ) -> Result<(), Self::Err> {
135        let style = self.node_type_to_style(node_type);
136
137        // Create the full label string using DebugExpr::Display for expressions
138        let full_label = match node_label {
139            super::render::NodeLabel::Static(s) => s.clone(),
140            super::render::NodeLabel::WithExprs { op_name, exprs } => {
141                if exprs.is_empty() {
142                    format!("{}()", op_name)
143                } else {
144                    // This is where DebugExpr::Display gets called with q! macro cleanup
145                    let expr_strs: Vec<String> = exprs.iter().map(|e| e.to_string()).collect();
146                    format!("{}({})", op_name, expr_strs.join(", "))
147                }
148            }
149        };
150
151        // Determine what label to display based on config
152        let display_label = if self.config.use_short_labels {
153            super::render::extract_short_label(&full_label)
154        } else {
155            full_label.clone()
156        };
157
158        // Always extract short label for UI toggle functionality
159        let short_label = super::render::extract_short_label(&full_label);
160
161        // If short and full labels are the same or very similar, enhance the full label
162        let enhanced_full_label = if short_label.len() >= full_label.len() - 2 {
163            // If they're nearly the same length, add more context to full label
164            match short_label.as_str() {
165                "inspect" => "inspect [debug output]".to_string(),
166                "persist" => "persist [state storage]".to_string(),
167                "tee" => "tee [branch dataflow]".to_string(),
168                "delta" => "delta [change detection]".to_string(),
169                "spin" => "spin [delay/buffer]".to_string(),
170                "send_bincode" => "send_bincode [send data to process/cluster]".to_string(),
171                "broadcast_bincode" => {
172                    "broadcast_bincode [send data to all cluster members]".to_string()
173                }
174                "source_iter" => "source_iter [iterate over collection]".to_string(),
175                "source_stream" => "source_stream [receive external data stream]".to_string(),
176                "network(recv)" => "network(recv) [receive from network]".to_string(),
177                "network(send)" => "network(send) [send to network]".to_string(),
178                "dest_sink" => "dest_sink [output destination]".to_string(),
179                _ => {
180                    if full_label.len() < 15 {
181                        format!("{} [{}]", node_label, "hydro operator")
182                    } else {
183                        node_label.to_string()
184                    }
185                }
186            }
187        } else {
188            node_label.to_string()
189        };
190
191        // Convert backtrace to JSON if available
192        let backtrace_json = if let Some(bt) = backtrace {
193            #[cfg(feature = "build")]
194            {
195                let elements = bt.elements();
196                serde_json::json!(elements.into_iter().map(|elem| {
197                    serde_json::json!({
198                        "fn_name": elem.fn_name,
199                        "filename": elem.filename,
200                        "lineno": elem.lineno,
201                        "addr": elem.addr
202                    })
203                }).collect::<Vec<_>>())
204            }
205            #[cfg(not(feature = "build"))]
206            {
207                serde_json::json!([])
208            }
209        } else {
210            serde_json::json!([])
211        };
212
213        let node = serde_json::json!({
214            "id": node_id.to_string(),
215            "type": "default",
216            "data": {
217                "label": display_label,
218                "shortLabel": short_label,
219                "fullLabel": enhanced_full_label,
220                "expanded": false,
221                "locationId": location_id,
222                "locationType": location_type,
223                "nodeType": match node_type {
224                    HydroNodeType::Source => "Source",
225                    HydroNodeType::Transform => "Transform",
226                    HydroNodeType::Join => "Join",
227                    HydroNodeType::Aggregation => "Aggregation",
228                    HydroNodeType::Network => "Network",
229                    HydroNodeType::Sink => "Sink",
230                    HydroNodeType::Tee => "Tee",
231                },
232                "backtrace": backtrace_json
233            },
234            "position": {
235                "x": 0,
236                "y": 0
237            },
238            "style": style
239        });
240        self.nodes.push(node);
241        Ok(())
242    }

340    fn write_epilogue(&mut self) -> Result<(), Self::Err> {
341        // Apply automatic layout using a simple algorithm
342        self.apply_layout();
343
344        // First, try to create backtrace-based hierarchy if available
345        let (hierarchy, node_assignments) = if self.has_backtrace_data() {
346            self.create_backtrace_hierarchy()
347        } else {
348            self.create_location_hierarchy()
349        };
350
351        // Create the final JSON structure in the format expected by the visualizer
352        let output = serde_json::json!({
353            "nodes": self.nodes,
354            "edges": self.edges,
355            "hierarchy": hierarchy,
356            "nodeAssignments": node_assignments
357        });
358
359        write!(
360            self.write,
361            "{}",
362            serde_json::to_string_pretty(&output).unwrap()
363        )
364    }

368    /// Check if any nodes have backtrace data
369    fn has_backtrace_data(&self) -> bool {
370        self.nodes.iter().any(|node| {
371            node["data"]["backtrace"]
372                .as_array()
373                .is_some_and(|bt| !bt.is_empty())
374        })
375    }
376
377    /// Create location-based hierarchy (original behavior)
378    fn create_location_hierarchy(&self) -> (Vec<serde_json::Value>, serde_json::Map<String, serde_json::Value>) {
379        // Create hierarchy structure (single level: locations as parents, nodes as children)
380        let hierarchy: Vec<serde_json::Value> = self
381            .locations
382            .iter()
383            .map(|(location_id, (label, _))| {
384                serde_json::json!({
385                    "id": format!("loc_{}", location_id),
386                    "name": label,
387                    "children": [] // Single level hierarchy - no nested children
388                })
389            })
390            .collect();
391
392        // Create node assignments (mapping nodes to their locations)
393        let mut node_assignments = serde_json::Map::new();
394        for (location_id, (_, node_ids)) in &self.locations {
395            let location_key = format!("loc_{}", location_id);
396            for &node_id in node_ids {
397                let node_key = format!("n{}", node_id);
398                node_assignments.insert(node_key, serde_json::Value::String(location_key.clone()));
399            }
400        }
401
402        (hierarchy, node_assignments)
403    }
404
405    /// Create backtrace-based hierarchy
406    fn create_backtrace_hierarchy(&self) -> (Vec<serde_json::Value>, serde_json::Map<String, serde_json::Value>) {
407        use std::collections::HashMap;
408
409        let mut hierarchy_map: HashMap<String, (String, usize, Option<String>)> = HashMap::new(); // path -> (name, depth, parent_path)
410        let mut path_to_node_assignments: HashMap<String, Vec<String>> = HashMap::new(); // path -> [node_ids]
411
412        // Process each node's backtrace
413        for node in &self.nodes {
414            if let (Some(node_id), Some(backtrace_array)) = (
415                node["id"].as_str(),
416                node["data"]["backtrace"].as_array(),
417            ) {
418                if backtrace_array.is_empty() {
419                    continue;
420                }
421
422                // Extract user-relevant frames from backtrace
423                let user_frames: Vec<_> = backtrace_array
424                    .iter()
425                    .filter_map(|frame| {
426                        let filename = frame["filename"].as_str().unwrap_or("");
427                        let fn_name = frame["fn_name"].as_str().unwrap_or("");
428                        
429                        // Include frames that are from user code
430                        if filename.contains("hydro_test") 
431                            || filename.contains("src/")
432                            || (!filename.contains(".cargo/") 
433                                && !filename.contains(".rustup/")
434                                && !fn_name.contains("tokio")) {
435                            Some((filename, fn_name))
436                        } else {
437                            None
438                        }
439                    })
440                    .take(3) // Take top 3 user frames
441                    .collect();
442
443                if user_frames.is_empty() {
444                    continue;
445                }
446
447                // Build hierarchy path from backtrace frames (reverse order for call stack)
448                let mut hierarchy_path = Vec::new();
449                for (i, (filename, fn_name)) in user_frames.iter().rev().enumerate() {
450                    let label = if i == 0 {
451                        // Top level: show file
452                        Self::extract_file_path(filename)
453                    } else {
454                        // Function levels: show function name
455                        Self::extract_function_name(fn_name)
456                    };
457                    hierarchy_path.push(label);
458                }
459
460                // Create hierarchy nodes for this path
461                let mut current_path = String::new();
462                let mut parent_path: Option<String> = None;
463                let mut deepest_path = String::new();
464
465                for (depth, label) in hierarchy_path.iter().enumerate() {
466                    current_path = if current_path.is_empty() {
467                        label.clone()
468                    } else {
469                        format!("{}/{}", current_path, label)
470                    };
471
472                    if !hierarchy_map.contains_key(&current_path) {
473                        hierarchy_map.insert(
474                            current_path.clone(),
475                            (label.clone(), depth, parent_path.clone()),
476                        );
477                    }
478
479                    deepest_path = current_path.clone();
480                    parent_path = Some(current_path.clone());
481                }
482
483                // Assign node to the deepest hierarchy level
484                if !deepest_path.is_empty() {
485                    path_to_node_assignments
486                        .entry(deepest_path)
487                        .or_default()
488                        .push(node_id.to_string());
489                }
490            }
491        }
492
493        // Build hierarchy tree and create proper ID mapping
494        let (hierarchy, path_to_id_map) = self.build_hierarchy_tree_with_ids(&hierarchy_map);
495        
496        // Create node assignments using the actual hierarchy IDs
497        let mut node_assignments = serde_json::Map::new();
498        for (path, node_ids) in path_to_node_assignments {
499            if let Some(hierarchy_id) = path_to_id_map.get(&path) {
500                for node_id in node_ids {
501                    node_assignments.insert(node_id, serde_json::Value::String(hierarchy_id.clone()));
502                }
503            }
504        }
505        
506        (hierarchy, node_assignments)
507    }
508
509    /// Build a tree structure and return both the tree and path-to-ID mapping
510    fn build_hierarchy_tree_with_ids(&self, hierarchy_map: &HashMap<String, (String, usize, Option<String>)>) -> (Vec<serde_json::Value>, HashMap<String, String>) {
511        let mut path_to_id: HashMap<String, String> = HashMap::new();
512        let mut id_counter = 1;
513        
514        for path in hierarchy_map.keys() {
515            path_to_id.insert(path.clone(), format!("bt_{}", id_counter));
516            id_counter += 1;
517        }
518
519        // Find root items (depth 0)
520        let mut roots = Vec::new();
521        
522        for (path, (name, depth, _)) in hierarchy_map {
523            if *depth == 0 {
524                let tree_node = Self::build_tree_node(path, name, hierarchy_map, &path_to_id);
525                roots.push(tree_node);
526            }
527        }
528        
529        (roots, path_to_id)
530    }
531
532    /// Build a single tree node recursively
533    fn build_tree_node(
534        current_path: &str,
535        name: &str,
536        hierarchy_map: &HashMap<String, (String, usize, Option<String>)>,
537        path_to_id: &HashMap<String, String>,
538    ) -> serde_json::Value {
539        let current_id = path_to_id.get(current_path).unwrap().clone();
540        
541        // Find children (paths that have this path as parent)
542        let mut children = Vec::new();
543        for (child_path, (child_name, _, parent_path)) in hierarchy_map {
544            if let Some(parent) = parent_path {
545                if parent == current_path {
546                    let child_node = Self::build_tree_node(child_path, child_name, hierarchy_map, path_to_id);
547                    children.push(child_node);
548                }
549            }
550        }
551
552        if children.is_empty() {
553            serde_json::json!({
554                "id": current_id,
555                "name": name
556            })
557        } else {
558            serde_json::json!({
559                "id": current_id,
560                "name": name,
561                "children": children
562            })
563        }
564    }

566    /// Extract meaningful function name from backtrace
567    fn extract_function_name(fn_name: &str) -> String {
568        if fn_name.is_empty() {
569            return "unknown".to_string();
570        }
571        
572        // Remove module paths and keep just the function name
573        let parts: Vec<&str> = fn_name.split("::").collect();
574        let last_part = parts.last().unwrap_or(&"unknown");
575        
576        // Handle closure syntax
577        if *last_part == "{{closure}}" && parts.len() > 1 {
578            parts[parts.len() - 2].to_string()
579        } else {
580            last_part.to_string()
581        }
582    }
583
584    /// Extract meaningful file path
585    fn extract_file_path(filename: &str) -> String {
586        if filename.is_empty() {
587            return "unknown".to_string();
588        }
589        
590        // Extract the most relevant part of the file path
591        let parts: Vec<&str> = filename.split('/').collect();
592        let file_name = parts.last().unwrap_or(&"unknown");
593        
594        // If it's a source file, include the parent directory for context
595        if file_name.ends_with(".rs") && parts.len() > 1 {
596            let parent_dir = parts[parts.len() - 2];
597            format!("{}/{}", parent_dir, file_name)
598        } else {
599            file_name.to_string()
600        }
601    }
602}

677/// Open JSON visualization in the docs browser with URL-encoded data
678#[cfg(feature = "viz")]
679fn open_json_docs_browser(
680    ir: &[crate::ir::HydroLeaf],
681    config: super::render::HydroWriteConfig,
682) -> Result<(), std::io::Error> {
683    // Generate JSON
684    let json_content = super::render::render_hydro_ir_json(ir, &config);
685
686    // Base64 encode the JSON for URL parameter
687    let encoded_data = data_encoding::BASE64URL.encode(json_content.as_bytes());
688
689    // Create the docs visualizer URL
690    let docs_url = format!(
691        "https://hydro-project.github.io/hydro/visualizer#data={}",
692        encoded_data
693    );
694
695    // For local development, use localhost instead
696    let local_url = format!("http://localhost:3000/visualizer#data={}", encoded_data);
697
698    // Try to open the local URL first (for development), fallback to docs URL
699    println!("Opening Hydro JSON visualization in browser...");
700
701    // First try localhost for development
702    match webbrowser::open(&local_url) {
703        Ok(_) => {
704            println!("Opened local visualizer: {}", local_url);
705            return Ok(());
706        }
707        Err(_) => {
708            // Local server not available, try docs URL
709            println!("Local server not available, opening docs visualizer...");
710        }
711    }
712
713    // Fallback to docs URL
714    match webbrowser::open(&docs_url) {
715        Ok(_) => {
716            println!("Opened docs visualizer: {}", docs_url);
717            Ok(())
718        }
719        Err(e) => {
720            eprintln!("Failed to open browser: {}", e);
721            eprintln!("You can manually open this URL: {}", docs_url);
722            Err(e)
723        }
724    }
725}
