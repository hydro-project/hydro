150    /// Optimize backtrace data for size efficiency
151    /// 1. Remove redundant/non-essential frames
152    /// 2. Truncate paths
153    /// 3. Remove memory addresses (not useful for visualization)
154    /// 4. Limit frame count for size efficiency
155    fn optimize_backtrace(&self, backtrace: &crate::backtrace::Backtrace) -> serde_json::Value {
156        #[cfg(feature = "build")]
157        {
158            let elements = backtrace.elements();
159
160            // filter out obviously internal frames
161            let relevant_frames: Vec<_> = elements
162                .iter()
163                .filter(|elem| {
164                    let filename = elem.filename.as_deref().unwrap_or("");
165                    let fn_name = &elem.fn_name;
166
167                    // Filter out obviously internal/system frames
168                    !(filename.contains(".cargo/registry/")
169                        || filename.contains(".rustup/toolchains/")
170                        || fn_name.starts_with("std::")
171                        || fn_name.starts_with("core::")
172                        || fn_name.starts_with("alloc::")
173                        || fn_name.contains("rust_begin_unwind")
174                        || fn_name.contains("rust_panic")
175                        || fn_name.contains("__rust_")
176                        || filename.ends_with("panic.rs")
177                        || filename.ends_with("/rustc/"))
178                })
179                .map(|elem| {
180                    // Truncate paths and function names for size
181                    let short_filename = elem
182                        .filename
183                        .as_deref()
184                        .map(|f| Self::truncate_path(f))
185                        .unwrap_or_else(|| "unknown".to_string());
186
187                    let short_fn_name = Self::truncate_function_name(&elem.fn_name);
188
189                    serde_json::json!({
190                        "fn": short_fn_name,
191                        "file": short_filename,
192                        "line": elem.lineno
193                        // Removed "addr" - not useful for visualization and saves space
194                    })
195                })
196                .collect();
197
198            serde_json::Value::Array(relevant_frames)
199        }
200        #[cfg(not(feature = "build"))]
201        {
202            serde_json::json!([])
203        }
204    }
205
206    /// Truncate file paths to keep only the relevant parts
207    fn truncate_path(path: &str) -> String {
208        let parts: Vec<&str> = path.split('/').collect();
209
210        // For paths like "/Users/foo/project/src/main.rs", keep "src/main.rs"
211        if let Some(src_idx) = parts.iter().rposition(|&p| p == "src") {
212            parts[src_idx..].join("/")
213        } else if parts.len() > 2 {
214            // Keep last 2 components
215            parts[parts.len().saturating_sub(2)..].join("/")
216        } else {
217            path.to_string()
218        }
219    }
220
221    /// Truncate function names to remove module paths
222    fn truncate_function_name(fn_name: &str) -> String {
223        // Remove everything before the last "::" to get just the function name
224        fn_name.split("::").last().unwrap_or(fn_name).to_string()
225    }

244    fn write_node_definition(
245        &mut self,
246        node_id: usize,
247        node_label: &super::render::NodeLabel,
248        node_type: HydroNodeType,
249        location_id: Option<usize>,
250        location_type: Option<&str>,
251        backtrace: Option<&crate::backtrace::Backtrace>,
252    ) -> Result<(), Self::Err> {
253        // Create the full label string using DebugExpr::Display for expressions
254        let full_label = match node_label {
255            super::render::NodeLabel::Static(s) => s.clone(),
256            super::render::NodeLabel::WithExprs { op_name, exprs } => {
257                if exprs.is_empty() {
258                    format!("{}()", op_name)
259                } else {
260                    // This is where DebugExpr::Display gets called with q! macro cleanup
261                    let expr_strs: Vec<String> = exprs.iter().map(|e| e.to_string()).collect();
262                    format!("{}({})", op_name, expr_strs.join(", "))
263                }
264            }
265        };
266
267        // Always extract short label for UI toggle functionality
268        let short_label = super::render::extract_short_label(&full_label);
269
270        // If short and full labels are the same or very similar, enhance the full label
271        let enhanced_full_label = if short_label.len() >= full_label.len() - 2 {
272            // If they're nearly the same length, add more context to full label
273            match short_label.as_str() {
274                "inspect" => "inspect [debug output]".to_string(),
275                "persist" => "persist [state storage]".to_string(),
276                "tee" => "tee [branch dataflow]".to_string(),
277                "delta" => "delta [change detection]".to_string(),
278                "spin" => "spin [delay/buffer]".to_string(),
279                "send_bincode" => "send_bincode [send data to process/cluster]".to_string(),
280                "broadcast_bincode" => {
281                    "broadcast_bincode [send data to all cluster members]".to_string()
282                }
283                "source_iter" => "source_iter [iterate over collection]".to_string(),
284                "source_stream" => "source_stream [receive external data stream]".to_string(),
285                "network(recv)" => "network(recv) [receive from network]".to_string(),
286                "network(send)" => "network(send) [send to network]".to_string(),
287                "dest_sink" => "dest_sink [output destination]".to_string(),
288                _ => {
289                    if full_label.len() < 15 {
290                        format!("{} [{}]", node_label, "hydro operator")
291                    } else {
292                        node_label.to_string()
293                    }
294                }
295            }
296        } else {
297            node_label.to_string()
298        };
299
300        // Convert backtrace to JSON if available (optimized for size)
301        let backtrace_json = if let Some(bt) = backtrace {
302            // Store backtrace for hierarchy generation
303            self.node_backtraces.insert(node_id, bt.clone());
304            self.optimize_backtrace(bt)
305        } else {
306            serde_json::json!([])
307        };
308
309        let node = serde_json::json!({
310            "id": node_id.to_string(),
311            "nodeType": Self::node_type_to_string(node_type),
312            "fullLabel": enhanced_full_label,
313            "shortLabel": short_label,
314            "data": {
315                "locationId": location_id,
316                "locationType": location_type,
317                "backtrace": backtrace_json
318            }
319        });
320        self.nodes.push(node);
321
322        // Track node location for cross-location edge detection
323        if let Some(loc_id) = location_id {
324            self.node_locations.insert(node_id, loc_id);
325        }
326
327        Ok(())
328    }

330    fn write_edge(
331        &mut self,
332        src_id: usize,
333        dst_id: usize,
334        edge_properties: &HashSet<HydroEdgeType>,
335        label: Option<&str>,
336    ) -> Result<(), Self::Err> {
337        let edge_id = format!("e{}", self.edge_count);
338        self.edge_count += 1;
339
340        // Convert edge properties to a JSON array
341        let properties: Vec<String> = edge_properties.iter().map(|p| format!("{:?}", p)).collect();
342
343        // Get location information for styling
344        let src_loc = self.node_locations.get(&src_id).copied();
345        let dst_loc = self.node_locations.get(&dst_id).copied();
346
347        // Get unified edge style
348        let style = super::render::get_unified_edge_style(edge_properties, src_loc, dst_loc);
349
350        let mut semantic_tags = properties.clone();
351        // Add Network tag if edge crosses locations
352        if let (Some(src), Some(dst)) = (src_loc, dst_loc)
353            && src != dst
354            && !semantic_tags.contains(&"Network".to_string())
355        {
356            semantic_tags.push("Network".to_string());
357        }
358
359        let mut edge = serde_json::json!({
360            "id": edge_id,
361            "source": src_id.to_string(),
362            "target": dst_id.to_string(),
363            "edgeProperties": properties,
364            "semanticTags": semantic_tags,
365            "style": {
366                "line-pattern": match style.line_pattern {
367                    super::render::LinePattern::Solid => "solid",
368                    super::render::LinePattern::Dotted => "dotted",
369                    super::render::LinePattern::Dashed => "dashed",
370                },
371                "line-width": style.line_width,
372                "arrowhead": match style.arrowhead {
373                    super::render::ArrowheadStyle::TriangleFilled => "triangle-filled",
374                    super::render::ArrowheadStyle::CircleFilled => "circle-filled",
375                    super::render::ArrowheadStyle::DiamondOpen => "diamond-open",
376                    super::render::ArrowheadStyle::Default => "triangle-filled",
377                },
378                "line-style": match style.line_style {
379                    super::render::LineStyle::Single => "single",
380                    super::render::LineStyle::Double => "double",
381                },
382                "halo": match style.halo {
383                    super::render::HaloStyle::None => "none",
384                    super::render::HaloStyle::LightRed => "light-red",
385                },
386                "waviness": match style.waviness {
387                    super::render::WavinessStyle::None => "none",
388                    super::render::WavinessStyle::Wavy => "wavy",
389                },
390                "animation": match style.animation {
391                    super::render::AnimationStyle::Static => "static",
392                    super::render::AnimationStyle::Animated => "animated",
393                },
394                "color": style.color,
395            }
396        });
397
398        if let Some(label_text) = label {
399            edge["label"] = serde_json::Value::String(label_text.to_string());
400        }
401
402        self.edges.push(edge);
403        Ok(())
404    }

454    fn write_epilogue(&mut self) -> Result<(), Self::Err> {
455        // Create multiple hierarchy options
456        let mut hierarchy_choices = Vec::new();
457        let mut node_assignments_choices = serde_json::Map::new();
458
459        // Always add location-based hierarchy
460        let (location_hierarchy, location_assignments) = self.create_location_hierarchy();
461        hierarchy_choices.push(serde_json::json!({
462            "id": "location",
463            "name": "Location",
464            "children": location_hierarchy
465        }));
466        node_assignments_choices.insert(
467            "location".to_string(),
468            serde_json::Value::Object(location_assignments),
469        );
470
471        // Add backtrace-based hierarchy if available
472        if self.has_backtrace_data() {
473            let (backtrace_hierarchy, backtrace_assignments) = self.create_backtrace_hierarchy();
474            hierarchy_choices.push(serde_json::json!({
475                "id": "backtrace",
476                "name": "Backtrace",
477                "children": backtrace_hierarchy
478            }));
479            node_assignments_choices.insert(
480                "backtrace".to_string(),
481                serde_json::Value::Object(backtrace_assignments),
482            );
483        }
484
485        // Create the final JSON structure in the format expected by the visualizer
486        let node_type_definitions = Self::get_node_type_definitions();
487        let legend_items = Self::get_legend_items();
488
489        // Build JSON string manually to guarantee field ordering
490        let mut json_parts = Vec::new();
491
492        // 1. nodes (required field first)
493        json_parts.push(format!(
494            "\"nodes\": {}",
495            serde_json::to_string_pretty(&self.nodes).unwrap()
496        ));
497
498        // 2. edges (required field second)
499        json_parts.push(format!(
500            "\"edges\": {}",
501            serde_json::to_string_pretty(&self.edges).unwrap()
502        ));
503
504        // 3. hierarchyChoices
505        json_parts.push(format!(
506            "\"hierarchyChoices\": {}",
507            serde_json::to_string_pretty(&hierarchy_choices).unwrap()
508        ));
509
510        // 4. nodeAssignments
511        json_parts.push(format!(
512            "\"nodeAssignments\": {}",
513            serde_json::to_string_pretty(&node_assignments_choices).unwrap()
514        ));
515
516        // 5. edgeStyleConfig
517        json_parts.push(format!(
518            "\"edgeStyleConfig\": {}",
519            serde_json::to_string_pretty(&Self::get_edge_style_config()).unwrap()
520        ));
521
522        // 6. nodeTypeConfig
523        let node_type_config = serde_json::json!({
524            "types": node_type_definitions,
525            "defaultType": "Transform"
526        });
527        json_parts.push(format!(
528            "\"nodeTypeConfig\": {}",
529            serde_json::to_string_pretty(&node_type_config).unwrap()
530        ));
531
532        // 7. legend
533        let legend = serde_json::json!({
534            "title": "Node Types",
535            "items": legend_items
536        });
537        json_parts.push(format!(
538            "\"legend\": {}",
539            serde_json::to_string_pretty(&legend).unwrap()
540        ));
541
542        let final_json = format!("{{\n  {}\n}}", json_parts.join(",\n  "));
543
544        write!(self.write, "{}", final_json)
545    }

549    /// Check if any nodes have meaningful backtrace data
550    fn has_backtrace_data(&self) -> bool {
551        self.nodes.iter().any(|node| {
552            if let Some(backtrace_array) = node["data"]["backtrace"].as_array() {
553                // Check if any frame has meaningful filename or fn_name data
554                backtrace_array.iter().any(|frame| {
555                    let filename = frame["file"].as_str().unwrap_or("");
556                    let fn_name = frame["fn"].as_str().unwrap_or("");
557                    !filename.is_empty() || !fn_name.is_empty()
558                })
559            } else {
560                false
561            }
562        })
563    }

565    /// Create location-based hierarchy (original behavior)
566    fn create_location_hierarchy(
567        &self,
568    ) -> (
569        Vec<serde_json::Value>,
570        serde_json::Map<String, serde_json::Value>,
571    ) {
572        // Create hierarchy structure (single level: locations as parents, nodes as children)
573        let hierarchy: Vec<serde_json::Value> = self
574            .locations
575            .iter()
576            .map(|(location_id, (label, _))| {
577                serde_json::json!({
578                    "id": format!("loc_{}", location_id),
579                    "name": label,
580                    "children": [] // Single level hierarchy - no nested children
581                })
582            })
583            .collect();
584
585        // Create node assignments by reading locationId from each node's data
586        // This is more reliable than using the write_node tracking which depends on HashMap iteration order
587        let mut node_assignments = serde_json::Map::new();
588        for node in &self.nodes {
589            if let (Some(node_id), Some(location_id)) =
590                (node["id"].as_str(), node["data"]["locationId"].as_u64())
591            {
592                let location_key = format!("loc_{}", location_id);
593                node_assignments
594                    .insert(node_id.to_string(), serde_json::Value::String(location_key));
595            }
596        }
597
598        (hierarchy, node_assignments)
599    }

601    /// Create backtrace-based hierarchy using structured backtrace data
602    fn create_backtrace_hierarchy(
603        &self,
604    ) -> (
605        Vec<serde_json::Value>,
606        serde_json::Map<String, serde_json::Value>,
607    ) {
608        use std::collections::HashMap;
609
610        let mut hierarchy_map: HashMap<String, (String, usize, Option<String>)> = HashMap::new(); // path -> (name, depth, parent_path)
611        let mut path_to_node_assignments: HashMap<String, Vec<String>> = HashMap::new(); // path -> [node_ids]
612
613        // Process each node's backtrace using the stored backtraces
614        for node in &self.nodes {
615            if let Some(node_id_str) = node["id"].as_str()
616                && let Ok(node_id) = node_id_str.parse::<usize>()
617                && let Some(backtrace) = self.node_backtraces.get(&node_id)
618            {
619                let elements = backtrace.elements();
620                if elements.is_empty() {
621                    continue;
622                }
623
624                // Filter to user-relevant frames using structured data
625                let user_frames: Vec<_> = elements
626                    .iter()
627                    .filter(|elem| {
628                        let filename = elem.filename.as_deref().unwrap_or("");
629                        let fn_name = &elem.fn_name;
630                        // Include frames that are from user code (more precise filtering)
631                        filename.contains("hydro_test")
632                            || filename.contains("/src/")
633                            || (filename.contains("examples/") && !filename.contains(".cargo/"))
634                            || (!filename.contains(".cargo/registry/")
635                                && !filename.contains(".rustup/toolchains/")
636                                && !fn_name.starts_with("std::")
637                                && !fn_name.starts_with("core::")
638                                && !fn_name.contains("tokio::"))
639                    })
640                    .take(5)
641                    .collect();
642                if user_frames.is_empty() {
643                    continue;
644                }
645
646                // Build hierarchy path from backtrace frames (reverse order for call stack)
647                let mut hierarchy_path = Vec::new();
648                for (i, elem) in user_frames.iter().rev().enumerate() {
649                    let label = if i == 0 {
650                        if let Some(filename) = &elem.filename {
651                            Self::extract_file_path(filename)
652                        } else {
653                            format!("fn_{}", Self::truncate_function_name(&elem.fn_name))
654                        }
655                    } else {
656                        Self::truncate_function_name(&elem.fn_name)
657                    };
658                    hierarchy_path.push(label);
659                }
660
661                // Create hierarchy nodes for this path
662                let mut current_path = String::new();
663                let mut parent_path: Option<String> = None;
664                let mut deepest_path = String::new();
665                for (depth, label) in hierarchy_path.iter().enumerate() {
666                    current_path = if current_path.is_empty() {
667                        label.clone()
668                    } else {
669                        format!("{}/{}", current_path, label)
670                    };
671                    if !hierarchy_map.contains_key(&current_path) {
672                        hierarchy_map.insert(
673                            current_path.clone(),
674                            (label.clone(), depth, parent_path.clone()),
675                        );
676                    }
677                    deepest_path = current_path.clone();
678                    parent_path = Some(current_path.clone());
679                }
680
681                if !deepest_path.is_empty() {
682                    path_to_node_assignments
683                        .entry(deepest_path)
684                        .or_default()
685                        .push(node_id_str.to_string());
686                }
687            }
688        }
689        // Build hierarchy tree and create proper ID mapping
690        let (hierarchy, path_to_id_map) = self.build_hierarchy_tree_with_ids(&hierarchy_map);
691
692        // Create node assignments using the actual hierarchy IDs
693        let mut node_assignments = serde_json::Map::new();
694        for (path, node_ids) in path_to_node_assignments {
695            if let Some(hierarchy_id) = path_to_id_map.get(&path) {
696                for node_id in node_ids {
697                    node_assignments
698                        .insert(node_id, serde_json::Value::String(hierarchy_id.clone()));
699                }
700            }
701        }
702        (hierarchy, node_assignments)
703    }
704
705    /// Build a tree structure and return both the tree and path-to-ID mapping
706    fn build_hierarchy_tree_with_ids(
707        &self,
708        hierarchy_map: &HashMap<String, (String, usize, Option<String>)>,
709    ) -> (Vec<serde_json::Value>, HashMap<String, String>) {
710        let mut path_to_id: HashMap<String, String> = HashMap::new();
711        let mut id_counter = 1;
712
713        for path in hierarchy_map.keys() {
714            path_to_id.insert(path.clone(), format!("bt_{}", id_counter));
715            id_counter += 1;
716        }
717
718        // Find root items (depth 0)
719        let mut roots = Vec::new();
720
721        for (path, (name, depth, _)) in hierarchy_map {
722            if *depth == 0 {
723                let tree_node = Self::build_tree_node(path, name, hierarchy_map, &path_to_id);
724                roots.push(tree_node);
725            }
726        }
727
728        (roots, path_to_id)
729    }
730
731    /// Build a single tree node recursively
732    fn build_tree_node(
733        current_path: &str,
734        name: &str,
735        hierarchy_map: &HashMap<String, (String, usize, Option<String>)>,
736        path_to_id: &HashMap<String, String>,
737    ) -> serde_json::Value {
738        let current_id = path_to_id.get(current_path).unwrap().clone();
739
740        // Find children (paths that have this path as parent)
741        let mut children = Vec::new();
742        for (child_path, (child_name, _, parent_path)) in hierarchy_map {
743            if let Some(parent) = parent_path
744                && parent == current_path
745            {
746                let child_node =
747                    Self::build_tree_node(child_path, child_name, hierarchy_map, path_to_id);
748                children.push(child_node);
749            }
750        }
751
752        if children.is_empty() {
753            serde_json::json!({
754                "id": current_id,
755                "name": name
756            })
757        } else {
758            serde_json::json!({
759                "id": current_id,
760                "name": name,
761                "children": children
762            })
763        }
764    }
765
766    /// Extract meaningful file path
767    fn extract_file_path(filename: &str) -> String {
768        if filename.is_empty() {
769            return "unknown".to_string();
770        }
771
772        // Extract the most relevant part of the file path
773        let parts: Vec<&str> = filename.split('/').collect();
774        let file_name = parts.last().unwrap_or(&"unknown");
775
776        // If it's a source file, include the parent directory for context
777        if file_name.ends_with(".rs") && parts.len() > 1 {
778            let parent_dir = parts[parts.len() - 2];
779            format!("{}/{}", parent_dir, file_name)
780        } else {
781            file_name.to_string()
782        }
783    }
784}
