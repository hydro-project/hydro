49fn open_mermaid_browser(mermaid_src: &str) -> Result<()> {
50    // Debug: Print the mermaid source being sent to browser
51    println!("=== MERMAID SOURCE BEING SENT TO BROWSER ===");
52    println!("{}", mermaid_src);
53    println!("=== END MERMAID SOURCE ===");
54
55    let state = serde_json::json!({
56        "code": mermaid_src,
57        "mermaid": serde_json::json!({
58            "theme": "default"
59        }),
60        "autoSync": true,
61        "updateDiagram": true
62    });
63    let state_json = serde_json::to_vec(&state)?;
64    let state_base64 = data_encoding::BASE64URL.encode(&state_json);
65    webbrowser::open(&format!(
66        "https://mermaid.live/edit#base64:{}",
67        state_base64
68    ))
69}

71fn open_dot_browser(dot_src: &str) -> Result<()> {
72    let mut url = "https://dreampuf.github.io/GraphvizOnline/#".to_owned();
73    for byte in dot_src.bytes() {
74        // Lazy percent encoding: https://en.wikipedia.org/wiki/Percent-encoding
75        write!(url, "%{:02x}", byte).unwrap();
76    }
77    webbrowser::open(&url)
78}

81/// Helper function to save content to a file with consistent path handling.
82/// If no filename is provided, saves to temporary directory with the default name.
83fn save_to_file(
84    content: String,
85    filename: Option<&str>,
86    default_name: &str,
87    content_type: &str,
88) -> Result<std::path::PathBuf> {
89    let file_path = if let Some(filename) = filename {
90        std::path::PathBuf::from(filename)
91    } else {
92        std::env::temp_dir().join(default_name)
93    };
94
95    std::fs::write(&file_path, content)?;
96    println!("Saved {} to {}", content_type, file_path.display());
97    Ok(file_path)
98}

113/// Compress JSON content using gzip compression.
114/// Returns the compressed bytes or an error if compression fails.
115#[cfg(feature = "viz")]
116fn compress_json(json_content: &str) -> Result<Vec<u8>> {
117    use flate2::write::GzEncoder;
118    use flate2::Compression;
119    
120    let mut encoder = GzEncoder::new(Vec::new(), Compression::best());
121    encoder.write_all(json_content.as_bytes())?;
122    encoder.finish()
123}

125/// Encode data to base64 URL-safe format without padding.
126/// This format is safe for use in URLs and doesn't require additional escaping.
127#[cfg(feature = "viz")]
128fn encode_base64_url_safe(data: &[u8]) -> String {
129    data_encoding::BASE64URL_NOPAD.encode(data)
130}
131
132/// Try to compress and encode JSON content for URL embedding.
133/// Returns (encoded_data, is_compressed, compression_ratio).
134/// 
135/// Compression is skipped for small JSON (<min_compression_size bytes).
136/// If compression fails or doesn't reduce size, falls back to uncompressed encoding.
137#[cfg(feature = "viz")]
138fn try_compress_and_encode(
139    json_content: &str,
140    config: &VisualizerConfig,
141) -> (String, bool, f64) {
142    let original_size = json_content.len();
143    
144    // Skip compression for small JSON
145    if !config.enable_compression || original_size < config.min_compression_size {
146        let encoded = encode_base64_url_safe(json_content.as_bytes());
147        return (encoded, false, 1.0);
148    }
149    
150    // Try compression
151    match compress_json(json_content) {
152        Ok(compressed) => {
153            let compressed_size = compressed.len();
154            let ratio = original_size as f64 / compressed_size as f64;
155            
156            // Only use compression if it actually reduces size
157            if compressed_size < original_size {
158                let encoded = encode_base64_url_safe(&compressed);
159                println!("📦 Compressed JSON: {} bytes → {} bytes (ratio: {:.2}x)", 
160                         original_size, compressed_size, ratio);
161                (encoded, true, ratio)
162            } else {
163                // Compression didn't help, use uncompressed
164                let encoded = encode_base64_url_safe(json_content.as_bytes());
165                (encoded, false, 1.0)
166            }
167        }
168        Err(e) => {
169            // Compression failed, fall back to uncompressed
170            println!("⚠️  Compression failed: {}, using uncompressed", e);
171            let encoded = encode_base64_url_safe(json_content.as_bytes());
172            (encoded, false, 1.0)
173        }
174    }
175}

177/// Calculate the total URL length for a given encoded data and parameter name.
178/// Returns the total length including base URL, parameter name, and encoded data.
179#[cfg(feature = "viz")]
180fn calculate_url_length(base_url: &str, param_name: &str, encoded_data: &str) -> usize {
181    // Format: base_url#param_name=encoded_data
182    base_url.len() + 1 + param_name.len() + 1 + encoded_data.len()
183}

185/// Generate a URL for the visualizer with the given JSON content.
186/// Automatically chooses between compressed and uncompressed encoding based on URL length.
187/// Returns (url, is_compressed) or None if the URL would be too long.
188#[cfg(feature = "viz")]
189fn generate_visualizer_url(
190    json_content: &str,
191    config: &VisualizerConfig,
192) -> Option<(String, bool)> {
193    let (encoded_data, is_compressed, _ratio) = try_compress_and_encode(json_content, config);
194    
195    // Determine parameter name based on compression
196    let param_name = if is_compressed { "compressed" } else { "data" };
197    
198    // Calculate total URL length
199    let url_length = calculate_url_length(&config.base_url, param_name, &encoded_data);
200    
201    println!("🔗 URL length: {} characters (max: {})", url_length, config.max_url_length);
202    
203    // Check if URL is within length limit
204    if url_length <= config.max_url_length {
205        let url = format!("{}#{}={}", config.base_url, param_name, encoded_data);
206        Some((url, is_compressed))
207    } else {
208        // If compressed URL is too long, try uncompressed as a fallback
209        if is_compressed {
210            println!("⚠️  Compressed URL too long, trying uncompressed...");
211            let uncompressed_encoded = encode_base64_url_safe(json_content.as_bytes());
212            let uncompressed_length = calculate_url_length(&config.base_url, "data", &uncompressed_encoded);
213            
214            if uncompressed_length <= config.max_url_length {
215                println!("✓ Uncompressed URL fits: {} characters", uncompressed_length);
216                let url = format!("{}#data={}", config.base_url, uncompressed_encoded);
217                return Some((url, false));
218            }
219        }
220        
221        println!("❌ URL too long even with compression, will use file-based fallback");
222        None
223    }
224}

226/// Generate a timestamped filename for temporary graph files.
227/// Format: hydro_graph_<timestamp>.json
228#[cfg(feature = "viz")]
229fn generate_timestamped_filename() -> String {
230    let timestamp = SystemTime::now()
231        .duration_since(UNIX_EPOCH)
232        .unwrap_or_else(|e| {
233            println!("⚠️  System time is before Unix epoch: {}. Using fallback timestamp 0.", e);
234            std::time::Duration::from_secs(0)
235        })
236        .as_secs();
237    format!("hydro_graph_{}.json", timestamp)
238}

240/// Save JSON content to a temporary file with a timestamped filename.

256/// URL-encode a file path for safe transmission in query parameters.
257/// Uses percent encoding to ensure special characters are properly escaped.
258/// 
259/// Requirements: 9.4
260#[cfg(feature = "viz")]
261fn url_encode_file_path(file_path: &std::path::Path) -> String {
262    let path_str = file_path.to_string_lossy();
263    urlencoding::encode(&path_str).to_string()
264}

270#[cfg(feature = "viz")]

276/// Print fallback instructions for manual loading of the graph file.
277/// Provides clear guidance if automatic browser opening fails.
278/// 
279/// Requirements: 9.6, 9.7
280#[cfg(feature = "viz")]
281fn print_fallback_instructions(file_path: &std::path::Path, url: &str) {
282    println!("\n📊 Graph Visualization Instructions");
283    println!("═══════════════════════════════════════════════════════════");
284    println!("The graph is too large to embed in a URL.");
285    println!("It has been saved to a temporary file:");
286    println!("  📁 {}", file_path.display());
287    println!();
288    println!("Opening visualizer in browser...");
289    println!("  🌐 {}", url);
290    println!();
291    println!("If the browser doesn't open automatically, you can:");
292    println!("  1. Manually open: {}", url);
293    println!("  2. Or visit {} and drag-and-drop the file", 
294             url.split('?').next().unwrap_or(url));
295    println!("═══════════════════════════════════════════════════════════\n");
296}

298/// Handle large graph visualization using file-based fallback.
299/// Saves the JSON to a temporary file and opens the visualizer with a file parameter.
300/// Uses the configured base URL from VisualizerConfig.
301/// 
302/// Requirements: 9.1, 9.2, 9.3, 9.4, 9.5, 9.6, 9.7, 9.8, 9.9
303#[cfg(feature = "viz")]
304fn handle_large_graph_fallback(
305    json_content: &str,
306    config: &VisualizerConfig,
307) -> Result<()> {
308    // Save JSON to temporary file with timestamped filename
309    let temp_file = save_json_to_temp_file(json_content)?;
310    
311    // Generate URL with file parameter using configured base URL
312    let url = generate_file_based_url(&temp_file, config);
313    
314    // Print fallback instructions
315    print_fallback_instructions(&temp_file, &url);
316    
317    // Try to open the visualizer in browser
318    match webbrowser::open(&url) {
319        Ok(_) => {
320            println!("✓ Successfully opened visualizer in browser");
321        }
322        Err(e) => {
323            println!("⚠️  Failed to open browser automatically: {}", e);
324            println!("Please manually open the URL above or drag-and-drop the file.");
325        }
326    }
327    
328    Ok(())
329}

331/// Open JSON visualizer with automatic fallback to file-based approach for large graphs.
332/// First attempts to embed the JSON in the URL using compression.
333/// If the URL is too long, falls back to saving the file and using a file parameter.
334/// 
335/// This is the main entry point for opening JSON visualizations.
336/// 
337/// Requirements: 8.1-8.9, 9.1-9.9
338#[cfg(feature = "viz")]
339fn open_json_visualizer_with_fallback(
340    json_content: &str,
341    config: &VisualizerConfig,
342) -> Result<()> {
343    // Try to generate a URL with embedded data
344    match generate_visualizer_url(json_content, config) {
345        Some((url, is_compressed)) => {
346            // URL fits within length limit, open it directly
347            let compression_msg = if is_compressed { " (compressed)" } else { "" };
348            println!("🌐 Opening visualizer with embedded data{}...", compression_msg);
349            webbrowser::open(&url)?;
350            println!("✓ Successfully opened visualizer in browser");
351            Ok(())
352        }
353        None => {
354            // URL too long, use file-based fallback
355            println!("📦 Graph too large for URL embedding, using file-based approach...");
356            handle_large_graph_fallback(json_content, config)
357        }
358    }
359}
