69/// Opens Hydro IR leaves in the new JSON visualizer in browser.
70/// Generates JSON and opens it via URL encoding in the docs visualizer.
71pub fn open_json_visualizer(leaves: &[HydroLeaf], config: Option<HydroWriteConfig>) -> Result<()> {
72    let json_content = render_with_config(leaves, config, render_hydro_ir_json);
73
74    // Debug: Print a snippet of the JSON to see if backtrace is included
75    println!("=== JSON CONTENT SAMPLE (for backtrace verification) ===");
76    if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(&json_content) {
77        if let Some(nodes) = parsed["nodes"].as_array() {
78            if let Some(first_node) = nodes.first() {
79                if let Some(backtrace) = first_node["data"]["backtrace"].as_array() {
80                    println!("First node has backtrace with {} elements", backtrace.len());
81                    if !backtrace.is_empty() {
82                        println!(
83                            "Sample backtrace element: {}",
84                            serde_json::to_string_pretty(&backtrace[0]).unwrap_or_default()
85                        );
86                    }
87                } else {
88                    println!("First node does not have backtrace data");
89                }
90            }
91        }
92    }
93    println!("=== END JSON SAMPLE ===");
94
95    open_json_browser_impl(&json_content)
96}

98fn open_mermaid_browser(mermaid_src: &str) -> Result<()> {
99    // Debug: Print the mermaid source being sent to browser
100    println!("=== MERMAID SOURCE BEING SENT TO BROWSER ===");
101    println!("{}", mermaid_src);
102    println!("=== END MERMAID SOURCE ===");
103
104    let state = serde_json::json!({
105        "code": mermaid_src,
106        "mermaid": serde_json::json!({
107            "theme": "default"
108        }),
109        "autoSync": true,
110        "updateDiagram": true
111    });
112    let state_json = serde_json::to_vec(&state)?;
113    let state_base64 = data_encoding::BASE64URL.encode(&state_json);
114    webbrowser::open(&format!(
115        "https://mermaid.live/edit#base64:{}",
116        state_base64
117    ))
118}

120fn open_dot_browser(dot_src: &str) -> Result<()> {
121    let mut url = "https://dreampuf.github.io/GraphvizOnline/#".to_owned();
122    for byte in dot_src.bytes() {
123        // Lazy percent encoding: https://en.wikipedia.org/wiki/Percent-encoding
124        write!(url, "%{:02x}", byte).unwrap();
125    }
126    webbrowser::open(&url)
127}

129fn open_json_browser_impl(json_content: &str) -> Result<()> {
130    #[cfg(feature = "viz")]
131    {
132        use data_encoding::BASE64URL_NOPAD;
133
134        // Encode the JSON data for URL
135        let encoded_data = BASE64URL_NOPAD.encode(json_content.as_bytes());
136
137        // Try localhost first (for development), then fall back to docs site
138        let localhost_url = format!("http://localhost:3000/visualizer#data={}", encoded_data);
139        let docs_url = format!("https://hydro.run/docs/visualizer#data={}", encoded_data);
140
141        // Try to open localhost first
142        match webbrowser::open(&localhost_url) {
143            Ok(_) => {
144                println!("Opened new JSON visualizer (localhost): {}", localhost_url);
145            }
146            Err(_) => {
147                // If localhost fails, try the main docs site
148                webbrowser::open(&docs_url)?;
149                println!("Opened new JSON visualizer: {}", docs_url);
150            }
151        }
152    }
153
154    #[cfg(not(feature = "viz"))]
155    {
156        println!("viz feature not enabled, cannot open browser");
157    }
158
159    Ok(())
160}

206/// Helper function to save content to a file with consistent path handling.
207/// If no filename is provided, saves to temporary directory with the default name.
208fn save_to_file(
209    content: String,
210    filename: Option<&str>,
211    default_name: &str,
212    content_type: &str,
213) -> Result<std::path::PathBuf> {
214    let file_path = if let Some(filename) = filename {
215        std::path::PathBuf::from(filename)
216    } else {
217        std::env::temp_dir().join(default_name)
218    };
219
220    std::fs::write(&file_path, content)?;
221    println!("Saved {} to {}", content_type, file_path.display());
222    Ok(file_path)
223}

225/// Helper function to handle config unwrapping and rendering.
226fn render_with_config<F>(
227    leaves: &[HydroLeaf],
228    config: Option<HydroWriteConfig>,

230) -> String
