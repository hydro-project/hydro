22struct TruncatedVecDebug<'a, T>(&'a Vec<T>, usize, fn(&T) -> Option<String>);
23impl<'a, T> Debug for TruncatedVecDebug<'a, T> {
24    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
25        let Self(vec, max, elem_debug) = self;
26        if vec.len() > *max {
27            f.debug_list()
28                .entries(
29                    vec[..*max]
30                        .iter()
31                        .map(|v| elem_debug(v).unwrap_or("?".to_string())),
32                )
33                .finish_non_exhaustive()?;
34            write!(f, " ({} total)", vec.len())
35        } else {
36            f.debug_list()
37                .entries(
38                    vec[..]
39                        .iter()
40                        .map(|v| elem_debug(v).unwrap_or("?".to_string())),
41                )
42                .finish()
43        }
44    }
45}

47pub struct StreamHook<T> {
48    pub input: Rc<RefCell<VecDeque<T>>>,
49    pub to_release: Option<Vec<T>>,
50    pub output: UnboundedSender<T>,
51    pub batch_location: (&'static str, &'static str, &'static str),
52    pub format_item_debug: fn(&T) -> Option<String>,
53}
54
55impl<T> SimHook for StreamHook<T> {
56    fn current_decision(&self) -> Option<bool> {
57        self.to_release.as_ref().map(|v| !v.is_empty())
58    }
59
60    fn can_make_nontrivial_decision(&self) -> bool {
61        !self.input.borrow().is_empty()
62    }
63
64    fn autonomous_decision<'a>(
65        &mut self,
66        driver: &mut Borrowed<'a>,
67        force_nontrivial: bool,
68    ) -> bool {
69        let mut current_input = self.input.borrow_mut();
70        let count = ((if force_nontrivial { 1 } else { 0 })..=current_input.len())
71            .generate(driver)
72            .unwrap();
73
74        self.to_release = Some(current_input.drain(0..count).collect());
75        count > 0
76    }
77
78    fn release_decision(&mut self, log_writer: &mut dyn std::fmt::Write) {
79        if let Some(to_release) = self.to_release.take() {
80            let (batch_location, line, caret_indent) = self.batch_location;
81            let note_str = if to_release.is_empty() {
82                "^ releasing no items".to_string()
83            } else {
84                format!(
85                    "^ releasing items: {:?}",
86                    TruncatedVecDebug(&to_release, 8, self.format_item_debug)
87                )
88            };
89
90            let _ = writeln!(
91                log_writer,
92                "{} {}",
93                "-->".color(colored::Color::Blue),
94                batch_location
95            );
96
97            let _ = writeln!(log_writer, " {}{}", "|".color(colored::Color::Blue), line);
98
99            let _ = writeln!(
100                log_writer,
101                " {}{}{}",
102                "|".color(colored::Color::Blue),
103                caret_indent,
104                note_str.color(colored::Color::Green)
105            );
106
107            for item in to_release {
108                self.output.send(item).unwrap();
109            }
110        } else {
111            panic!("No decision to release");
112        }
113    }
114}
