10/// A builder for DFIR graphs used in simulations.
11///
12/// Instead of emitting one DFIR graph per location, we emit one big DFIR graph in `async_level`,
13/// which contains all asynchronously executed top-level operators in the Hydro program. Because
14/// "top-level" operators guarantee "eventual determinism" (per Flo), we do not need to simulate
15/// every possible interleaving of message arrivals and processing. Instead, we only need to
16/// simulate sources of non-determinism at the points in the program where a user intentionally
17/// observes them (such as batch or assume_ordering).
18///
19/// Because each tick relies on a set of decisions being made to select their inputs (batch,
20/// snapshot), we emit each tick's code into a separate DFIR graph. Each non-deterministic input
21/// to a tick has a corresponding "hook" that the simulation runtime can use to control the
22/// non-deterministic decision made at that boundary. This hook interacts with the DFIR program
23/// by accumulating inputs from the async level into a buffer, and then the hook can send selected
24/// elements from that buffer into the tick's DFIR graph with a separate handoff channel.
25pub struct SimBuilder {
26    pub extra_stmts: Vec<syn::Stmt>,
27    pub async_level: FlatGraphBuilder,
28    pub tick_dfirs: BTreeMap<LocationId, FlatGraphBuilder>,
29    pub next_hoff_id: usize,
30}
31
32impl DfirBuilder for SimBuilder {
33    fn get_dfir_mut(&mut self, location: &LocationId) -> &mut FlatGraphBuilder {
34        match location {
35            LocationId::Process(_) => &mut self.async_level,
36            LocationId::Cluster(_) => panic!("SimBuilder does not support clusters"),
37            LocationId::Atomic(_) => panic!("SimBuilder does not support atomic locations"),
38            LocationId::Tick(_, _) => self.tick_dfirs.entry(location.clone()).or_default(),
39        }
40    }
41
42    fn batch(
43        &mut self,
44        in_ident: syn::Ident,
45        in_location: &LocationId,
46        in_kind: &CollectionKind,
47        out_ident: &syn::Ident,
48        out_location: &LocationId,
49    ) {
50        match in_kind {
51            CollectionKind::Stream {
52                order: StreamOrder::TotalOrder,
53                retry: StreamRetry::ExactlyOnce,
54                ..
55            } => {
56                debug_assert!(in_location.is_top_level());
57
58                let hoff_id = self.next_hoff_id;
59                self.next_hoff_id += 1;
60
61                let out_location_ser = serde_json::to_string(out_location).unwrap();
62                let buffered_ident =
63                    syn::Ident::new(&format!("__buffered_{hoff_id}"), Span::call_site());
64                let hoff_send_ident =
65                    syn::Ident::new(&format!("__hoff_send_{hoff_id}"), Span::call_site());
66                let hoff_recv_ident =
67                    syn::Ident::new(&format!("__hoff_recv_{hoff_id}"), Span::call_site());
68
69                self.extra_stmts.push(syn::parse_quote! {
70                    let (#hoff_send_ident, #hoff_recv_ident) = __root_dfir_rs::util::unbounded_channel();
71                });
72                self.extra_stmts.push(syn::parse_quote! {
73                    let #buffered_ident = ::std::rc::Rc::new(::std::cell::RefCell::new(::std::collections::VecDeque::new()));
74                });
75                self.extra_stmts.push(syn::parse_quote! {
76                    __hydro_hooks.entry(#out_location_ser).or_default().push(Box::new(hydro_lang::sim::runtime::StreamHook {
77                        input: #buffered_ident.clone(),
78                        to_release: None,
79                        output: #hoff_send_ident,
80                    }));
81                });
82
83                self.async_level.add_dfir(
84                    parse_quote! {
85                        #in_ident -> for_each(|v| #buffered_ident.borrow_mut().push_back(v));
86                    },
87                    None,
88                    None,
89                );
90
91                self.get_dfir_mut(out_location).add_dfir(
92                    parse_quote! {
93                        #out_ident = source_stream(#hoff_recv_ident);
94                    },
95                    None,
96                    None,
97                );
98            }
99            _ => todo!(),
100        }
101    }
102
103    fn yield_from_tick(
104        &mut self,
105        in_ident: syn::Ident,
106        in_location: &LocationId,
107        in_kind: &CollectionKind,
108        out_ident: &syn::Ident,
109    ) {
110        match in_kind {
111            CollectionKind::Stream { .. } => {
112                if let LocationId::Tick(_, outer) = in_location {
113                    debug_assert!(outer.is_top_level());
114
115                    let hoff_id = self.next_hoff_id;
116                    self.next_hoff_id += 1;
117
118                    let hoff_send_ident =
119                        syn::Ident::new(&format!("__hoff_send_{hoff_id}"), Span::call_site());
120                    let hoff_recv_ident =
121                        syn::Ident::new(&format!("__hoff_recv_{hoff_id}"), Span::call_site());
122
123                    self.extra_stmts.push(syn::parse_quote! {
124                        let (#hoff_send_ident, #hoff_recv_ident) = __root_dfir_rs::util::unbounded_channel();
125                    });
126
127                    self.get_dfir_mut(in_location).add_dfir(
128                        parse_quote! {
129                            #in_ident -> for_each(|v| #hoff_send_ident.send(v).unwrap());
130                        },
131                        None,
132                        None,
133                    );
134
135                    self.async_level.add_dfir(
136                        parse_quote! {
137                            #out_ident = source_stream(#hoff_recv_ident);
138                        },
139                        None,
140                        None,
141                    );
142                } else {
143                    panic!()
144                }
145            }
146            _ => todo!(),
147        }
148    }

176    fn create_external_source(
177        &mut self,
178        _on: &LocationId,
179        source_expr: syn::Expr,
180        out_ident: &syn::Ident,
181        deserialize: &Option<DebugExpr>,
182        tag_id: usize,
183    ) {
184        if let Some(deserialize_pipeline) = deserialize {
185            self.async_level.add_dfir(
186                parse_quote! {
187                    #out_ident = source_stream(#source_expr) -> map(|v| -> ::std::result::Result<_, ()> { Ok(v) }) -> map(#deserialize_pipeline);
188                },
189                None,
190                Some(&format!("recv{}", tag_id)),
191            );
192        } else {
193            self.async_level.add_dfir(
194                parse_quote! {
195                    #out_ident = source_stream(#source_expr);
196                },
197                None,
198                Some(&format!("recv{}", tag_id)),
199            );
200        }
201    }
202
203    fn create_external_output(
204        &mut self,
205        _on: &LocationId,
206        sink_expr: syn::Expr,
207        input_ident: &syn::Ident,
208        serialize: &Option<DebugExpr>,
209        tag_id: usize,
210    ) {
211        let grabbed_ident = syn::Ident::new(&format!("__sink_{tag_id}"), Span::call_site());
212        self.extra_stmts.push(syn::parse_quote! {
213            let #grabbed_ident = #sink_expr;
214        });
215
216        if let Some(serialize_pipeline) = serialize {
217            self.async_level.add_dfir(
218                parse_quote! {
219                    #input_ident -> map(#serialize_pipeline) -> for_each(|v| #grabbed_ident.send(v).unwrap());
220                },
221                None,
222                Some(&format!("send{}", tag_id)),
223            );
224        } else {
225            self.async_level.add_dfir(
226                parse_quote! {
227                    #input_ident -> for_each(|v| #grabbed_ident.send(v).unwrap());
228                },
229                None,
230                Some(&format!("send{}", tag_id)),
231            );
232        }
233    }
