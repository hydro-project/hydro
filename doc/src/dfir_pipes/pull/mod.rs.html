100/// The result of polling a [`Pull`].
101///
102/// `PullStep` represents the three possible outcomes when pulling from a stream:
103/// - `Ready(item, meta)`: An item is available along with associated metadata.
104/// - `Pending(can_pend)`: No item is available yet, but more may come (async).
105/// - `Ended(can_end)`: The stream has terminated and will produce no more items.
106///
107/// The `CanPend` and `CanEnd` type parameters use [`Toggle`] to statically encode
108/// which variants are possible. When a variant is impossible (e.g., `CanPend = No`),
109/// its payload type becomes [`No`], making it a compile error to construct.
110#[derive(Clone, Copy, Debug, Eq, Hash, Ord, PartialEq, PartialOrd)]
111pub enum PullStep<Item, Meta, CanPend: Toggle, CanEnd: Toggle> {
112    /// An item is ready with associated metadata.
113    Ready(Item, Meta),
114    /// The pull is not ready yet (only possible when `CanPend = Yes`).
115    Pending(CanPend),
116    /// The pull has ended (only possible when `CanEnd = Yes`).
117    Ended(CanEnd),
118}
119
120impl<Item, Meta, CanPend: Toggle, CanEnd: Toggle> PullStep<Item, Meta, CanPend, CanEnd> {
121    /// Creates a new `PullStep::Ended`, or panics if `CanEnd = No`.
122    pub fn ended() -> Self {
123        PullStep::Ended(Toggle::create())
124    }
125
126    /// Creates a new `PullStep::Pending`, or panics if `CanPend = No`.
127    pub fn pending() -> Self {
128        PullStep::Pending(Toggle::create())
129    }
130
131    /// Returns `true` if the step is a [`PullStep::Ready`].
132    pub const fn is_ready(&self) -> bool {
133        matches!(self, PullStep::Ready(_, _))
134    }
135
136    /// Returns `true` if the step is a [`PullStep::Pending`].
137    pub const fn is_pending(&self) -> bool {
138        matches!(self, PullStep::Pending(_))
139    }
140
141    /// Returns `true` if the step is a [`PullStep::Ended`].
142    pub const fn is_ended(&self) -> bool {
143        matches!(self, PullStep::Ended(_))
144    }
145
146    /// Tries to convert the `CanPend` and `CanEnd` type parameters, returning `None` if the conversion is invalid.
147    pub fn try_convert_into<NewPend: Toggle, NewEnd: Toggle>(
148        self,
149    ) -> Option<PullStep<Item, Meta, NewPend, NewEnd>> {
150        Some(match self {
151            Self::Ready(item, meta) => PullStep::Ready(item, meta),
152            Self::Pending(_) => PullStep::Pending(Toggle::try_create()?),
153            Self::Ended(_) => PullStep::Ended(Toggle::try_create()?),
154        })
155    }
156
157    /// Converts the `CanPend` and `CanEnd` type parameters, panicking if the conversion is invalid.
158    pub fn convert_into<NewPend: Toggle, NewEnd: Toggle>(
159        self,
160    ) -> PullStep<Item, Meta, NewPend, NewEnd> {
161        match self {
162            Self::Ready(item, meta) => PullStep::Ready(item, meta),
163            Self::Pending(_) => PullStep::pending(),
164            Self::Ended(_) => PullStep::ended(),
165        }
166    }
167
168    /// Converts this `PullStep` into a [`Poll`]`<Option<(Item, Meta)>>`.
169    pub fn into_poll(self) -> Poll<Option<(Item, Meta)>> {
170        match self {
171            PullStep::Ready(item, meta) => Poll::Ready(Some((item, meta))),
172            PullStep::Pending(_) => Poll::Pending,
173            PullStep::Ended(_) => Poll::Ready(None),
174        }
175    }
176}

178/// The `Pull` trait represents a pull-based stream that can be polled for items.
179///
180/// The `Ctx` type parameter allows operators to be generic over the context type.
181/// Most operators don't use the context and just forward it to their predecessor,
182/// so they can be generic over `Ctx`. Operators that need `std::task::Context`
183/// (like `StreamReady`) will use `Ctx = &mut Context<'_>`.
184///
185/// Setting `Ctx = ()` allows most pull pipelines to be used without any context.
186pub trait Pull {
187    /// The context type required to poll this pull.
188    type Ctx<'ctx>: Context<'ctx>;
189
190    /// The type of items yielded by this pull.
191    type Item;
192    /// The metadata type associated with each item.
193    type Meta: Copy;
194    /// Whether this pull can return [`PullStep::Pending`].
195    type CanPend: Toggle;
196    /// Whether this pull can return [`PullStep::Ended`].
197    type CanEnd: Toggle;
198
199    /// Attempts to pull the next item from this stream.
200    fn pull(
201        self: Pin<&mut Self>,
202        ctx: &mut Self::Ctx<'_>,
203    ) -> PullStep<Self::Item, Self::Meta, Self::CanPend, Self::CanEnd>;
204
205    /// Returns the bounds on the remaining length of the pull.
206    ///
207    /// Specifically, `size_hint()` returns a tuple where the first element
208    /// is the lower bound, and the second element is the upper bound.
209    ///
210    /// The second half of the tuple that is returned is an [`Option`]`<`[`usize`]`>`.
211    /// A [`None`] here means that either there is no known upper bound, or the
212    /// upper bound is larger than [`usize`].
213    ///
214    /// # Implementation notes
215    ///
216    /// It is not enforced that a pull implementation yields the declared
217    /// number of elements. A buggy pull may yield less than the lower bound
218    /// or more than the upper bound of elements.
219    ///
220    /// `size_hint()` is primarily intended to be used for optimizations such as
221    /// reserving space for the elements of the pull, but must not be trusted
222    /// to e.g., omit bounds checks in unsafe code. An incorrect implementation
223    /// of `size_hint()` should not lead to memory safety violations.
224    ///
225    /// That said, the implementation should provide a correct estimation,
226    /// because otherwise it would be a violation of the trait's protocol.
227    ///
228    /// A default implementation should return `(0, None)` which is correct for any
229    /// pull. However this is not provided, to prevent oversight.
230    fn size_hint(&self) -> (usize, Option<usize>);
231
232    /// Borrows this pull, allowing it to be used by reference.
233    fn by_ref(&mut self) -> &mut Self {
234        self
235    }
236
237    /// Takes two pulls and creates a new pull over both in sequence.
238    ///
239    /// `chain()` will return a new pull which will first iterate over
240    /// values from the first pull and then over values from the second pull.
241    ///
242    /// The first pull must be finite (`CanEnd = Yes`) and fused ([`FusedPull`]).
243    fn chain<U>(self, other: U) -> Chain<Self, U>
244    where
245        Self: Sized + FusedPull<CanEnd = Yes>,
246        U: Pull<Item = Self::Item, Meta = Self::Meta>,
247    {
248        Chain::new(self, other)
249    }

609/// A marker trait for pulls that are "fused".
610///
611/// A fused pull guarantees that once it returns [`PullStep::Ended`], all subsequent
612/// calls to [`Pull::pull`] will also return [`PullStep::Ended`]. This property allows
613/// downstream operators like [`Pull::chain`] to avoid tracking whether
614/// the upstream has ended.
615///
616/// Implementors should ensure this invariant is upheld. The [`Pull::fuse`]
617/// adapter can be used to make any pull fused.
618pub trait FusedPull: Pull {}
619
620impl<P> FusedPull for &mut P where P: FusedPull + Unpin + ?Sized {}

656/// The closure is called each time the pull is polled and should return a `PullStep`.
657pub fn from_fn<F, Item, Meta, CanEnd>(func: F) -> FromFn<F, Item, Meta, CanEnd>
658where
659    F: FnMut() -> PullStep<Item, Meta, No, CanEnd>,
660    Meta: Copy,
661    CanEnd: Toggle,
662{
663    FromFn::new(func)
664}
665
666/// Creates an asynchronous or synchronous pull from a closure.
667///
668/// The closure is called each time the pull is polled and should return a `PullStep`.
669pub fn poll_fn<F, Item, Meta, CanPend, CanEnd>(func: F) -> PollFn<F, Item, Meta, CanPend, CanEnd>
670where
671    F: FnMut(&mut core::task::Context<'_>) -> PullStep<Item, Meta, CanPend, CanEnd>,
672    Meta: Copy,
673    CanPend: Toggle,
674    CanEnd: Toggle,
675{
676    PollFn::new(func)
677}

702/// A macro to override [`Pull::fuse`] for pulls that are already fused.
703///
704/// This macro should be used in the `impl` block for a pull type that is already fused.
705/// It provides a default implementation of `fuse` that simply returns `self`.
706macro_rules! fuse_self {
707    () => {
708        fn fuse(
709            self,
710        ) -> impl for<'ctx> FusedPull<
711            Ctx<'ctx> = Self::Ctx<'ctx>,
712            Item = Self::Item,
713            Meta = Self::Meta,
714            CanPend = Self::CanPend,
715            CanEnd = Self::CanEnd,
716        >
717        where
718            Self: Sized,
719        {
720            self
721        }
722    };
723}
724use fuse_self;
