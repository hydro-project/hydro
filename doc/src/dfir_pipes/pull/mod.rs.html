95/// The result of polling a [`Pull`].
96///
97/// `PullStep` represents the three possible outcomes when pulling from a stream:
98/// - `Ready(item, meta)`: An item is available along with associated metadata.
99/// - `Pending(can_pend)`: No item is available yet, but more may come (async).
100/// - `Ended(can_end)`: The stream has terminated and will produce no more items.
101///
102/// The `CanPend` and `CanEnd` type parameters use [`Toggle`] to statically encode
103/// which variants are possible. When a variant is impossible (e.g., `CanPend = No`),
104/// its payload type becomes [`No`], making it a compile error to construct.
105pub enum PullStep<Item, Meta, CanPend: Toggle, CanEnd: Toggle> {
106    /// An item is ready with associated metadata.
107    Ready(Item, Meta),
108    /// The pull is not ready yet (only possible when `CanPend = Yes`).
109    Pending(CanPend),
110    /// The pull has ended (only possible when `CanEnd = Yes`).
111    Ended(CanEnd),
112}
113
114impl<Item, Meta, CanPend: Toggle, CanEnd: Toggle> PullStep<Item, Meta, CanPend, CanEnd> {
115    /// Creates a new `PullStep::Ended`, or panics if `CanEnd = No`.
116    pub fn ended() -> Self {
117        PullStep::Ended(Toggle::create())
118    }
119
120    /// Creates a new `PullStep::Pending`, or panics if `CanPend = No`.
121    pub fn pending() -> Self {
122        PullStep::Pending(Toggle::create())
123    }
124
125    /// Returns `true` if the step is a [`PullStep::Ended`].
126    pub const fn is_ended(&self) -> bool {
127        matches!(self, PullStep::Ended(_))
128    }
129
130    /// Returns `true` if the step is a [`PullStep::Pending`].
131    pub const fn is_pending(&self) -> bool {
132        matches!(self, PullStep::Pending(_))
133    }
134
135    /// Tries to convert the `CanPend` and `CanEnd` type parameters, returning `None` if the conversion is invalid.
136    pub fn try_convert_into<NewPend: Toggle, NewEnd: Toggle>(
137        self,
138    ) -> Option<PullStep<Item, Meta, NewPend, NewEnd>> {
139        Some(match self {
140            Self::Ready(item, meta) => PullStep::Ready(item, meta),
141            Self::Pending(_) => PullStep::Pending(Toggle::try_create()?),
142            Self::Ended(_) => PullStep::Ended(Toggle::try_create()?),
143        })
144    }
145
146    /// Converts the `CanPend` and `CanEnd` type parameters, panicking if the conversion is invalid.
147    pub fn convert_into<NewPend: Toggle, NewEnd: Toggle>(
148        self,
149    ) -> PullStep<Item, Meta, NewPend, NewEnd> {
150        match self {
151            Self::Ready(item, meta) => PullStep::Ready(item, meta),
152            Self::Pending(_) => PullStep::pending(),
153            Self::Ended(_) => PullStep::ended(),
154        }
155    }
156
157    /// Converts this `PullStep` into a [`Poll`]`<Option<(Item, Meta)>>`.
158    pub fn into_poll(self) -> Poll<Option<(Item, Meta)>> {
159        match self {
160            PullStep::Ready(item, meta) => Poll::Ready(Some((item, meta))),
161            PullStep::Pending(_) => Poll::Pending,
162            PullStep::Ended(_) => Poll::Ready(None),
163        }
164    }
165}

167/// The `Pull` trait represents a pull-based stream that can be polled for items.
168///
169/// The `Ctx` type parameter allows operators to be generic over the context type.
170/// Most operators don't use the context and just forward it to their predecessor,
171/// so they can be generic over `Ctx`. Operators that need `std::task::Context`
172/// (like `StreamReady`) will use `Ctx = &mut Context<'_>`.
173///
174/// Setting `Ctx = ()` allows most pull pipelines to be used without any context.
175pub trait Pull {
176    /// The context type required to poll this pull.
177    type Ctx<'ctx>: Context<'ctx>;
178
179    /// The type of items yielded by this pull.
180    type Item;
181    /// The metadata type associated with each item.
182    type Meta: Copy;
183    /// Whether this pull can return [`PullStep::Pending`].
184    type CanPend: Toggle;
185    /// Whether this pull can return [`PullStep::Ended`].
186    type CanEnd: Toggle;
187
188    /// Attempts to pull the next item from this stream.
189    fn pull(
190        self: Pin<&mut Self>,
191        ctx: &mut Self::Ctx<'_>,
192    ) -> PullStep<Self::Item, Self::Meta, Self::CanPend, Self::CanEnd>;
193
194    /// Returns the bounds on the remaining length of the pull.
195    ///
196    /// Specifically, `size_hint()` returns a tuple where the first element
197    /// is the lower bound, and the second element is the upper bound.
198    ///
199    /// The second half of the tuple that is returned is an [`Option`]`<`[`usize`]`>`.
200    /// A [`None`] here means that either there is no known upper bound, or the
201    /// upper bound is larger than [`usize`].
202    ///
203    /// # Implementation notes
204    ///
205    /// It is not enforced that a pull implementation yields the declared
206    /// number of elements. A buggy pull may yield less than the lower bound
207    /// or more than the upper bound of elements.
208    ///
209    /// `size_hint()` is primarily intended to be used for optimizations such as
210    /// reserving space for the elements of the pull, but must not be trusted
211    /// to e.g., omit bounds checks in unsafe code. An incorrect implementation
212    /// of `size_hint()` should not lead to memory safety violations.
213    ///
214    /// That said, the implementation should provide a correct estimation,
215    /// because otherwise it would be a violation of the trait's protocol.
216    ///
217    /// The default implementation returns `(0, None)` which is correct for any
218    /// pull.
219    #[inline]
220    fn size_hint(self: Pin<&Self>) -> (usize, Option<usize>) {
221        (0, None)
222    }
223
224    /// Borrows this pull, allowing it to be used by reference.
225    fn by_ref(&mut self) -> &mut Self {
226        self
227    }
228
229    /// Takes two pulls and creates a new pull over both in sequence.
230    ///
231    /// `chain()` will return a new pull which will first iterate over
232    /// values from the first pull and then over values from the second pull.
233    ///
234    /// The first pull must be finite (`CanEnd = Yes`) and fused ([`FusedPull`]).
235    fn chain<U>(self, other: U) -> Chain<Self, U>
236    where
237        Self: Sized + FusedPull<CanEnd = Yes>,
238        U: Pull<Item = Self::Item, Meta = Self::Meta>,
239    {
240        Chain::new(self, other)
241    }
242
243    /// Creates a pull which gives the current iteration count as well as the next value.
244    ///
245    /// The pull returned yields pairs `(i, val)`, where `i` is the current index
246    /// of iteration and `val` is the value returned by the pull.
247    fn enumerate(self) -> Enumerate<Self>
248    where
249        Self: Sized,
250    {
251        Enumerate::new(self)
252    }
253
254    /// Creates a pull which uses a closure to determine if an element should be yielded.
255    ///
256    /// Given an element the closure must return `true` or `false`. The returned pull
257    /// will yield only the elements for which the closure returns `true`.
258    fn filter<P>(self, predicate: P) -> Filter<Self, P>
259    where
260        Self: Sized,
261        P: FnMut(&Self::Item) -> bool,
262    {
263        Filter::new(self, predicate)
264    }

266    /// Creates a pull that both filters and maps.
267    ///
268    /// The returned pull yields only the values for which the supplied closure

270    fn filter_map<B, F>(self, f: F) -> FilterMap<Self, F>
271    where
272        Self: Sized,
273        F: FnMut(Self::Item) -> Option<B>,
274    {
275        FilterMap::new(self, f)
276    }

278    /// Creates a pull that works like map, but flattens nested structure.
279    ///
280    /// The `flat_map()` method is useful when you have a pull of items, and you
281    /// want to apply a function that returns an iterator for each item, then
282    /// flatten all those iterators into a single pull.

284    where

290    }

297    where

324    /// Creates a pull that ends after the first `None`.
325    ///
326    /// After a pull returns `Ended` for the first time, the behavior of calling
327    /// `pull` again is implementation-defined. `fuse()` adapts any pull,

338        Meta = Self::Meta,

570/// A marker trait for pulls that are "fused".
571///
572/// A fused pull guarantees that once it returns [`PullStep::Ended`], all subsequent
573/// calls to [`Pull::pull`] will also return [`PullStep::Ended`]. This property allows
574/// downstream operators like [`Pull::chain`] to avoid tracking whether
575/// the upstream has ended.
576///
577/// Implementors should ensure this invariant is upheld. The [`Pull::fuse`]
578/// adapter can be used to make any pull fused.
579pub trait FusedPull: Pull {}

656/// A macro to override [`Pull::fuse`] for pulls that are already fused.
657///
658/// This macro should be used in the `impl` block for a pull type that is already fused.
659/// It provides a default implementation of `fuse` that simply returns `self`.
660macro_rules! fuse_self {
661    () => {
662        fn fuse(
