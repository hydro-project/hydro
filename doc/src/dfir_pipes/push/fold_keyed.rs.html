12pin_project! {
13    /// Push combinator that folds items by key into a hashmap, then emits all
14    /// (key, accumulator) pairs downstream on finalize.
15    ///
16    /// Input items are `(K, V)`, accumulated into `HashMap<K, Acc>` via
17    /// `InitFn: FnMut() -> Acc` and `CombFn: FnMut(&mut Acc, V)`, then
18    /// emits `(K, Acc)` downstream.
19    #[must_use = "`Push`es do nothing unless items are pushed into them"]
20    pub struct FoldKeyed<MapRef, InitFn, CombFn, Next, K, V, Acc> {
21        #[pin]
22        next: Next,
23        map: MapRef,
24        init_fn: InitFn,
25        comb_fn: CombFn,
26        flush_items: Vec<(K, Acc)>,
27        flush_idx: usize,
28        _phantom: core::marker::PhantomData<fn(V)>,
29    }
30}
31
32impl<MapRef, InitFn, CombFn, Next, K, V, Acc> FoldKeyed<MapRef, InitFn, CombFn, Next, K, V, Acc> {
33    /// Creates a new `FoldKeyed` push combinator.
34    pub const fn new(map: MapRef, init_fn: InitFn, comb_fn: CombFn, next: Next) -> Self {
35        Self {
36            next,
37            map,
38            init_fn,
39            comb_fn,
40            flush_items: Vec::new(),
41            flush_idx: 0,
42            _phantom: core::marker::PhantomData,
43        }
44    }
45}
46
47// Impl for `&mut HashMap<K, Acc, S>` (works with any hasher including FxHashMap).
48// TODO(mingwei): support arbitrary metadata.
49impl<K, V, Acc, S, InitFn, CombFn, Next> Push<(K, V), ()>
50    for FoldKeyed<&mut std::collections::HashMap<K, Acc, S>, InitFn, CombFn, Next, K, V, Acc>
51where
52    K: Eq + Hash + Clone,
53    Acc: Clone,
54    S: BuildHasher,
55    InitFn: FnMut() -> Acc,
56    CombFn: FnMut(&mut Acc, V),
57    Next: Push<(K, Acc), ()>,
58{
59    type Ctx<'ctx> = Next::Ctx<'ctx>;
60
61    type CanPend = Next::CanPend;
62
63    fn poll_ready(self: Pin<&mut Self>, _ctx: &mut Self::Ctx<'_>) -> PushStep<Self::CanPend> {
64        PushStep::Done
65    }
66
67    fn start_send(self: Pin<&mut Self>, item: (K, V), _meta: ()) {
68        let this = self.project();
69        let entry = this.map.entry(item.0).or_insert_with(|| (this.init_fn)());
70        (this.comb_fn)(entry, item.1);
71    }
72
73    fn poll_finalize(self: Pin<&mut Self>, ctx: &mut Self::Ctx<'_>) -> PushStep<Self::CanPend> {
74        let mut this = self.project();
75        if this.flush_items.is_empty() && *this.flush_idx == 0 {
76            #[expect(
77                clippy::disallowed_methods,
78                reason = "collected into a Vec; key order is irrelevant"
79            )]
80            {
81                this.flush_items
82                    .extend(this.map.iter().map(|(k, v)| (k.clone(), v.clone())));
83            }
84            *this.flush_idx = 1; // mark as initialized
85        }
86        while !this.flush_items.is_empty() {
87            ready!(this.next.as_mut().poll_ready(ctx));
88            let item = this.flush_items.pop().unwrap();
89            this.next.as_mut().start_send(item, ());
90        }
91        let step = this.next.poll_finalize(ctx);
92        if step.is_done() {
93            *this.flush_idx = 0;
94        }
95        step
96    }
97
98    fn size_hint(self: Pin<&mut Self>, _hint: (usize, Option<usize>)) {}
99}

103    use alloc::string::String;

113    #[test]
114    fn fold_keyed_emits_on_finalize() {
115        let mut map = HashMap::new();
116        let mut tp = TestPush::no_pend();
117        let mut fk = FoldKeyed::new(&mut map, || 0i32, |acc: &mut i32, v| *acc += v, &mut tp);
118        let mut fk = Pin::new(&mut fk);
119        fk.as_mut().poll_ready(&mut ());
120        fk.as_mut().start_send((1, 10), ());
121        fk.as_mut().poll_ready(&mut ());
122        fk.as_mut().start_send((1, 20), ());
123        fk.as_mut().poll_ready(&mut ());
124        fk.as_mut().start_send((2, 30), ());
125        fk.as_mut().poll_finalize(&mut ());
126        let mut items = tp.items();
127        items.sort();
128        assert_eq!(items, vec![(1, 30), (2, 30)]);
129    }
130
131    #[test]
132    fn fold_keyed_different_acc_type() {
133        // Input (K, V) = (i32, &str), Acc = String
134        let mut map: HashMap<i32, String> = HashMap::new();
135        let mut tp = TestPush::no_pend();
136        let mut fk = FoldKeyed::new(
137            &mut map,
138            String::new,
139            |acc: &mut String, v: &str| acc.push_str(v),
140            &mut tp,
141        );
142        let mut fk = Pin::new(&mut fk);
143        fk.as_mut().start_send((1, "hello"), ());
144        fk.as_mut().start_send((1, " world"), ());
145        fk.as_mut().poll_finalize(&mut ());
146        assert_eq!(tp.items(), vec![(1, String::from("hello world"))]);
147    }

164    #[test]
165    fn fold_keyed_resumes_after_pending() {
166        let mut map = HashMap::new();
167        let mut tp: TestPush<(i32, i32), Yes, true> =
168            TestPush::new_fused([PushStep::Done, PushStep::pending(), PushStep::Done], []);
169        let mut fk = FoldKeyed::new(&mut map, || 0i32, |acc: &mut i32, v| *acc += v, &mut tp);
170        let mut fk = Pin::new(&mut fk);
171        fk.as_mut().start_send((1, 10), ());
172        fk.as_mut().start_send((2, 20), ());
173        // First finalize: sends one item, then Pending
174        let step = fk.as_mut().poll_finalize(&mut ());
175        assert!(step.is_pending());
176        // Second finalize: resumes and sends remaining
177        let step = fk.as_mut().poll_finalize(&mut ());
178        assert!(step.is_done());
179        let mut items = tp.items();
180        items.sort();
181        assert_eq!(items, vec![(1, 10), (2, 20)]);
182    }
