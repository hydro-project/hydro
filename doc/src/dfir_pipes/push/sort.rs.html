1//! [`Sort`] push combinator.
2use alloc::vec::Vec;
3use core::pin::Pin;
4
5use pin_project_lite::pin_project;
6
7use crate::push::{Push, PushStep, ready};
8
9pin_project! {
10    /// Push combinator that collects all items, sorts them, then emits them
11    /// downstream in sorted order on finalize.
12    #[must_use = "`Push`es do nothing unless items are pushed into them"]
13    #[derive(Clone, Debug)]
14    pub struct Sort<Item, Next> {
15        #[pin]
16        next: Next,
17        buf: Vec<Item>,
18        sorted: bool,
19    }
20}
21
22impl<Item, Next> Sort<Item, Next> {
23    /// Creates a new `Sort` push combinator.
24    pub const fn new(next: Next) -> Self {
25        Self {
26            next,
27            buf: Vec::new(),
28            sorted: false,
29        }
30    }
31}
32
33// TODO(mingwei): support arbitrary metadata.
34impl<Item, Next> Push<Item, ()> for Sort<Item, Next>
35where
36    Item: Ord,
37    Next: Push<Item, ()>,
38{
39    type Ctx<'ctx> = Next::Ctx<'ctx>;
40
41    type CanPend = Next::CanPend;
42
43    fn poll_ready(self: Pin<&mut Self>, _ctx: &mut Self::Ctx<'_>) -> PushStep<Self::CanPend> {
44        PushStep::Done
45    }
46
47    fn start_send(self: Pin<&mut Self>, item: Item, _meta: ()) {
48        let this = self.project();
49        this.buf.push(item);
50        *this.sorted = false;
51    }
52
53    fn poll_finalize(self: Pin<&mut Self>, ctx: &mut Self::Ctx<'_>) -> PushStep<Self::CanPend> {
54        let mut this = self.project();
55        if !*this.sorted {
56            this.buf.sort_unstable_by(|a, b| b.cmp(a));
57            *this.sorted = true;
58        }
59        while !this.buf.is_empty() {
60            ready!(this.next.as_mut().poll_ready(ctx));
61            let item = this.buf.pop().unwrap();
62            this.next.as_mut().start_send(item, ());
63        }
64        this.next.poll_finalize(ctx)
65    }
66
67    fn size_hint(self: Pin<&mut Self>, hint: (usize, Option<usize>)) {
68        let this = self.project();
69        this.buf.reserve(hint.0);
70        this.next.size_hint(hint);
71    }
72}

74#[cfg(test)]
75mod tests {
76    use alloc::vec;
77    use core::pin::Pin;
78
79    use crate::Yes;
80    use crate::push::test_utils::TestPush;
81    use crate::push::{Push, PushStep};
82
83    #[test]
84    fn sort_emits_sorted_on_finalize() {
85        let mut tp = TestPush::no_pend();
86        let mut s = crate::push::sort(&mut tp);
87        let mut s = Pin::new(&mut s);
88        s.as_mut().poll_ready(&mut ());
89        s.as_mut().start_send(3, ());
90        s.as_mut().poll_ready(&mut ());
91        s.as_mut().start_send(1, ());
92        s.as_mut().poll_ready(&mut ());
93        s.as_mut().start_send(2, ());
94        s.as_mut().poll_finalize(&mut ());
95        assert_eq!(tp.items(), vec![1, 2, 3]);
96    }
97
98    #[test]
99    fn sort_resumes_on_pending_without_dropping_items() {
100        // poll_ready returns Pending on the second item, then Done on retry.
101        let mut tp: TestPush<i32, Yes, true> = TestPush::new_fused(
102            [
103                PushStep::Done,
104                PushStep::pending(),
105                PushStep::Done,
106                PushStep::Done,
107            ],
108            [],
109        );
110        {
111            let mut s = crate::push::sort(&mut tp);
112            let mut s = Pin::new(&mut s);
113            s.as_mut().poll_ready(&mut ());
114            s.as_mut().start_send(3, ());
115            s.as_mut().poll_ready(&mut ());
116            s.as_mut().start_send(1, ());
117            s.as_mut().poll_ready(&mut ());
118            s.as_mut().start_send(2, ());
119            // First call: sends item 1, then poll_ready returns Pending.
120            let step = s.as_mut().poll_finalize(&mut ());
121            assert!(step.is_pending());
122            // Second call: resumes, sends remaining items.
123            let step = s.as_mut().poll_finalize(&mut ());
124            assert!(step.is_done());
125        }
126        assert_eq!(tp.items(), vec![1, 2, 3]);
127    }
