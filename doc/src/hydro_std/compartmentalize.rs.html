10pub trait PartitionStream<'a, T, C1, C2, Order> {
11    fn send_partitioned<F: Fn((MemberId<C2>, T)) -> (MemberId<C2>, T) + 'a>(
12        self,
13        other: &Cluster<'a, C2>,
14        dist_policy: impl IntoQuotedMut<'a, F, Cluster<'a, C1>>,
15    ) -> Stream<T, Cluster<'a, C2>, Unbounded, NoOrder>
16    where
17        T: Clone + Serialize + DeserializeOwned;
18}
19
20impl<'a, T, C1, C2, Order: Ordering> PartitionStream<'a, T, C1, C2, Order>
21    for Stream<(MemberId<C2>, T), Cluster<'a, C1>, Unbounded, Order>
22{
23    fn send_partitioned<F: Fn((MemberId<C2>, T)) -> (MemberId<C2>, T) + 'a>(
24        self,
25        other: &Cluster<'a, C2>,
26        dist_policy: impl IntoQuotedMut<'a, F, Cluster<'a, C1>>,
27    ) -> Stream<T, Cluster<'a, C2>, Unbounded, NoOrder>
28    where
29        T: Clone + Serialize + DeserializeOwned,
30    {
31        self.map(dist_policy).demux_bincode(other).values()
32    }
33}

35pub trait DecoupleClusterStream<'a, T, C1, B, Order: Ordering> {
36    fn decouple_cluster<C2: 'a>(
37        self,
38        other: &Cluster<'a, C2>,
39    ) -> Stream<T, Cluster<'a, C2>, Unbounded, Order>
40    where
41        T: Clone + Serialize + DeserializeOwned;
42}
43
44impl<'a, T, C1, B: Boundedness, Order: Ordering> DecoupleClusterStream<'a, T, C1, B, Order>
45    for Stream<T, Cluster<'a, C1>, B, Order>
46{
47    fn decouple_cluster<C2: 'a>(
48        self,
49        other: &Cluster<'a, C2>,
50    ) -> Stream<T, Cluster<'a, C2>, Unbounded, Order>
51    where
52        T: Clone + Serialize + DeserializeOwned,
53    {
54        let sent = self
55            .map(q!(move |b| (
56                MemberId::from_raw(CLUSTER_SELF_ID.raw_id),
57                b.clone()
58            )))
59            .demux_bincode(other)
60            .values();
61
62        sent.assume_ordering(
63            nondet!(/** this is safe because we are only receiving from one sender */),
64        )
65    }
66}
