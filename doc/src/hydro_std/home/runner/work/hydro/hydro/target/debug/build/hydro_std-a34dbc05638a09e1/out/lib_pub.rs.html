10    /// Benchmarks transactional workloads by concurrently submitting workloads
11    /// (up to `num_clients_per_node` per machine), measuring the latency
12    /// of each transaction and throughput over the entire workload.
13    ///
14    /// # Safety
15    /// This function uses non-deterministic time-based samples, and also updates results
16    /// at non-deterministic points in time.
17    pub unsafe fn bench_client<'a, Client>(
18        clients: &Cluster<'a, Client>,
19        transaction_cycle: impl FnOnce(
20            Stream<(u32, u32), Cluster<'a, Client>, Unbounded>,
21        ) -> Stream<(u32, u32), Cluster<'a, Client>, Unbounded, NoOrder>,
22        num_clients_per_node: usize,
23    ) -> BenchResult<'a, Client> {
24        let client_tick = clients.tick();
25        let start_this_tick = client_tick.optional_first_tick(q!(()));
26        let c_new_payloads_on_start = start_this_tick
27            .clone()
28            .flat_map_ordered(
29                q!(
30                    move | _ | (0..num_clients_per_node).map(move | i | ((CLUSTER_SELF_ID
31                    .raw_id * (num_clients_per_node as u32)) + i as u32, 0))
32                ),
33            );
34        let (c_to_proposers_complete_cycle, c_to_proposers) = clients
35            .forward_ref::<Stream<_, _, _, TotalOrder>>();
36        let c_received_quorum_payloads = unsafe {
37            transaction_cycle(c_to_proposers).tick_batch(&client_tick)
38        };
39        let c_new_payloads_when_committed = c_received_quorum_payloads
40            .clone()
41            .map(q!(| payload | (payload.0, payload.1 + 1)));
42        c_to_proposers_complete_cycle
43            .complete(
44                c_new_payloads_on_start
45                    .chain(unsafe {
46                        c_new_payloads_when_committed.assume_ordering::<TotalOrder>()
47                    })
48                    .all_ticks(),
49            );
50        let (c_timers_complete_cycle, c_timers) = client_tick
51            .cycle::<Stream<(usize, Instant), _, _, NoOrder>>();
52        let c_new_timers_when_leader_elected = start_this_tick
53            .map(q!(| _ | Instant::now()))
54            .flat_map_ordered(
55                q!(
56                    move | now | (0..num_clients_per_node).map(move | virtual_id |
57                    (virtual_id, now))
58                ),
59            );
60        let c_updated_timers = c_received_quorum_payloads
61            .clone()
62            .map(q!(| (key, _prev_count) | (key as usize, Instant::now())));
63        let c_new_timers = c_timers
64            .clone()
65            .chain(c_new_timers_when_leader_elected)
66            .chain(c_updated_timers.clone())
67            .reduce_keyed_commutative(
68                q!(
69                    | curr_time, new_time | { if new_time > * curr_time { * curr_time =
70                    new_time; } }
71                ),
72            );
73        c_timers_complete_cycle.complete_next_tick(c_new_timers);
74        let c_stats_output_timer = unsafe {
75            clients.source_interval(q!(Duration::from_secs(1))).tick_batch(&client_tick)
76        }
77            .first();
78        let c_latencies = c_timers
79            .join(c_updated_timers)
80            .map(
81                q!(
82                    | (_virtual_id, (prev_time, curr_time)) | curr_time
83                    .duration_since(prev_time)
84                ),
85            )
86            .all_ticks()
87            .fold_commutative(
88                q!(move || Rc::new(RefCell::new(Histogram::< u64 >::new(3).unwrap()))),
89                q!(
90                    move | latencies, latency | { latencies.borrow_mut().record(latency
91                    .as_nanos() as u64).unwrap(); }
92                ),
93            );
94        let c_throughput_new_batch = c_received_quorum_payloads
95            .clone()
96            .count()
97            .continue_unless(c_stats_output_timer.clone())
98            .map(q!(| batch_size | (batch_size, false)));
99        let c_throughput_reset = c_stats_output_timer
100            .clone()
101            .map(q!(| _ | (0, true)))
102            .defer_tick();
103        let c_throughput = c_throughput_new_batch
104            .union(c_throughput_reset)
105            .all_ticks()
106            .fold(
107                q!(|| (0, { stats_ci::mean::Arithmetic::new() })),
108                q!(
109                    | (total, stats), (batch_size, reset) | { if reset { if * total > 0 {
110                    stats.extend(& [* total as f64]).unwrap(); } * total = 0; } else { *
111                    total += batch_size; } }
112                ),
113            )
114            .map(q!(| (_, stats) | { stats }));
115        BenchResult {
116            latency_histogram: c_latencies,
117            throughput: c_throughput,
118        }
119    }
120    /// Prints transaction latency and throughput results to stdout,
121    /// with percentiles for latency and a confidence interval for throughput.
122    pub fn print_bench_results<Client>(results: BenchResult<Client>) {
123        unsafe {
124            results.latency_histogram.sample_every(q!(Duration::from_millis(1000)))
125        }
126            .for_each(
127                q!(
128                    move | latencies | { let latencies = latencies.borrow();
129                    println!("Latency p50: {:.3} | p99 {:.3} ms | p999 {:.3} ms ({:} samples)",
130                    Duration::from_nanos(latencies.value_at_quantile(0.5)).as_micros() as
131                    f64 / 1000.0, Duration::from_nanos(latencies.value_at_quantile(0.99))
132                    .as_micros() as f64 / 1000.0, Duration::from_nanos(latencies
133                    .value_at_quantile(0.999)).as_micros() as f64 / 1000.0, latencies
134                    .len()); }
135                ),
136            );
137        unsafe { results.throughput.sample_every(q!(Duration::from_millis(1000))) }
138            .for_each(
139                q!(
140                    move | throughputs | { let confidence = Confidence::new(0.99); if
141                    throughputs.sample_count() >= 2 { if let Ok(interval) = throughputs
142                    .ci_mean(confidence) { if let Some(lower) = interval.left() { if let
143                    Some(upper) = interval.right() {
144                    println!("Throughput 99% interval: {:.2} - {:.2} requests/s", lower,
145                    upper); } } } } }
146                ),
147            );
148    }

256    #[expect(clippy::type_complexity, reason = "stream types with ordering")]
257    pub fn collect_quorum_with_response<
258        'a,
259        L: Location<'a> + NoTick,
260        Order,
261        K: Clone + Eq + Hash,
262        V: Clone,
263        E: Clone,
264    >(
265        responses: Stream<(K, Result<V, E>), Atomic<L>, Unbounded, Order>,
266        min: usize,
267        max: usize,
268    ) -> (
269        Stream<(K, V), Atomic<L>, Unbounded, Order>,
270        Stream<(K, E), Atomic<L>, Unbounded, Order>,
271    ) {
272        let tick = responses.atomic_source();
273        let (not_all_complete_cycle, not_all) = tick.cycle::<Stream<_, _, _, Order>>();
274        let current_responses = not_all.chain(unsafe { responses.clone().tick_batch() });
275        let count_per_key = current_responses
276            .clone()
277            .fold_keyed_commutative(
278                q!(move || (0, 0)),
279                q!(
280                    move | accum, value | { if value.is_ok() { accum.0 += 1; } else {
281                    accum.1 += 1; } }
282                ),
283            );
284        let not_reached_min_count = count_per_key
285            .clone()
286            .filter_map(
287                q!(
288                    move | (key, (success, _error)) | if success < min { Some(key) } else
289                    { None }
290                ),
291            );
292        let reached_min_count = count_per_key
293            .clone()
294            .filter_map(
295                q!(
296                    move | (key, (success, _error)) | if success >= min { Some(key) }
297                    else { None }
298                ),
299            );
300        let just_reached_quorum = if max == min {
301            not_all_complete_cycle
302                .complete_next_tick(
303                    current_responses.clone().anti_join(reached_min_count),
304                );
305            current_responses.anti_join(not_reached_min_count)
306        } else {
307            let (min_but_not_max_complete_cycle, min_but_not_max) = tick.cycle();
308            let received_from_all = count_per_key
309                .filter_map(
310                    q!(
311                        move | (key, (success, error)) | if (success + error) >= max {
312                        Some(key) } else { None }
313                    ),
314                );
315            min_but_not_max_complete_cycle
316                .complete_next_tick(
317                    reached_min_count.filter_not_in(received_from_all.clone()),
318                );
319            not_all_complete_cycle
320                .complete_next_tick(
321                    current_responses.clone().anti_join(received_from_all),
322                );
323            current_responses.anti_join(not_reached_min_count).anti_join(min_but_not_max)
324        };
325        (
326            just_reached_quorum
327                .filter_map(
328                    q!(
329                        move | (key, res) | match res { Ok(v) => Some((key, v)), Err(_)
330                        => None, }
331                    ),
332                )
333                .all_ticks_atomic(),
334            responses
335                .filter_map(
336                    q!(
337                        move | (key, res) | match res { Ok(_) => None, Err(e) =>
338                        Some((key, e)), }
339                    ),
340                ),
341        )
342    }

417    type JoinResponses<K, M, V, L> = Stream<(K, (M, V)), Atomic<L>, Unbounded, NoOrder>;
418    /// Given an incoming stream of request-response responses, joins with metadata generated
419    /// at request time that is stored in-memory.
420    ///
421    /// The metadata must be generated in the same or a previous tick than the response,
422    /// typically at request time. Only one response element should be produced with a given
423    /// key, same for the metadata stream.
424    pub fn join_responses<
425        'a,
426        K: Clone + Eq + Hash,
427        M: Clone,
428        V: Clone,
429        L: Location<'a> + NoTick,
430    >(
431        tick: &Tick<L>,
432        responses: Stream<(K, V), Atomic<L>, Unbounded, NoOrder>,
433        metadata: Stream<(K, M), Tick<L>, Bounded, NoOrder>,
434    ) -> JoinResponses<K, M, V, L> {
435        let (remaining_to_join_complete_cycle, remaining_to_join) = tick
436            .cycle::<Stream<_, _, _, NoOrder>>();
437        let remaining_and_new: Stream<(K, M), Tick<L>, Bounded, _> = remaining_to_join
438            .chain(metadata);
439        let responses = unsafe { responses.tick_batch() };
440        let joined_this_tick = remaining_and_new
441            .clone()
442            .join(responses.clone())
443            .map(q!(| (key, (meta, resp)) | (key, (meta, resp))));
444        remaining_to_join_complete_cycle
445            .complete_next_tick(
446                remaining_and_new.anti_join(responses.map(q!(| (key, _) | key))),
447            );
448        joined_this_tick.all_ticks_atomic()
449    }
