7#[expect(clippy::type_complexity, reason = "stream types with ordering")]
8pub fn collect_quorum_with_response<
9    'a,
10    L: Location<'a> + NoTick,
11    Order: Ordering,
12    K: Clone + Eq + Hash,
13    V: Clone,
14    E: Clone,
15>(
16    responses: Stream<(K, Result<V, E>), Atomic<L>, Unbounded, Order>,
17    min: usize,
18    max: usize,
19) -> (
20    Stream<(K, V), Atomic<L>, Unbounded, Order>,
21    Stream<(K, E), Atomic<L>, Unbounded, Order>,
22) {
23    let tick = responses.atomic_source();
24    let (not_all_complete_cycle, not_all) = tick.cycle::<Stream<_, _, _, Order>>();
25
26    let current_responses = not_all.chain(responses.clone().batch(nondet!(
27        /// We always persist values that have not reached quorum, so even
28        /// with arbitrary batching we always produce deterministic quorum results.
29    )));
30
31    let count_per_key = current_responses.clone().into_keyed().fold_commutative(
32        q!(move || (0, 0)),
33        q!(move |accum, value| {
34            if value.is_ok() {
35                accum.0 += 1;
36            } else {
37                accum.1 += 1;
38            }
39        }),
40    );
41
42    let not_reached_min_count = count_per_key
43        .clone()
44        .filter(q!(move |(success, _error)| success < &min))
45        .keys();
46
47    let reached_min_count = count_per_key
48        .clone()
49        .filter(q!(move |(success, _error)| success >= &min))
50        .keys();
51
52    let just_reached_quorum = if max == min {
53        not_all_complete_cycle
54            .complete_next_tick(current_responses.clone().anti_join(reached_min_count));
55
56        current_responses.anti_join(not_reached_min_count)
57    } else {
58        let (min_but_not_max_complete_cycle, min_but_not_max) =
59            tick.cycle::<Stream<K, Tick<L>, Bounded, NoOrder>>();
60
61        let received_from_all = count_per_key
62            .filter(q!(move |(success, error)| (success + error) >= max))
63            .keys();
64
65        min_but_not_max_complete_cycle
66            .complete_next_tick(reached_min_count.filter_not_in(received_from_all.clone()));
67
68        not_all_complete_cycle
69            .complete_next_tick(current_responses.clone().anti_join(received_from_all));
70
71        current_responses
72            .anti_join(not_reached_min_count)
73            .anti_join(min_but_not_max)
74    };
75
76    (
77        just_reached_quorum
78            .filter_map(q!(move |(key, res)| match res {
79                Ok(v) => Some((key, v)),
80                Err(_) => None,
81            }))
82            .all_ticks_atomic(),
83        responses.filter_map(q!(move |(key, res)| match res {
84            Ok(_) => None,
85            Err(e) => Some((key, e)),
86        })),
87    )
88}

90#[expect(clippy::type_complexity, reason = "stream types with ordering")]
91pub fn collect_quorum<
92    'a,
93    L: Location<'a> + NoTick,
94    Order: Ordering,
95    K: Clone + Eq + Hash,
96    E: Clone,
97>(
98    responses: Stream<(K, Result<(), E>), Atomic<L>, Unbounded, Order>,
99    min: usize,
100    max: usize,
101) -> (
102    Stream<K, Atomic<L>, Unbounded, NoOrder>,
103    Stream<(K, E), Atomic<L>, Unbounded, Order>,
104) {
105    let tick = responses.atomic_source();
106    let (not_all_complete_cycle, not_all) = tick.cycle::<Stream<_, _, _, Order>>();
107
108    let current_responses = not_all.chain(responses.clone().batch(nondet!(
109        /// We always persist values that have not reached quorum, so even
110        /// with arbitrary batching we always produce deterministic quorum results.
111    )));
112
113    let count_per_key = current_responses.clone().into_keyed().fold_commutative(
114        q!(move || (0, 0)),
115        q!(move |accum, value| {
116            if value.is_ok() {
117                accum.0 += 1;
118            } else {
119                accum.1 += 1;
120            }
121        }),
122    );
123
124    let reached_min_count =
125        count_per_key
126            .clone()
127            .entries()
128            .filter_map(q!(move |(key, (success, _error))| if success >= min {
129                Some(key)
130            } else {
131                None
132            }));
133
134    let just_reached_quorum = if max == min {
135        not_all_complete_cycle.complete_next_tick(
136            current_responses
137                .clone()
138                .anti_join(reached_min_count.clone()),
139        );
140
141        reached_min_count
142    } else {
143        let (min_but_not_max_complete_cycle, min_but_not_max) = tick.cycle();
144
145        let received_from_all = count_per_key
146            .filter(q!(move |(success, error)| (success + error) >= max))
147            .keys();
148
149        min_but_not_max_complete_cycle.complete_next_tick(
150            reached_min_count
151                .clone()
152                .filter_not_in(received_from_all.clone()),
153        );
154
155        not_all_complete_cycle.complete_next_tick(current_responses.anti_join(received_from_all));
156
157        reached_min_count.filter_not_in(min_but_not_max)
158    };
159
160    (
161        just_reached_quorum.all_ticks_atomic(),
162        responses.filter_map(q!(move |(key, res)| match res {
163            Ok(_) => None,
164            Err(e) => Some((key, e)),
165        })),
166    )
167}
