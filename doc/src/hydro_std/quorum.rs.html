7#[expect(clippy::type_complexity, reason = "stream types with ordering")]
8pub fn collect_quorum_with_response<
9    'a,
10    L: Location<'a> + NoTick,
11    Order: Ordering,
12    K: Clone + Eq + Hash,
13    V: Clone,
14    E: Clone,
15>(
16    responses: Stream<(K, Result<V, E>), L, Unbounded, Order>,
17    min: usize,
18    max: usize,
19) -> (
20    Stream<(K, V), L, Unbounded, Order>,
21    Stream<(K, E), L, Unbounded, Order>,
22) {
23    let tick = responses.location().tick();
24    let (not_all_complete_cycle, not_all) = tick.cycle::<Stream<_, _, _, Order>>();
25
26    let current_responses = not_all.chain(responses.clone().batch(
27        &tick,
28        nondet!(
29            /// We always persist values that have not reached quorum, so even
30            /// with arbitrary batching we always produce deterministic quorum results.
31        ),
32    ));
33
34    let count_per_key = current_responses.clone().into_keyed().fold_commutative(
35        q!(move || (0, 0)),
36        q!(move |accum, value| {
37            if value.is_ok() {
38                accum.0 += 1;
39            } else {
40                accum.1 += 1;
41            }
42        }),
43    );
44
45    let not_reached_min_count = count_per_key
46        .clone()
47        .filter(q!(move |(success, _error)| success < &min))
48        .keys();
49
50    let reached_min_count = count_per_key
51        .clone()
52        .filter(q!(move |(success, _error)| success >= &min))
53        .keys();
54
55    let just_reached_quorum = if max == min {
56        not_all_complete_cycle
57            .complete_next_tick(current_responses.clone().anti_join(reached_min_count));
58
59        current_responses.anti_join(not_reached_min_count)
60    } else {
61        let (min_but_not_max_complete_cycle, min_but_not_max) =
62            tick.cycle::<Stream<K, _, Bounded, NoOrder>>();
63
64        let received_from_all = count_per_key
65            .filter(q!(move |(success, error)| (success + error) >= max))
66            .keys();
67
68        min_but_not_max_complete_cycle
69            .complete_next_tick(reached_min_count.filter_not_in(received_from_all.clone()));
70
71        not_all_complete_cycle
72            .complete_next_tick(current_responses.clone().anti_join(received_from_all));
73
74        current_responses
75            .anti_join(not_reached_min_count)
76            .anti_join(min_but_not_max)
77    };
78
79    (
80        just_reached_quorum
81            .filter_map(q!(move |(key, res)| match res {
82                Ok(v) => Some((key, v)),
83                Err(_) => None,
84            }))
85            .all_ticks(),
86        responses.filter_map(q!(move |(key, res)| match res {
87            Ok(_) => None,
88            Err(e) => Some((key, e)),
89        })),
90    )
91}

93#[expect(clippy::type_complexity, reason = "stream types with ordering")]
94pub fn collect_quorum<
95    'a,
96    L: Location<'a> + NoTick,
97    Order: Ordering,
98    K: Clone + Eq + Hash,
99    E: Clone,
100>(
101    responses: Stream<(K, Result<(), E>), L, Unbounded, Order>,
102    min: usize,
103    max: usize,
104) -> (
105    Stream<K, L, Unbounded, NoOrder>,
106    Stream<(K, E), L, Unbounded, Order>,
107) {
108    let tick = responses.location().tick();
109    let (not_all_complete_cycle, not_all) = tick.cycle::<Stream<_, _, _, Order>>();
110
111    let current_responses = not_all.chain(responses.clone().batch(
112        &tick,
113        nondet!(
114            /// We always persist values that have not reached quorum, so even
115            /// with arbitrary batching we always produce deterministic quorum results.
116        ),
117    ));
118
119    let count_per_key = current_responses.clone().into_keyed().fold_commutative(
120        q!(move || (0, 0)),
121        q!(move |accum, value| {
122            if value.is_ok() {
123                accum.0 += 1;
124            } else {
125                accum.1 += 1;
126            }
127        }),
128    );
129
130    let reached_min_count =
131        count_per_key
132            .clone()
133            .entries()
134            .filter_map(q!(move |(key, (success, _error))| if success >= min {
135                Some(key)
136            } else {
137                None
138            }));
139
140    let just_reached_quorum = if max == min {
141        not_all_complete_cycle.complete_next_tick(
142            current_responses
143                .clone()
144                .anti_join(reached_min_count.clone()),
145        );
146
147        reached_min_count
148    } else {
149        let (min_but_not_max_complete_cycle, min_but_not_max) = tick.cycle();
150
151        let received_from_all = count_per_key
152            .filter(q!(move |(success, error)| (success + error) >= max))
153            .keys();
154
155        min_but_not_max_complete_cycle.complete_next_tick(
156            reached_min_count
157                .clone()
158                .filter_not_in(received_from_all.clone()),
159        );
160
161        not_all_complete_cycle.complete_next_tick(current_responses.anti_join(received_from_all));
162
163        reached_min_count.filter_not_in(min_but_not_max)
164    };
165
166    (
167        just_reached_quorum.all_ticks(),
168        responses.filter_map(q!(move |(key, res)| match res {
169            Ok(_) => None,
170            Err(e) => Some((key, e)),
171        })),
172    )
173}

217
218        let (port, input) = node.source_external_bincode(&external);
219        let (success_port, error_port) = {

226
227        let compiled_sim = flow.sim().compiled();

322    fn collect_quorum_min_equals_max() {
323        let flow = FlowBuilder::new();
324        let external = flow.external::<()>();
325        let node = flow.process::<()>();
326
327        let (port, input) = node.source_external_bincode::<_, _, TotalOrder, _>(&external);
328        let success_port = collect_quorum(input, 2, 2)
329            .0
330            .send_bincode_external(&external);
331
332        flow.sim().exhaustive(async |mut compiled| {
333            let in_send = compiled.connect(&port);
334            let success_recv = compiled.connect(&success_port);
335            compiled.launch();
336
337            // When min == max, we need exactly that many responses
338            in_send.send((1, Ok::<(), ()>(()))).unwrap();
339            in_send.send((1, Ok(()))).unwrap();
340
341            // This key gets exactly 2 responses (1 success, 1 error) - should not reach quorum
342            in_send.send((2, Ok(()))).unwrap();
343            in_send.send((2, Err(()))).unwrap();
344
345            // This key gets 2 successes - should reach quorum
346            in_send.send((3, Ok(()))).unwrap();
347            in_send.send((3, Ok(()))).unwrap();
348
349            // Only keys 1 and 3 should reach quorum (both have 2 successes)
350            success_recv.assert_yields_only_unordered([1, 3]).await;
351        });
352    }

415            let success_recv = compiled.connect(&success_port);

419            in_send.send((1, Ok::<(), ()>(()))).unwrap();
420            in_send.send((1, Ok(()))).unwrap();

433            // additional responses after reaching quorum
