13pub struct SerializableHistogramWrapper {
14    pub histogram: Rc<RefCell<Histogram<u64>>>,
15}
16
17impl Serialize for SerializableHistogramWrapper {
18    fn serialize<S: serde::Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
19        let mut vec = Vec::new();
20        V2Serializer::new()
21            .serialize(&self.histogram.borrow(), &mut vec)
22            .unwrap();
23        serializer.serialize_bytes(&vec)
24    }
25}
26impl<'a> Deserialize<'a> for SerializableHistogramWrapper {
27    fn deserialize<D: serde::Deserializer<'a>>(deserializer: D) -> Result<Self, D::Error> {
28        let mut bytes: &[u8] = Deserialize::deserialize(deserializer)?;
29        let mut histogram = Deserializer::new().deserialize(&mut bytes).unwrap();
30        // Allow auto-resizing to prevent error when combining
31        histogram.auto(true);
32        Ok(SerializableHistogramWrapper {
33            histogram: Rc::new(RefCell::new(histogram)),
34        })
35    }
36}

44/// Benchmarks transactional workloads by concurrently submitting workloads
45/// (up to `num_clients_per_node` per machine)
46/// * `workload_generator`: Converts previous output (or None, for new virtual clients) into the next input payload
47/// * `protocol`: The protocol to benchmark
48///
49/// ## Returns
50/// A stream of latencies per completed client request
51pub fn bench_client<'a, Client, Input, Output>(
52    clients: &Cluster<'a, Client>,
53    num_clients_per_node: usize,
54    workload_generator: impl FnOnce(
55        KeyedStream<u32, Option<Output>, Cluster<'a, Client>, Unbounded, NoOrder>,
56    )
57        -> KeyedStream<u32, Input, Cluster<'a, Client>, Unbounded, NoOrder>,
58    protocol: impl FnOnce(
59        KeyedStream<u32, Input, Cluster<'a, Client>, Unbounded, NoOrder>,
60    ) -> KeyedStream<u32, Output, Cluster<'a, Client>, Unbounded, NoOrder>,
61) -> KeyedStream<u32, (Output, Duration), Cluster<'a, Client>, Unbounded, NoOrder>
62where
63    Input: Clone,
64    Output: Clone,
65{
66    let dummy = clients.singleton(q!(0));
67    let new_payload_ids = sliced! {
68        let _dummy_batched = use(dummy, nondet!(/** temp */));
69        let mut next_virtual_client = use::state(|l| Optional::from(l.singleton(q!((0u32, None)))));
70
71        // Set up virtual clients - spawn new ones each tick until we reach the limit
72        let new_virtual_client = next_virtual_client.clone();
73        next_virtual_client = new_virtual_client.clone().filter_map(
74            q!(move |(virtual_id, _)| {
75                if virtual_id < num_clients_per_node as u32 {
76                    Some((virtual_id + 1, None))
77                } else {
78                    None
79                }
80            }),
81        );
82
83        new_virtual_client.into_stream().into_keyed()
84    };
85
86    let (protocol_outputs_complete, protocol_outputs) =
87        clients.forward_ref::<KeyedStream<u32, Output, Cluster<'a, Client>, Unbounded, NoOrder>>();
88    // Use new payload IDS and previous outputs to generate new payloads
89    let protocol_inputs = workload_generator(
90        new_payload_ids.interleave(protocol_outputs.map(q!(|payload| Some(payload)))),
91    );
92    // Feed new payloads to the protocol
93    let protocol_outputs = protocol(protocol_inputs.clone());
94    protocol_outputs_complete.complete(protocol_outputs.clone());
95
96    // Persist start latency, overwrite on new value. Memory footprint = O(num_clients_per_node)
97    let start_times = protocol_inputs
98        .reduce(q!(
99            |curr, new| {
100                *curr = new;
101            },
102            commutative = ManualProof(/* The value will be thrown away */)
103        ))
104        .map(q!(|_input| SystemTime::now()));
105
106    sliced! {
107        let start_times = use(start_times, nondet!(/** Only one in-flight message per virtual client at any time, and outputs happen-after inputs, so if an output is received the start_times must contain its input time. */));
108        let current_outputs = use(protocol_outputs, nondet!(/** Batching is required to compare output to input time, but does not actually affect the result. */));
109
110        let end_times_and_output = current_outputs
111            .assume_ordering(nondet!(/** Only one in-flight message per virtual client at any time, and they are causally dependent, so this just casts to KeyedSingleton */))
112            .reduce(
113                q!(
114                    |curr, new| {
115                        *curr = new;
116                    },
117                ),
118            )
119            .map(q!(|output| (SystemTime::now(), output)));
120
121        start_times
122            .defer_tick() // Get the start_times before they were overwritten with the newly generated input
123            .join_keyed_singleton(end_times_and_output)
124            .map(q!(|(start_time, (end_time, output))| (output, end_time.duration_since(start_time).unwrap())))
125            .into_keyed_stream()
126            .weaken_ordering()
127    }
128}

130/// Computes the throughput and latency of transactions and outputs it every `interval_millis`.
131/// An output is produced even if there are no transactions.
132///
133/// # Non-Determinism
134/// This function uses non-deterministic wall-clock windows for measuring throughput.
135pub fn compute_throughput_latency<'a, Client: 'a>(
136    clients: &Cluster<'a, Client>,
137    latencies: Stream<Duration, Cluster<'a, Client>, Unbounded, NoOrder>,
138    interval_millis: u64,
139    nondet_measurement_window: NonDet,
140) -> BenchResult<Cluster<'a, Client>> {
141    let punctuation = clients.source_interval(
142        q!(Duration::from_millis(interval_millis)),
143        nondet_measurement_window,
144    );
145
146    let (interval_throughput, interval_latency) = sliced! {
147        let punctuation = use(punctuation, nondet_measurement_window);
148        let latencies = use(latencies, nondet_measurement_window);
149        let mut latency_histogram = use::state(|l| l.singleton(q!(Rc::new(RefCell::new(Histogram::<u64>::new(3).unwrap())))));
150        let mut throughput = use::state(|l| l.singleton(q!(0usize)));
151
152        let punctuation_option = punctuation.first();
153        let batched_latency_histogram = latencies.clone().fold(
154            q!(move || Histogram::<u64>::new(3).unwrap()),
155            q!(move |latencies, latency| {
156                    latencies
157                        .record(latency.as_nanos() as u64)
158                        .unwrap();
159                },
160                commutative = ManualProof(
161                    /* adding elements to histogram is commutative */
162                )
163            ),
164        );
165
166        // Output every punctuation
167        let interval_throughput = throughput.clone().filter_if_some(punctuation_option.clone());
168        let interval_latency = latency_histogram.clone().filter_if_some(punctuation_option.clone());
169
170        let batched_throughput = latencies.count();
171        // Clear every punctuation
172        let prev_throughput = throughput.filter_if_none(punctuation_option.clone());
173        // Merge new values
174        throughput = batched_throughput
175            .clone()
176            .zip(prev_throughput.clone())
177            .map(q!(|(new, old)| new + old))
178            .unwrap_or(batched_throughput.clone());
179
180        // Clear every punctuation
181        let prev_histogram = latency_histogram.filter_if_none(punctuation_option);
182        // Merge new values
183        latency_histogram = batched_latency_histogram
184            .clone()
185            .zip(prev_histogram.clone())
186            .map(q!(|(new, old)| {
187                old.borrow_mut().add(new).expect("Error adding value to histogram");
188                old
189            }))
190            .unwrap_or(batched_latency_histogram.map(q!(|histogram| Rc::new(RefCell::new(histogram)))));
191
192        (interval_throughput.into_stream(), interval_latency.into_stream())
193    };
194
195    BenchResult {
196        latency_histogram: interval_latency,
197        throughput: interval_throughput,
198    }
199}

201/// Returns transaction throughput and latency results.
202/// Aggregates results from clients and outputs every `output_interval_millis`.
203///
204/// Note: Inconsistent windowing may result in unexpected outputs unless `output_interval_millis` >> `interval_millis`.
205pub fn aggregate_bench_results<'a, Client: 'a, Aggregator>(
206    results: BenchResult<Cluster<'a, Client>>,
207    aggregator: &Process<'a, Aggregator>,
208    output_interval_millis: u64,
209) -> BenchResult<Process<'a, Aggregator>> {
210    let nondet_sampling = nondet!(/** non-deterministic samping only affects logging */);
211    let punctuation = aggregator.source_interval(
212        q!(Duration::from_millis(output_interval_millis)),
213        nondet_sampling,
214    );
215
216    let a_throughputs = results
217        .throughput
218        .send(aggregator, TCP.fail_stop().bincode())
219        .values();
220
221    let a_latencies = results
222        .latency_histogram
223        .map(q!(|latencies| {
224            SerializableHistogramWrapper {
225                histogram: latencies,
226            }
227        }))
228        .send(aggregator, TCP.fail_stop().bincode())
229        .values()
230        .map(q!(|wrapper| wrapper.histogram));
231
232    let (combined_throughputs, combined_latencies) = sliced! {
233        let punctuation = use(punctuation, nondet_sampling);
234        let a_throughputs = use(a_throughputs, nondet_sampling);
235        let a_latencies = use(a_latencies, nondet_sampling);
236        let mut latency_histogram = use::state(|l| l.singleton(q!(Rc::new(RefCell::new(Histogram::<u64>::new(3).unwrap())))));
237        let mut throughput = use::state(|l| l.singleton(q!(0usize)));
238
239        let punctuation_option = punctuation.first();
240
241        // Output every punctuation
242        let interval_throughput = throughput.clone().filter_if_some(punctuation_option.clone());
243        let interval_latency = latency_histogram.clone().filter_if_some(punctuation_option.clone());
244
245        // Clear every punctuation
246        let prev_throughput = throughput.filter_if_none(punctuation_option.clone()).into_stream();
247        // Merge new values
248        throughput = a_throughputs
249            .chain(prev_throughput)
250            .fold(q!(|| 0usize), q!(|curr, new| {
251                    *curr += new;
252                },
253                commutative = ManualProof(/* Addition is commutative */)
254            ));
255
256        // Merge new values
257        let merged_new_histograms = a_latencies
258            .reduce(
259                q!(|curr, new| {
260                    curr.borrow_mut().add(&*new.borrow_mut()).expect("Error adding value to histogram");
261                },
262                commutative = ManualProof(/* Merge is commutative */)
263            ));
264        // Clear every punctuation
265        latency_histogram = latency_histogram
266            .zip(merged_new_histograms.into_singleton())
267            .zip(punctuation_option.into_singleton())
268            .map(q!(|((old, new), reset)| {
269                if reset.is_some() {
270                    // Use replace instead of clear, since interval_latency is pointing to the Histogram too
271                    old.replace(Histogram::<u64>::new(3).unwrap());
272                }
273                if let Some(new) = new {
274                    old.borrow_mut().add(&*new.borrow_mut()).expect("Error adding value to histogram");
275                }
276                old
277            }));
278
279        (interval_throughput.into_stream(), interval_latency.into_stream())
280    };
281
282    BenchResult {
283        throughput: combined_throughputs,
284        latency_histogram: combined_latencies,
285    }
286}

288/// Pretty prints output of `aggregate_bench_results`.
289///
290/// Prints the throughput, and the 50th, 99th, and 99.9th percentile latencies.
291pub fn pretty_print_bench_results<'a, Aggregator>(
292    aggregate_results: BenchResult<Process<'a, Aggregator>>,
293) {
294    aggregate_results.throughput.for_each(q!(|throughput| {
295        println!("Throughput: {:.2} requests/s", throughput);
296    }));
297    aggregate_results
298        .latency_histogram
299        .map(q!(move |latencies| (
300            // Convert to milliseconds but include floating point (as_millis is for whole numbers only)
301            Duration::from_nanos(latencies.borrow().value_at_quantile(0.5)).as_micros() as f64
302                / 1000.0,
303            Duration::from_nanos(latencies.borrow().value_at_quantile(0.99)).as_micros() as f64
304                / 1000.0,
305            Duration::from_nanos(latencies.borrow().value_at_quantile(0.999)).as_micros() as f64
306                / 1000.0,
307            latencies.borrow().len(),
308        )))
309        .for_each(q!(move |(p50, p99, p999, num_samples)| {
310            println!(
311                "Latency p50: {:.3} | p99 {:.3} | p999 {:.3} ms ({:} samples)",
312                p50, p99, p999, num_samples
313            );
314        }));
