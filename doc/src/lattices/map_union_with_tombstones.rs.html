1//! Module containing the [`MapUnionWithTombstones`] lattice and aliases for different datastructures.
2//!
3//! # Choosing a Tombstone Implementation
4//!
5//! This module provides several specialized tombstone storage implementations optimized for different key types:
6//!
7//! ## For Integer Keys (u64)
8//! Use [`MapUnionWithTombstonesRoaring`] with [`crate::tombstone::RoaringTombstoneSet`]:
9//! - Extremely space-efficient bitmap compression
10//! - Fast O(1) lookups and efficient bitmap OR operations during merge
11//! - Works with u64 keys (other integer types can be cast to u64)
12//! - Example: `MapUnionWithTombstonesRoaring::new_from(HashMap::from([(1u64, value)]), RoaringTombstoneSet::new())`
13//!
14//! ## For String Keys
15//! Use [`MapUnionWithTombstonesFstString`] with [`crate::tombstone::FstTombstoneSet<String>`]:
16//! - Compressed finite state transducer storage
17//! - Zero false positives (collision-free)
18//! - Efficient union operations for merging
19//! - Maintains sorted order
20//! - Example: `MapUnionWithTombstonesFstString::new_from(HashMap::from([("key".to_string(), value)]), FstTombstoneSet::new())`
21
22//! ## For Other Types
23//! Use the generic [`MapUnionWithTombstones`] with [`HashSet`] for tombstones:
24//! - Works with any `Hash + Eq` key type
25//! - No compression, but simple and flexible
26//! - Example: `MapUnionHashMapWithTombstoneHashSet::new_from([(custom_key, value)], [])`
27//!
28//! Alternatively, you can hash to 64-bit integers and follow instructions for that case above,
29//! understanding there's a tiny risk of hash collision which could result in keys being
30//! tombstoned (deleted) incorrectly.
31//!
32//! ## Performance Characteristics
33//!
34//! | Implementation | Space Efficiency | Merge Speed | Lookup Speed | False Positives |
35//! |----------------|------------------|-------------|--------------|-----------------|
36//! | RoaringBitmap  | Excellent        | Excellent   | Excellent    | None            |
37//! | FST            | Very Good        | Good        | Very Good    | None            |
38//! | HashSet        | Poor             | Good        | Excellent    | None            |

51/// Map-union-with-tombstones compound lattice.
52///
53/// When a key is deleted from the map-union-with-tombstones lattice, it is removed from the underlying `map` and placed into
54/// the `tombstones` set.
55///
56/// This forms the first invariant for this data structure. A key should appear either nowhere, in `map` or in `tombstones`.
57/// but never in `map` and `tombstones` at the same time.
58///
59/// merging is done by merging the underlying `map` and then merging the `tombstones` set, then doing `map` = `map` - `tombstones`.
60///
61/// The implementation of `tombstones` can be any set-like thing. This allows a user to plug in their own set-like implementation.
62/// For example, if the user knows that keys will be created and deleted strictly sequentially, then they could create a highly optimized set implementation
63/// which would just be a single integer, correpsonding to the current key value that the set is up to. Queries for keys below that integer would return true,
64/// queries for keys above it would return false.
65#[derive(Copy, Clone, Debug, Default)]
66#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
67pub struct MapUnionWithTombstones<Map, TombstoneSet> {
68    map: Map,
69    tombstones: TombstoneSet,
70}

72impl<Map, TombstoneSet> MapUnionWithTombstones<Map, TombstoneSet> {
73    /// Create a new `MapUnionWithTombstones` from a `Map` and a `TombstoneSet`.
74    pub fn new(map: Map, tombstones: TombstoneSet) -> Self {
75        Self { map, tombstones }
76    }
77
78    /// Create a new `MapUnionWithTombstones` from an `Into<Map>` and an `Into<TombstoneSet>`.
79    pub fn new_from(map: impl Into<Map>, tombstones: impl Into<TombstoneSet>) -> Self {
80        Self::new(map.into(), tombstones.into())
81    }
82
83    /// Reveal the inner value as a shared reference.
84    pub fn as_reveal_ref(&self) -> (&Map, &TombstoneSet) {
85        (&self.map, &self.tombstones)
86    }
87
88    /// Reveal the inner value as an exclusive reference.
89    pub fn as_reveal_mut(&mut self) -> (&mut Map, &mut TombstoneSet) {
90        (&mut self.map, &mut self.tombstones)
91    }
92
93    /// Gets the inner by value, consuming self.
94    pub fn into_reveal(self) -> (Map, TombstoneSet) {
95        (self.map, self.tombstones)
96    }
97}

99impl<MapSelf, MapOther, K, ValSelf, ValOther, TombstoneSetSelf, TombstoneSetOther>
100    Merge<MapUnionWithTombstones<MapOther, TombstoneSetOther>>
101    for MapUnionWithTombstones<MapSelf, TombstoneSetSelf>
102where
103    MapSelf: Keyed<Key = K, Item = ValSelf>
104        + Extend<(K, ValSelf)>
105        + for<'a> GetMut<&'a K, Item = ValSelf>
106        + for<'b> Remove<&'b K>,
107    MapOther: IntoIterator<Item = (K, ValOther)>,
108    ValSelf: Merge<ValOther> + LatticeFrom<ValOther>,
109    ValOther: IsBot,
110    TombstoneSetSelf: Extend<K> + Len + for<'a> Get<&'a K> + Iter<Item = K>,
111    TombstoneSetOther: IntoIterator<Item = K>,
112{
113    fn merge(&mut self, other: MapUnionWithTombstones<MapOther, TombstoneSetOther>) -> bool {
114        let mut changed = false;
115        // This vec collect is needed to prevent simultaneous mut references `self.0.extend` and
116        // `self.0.get_mut`.
117        // TODO(mingwei): This could be fixed with a different structure, maybe some sort of
118        // `Collection` entry API.
119        let iter: Vec<_> = other
120            .map
121            .into_iter()
122            .filter(|(k_other, val_other)| {
123                !val_other.is_bot() && !self.tombstones.contains(k_other)
124            })
125            .filter_map(|(k_other, val_other)| {
126                match self.map.get_mut(&k_other) {
127                    // Key collision, merge into `self`.
128                    Some(mut val_self) => {
129                        changed |= val_self.merge(val_other);
130                        None
131                    }
132                    // New value, convert for extending.
133                    None => {
134                        changed = true;
135                        Some((k_other, ValSelf::lattice_from(val_other)))
136                    }
137                }
138            })
139            .collect();
140        self.map.extend(iter);
141
142        let old_self_tombstones_len = self.tombstones.len();
143
144        self.tombstones
145            .extend(other.tombstones.into_iter().inspect(|k| {
146                self.map.remove(k);
147            }));
148
149        if old_self_tombstones_len != self.tombstones.len() {
150            changed = true;
151        }
152
153        changed
154    }
155}

179impl<MapSelf, MapOther, K, ValSelf, ValOther, TombstoneSetSelf, TombstoneSetOther>
180    PartialOrd<MapUnionWithTombstones<MapOther, TombstoneSetOther>>
181    for MapUnionWithTombstones<MapSelf, TombstoneSetSelf>
182where
183    MapSelf: Map<K, ValSelf, Key = K, Item = ValSelf> + MapIter + SimpleKeyedRef,
184    MapOther: Map<K, ValOther, Key = K, Item = ValOther> + MapIter + SimpleKeyedRef,
185    ValSelf: PartialOrd<ValOther> + IsBot,
186    ValOther: IsBot,
187    TombstoneSetSelf: Len + Iter<Item = K> + for<'a> Get<&'a K>,
188    TombstoneSetOther: Len + Iter<Item = K> + for<'a> Get<&'a K>,
189{
190    fn partial_cmp(
191        &self,
192        other: &MapUnionWithTombstones<MapOther, TombstoneSetOther>,
193    ) -> Option<Ordering> {
194        let self_tombstones_greater = self
195            .tombstones
196            .iter()
197            .any(|k| !other.tombstones.contains(&*k));
198
199        let other_tombstones_greater = other
200            .tombstones
201            .iter()
202            .any(|k| !self.tombstones.contains(&*k));
203
204        if self_tombstones_greater && other_tombstones_greater {
205            return None;
206        }
207
208        let mut self_any_greater = false;
209        let mut other_any_greater = false;
210        let self_keys = self
211            .map
212            .iter()
213            .filter(|(k, v)| {
214                !v.is_bot() && !self.tombstones.contains(k) && !other.tombstones.contains(k)
215            })
216            .map(|(k, _v)| <MapSelf as SimpleKeyedRef>::into_ref(k));
217        let other_keys = other
218            .map
219            .iter()
220            .filter(|(k, v)| {
221                !v.is_bot() && !self.tombstones.contains(k) && !other.tombstones.contains(k)
222            })
223            .map(|(k, _v)| <MapOther as SimpleKeyedRef>::into_ref(k));
224
225        for k in self_keys.chain(other_keys) {
226            match (self.map.get(k), other.map.get(k)) {
227                (Some(self_value), Some(other_value)) => {
228                    match self_value.partial_cmp(&*other_value)? {
229                        Less => {
230                            other_any_greater = true;
231                        }
232                        Greater => {
233                            self_any_greater = true;
234                        }
235                        Equal => {}
236                    }
237                }
238                (Some(_), None) => {
239                    self_any_greater = true;
240                }
241                (None, Some(_)) => {
242                    other_any_greater = true;
243                }
244                (None, None) => unreachable!(),
245            }
246            if self_any_greater && other_any_greater {
247                return None;
248            }
249        }
250        match (
251            self_any_greater,
252            other_any_greater,
253            self_tombstones_greater,
254            other_tombstones_greater,
255        ) {
256            (false, false, false, false) => Some(Equal),
257
258            (false, false, true, false) => Some(Greater),
259            (false, false, false, true) => Some(Less),
260
261            (true, false, false, false) => Some(Greater),
262            (false, true, false, false) => Some(Less),
263
264            (true, false, true, false) => Some(Greater),
265            (false, true, false, true) => Some(Less),
266
267            (true, false, false, true) => None,
268            (false, true, true, false) => None,
269
270            (true, true, _, _) => unreachable!(),
271            (_, _, true, true) => unreachable!(),
272        }
273    }
274}

283impl<MapSelf, MapOther, K, ValSelf, ValOther, TombstoneSetSelf, TombstoneSetOther>
284    PartialEq<MapUnionWithTombstones<MapOther, TombstoneSetOther>>
285    for MapUnionWithTombstones<MapSelf, TombstoneSetSelf>
286where
287    MapSelf: Map<K, ValSelf, Key = K, Item = ValSelf> + MapIter + SimpleKeyedRef,
288    MapOther: Map<K, ValOther, Key = K, Item = ValOther> + MapIter + SimpleKeyedRef,
289    ValSelf: PartialEq<ValOther> + IsBot,
290    ValOther: IsBot,
291    TombstoneSetSelf: Len + Iter<Item = K> + for<'a> Get<&'a K>,
292    TombstoneSetOther: Len + Iter<Item = K> + for<'b> Get<&'b K>,
293{
294    fn eq(&self, other: &MapUnionWithTombstones<MapOther, TombstoneSetOther>) -> bool {
295        if self.tombstones.len() != other.tombstones.len() {
296            return false;
297        }
298
299        if self
300            .tombstones
301            .iter()
302            .any(|k| !other.tombstones.contains(&*k))
303        {
304            return false;
305        }
306
307        if other
308            .tombstones
309            .iter()
310            .any(|k| !self.tombstones.contains(&*k))
311        {
312            return false;
313        }
314
315        let self_keys = self
316            .map
317            .iter()
318            .filter(|(_k, v)| !v.is_bot())
319            .map(|(k, _v)| <MapSelf as SimpleKeyedRef>::into_ref(k));
320        let other_keys = other
321            .map
322            .iter()
323            .filter(|(_k, v)| !v.is_bot())
324            .map(|(k, _v)| <MapOther as SimpleKeyedRef>::into_ref(k));
325        for k in self_keys.chain(other_keys) {
326            match (self.map.get(k), other.map.get(k)) {
327                (Some(self_value), Some(other_value)) => {
328                    if *self_value != *other_value {
329                        return false;
330                    }
331                }
332                (None, None) => unreachable!(),
333                _ => {
334                    return false;
335                }
336            }
337        }
338
339        true
340    }

386// Specialized merge implementation for RoaringTombstoneSet with HashMap<u64, Val>
387// This is highly efficient because:
388// 1. We can OR the roaring bitmaps directly (very fast)
389// 2. We can use the bitmap's efficient contains() for lookups
390impl<Val, ValOther> Merge<MapUnionWithTombstones<HashMap<u64, ValOther>, RoaringTombstoneSet>>
391    for MapUnionWithTombstones<HashMap<u64, Val>, RoaringTombstoneSet>
392where
393    Val: Merge<ValOther> + LatticeFrom<ValOther>,
394    ValOther: IsBot,
395{
396    fn merge(
397        &mut self,
398        other: MapUnionWithTombstones<HashMap<u64, ValOther>, RoaringTombstoneSet>,
399    ) -> bool {
400        let mut changed = false;
401
402        // OR the roaring bitmaps together - O(n) where n is bitmap size, very fast!
403        let old_tombstones_len = self.tombstones.len();
404        self.tombstones.union_with(&other.tombstones);
405        if old_tombstones_len != self.tombstones.len() {
406            changed = true;
407        }
408
409        // Merge the maps, filtering out tombstoned keys
410        let iter: Vec<_> = other
411            .map
412            .into_iter()
413            .filter(|(k_other, val_other)| {
414                !val_other.is_bot() && !self.tombstones.contains(k_other)
415            })
416            .filter_map(|(k_other, val_other)| match self.map.get_mut(&k_other) {
417                Some(val_self) => {
418                    changed |= val_self.merge(val_other);
419                    None
420                }
421                None => {
422                    changed = true;
423                    Some((k_other, Val::lattice_from(val_other)))
424                }
425            })
426            .collect();
427        self.map.extend(iter);
428
429        // Remove any keys from the map that are now tombstoned
430        self.map.retain(|k, _| !self.tombstones.contains(k));
431
432        changed
433    }
434}
