1//! Shared locking and job directory utilities for parallel cargo builds in Hydro.
2//!
3//! This crate provides the coordination primitives for running multiple cargo
4//! builds concurrently against a shared target directory using per-job symlinked
5//! directories.

18/// Log a timestamped message to the build coordination log.
19pub fn log_build_event(project_dir: &Path, msg: &str) {
20    let log_path = project_dir.join("build-coordination.log");
21    let mut file = fs::OpenOptions::new()
22        .create(true)
23        .append(true)
24        .open(log_path)
25        .unwrap();
26    let now = std::time::SystemTime::now()
27        .duration_since(std::time::UNIX_EPOCH)
28        .unwrap()
29        .as_millis();
30    let thread_id = std::thread::current().id();
31    let _ = writeln!(file, "[{now}ms] [{thread_id:?}] {msg}");
32}

38/// Global prebuild serialization lock (in-process).
39static GLOBAL_PREBUILD_LOCK: parking_lot::RwLock<()> = parking_lot::RwLock::new(());
40
41/// Per-feature-hash locks for when `__CARGO_DEFAULT_LIB_METADATA` is set.
42static DEP_BUILD_LOCK: parking_lot::RwLock<()> = parking_lot::RwLock::new(());
43static DEP_BUILD_LOCKS_PER_HASH: LazyLock<
44    Mutex<HashMap<String, &'static parking_lot::RwLock<()>>>,
45> = LazyLock::new(|| Mutex::new(HashMap::new()));
46
47fn get_dep_lock(features_hash: Option<&str>) -> &'static parking_lot::RwLock<()> {
48    match features_hash {
49        None => &DEP_BUILD_LOCK,
50        Some(hash) => {
51            let mut map = DEP_BUILD_LOCKS_PER_HASH.lock().unwrap();
52            map.entry(hash.to_owned())
53                .or_insert_with(|| Box::leak(Box::new(parking_lot::RwLock::new(()))))
54        }
55    }
56}
57
58/// Per-job-dir mutexes for serializing job dir setup/population within a process.
59static JOB_DIR_LOCKS: LazyLock<Mutex<HashMap<PathBuf, &'static Mutex<()>>>> =
60    LazyLock::new(|| Mutex::new(HashMap::new()));
61
62fn get_job_dir_lock(job_dir: &Path) -> &'static Mutex<()> {
63    let mut map = JOB_DIR_LOCKS.lock().unwrap();
64    map.entry(job_dir.to_owned())
65        .or_insert_with(|| Box::leak(Box::new(Mutex::new(()))))
66}

72/// Guard holding the cargo build directory file lock (shared).
73pub struct CargoBuildLock {
74    _file: fs::File,
75}
76
77impl CargoBuildLock {
78    pub fn lock_shared(lock_path: &Path) -> Self {
79        eprintln!(
80            "[hydro-build] acquiring .cargo-build-lock shared: {}",
81            lock_path.display()
82        );
83        let file = fs::OpenOptions::new()
84            .read(true)
85            .write(true)
86            .create(true)
87            .truncate(false)
88            .open(lock_path)
89            .unwrap();
90        file.lock_shared().unwrap();
91        eprintln!("[hydro-build] acquired .cargo-build-lock shared");
92        CargoBuildLock { _file: file }
93    }
94}

100/// Guard holding both in-process and file locks for prebuild coordination.
101pub struct PrebuildGuard {
102    _rw_guard: RwGuard,
103    _global_guard: Option<parking_lot::RwLockWriteGuard<'static, ()>>,
104    _global_file: Option<fs::File>,
105    _file: fs::File,
106    lock_dir: PathBuf,
107}
108
109#[expect(
110    dead_code,
111    reason = "variants hold lock guards that are released on drop"
112)]
113enum RwGuard {
114    Read(parking_lot::RwLockReadGuard<'static, ()>),
115    Upgradable(parking_lot::RwLockUpgradableReadGuard<'static, ()>),
116    Write(parking_lot::RwLockWriteGuard<'static, ()>),
117}
118
119impl PrebuildGuard {
120    /// Acquire an upgradable lock (in-process upgradable read + file shared).
121    /// When `features_hash` is Some, uses a per-hash lock (for `__CARGO_DEFAULT_LIB_METADATA` mode).
122    pub fn lock_upgradable(lock_path: &Path, features_hash: Option<&str>) -> Self {
123        eprintln!(
124            "[hydro-build] acquiring prebuild upgradable lock: {}",
125            lock_path.display()
126        );
127        let rw_guard = get_dep_lock(features_hash).upgradable_read();
128        let lock_dir = lock_path.parent().unwrap().to_owned();
129        let file = fs::OpenOptions::new()
130            .read(true)
131            .write(true)
132            .create(true)
133            .truncate(false)
134            .open(lock_path)
135            .unwrap();
136        file.lock_shared().unwrap();
137        eprintln!("[hydro-build] acquired prebuild upgradable lock");
138        PrebuildGuard {
139            _rw_guard: RwGuard::Upgradable(rw_guard),
140            _global_guard: None,
141            _global_file: None,
142            _file: file,
143            lock_dir,
144        }
145    }
146
147    /// Upgrade to exclusive (in-process write + file exclusive + global locks).
148    pub fn upgrade(self) -> Self {
149        eprintln!("[hydro-build] upgrading prebuild lock to exclusive");
150        let file = self._file;
151        let rw_guard = match self._rw_guard {
152            RwGuard::Upgradable(u) => parking_lot::RwLockUpgradableReadGuard::upgrade(u),
153            _ => panic!("can only upgrade from upgradable"),
154        };
155        // Release per-feature shared lock before acquiring global exclusive
156        // to avoid deadlock (other processes hold per-feature shared and wait for global).
157        file.unlock().unwrap();
158        eprintln!("[hydro-build] acquiring global prebuild file lock");
159        let global_guard = GLOBAL_PREBUILD_LOCK.write();
160        let global_file = fs::OpenOptions::new()
161            .read(true)
162            .write(true)
163            .create(true)
164            .truncate(false)
165            .open(self.lock_dir.join(".global-prebuild.lock"))
166            .unwrap();
167        global_file.lock().unwrap();
168        eprintln!("[hydro-build] acquiring exclusive on per-feature lock");
169        file.lock().unwrap();
170        eprintln!("[hydro-build] acquired prebuild exclusive lock");
171        PrebuildGuard {
172            _rw_guard: RwGuard::Write(rw_guard),
173            _global_guard: Some(global_guard),
174            _global_file: Some(global_file),
175            _file: file,
176            lock_dir: self.lock_dir,
177        }
178    }
179
180    /// Downgrade from exclusive to shared (releases global lock).
181    pub fn downgrade(self) -> Self {
182        let file = self._file;
183        file.lock_shared().unwrap();
184        let rw_guard = match self._rw_guard {
185            RwGuard::Write(w) => RwGuard::Read(parking_lot::RwLockWriteGuard::downgrade(w)),
186            RwGuard::Upgradable(u) => {
187                RwGuard::Read(parking_lot::RwLockUpgradableReadGuard::downgrade(u))
188            }
189            r @ RwGuard::Read(_) => r,
190        };
191        PrebuildGuard {
192            _rw_guard: rw_guard,
193            _global_guard: None,
194            _global_file: None,
195            _file: file,
196            lock_dir: self.lock_dir,
197        }
198    }
199
200    /// Get mutable access to the underlying file (for writing timestamps).
201    pub fn file_mut(&mut self) -> &mut fs::File {
202        &mut self._file
203    }
204}

210/// Set up a job directory with symlinked .fingerprint and deps subdirs.
211/// Does NOT set up build/ — use `populate_job_build_dir` for final builds
212/// or manually symlink for prebuild.
213pub fn setup_job_dir(jobs_dir: &Path, name: &str, shared_debug: &Path) -> PathBuf {
214    let job = jobs_dir.join(name);
215    let _lock = get_job_dir_lock(&job);
216    let _lock = _lock.lock().unwrap();
217
218    // File lock for cross-process serialization on the same job dir.
219    fs::create_dir_all(&job).unwrap();
220    let job_lock_file = fs::OpenOptions::new()
221        .read(true)
222        .write(true)
223        .create(true)
224        .truncate(false)
225        .open(job.join(".job.lock"))
226        .unwrap();
227    job_lock_file.lock().unwrap();
228
229    let job_debug = job.join("debug");
230    fs::create_dir_all(&job_debug).unwrap();
231    for subdir in [".fingerprint", "deps"] {
232        let link = job_debug.join(subdir);
233        if !link.exists() {
234            let target = shared_debug.join(subdir);
235            fs::create_dir_all(&target).unwrap();
236            #[cfg(unix)]
237            std::os::unix::fs::symlink(&target, &link).unwrap();
238            #[cfg(windows)]
239            std::os::windows::fs::symlink_dir(&target, &link).unwrap();
240        }
241    }
242    job
243}

245/// Symlink the build/ directory for prebuild jobs (writes through to shared).
246pub fn symlink_prebuild_build_dir(prebuild_target: &Path, shared_debug: &Path) {
247    let link = prebuild_target.join("debug").join("build");
248    if !link.exists() {
249        let target = shared_debug.join("build");
250        fs::create_dir_all(&target).unwrap();
251        #[cfg(unix)]
252        std::os::unix::fs::symlink(&target, &link).unwrap();
253        #[cfg(windows)]
254        std::os::windows::fs::symlink_dir(&target, &link).unwrap();
255    }
256}

258/// Guard that holds the job directory lock for the duration of a build.
259pub struct JobBuildGuard {
260    _mutex_guard: std::sync::MutexGuard<'static, ()>,
261    _file: fs::File,
262}
263
264/// Populate the per-job build/ directory from the shared build/ directory.
265/// Returns a guard that holds the job lock — keep alive for the entire final build.
266/// This prevents races from cargo's `link_or_copy` on build script binaries.
267pub fn populate_job_build_dir(job_debug: &Path, shared_debug: &Path) -> JobBuildGuard {
268    let shared_build = shared_debug.join("build");
269    let job_build = job_debug.join("build");
270
271    let job_dir = job_debug.parent().unwrap();
272    let mutex_guard = get_job_dir_lock(job_dir);
273    let mutex_guard = mutex_guard.lock().unwrap();
274
275    // Also hold a file lock for cross-process serialization on the same job dir.
276    let job_lock_path = job_dir.join(".job.lock");
277    let job_lock_file = fs::OpenOptions::new()
278        .read(true)
279        .write(true)
280        .create(true)
281        .truncate(false)
282        .open(&job_lock_path)
283        .unwrap();
284    job_lock_file.lock().unwrap();
285
286    let _ = fs::remove_dir_all(&job_build);
287    fs::create_dir_all(&job_build).unwrap();
288    if shared_build.exists() {
289        for entry in fs::read_dir(&shared_build).unwrap() {
290            let entry = entry.unwrap();
291            if !entry.file_type().unwrap().is_dir() {
292                continue;
293            }
294            let dest = job_build.join(entry.file_name());
295            // Create dir and symlink each child individually so cargo can
296            // safely remove+relink build-script-build without racing other jobs.
297            fs::create_dir_all(&dest).unwrap();
298            for file in fs::read_dir(entry.path()).unwrap() {
299                let file = file.unwrap();
300                let file_dest = dest.join(file.file_name());
301                #[cfg(unix)]
302                std::os::unix::fs::symlink(file.path(), &file_dest).unwrap();
303                #[cfg(windows)]
304                {
305                    if file.file_type().unwrap().is_dir() {
306                        std::os::windows::fs::symlink_dir(file.path(), &file_dest).unwrap();
307                    } else {
308                        std::os::windows::fs::symlink_file(file.path(), &file_dest).unwrap();
309                    }
310                }
311            }
312        }
313    }
314
315    JobBuildGuard {
316        _mutex_guard: mutex_guard,
317        _file: job_lock_file,
318    }
319}

325/// Run the prebuild phase with proper locking and freshness checking.
326///
327/// - `target_dir`: the shared target directory (e.g. `target/`)
328/// - `crate_name`: unique identifier for the crate being built (included in hash)
329/// - `features`: list of features for hashing
330/// - `staged_paths`: paths to check mtime against for freshness
331/// - `build_fn`: closure called with `prebuild_target` path; should run cargo build(s)
332///
333/// Returns `(PrebuildGuard, CargoBuildLock)` both held in shared mode.
334/// Caller should keep both alive during the final build.
335pub fn run_prebuild(
336    target_dir: &Path,
337    crate_name: &str,
338    features: &[String],
339    staged_paths: &[PathBuf],
340    build_fn: impl FnOnce(&Path),
341) -> (PrebuildGuard, CargoBuildLock) {
342    // Acquire cargo build lock shared for the entire prebuild + final build duration.
343    let shared_debug = target_dir.join("debug");
344    fs::create_dir_all(&shared_debug).ok();
345    let cargo_lock = CargoBuildLock::lock_shared(&shared_debug.join(".cargo-build-lock"));
346
347    let features_hash = {
348        use std::hash::{Hash, Hasher};
349        let mut hasher = std::collections::hash_map::DefaultHasher::new();
350        crate_name.hash(&mut hasher);
351        let mut sorted = features.to_vec();
352        sorted.sort();
353        sorted.hash(&mut hasher);
354        format!("{:016x}", hasher.finish())
355    };
356
357    let has_lib_metadata = std::env::var("__CARGO_DEFAULT_LIB_METADATA").is_ok();
358    let lock_path = if has_lib_metadata {
359        target_dir.join(format!(".prebuild-{features_hash}.lock"))
360    } else {
361        target_dir.join(".prebuild.lock")
362    };
363
364    let staged_mtime = staged_paths
365        .iter()
366        .filter_map(|p| fs::metadata(p).and_then(|m| m.modified()).ok())
367        .max()
368        .unwrap_or(SystemTime::UNIX_EPOCH);
369
370    let prebuild_is_fresh = |path: &Path, expected_hash: &str| -> bool {
371        fs::read_to_string(path)
372            .ok()
373            .and_then(|s| {
374                let mut parts = s.trim().splitn(2, ':');
375                let hash = parts.next()?;
376                let nanos = parts.next()?.parse::<u128>().ok()?;
377                let ts = SystemTime::UNIX_EPOCH
378                    .checked_add(std::time::Duration::from_nanos(nanos as u64))?;
379                Some((hash.to_owned(), ts))
380            })
381            .is_some_and(|(hash, ts)| hash == expected_hash && ts >= staged_mtime)
382    };
383
384    let lock_hash = if has_lib_metadata {
385        Some(features_hash.as_str())
386    } else {
387        None
388    };
389
390    log_build_event(
391        target_dir,
392        &format!("prebuild: lock_upgradable, hash={features_hash}"),
393    );
394    let guard = PrebuildGuard::lock_upgradable(&lock_path, lock_hash);
395    if prebuild_is_fresh(&lock_path, &features_hash) {
396        log_build_event(target_dir, "prebuild: fresh, downgrading to shared");
397        return (guard.downgrade(), cargo_lock);
398    }
399
400    log_build_event(target_dir, "prebuild: not fresh, upgrading to exclusive");
401    let mut guard = guard.upgrade();
402    log_build_event(target_dir, "prebuild: exclusive acquired");
403
404    // Re-check after acquiring exclusive.
405    if !prebuild_is_fresh(&lock_path, &features_hash) {
406        let shared_debug = target_dir.join("debug");
407        let jobs_dir = target_dir.join("jobs");
408        let prebuild_target = setup_job_dir(&jobs_dir, "prebuild", &shared_debug);
409        symlink_prebuild_build_dir(&prebuild_target, &shared_debug);
410
411        build_fn(&prebuild_target);
412
413        // Write features_hash:timestamp.
414        use std::io::Seek;
415        let now_nanos = SystemTime::now()
416            .duration_since(SystemTime::UNIX_EPOCH)
417            .unwrap()
418            .as_nanos();
419        let file = guard.file_mut();
420        file.set_len(0).unwrap();
421        file.seek(std::io::SeekFrom::Start(0)).unwrap();
422        write!(file, "{}:{}", features_hash, now_nanos).unwrap();
423    }
424
425    (guard.downgrade(), cargo_lock)
426}
