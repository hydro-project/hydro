23/// The delay (soft barrier) type, for each input to an operator if needed.
24#[derive(Clone, Copy, PartialOrd, Ord, PartialEq, Eq, Debug)]
25pub enum DelayType {
26    /// Input must be collected over the preceding stratum.
27    Stratum,
28    /// Monotone accumulation: can delay to reduce flow rate, but also correct to emit "early"
29    MonotoneAccum,
30    /// Input must be collected over the previous tick.
31    Tick,
32    /// Input must be collected over the previous tick but also not cause a new tick to occur.
33    TickLazy,
34}

113/// The code generated and returned by a [`OperatorConstraints::write_fn`].
114#[derive(Default)]
115#[non_exhaustive]
116pub struct OperatorWriteOutput {
117    /// Code which runs once outside any subgraphs, BEFORE subgraphs are initialized,
118    /// to set up any external state (state API, chanels, network connections, etc.)
119    /// to be used by the subgraph.
120    pub write_prologue: TokenStream,
121    /// Code which runs once outside the subgraph, AFTER subgraphs are initialized,
122    /// to set up state hooks which may need the subgraph ID.
123    pub write_prologue_after: TokenStream,
124    /// Iterator (or pusherator) code inside the subgraphs. The code for each
125    /// operator is emitted in order.
126    ///
127    /// Emitted code should assign to [`WriteContextArgs::ident`] and use
128    /// [`WriteContextArgs::inputs`] (pull `Stream`s) or
129    /// [`WriteContextArgs::outputs`] (push `Sink`s).
130    pub write_iterator: TokenStream,
131    /// Code which runs after `Stream`s/`Sink`s have been run. Mainly for flushing IO.
132    pub write_iterator_after: TokenStream,
133}

142/// Helper to write the `write_iterator` portion of [`OperatorConstraints::write_fn`] output for
143/// unary identity operators.
144pub fn identity_write_iterator_fn(
145    &WriteContextArgs {
146        root,
147        op_span,
148        ident,
149        inputs,
150        outputs,
151        is_pull,
152        op_inst:
153            OperatorInstance {
154                generics: OpInstGenerics { type_args, .. },
155                ..
156            },
157        ..
158    }: &WriteContextArgs,
159) -> TokenStream {
160    let generic_type = type_args
161        .first()
162        .map(quote::ToTokens::to_token_stream)
163        .unwrap_or(quote_spanned!(op_span=> _));
164
165    if is_pull {
166        let input = &inputs[0];
167        quote_spanned! {op_span=>
168            let #ident = {
169                fn check_input<St, Item>(stream: St) -> impl #root::futures::stream::Stream<Item = Item>
170                where
171                    St: #root::futures::stream::Stream<Item = Item>,
172                {
173                    stream
174                }
175                check_input::<_, #generic_type>(#input)
176            };
177        }
178    } else {
179        let output = &outputs[0];
180        quote_spanned! {op_span=>
181            let #ident = {
182                fn check_output<Si, Item>(sink: Si) -> impl #root::futures::sink::Sink<Item, Error = #root::Never>
183                where
184                    Si: #root::futures::sink::Sink<Item, Error = #root::Never>,
185                {
186                    sink
187                }
188                check_output::<_, #generic_type>(#output)
189            };
190        }
191    }
192}

203/// Helper to write the `write_iterator` portion of [`OperatorConstraints::write_fn`] output for
204/// the null operator - an operator that ignores all inputs and produces no output.
205pub fn null_write_iterator_fn(
206    &WriteContextArgs {
207        root,
208        op_span,
209        ident,
210        inputs,
211        outputs,
212        is_pull,
213        op_inst:
214            OperatorInstance {
215                generics: OpInstGenerics { type_args, .. },
216                ..
217            },
218        ..
219    }: &WriteContextArgs,
220) -> TokenStream {
221    let default_type = parse_quote_spanned! {op_span=> _};
222    let iter_type = type_args.first().unwrap_or(&default_type);
223
224    if is_pull {
225        quote_spanned! {op_span=>
226            let #ident = #root::futures::stream::poll_fn(move |_cx| {
227                // Make sure to poll all #inputs to completion.
228                #(
229                    let #inputs = #root::futures::stream::Stream::poll_next(::std::pin::pin!(#inputs), _cx);
230                )*
231                #(
232                    let _ = ::std::task::ready!(#inputs);
233                )*
234                ::std::task::Poll::Ready(::std::option::Option::None)
235            });
236        }
237    } else {
238        quote_spanned! {op_span=>
239            #[allow(clippy::let_unit_value)]
240            let _ = (#(#outputs),*);
241            let #ident = #root::sinktools::for_each::ForEach::new::<#iter_type>(::std::mem::drop::<#iter_type>);
242        }
243    }
244}

393    pub work_fn_async: &'a Ident,

395    /// Ident the `Stream` or `Sink` should be assigned to.
396    pub ident: &'a Ident,
397    /// If a pull `Stream` (true) or push `Sink` (false) should be used.
398    pub is_pull: bool,
399    /// Input `Stream` operator idents (or ref idents; used for pull).
400    pub inputs: &'a [Ident],
401    /// Output `Sink` operator idents (or ref idents; used for push).
402    pub outputs: &'a [Ident],
403    /// Ident for the singleton output of this operator, if any.
404    pub singleton_output_ident: &'a Ident,
405
406    /// Operator name.
407    pub op_name: &'static str,
408    /// Operator instance arguments object.
409    pub op_inst: &'a OperatorInstance,
410    /// Arguments provided by the user into the operator as arguments.
411    /// I.e. the `a, b, c` in `-> my_op(a, b, c) -> `.
412    ///
413    /// These arguments include singleton postprocessing codegen, with
414    /// [`std::cell::RefCell::borrow_mut`] code pre-generated.
415    pub arguments: &'a Punctuated<Expr, Token![,]>,
416    /// Same as [`Self::arguments`] but with only `StateHandle`s, no borrowing code.
417    pub arguments_handles: &'a Punctuated<Expr, Token![,]>,

419impl WriteContextArgs<'_> {
420    /// Generate a (almost certainly) unique identifier with the given suffix.
421    ///
422    /// Includes the subgraph and node IDs in the generated identifier.
423    ///
424    /// This will always return the same identifier for a given `suffix`.
425    pub fn make_ident(&self, suffix: impl AsRef<str>) -> Ident {
426        Ident::new(
427            &format!(
428                "sg_{:?}_node_{:?}_{}",
429                self.subgraph_id.data(),
430                self.node_id.data(),
431                suffix.as_ref(),
432            ),
433            self.op_span,
434        )
435    }
436
437    /// Returns `#root::scheduled::graph::StateLifespan::#variant` corresponding to the given
438    /// peristence.
439    pub fn persistence_as_state_lifespan(&self, persistence: Persistence) -> Option<TokenStream> {
440        let root = self.root;
441        let variant =
442            persistence.as_state_lifespan_variant(self.subgraph_id, self.loop_id, self.op_span)?;
443        Some(quote_spanned! {self.op_span=>
444            #root::scheduled::graph::StateLifespan::#variant
445        })
446    }
447
448    /// Returns the given number of persistence arguments, disallowing mutable lifetimes.
449    pub fn persistence_args_disallow_mutable<const N: usize>(
450        &self,
451        diagnostics: &mut Diagnostics,
452    ) -> [Persistence; N] {
453        let len = self.op_inst.generics.persistence_args.len();
454        if 0 != len && 1 != len && N != len {
455            diagnostics.push(Diagnostic::spanned(
456                self.op_span,
457                Level::Error,
458                format!(
459                    "The operator `{}` only accepts 0, 1, or {} persistence arguments",
460                    self.op_name, N
461                ),
462            ));
463        }
464
465        let default_persistence = if self.loop_id.is_some() {
466            Persistence::None
467        } else {
468            Persistence::Tick
469        };
470        let mut out = [default_persistence; N];
471        self.op_inst
472            .generics
473            .persistence_args
474            .iter()
475            .copied()
476            .cycle() // Re-use the first element for both persistences.
477            .take(N)
478            .enumerate()
479            .filter(|&(_i, p)| {
480                if p == Persistence::Mutable {
481                    diagnostics.push(Diagnostic::spanned(
482                        self.op_span,
483                        Level::Error,
484                        format!(
485                            "An implementation of `'{}` does not exist",
486                            p.to_str_lowercase()
487                        ),
488                    ));
489                    false
490                } else {
491                    true
492                }
493            })
494            .for_each(|(i, p)| {
495                out[i] = p;
496            });
497        out
498    }
499}

501/// An object-safe version of [`RangeBounds`].
502pub trait RangeTrait<T>: Send + Sync + Debug
503where
504    T: ?Sized,
505{
506    /// Start (lower) bound.
507    fn start_bound(&self) -> Bound<&T>;
508    /// End (upper) bound.
509    fn end_bound(&self) -> Bound<&T>;
510    /// Returns if `item` is contained in this range.
511    fn contains(&self, item: &T) -> bool
512    where
513        T: PartialOrd<T>;
514
515    /// Turn this range into a human-readable string.
516    fn human_string(&self) -> String
517    where
518        T: Display + PartialEq,
519    {
520        match (self.start_bound(), self.end_bound()) {
521            (Bound::Unbounded, Bound::Unbounded) => "any number of".to_owned(),
522
523            (Bound::Included(n), Bound::Included(x)) if n == x => {
524                format!("exactly {}", n)
525            }
526            (Bound::Included(n), Bound::Included(x)) => {
527                format!("at least {} and at most {}", n, x)
528            }
529            (Bound::Included(n), Bound::Excluded(x)) => {
530                format!("at least {} and less than {}", n, x)
531            }
532            (Bound::Included(n), Bound::Unbounded) => format!("at least {}", n),
533            (Bound::Excluded(n), Bound::Included(x)) => {
534                format!("more than {} and at most {}", n, x)
535            }
536            (Bound::Excluded(n), Bound::Excluded(x)) => {
537                format!("more than {} and less than {}", n, x)
538            }
539            (Bound::Excluded(n), Bound::Unbounded) => format!("more than {}", n),
540            (Bound::Unbounded, Bound::Included(x)) => format!("at most {}", x),
541            (Bound::Unbounded, Bound::Excluded(x)) => format!("less than {}", x),
542        }
543    }
544}

546impl<R, T> RangeTrait<T> for R
547where
548    R: RangeBounds<T> + Send + Sync + Debug,
549{
550    fn start_bound(&self) -> Bound<&T> {
551        self.start_bound()
552    }
553
554    fn end_bound(&self) -> Bound<&T> {
555        self.end_bound()
556    }
557
558    fn contains(&self, item: &T) -> bool
559    where
560        T: PartialOrd<T>,
561    {
562        self.contains(item)
563    }
564}

566/// Persistence lifetimes: `'none`, `'tick`, `'static`, or `'mutable`.
567#[derive(Clone, Copy, PartialOrd, Ord, PartialEq, Eq, Debug, Serialize, Deserialize)]
568pub enum Persistence {
569    /// No persistence, for within a loop iteration.
570    None,
571    /// Persistence throughout a single loop execution, across iterations.
572    Loop,
573    /// Persistence for one tick, at the top-level only (outside any loops).
574    Tick,
575    /// Persistence across all ticks.
576    Static,
577    /// The static lifetime but allowing non-monotonic mutability.
578    Mutable,
579}
580impl Persistence {
581    /// Returns just the variant of `#root::scheduled::graph::StateLifespan::VARIANT` for use in macros.
582    pub fn as_state_lifespan_variant(
583        self,
584        subgraph_id: GraphSubgraphId,
585        loop_id: Option<GraphLoopId>,
586        span: Span,
587    ) -> Option<TokenStream> {
588        match self {
589            Persistence::None => {
590                let sg_ident = subgraph_id.as_ident(span);
591                Some(quote_spanned!(span=> Subgraph(#sg_ident)))
592            }
593            Persistence::Loop => {
594                let loop_ident = loop_id
595                    .expect("`Persistence::Loop` outside of a loop context.")
596                    .as_ident(span);
597                Some(quote_spanned!(span=> Loop(#loop_ident)))
598            }
599            Persistence::Tick => Some(quote_spanned!(span=> Tick)),
600            Persistence::Static => None,
601            Persistence::Mutable => None,
602        }
603    }
604
605    /// Returns a lowercase string for the persistence type.
606    pub fn to_str_lowercase(self) -> &'static str {
607        match self {
608            Persistence::None => "none",
609            Persistence::Tick => "tick",
610            Persistence::Loop => "loop",
611            Persistence::Static => "static",
612            Persistence::Mutable => "mutable",
613        }
614    }
615}

626#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Hash, DocumentedVariants)]
627pub enum OperatorCategory {
628    /// Maps: Simple one-in-one-out operators.
629    Map,
630    /// Filters: One-in zero-or-one-out operators.
631    Filter,
632    /// Flattens: One-in multiple-out operators.
633    Flatten,
634    /// Folds: Operators which accumulate elements together.
635    Fold,
636    /// Keyed Folds: Operators which accumulate elements together by key.
637    KeyedFold,
638    /// Lattice Folds: Folds based on lattice-merge.
639    LatticeFold,
640    /// Persistent Operators: Persistent (stateful) operators.
641    Persistence,
642    /// Multi-Input Operators: Operators with multiple inputs.
643    MultiIn,
644    /// Multi-Output Operators: Operators with multiple outputs.
645    MultiOut,
646    /// Sources: Operators which produce output elements (and consume no inputs).
647    Source,
648    /// Sinks: Operators which consume input elements (and produce no outputs).
649    Sink,
650    /// Control Flow Operators: Operators which affect control flow/scheduling.
651    Control,
652    /// Compiler Fusion Operators: Operators which are necessary to implement certain optimizations and rewrite rules.
653    CompilerFusionOperator,
654    /// Windowing Operators: Operators for windowing `loop` inputs.
655    Windowing,
656    /// Un-Windowing Operators: Operators for collecting `loop` outputs.
657    Unwindowing,
658}
659impl OperatorCategory {
660    /// Human-readible heading name, for docs.
661    pub fn name(self) -> &'static str {
662        self.get_variant_docs().split_once(":").unwrap().0
663    }
664    /// Human description, for docs.
665    pub fn description(self) -> &'static str {
666        self.get_variant_docs().split_once(":").unwrap().1
667    }
668}
