11/// > 2 input streams of type `<(K, V1)>` and `<(K, V2)>`, 1 output stream of type `<(K, (V1, V2))>`
12///
13/// `join_fused` takes two arguments, they are the aggregators for the left hand side and right hand side inputs respectively.
14/// There are three main aggregators available, they are `Reduce`: if the input type is the same as the accumulator type,
15/// `Fold`: if the input type is different from the accumulator type, and the accumulator type has a sensible default value, and
16/// `FoldFrom`: if the input type is different from the accumulator type, and the accumulator needs to be derived from the first input value.
17/// Examples of all three configuration options are below:
18/// ```dfir,ignore
19/// // Left hand side input will use `Fold`, right hand side input will use `Reduce`,
20/// join_fused(Fold::new(|| "default value", |x, y| *x += y), Reduce::new(|x, y| *x -= y))
21///
22/// // Left hand side input will use `FoldFrom`, and the right hand side input will use `Reduce` again
23/// join_fused(FoldFrom::new(|x| "conversion function", |x, y| *x += y), Reduce::new(|x, y| *x *= y))
24/// ```
25/// The three currently supported fused operator types are `Fold(Fn() -> A, Fn(A, T) -> A)`, `Reduce(Fn(A, A) -> A)`, and `FoldFrom(Fn(T) -> A, Fn(A, T) -> A)`
26///
27/// `join_fused` first performs a fold_keyed/reduce_keyed operation on each input stream before performing joining. See `join()`. There is currently no equivalent for `FoldFrom` in dfir operators.
28///
29/// For example, the following two dfir programs are equivalent, the former would optimize into the latter:
30///
31/// ```dfir
32/// source_iter(vec![("key", 0), ("key", 1), ("key", 2)])
33///     -> reduce_keyed(|x: &mut _, y| *x += y)
34///     -> [0]my_join;
35/// source_iter(vec![("key", 2), ("key", 3)])
36///     -> fold_keyed(|| 1, |x: &mut _, y| *x *= y)
37///     -> [1]my_join;
38/// my_join = join_multiset()
39///     -> assert_eq([("key", (3, 6))]);
40/// ```
41///
42/// ```dfir
43/// use dfir_rs::util::accumulator::{Fold, Reduce};
44///
45/// source_iter(vec![("key", 0), ("key", 1), ("key", 2)])
46///     -> [0]my_join;
47/// source_iter(vec![("key", 2), ("key", 3)])
48///     -> [1]my_join;
49/// my_join = join_fused(Reduce::new(|x, y| *x += y), Fold::new(|| 1, |x, y| *x *= y))
50///     -> assert_eq([("key", (3, 6))]);
51/// ```
52///
53/// Here is an example of using `FoldFrom` to derive the accumulator from the first value:
54///
55/// ```dfir
56/// use dfir_rs::util::accumulator::{Fold, FoldFrom};
57///
58/// source_iter(vec![("key", 0), ("key", 1), ("key", 2)])
59///     -> [0]my_join;
60/// source_iter(vec![("key", 2), ("key", 3)])
61///     -> [1]my_join;
62/// my_join = join_fused(FoldFrom::new(|x: u32| x + 3, |x, y| *x += y), Fold::new(|| 1, |x, y| *x *= y))
63///     -> assert_eq([("key", (6, 6))]);
64/// ```
65///
66/// The benefit of this is that the state between the reducing/folding operator and the join is merged together.
67///
68/// `join_fused` follows the same persistence rules as `join` and all other operators. By default, both the left hand side and right hand side are `'tick` persistence. They can be set to `'static` persistence
69/// by specifying `'static` in the type arguments of the operator.
70///
71/// for `join_fused::<'static>`, the operator will replay all _keys_ that the join has ever seen each tick, and not only the new matches from that specific tick.
72/// This means that it behaves identically to if `persist::<'static>()` were placed before the inputs and the persistence of
73/// for example, the two following examples have identical behavior:
74///
75/// ```dfir
76/// use dfir_rs::util::accumulator::{Fold, Reduce};
77///
78/// source_iter(vec![("key", 0), ("key", 1), ("key", 2)]) -> persist::<'static>() -> [0]my_join;
79/// source_iter(vec![("key", 2)]) -> my_union;
80/// source_iter(vec![("key", 3)]) -> defer_tick() -> my_union;
81/// my_union = union() -> persist::<'static>() -> [1]my_join;
82///
83/// my_join = join_fused(Reduce::new(|x, y| *x += y), Fold::new(|| 1, |x, y| *x *= y))
84///     -> assert_eq([("key", (3, 2)), ("key", (3, 6))]);
85/// ```
86///
87/// ```dfir
88/// use dfir_rs::util::accumulator::{Fold, Reduce};
89///
90/// source_iter(vec![("key", 0), ("key", 1), ("key", 2)]) -> [0]my_join;
91/// source_iter(vec![("key", 2)]) -> my_union;
92/// source_iter(vec![("key", 3)]) -> defer_tick() -> my_union;
93/// my_union = union() -> [1]my_join;
94///
95/// my_join = join_fused::<'static>(Reduce::new(|x, y| *x += y), Fold::new(|| 1, |x, y| *x *= y))
96///     -> assert_eq([("key", (3, 2)), ("key", (3, 6))]);
97/// ```
98pub const JOIN_FUSED: OperatorConstraints = OperatorConstraints {
99    name: "join_fused",
100    categories: &[OperatorCategory::MultiIn],
101    hard_range_inn: &(2..=2),
102    soft_range_inn: &(2..=2),
103    hard_range_out: RANGE_1,
104    soft_range_out: RANGE_1,
105    num_args: 2,
106    persistence_args: &(0..=2),
107    type_args: RANGE_0,
108    is_external_input: false,
109    has_singleton_output: false,
110    flo_type: None,
111    ports_inn: Some(|| super::PortListSpec::Fixed(parse_quote! { 0, 1 })),
112    ports_out: None,
113    input_delaytype_fn: |_| Some(DelayType::Stratum),
114    write_fn: |wc @ &WriteContextArgs {
115                   root,
116                   context,
117                   op_span,
118                   work_fn_async,
119                   ident,
120                   inputs,
121                   is_pull,
122                   arguments,
123                   ..
124               },
125               diagnostics| {
126        assert!(is_pull);
127
128        let persistences: [_; 2] = wc.persistence_args_disallow_mutable(diagnostics);
129
130        let (lhs_prologue, lhs_prologue_after, lhs_pre_write_iter, lhs_borrow) =
131            make_joindata(wc, persistences[0], "lhs").map_err(|err| diagnostics.push(err))?;
132
133        let (rhs_prologue, rhs_prologue_after, rhs_pre_write_iter, rhs_borrow) =
134            make_joindata(wc, persistences[1], "rhs").map_err(|err| diagnostics.push(err))?;
135
136        let lhs = &inputs[0];
137        let rhs = &inputs[1];
138
139        let lhs_accum = &arguments[0];
140        let rhs_accum = &arguments[1];
141
142        // Since both input arguments are stratum blocking then we don't need to keep track of ticks to avoid emitting the same thing twice in the same tick.
143        let write_iterator = quote_spanned! {op_span=>
144            #lhs_pre_write_iter
145            #rhs_pre_write_iter
146
147            let #ident = {
148                async fn __check_accum<Accumulator, Key, Accum, St, Hasher, Item>(accum: &mut Accumulator, borrow: &mut ::std::collections::HashMap<Key, Accum, Hasher>, st: St)
149                where
150                    Accumulator: #root::util::accumulator::Accumulator<Accum, Item>,
151                    Key: ::std::cmp::Eq + ::std::hash::Hash + ::std::clone::Clone,
152                    St: #root::futures::stream::Stream<Item = (Key, Item)>,
153                    Hasher: ::std::hash::BuildHasher,
154                    Item: ::std::clone::Clone,
155                {
156                    #root::compiled::pull::accumulate_all(accum, borrow, st).await;
157                }
158                #work_fn_async(__check_accum(&mut #lhs_accum, &mut *#lhs_borrow, #lhs)).await;
159                #work_fn_async(__check_accum(&mut #rhs_accum, &mut *#rhs_borrow, #rhs)).await;
160
161                // TODO: start the iterator with the smallest len() table rather than always picking rhs.
162                #[allow(suspicious_double_ref_op, clippy::clone_on_copy)]
163                #[allow(clippy::disallowed_methods, reason = "FxHasher is deterministic")]
164                let iter = #rhs_borrow
165                    .iter()
166                    .filter_map(|(k, v2)| {
167                        #lhs_borrow.get(k).map(|v1| (k.clone(), (v1.clone(), v2.clone())))
168                    });
169                #root::futures::stream::iter(iter)
170            };
171        };
172
173        let write_iterator_after =
174            if persistences[0] == Persistence::Static || persistences[1] == Persistence::Static {
175                quote_spanned! {op_span=>
176                    // TODO: Probably only need to schedule if #*_borrow.len() > 0?
177                    #context.schedule_subgraph(#context.current_subgraph(), false);
178                }
179            } else {
180                quote_spanned! {op_span=>}
181            };
182
183        Ok(OperatorWriteOutput {
184            write_prologue: quote_spanned! {op_span=>
185                #lhs_prologue
186                #rhs_prologue
187            },
188            write_prologue_after: quote_spanned! {op_span=>
189                #lhs_prologue_after
190                #rhs_prologue_after
191            },
192            write_iterator,
193            write_iterator_after,
194        })
195    },
196};

198/// Returns `(prologue, prologue_after, pre_write_iter, borrow)`.
199pub(crate) fn make_joindata(
200    wc: &WriteContextArgs,
201    persistence: Persistence,
202    side: &str,
203) -> Result<(TokenStream, TokenStream, TokenStream, Ident), Diagnostic> {
204    let joindata_ident = wc.make_ident(format!("joindata_{}", side));
205    let borrow_ident = wc.make_ident(format!("joindata_{}_borrow", side));
206
207    let &WriteContextArgs {
208        context,
209        df_ident,
210        root,
211        op_span,
212        ..
213    } = wc;
214
215    Ok(match persistence {
216        Persistence::None => (
217            Default::default(),
218            Default::default(),
219            quote_spanned! {op_span=>
220                let #borrow_ident = &mut #root::rustc_hash::FxHashMap::default();
221            },
222            borrow_ident,
223        ),
224        Persistence::Tick | Persistence::Loop | Persistence::Static => {
225            let lifespan = wc.persistence_as_state_lifespan(persistence);
226            (
227                quote_spanned! {op_span=>
228                    let #joindata_ident = #df_ident.add_state(::std::cell::RefCell::new(#root::rustc_hash::FxHashMap::default()));
229                },
230                lifespan.map(|lifespan| quote_spanned! {op_span=>
231                    // Reset the value to the initializer fn at the end of each tick/loop execution.
232                    #df_ident.set_state_lifespan_hook(#joindata_ident, #lifespan, |rcell| { rcell.take(); });
233                }).unwrap_or_default(),
234                quote_spanned! {op_span=>
235                    let mut #borrow_ident = unsafe {
236                        // SAFETY: handles from `#df_ident`.
237                        #context.state_ref_unchecked(#joindata_ident)
238                    }.borrow_mut();
239                },
240                borrow_ident
241            )
242        }
243        Persistence::Mutable => panic!(),
244    })
