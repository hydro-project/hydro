9/// `persist_mut_keyed()` is similar to `persist_mut()` except that it also enables key-based deletions
10/// `persist_mut()` expects an input of type [`PersistenceKeyed<T>`](https://docs.rs/dfir_rs/latest/dfir_rs/util/enum.PersistenceKeyed.html),
11/// and it is this enumeration that enables the user to communicate deletion.
12/// Deletions/persists happen in the order they are received in the stream.
13/// For example, `[Persist(1), Delete(1), Persist(1)]` will result in a a single `1` value being stored.
14///
15/// ```dfir
16/// use dfir_rs::util::PersistenceKeyed;
17///
18/// source_iter([
19///         PersistenceKeyed::Persist(0, 1),
20///         PersistenceKeyed::Persist(1, 1),
21///         PersistenceKeyed::Delete(1),
22///     ])
23///     -> persist_mut_keyed::<'mutable>()
24///     -> assert_eq([(0, 1)]);
25/// ```
26pub const PERSIST_MUT_KEYED: OperatorConstraints = OperatorConstraints {
27    name: "persist_mut_keyed",
28    categories: &[OperatorCategory::Persistence],
29    hard_range_inn: RANGE_1,
30    soft_range_inn: RANGE_1,
31    hard_range_out: RANGE_1,
32    soft_range_out: RANGE_1,
33    num_args: 0,
34    persistence_args: RANGE_1,
35    type_args: RANGE_0,
36    is_external_input: false,
37    // If this is set to true, the state will need to be cleared using `#context.set_state_lifespan_hook`
38    // to prevent reading uncleared data if this subgraph doesn't run.
39    // https://github.com/hydro-project/hydro/issues/1298
40    // If `'tick` lifetimes are added.
41    has_singleton_output: false,
42    flo_type: None,
43    ports_inn: None,
44    ports_out: None,
45    input_delaytype_fn: |_| Some(DelayType::Stratum),
46    write_fn: |wc @ &WriteContextArgs {
47                   root,
48                   context,
49                   df_ident,
50                   op_span,
51                   work_fn_async,
52                   ident,
53                   inputs,
54                   is_pull,
55                   op_name,
56                   op_inst:
57                       OperatorInstance {
58                           generics:
59                               OpInstGenerics {
60                                   persistence_args, ..
61                               },
62                           ..
63                       },
64                   ..
65               },
66               diagnostics| {
67        assert!(is_pull);
68
69        if [Persistence::Mutable] != persistence_args[..] {
70            diagnostics.push(Diagnostic::spanned(
71                op_span,
72                Level::Error,
73                format!(
74                    "{} only supports `'{}`.",
75                    op_name,
76                    Persistence::Mutable.to_str_lowercase()
77                ),
78            ));
79        }
80
81        let persistdata_ident = wc.make_ident("persistdata");
82        let map_ident = wc.make_ident("persistmap");
83        let write_prologue = quote_spanned! {op_span=>
84            let #persistdata_ident = #df_ident.add_state(::std::cell::RefCell::new(
85                #root::rustc_hash::FxHashMap::<_, #root::util::sparse_vec::SparseVec<_>>::default()
86            ));
87        };
88
89        let write_iterator = {
90            let input = &inputs[0];
91            quote_spanned! {op_span=>
92                let mut #map_ident = unsafe {
93                    // SAFETY: handle from `#df_ident.add_state(..)`.
94                    #context.state_ref_unchecked(#persistdata_ident)
95                }.borrow_mut();
96
97                let #ident = {
98                    #[inline(always)]
99                    fn check_stream<K, V>(st: impl #root::futures::stream::Stream<Item = #root::util::PersistenceKeyed::<K, V>>) -> impl #root::futures::stream::Stream<Item = #root::util::PersistenceKeyed::<K, V>> {
100                        st
101                    }
102
103                    let iter = if context.is_first_run_this_tick() {
104                        let fut = #root::compiled::pull::ForEach::new(check_stream(#input), |item| {
105                            match item {
106                                #root::util::PersistenceKeyed::Persist(k, v) => {
107                                    #map_ident.entry(k).or_default().push(v);
108                                },
109                                #root::util::PersistenceKeyed::Delete(k) => {
110                                    #map_ident.remove(&k);
111                                }
112                            }
113                        });
114                        let () = #work_fn_async(fut).await;
115
116                        #[allow(clippy::clone_on_copy)]
117                        #[allow(clippy::disallowed_methods, reason = "FxHasher is deterministic")]
118                        Some(
119                            #map_ident
120                                .iter()
121                                .flat_map(|(k, v)| v.iter().map(move |v| (k.clone(), v.clone())))
122                        )
123                        .into_iter()
124                        .flatten()
125                    } else {
126                        None.into_iter().flatten()
127                    };
128                    #root::futures::stream::iter(iter)
129                };
130            }
131        };
132
133        let write_iterator_after = quote_spanned! {op_span=>
134            #context.schedule_subgraph(#context.current_subgraph(), false);
135        };
136
137        Ok(OperatorWriteOutput {
138            write_prologue,
139            write_iterator,
140            write_iterator_after,
141            ..Default::default()
142        })
143    },
144};
