10// This implementation is largely redundant to ANTI_JOIN and should be DRY'ed
11/// > 2 input streams the first of type (K, T), the second of type K,
12/// > with output type (K, T)
13///
14/// For a given tick, computes the anti-join of the items in the input
15/// streams, returning items in the `pos` input that do not have matching keys
16/// in the `neg` input. NOTE this uses multiset semantics only on the positive side,
17/// so duplicated positive inputs will appear in the output either 0 times (if matched in `neg`)
18/// or as many times as they appear in the input (if not matched in `neg`)
19///
20/// ```dfir
21/// source_iter(vec![("cat", 2), ("cat", 2), ("elephant", 3), ("elephant", 3)]) -> [pos]diff;
22/// source_iter(vec!["dog", "cat", "gorilla"]) -> [neg]diff;
23/// diff = anti_join() -> assert_eq([("elephant", 3), ("elephant", 3)]);
24/// ```
25pub const ANTI_JOIN: OperatorConstraints = OperatorConstraints {
26    name: "anti_join",
27    categories: &[OperatorCategory::MultiIn],
28    hard_range_inn: &(2..=2),
29    soft_range_inn: &(2..=2),
30    hard_range_out: RANGE_1,
31    soft_range_out: RANGE_1,
32    num_args: 0,
33    persistence_args: &(0..=2),
34    type_args: RANGE_0,
35    is_external_input: false,
36    has_singleton_output: false,
37    flo_type: None,
38    ports_inn: Some(|| super::PortListSpec::Fixed(parse_quote! { pos, neg })),
39    ports_out: None,
40    input_delaytype_fn: |idx| match idx {
41        PortIndexValue::Path(path) if "neg" == path.to_token_stream().to_string() => {
42            Some(DelayType::Stratum)
43        }
44        _else => None,
45    },
46    write_fn: |wc @ &WriteContextArgs {
47                   root,
48                   context,
49                   df_ident,
50                   op_span,
51                   work_fn_async,
52                   ident,
53                   is_pull,
54                   inputs,
55                   ..
56               },
57               diagnostics| {
58        assert!(is_pull);
59
60        let persistences: [_; 2] = wc.persistence_args_disallow_mutable(diagnostics);
61
62        let pos_ident = wc.make_ident("pos");
63        let neg_ident = wc.make_ident("neg");
64        let pos_borrow = wc.make_ident("pos_borrow");
65        let neg_borrow = wc.make_ident("neg_borrow");
66
67        let pos_persist = match persistences[0] {
68            Persistence::None | Persistence::Tick => false,
69            Persistence::Loop | Persistence::Static => true,
70            Persistence::Mutable => unreachable!(),
71        };
72
73        let write_prologue_pos = pos_persist.then(|| {
74            quote_spanned! {op_span=>
75                let #pos_ident = #df_ident.add_state(std::cell::RefCell::new(
76                    ::std::vec::Vec::new()
77                ));
78            }
79        });
80        let write_prologue_after_pos = pos_persist.then(|| wc
81            .persistence_as_state_lifespan(persistences[0])
82            .map(|lifespan| quote_spanned! {op_span=>
83                #[allow(clippy::redundant_closure_call)]
84                #df_ident.set_state_lifespan_hook(
85                    #pos_ident, #lifespan, move |rcell| { rcell.borrow_mut().clear(); },
86                );
87            })).flatten();
88
89        let write_prologue_neg = quote_spanned! {op_span=>
90            let #neg_ident = #df_ident.add_state(std::cell::RefCell::new(
91                #root::rustc_hash::FxHashSet::default()
92            ));
93        };
94        let write_prologue_after_neg = wc
95            .persistence_as_state_lifespan(persistences[1])
96            .map(|lifespan| quote_spanned! {op_span=>
97                #[allow(clippy::redundant_closure_call)]
98                #df_ident.set_state_lifespan_hook(
99                    #neg_ident, #lifespan, move |rcell| { rcell.borrow_mut().clear(); },
100                );
101            }).unwrap_or_default();
102
103        let input_neg = &inputs[0]; // N before P
104        let input_pos = &inputs[1];
105
106        let accum_neg = quote_spanned! {op_span=>
107            let fut = #root::compiled::pull::ForEach::new(#input_neg, |k| {
108                #neg_borrow.insert(k);
109            });
110            let () = #work_fn_async(fut).await;
111        };
112
113        let write_iterator = if !pos_persist {
114            quote_spanned! {op_span=>
115                let mut #neg_borrow = unsafe {
116                    // SAFETY: handle from `#df_ident`.
117                    #context.state_ref_unchecked(#neg_ident)
118                }.borrow_mut();
119
120                let #ident = {
121                    #accum_neg
122
123                    #root::tokio_stream::StreamExt::filter(#input_pos, |(k, _)| {
124                        !#neg_borrow.contains(k)
125                    })
126                };
127            }
128        } else {
129            quote_spanned! {op_span =>
130                let (mut #neg_borrow, mut #pos_borrow) = unsafe {
131                    // SAFETY: handles from `#df_ident`.
132                    (
133                        #context.state_ref_unchecked(#neg_ident).borrow_mut(),
134                        #context.state_ref_unchecked(#pos_ident).borrow_mut(),
135                    )
136                };
137
138
139                let #ident = {
140                    #accum_neg
141
142                    let replay_idx = if #context.is_first_run_this_tick() {
143                        0
144                    } else {
145                        #pos_borrow.len()
146                    };
147
148                    // Accum into pos vec
149                    let fut = #root::compiled::pull::ForEach::new(#input_pos, |kv| {
150                        #pos_borrow.push(kv);
151                    });
152                    let () = #work_fn_async(fut).await;
153
154                    // Replay out of pos vec
155                    let iter = #pos_borrow[replay_idx..].iter();
156                    let iter = ::std::iter::Iterator::filter(iter, |(k, _)| {
157                        !#neg_borrow.contains(k)
158                    });
159                    let iter = ::std::iter::Iterator::cloned(iter);
160                    #root::futures::stream::iter(iter)
161                };
162            }
163        };
164
165        Ok(OperatorWriteOutput {
166            write_prologue: quote_spanned! {op_span=>
167                #write_prologue_pos
168                #write_prologue_neg
169            },
170            write_prologue_after: quote_spanned! {op_span=>
171                #write_prologue_after_pos
172                #write_prologue_after_neg
173            },
174            write_iterator,
175            ..Default::default()
176        })
177    },
178};
