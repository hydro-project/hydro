8/// > Arguments: A `tag` string and a `Duration` for how long to wait between printing. A third optional
9/// > parameter controls the prefix used for logging (defaults to "_counter").
10///
11/// Counts the number of items passing through and prints to stdout whenever the stream trigger activates.
12///
13/// ```dfir
14/// source_stream(dfir_rs::util::iter_batches_stream(0..=100_000, 1))
15///     -> _counter("nums", std::time::Duration::from_millis(100));
16/// ```
17/// stdout:
18/// ```text
19/// _counter(nums): 1
20/// _counter(nums): 6202
21/// _counter(nums): 12540
22/// _counter(nums): 18876
23/// _counter(nums): 25218
24/// _counter(nums): 31557
25/// _counter(nums): 37893
26/// _counter(nums): 44220
27/// _counter(nums): 50576
28/// _counter(nums): 56909
29/// _counter(nums): 63181
30/// _counter(nums): 69549
31/// _counter(nums): 75914
32/// _counter(nums): 82263
33/// _counter(nums): 88638
34/// _counter(nums): 94980
35/// ```
36pub const _COUNTER: OperatorConstraints = OperatorConstraints {
37    name: "_counter",
38    categories: &[OperatorCategory::Map],
39    hard_range_inn: RANGE_1,
40    soft_range_inn: RANGE_1,
41    hard_range_out: &(0..=1),
42    soft_range_out: &(0..=1),
43    num_args: 2,
44    persistence_args: RANGE_0,
45    type_args: RANGE_0,
46    is_external_input: false,
47    has_singleton_output: false,
48    flo_type: None,
49    ports_inn: None,
50    ports_out: None,
51    input_delaytype_fn: |_| None,
52    write_fn: |wc @ &WriteContextArgs {
53                   root,
54                   df_ident,
55                   op_span,
56                   ident,
57                   inputs,
58                   outputs,
59                   is_pull,
60                   arguments,
61                   ..
62               },
63               _| {
64        let read_ident = wc.make_ident("read");
65        let write_ident = wc.make_ident("write");
66
67        let tag_expr = &arguments[0];
68        let tag_ident = wc.make_ident("tag");
69        let duration_expr = &arguments[1];
70        let duration_ident = wc.make_ident("duration");
71        let prefix = arguments.get(2).cloned().unwrap_or(parse_quote_spanned!(op_span=> "_counter"));
72
73        let write_prologue = quote_spanned! {op_span=>
74            let #write_ident = ::std::rc::Rc::new(::std::cell::Cell::new(0_u64));
75
76            let #read_ident = ::std::rc::Rc::clone(&#write_ident);
77            let #duration_ident = #duration_expr;
78            let #tag_ident = #tag_expr;
79            #df_ident.request_task(async move {
80                loop {
81                    println!("{}({}): {}", #prefix, #tag_ident, #read_ident.get());
82                    #root::tokio::time::sleep(#duration_ident).await;
83                }
84            });
85        };
86
87        let count_ident = wc.make_ident("count");
88        let write_iterator = if is_pull {
89            let input = &inputs[0];
90            quote_spanned! {op_span=>
91                let #ident = #root::futures::stream::StreamExt::inspect(#input, |_| { #count_ident += 1; });
92            }
93        } else if outputs.is_empty() {
94            quote_spanned! {op_span=>
95                let #ident = #root::sinktools::inspect(|_| { #count_ident += 1; }, #root::sinktools::for_each::ForEach::new(::std::mem::drop));
96            }
97        } else {
98            let output = &outputs[0];
99            quote_spanned! {op_span=>
100                let #ident = #root::sinktools::inspect(|_| { #count_ident += 1; }, #output);
101            }
102        };
103        let write_iterator = quote_spanned! {op_span=>
104            let mut #count_ident = 0;
105            #write_iterator
106        };
107
108        let write_iterator_after = quote_spanned! {op_span=>
109            #write_ident.set(#write_ident.get() + #count_ident);
110        };
111
112        Ok(OperatorWriteOutput {
113            write_prologue,
114            write_iterator,
115            write_iterator_after,
116            ..Default::default()
117        })
118    },
119};
