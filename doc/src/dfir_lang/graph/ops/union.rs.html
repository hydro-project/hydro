8/// > *n* input streams of the same type, 1 output stream of the same type
9///
10/// Unions an arbitrary number of input streams into a single stream. Each input sequence is a subsequence of the output, but no guarantee is given on how the inputs are interleaved.
11///
12/// Since `union` has multiple input streams, it needs to be assigned to
13/// a variable to reference its multiple input ports across statements.
14///
15/// ```dfir
16/// source_iter(vec!["hello", "world"]) -> my_union;
17/// source_iter(vec!["stay", "gold"]) -> my_union;
18/// source_iter(vec!["don't", "give", "up"]) -> my_union;
19/// my_union = union()
20///     -> map(|x| x.to_uppercase())
21///     -> assert_eq(["HELLO", "WORLD", "STAY", "GOLD", "DON'T", "GIVE", "UP"]);
22/// ```
23pub const UNION: OperatorConstraints = OperatorConstraints {
24    name: "union",
25    categories: &[OperatorCategory::MultiIn],
26    hard_range_inn: RANGE_ANY,
27    soft_range_inn: &(2..),
28    hard_range_out: RANGE_1,
29    soft_range_out: RANGE_1,
30    num_args: 0,
31    persistence_args: RANGE_0,
32    type_args: RANGE_0,
33    is_external_input: false,
34    has_singleton_output: false,
35    flo_type: None,
36    ports_inn: None,
37    ports_out: None,
38    input_delaytype_fn: |_| None,
39    write_fn: |&WriteContextArgs {
40        root,
41                   op_span,
42                   ident,
43                   inputs,
44                   outputs,
45                   is_pull,
46                   ..
47               },
48               _| {
49        let write_iterator = if is_pull {
50            let chains = inputs
51                .iter()
52                .map(|i| i.to_token_stream())
53                .reduce(|a, b| quote_spanned! {op_span=> check_inputs(#a, #b) })
54                .unwrap_or_else(|| quote_spanned! {op_span=> #root::dfir_pipes::pull::empty() });
55            quote_spanned! {op_span=>
56                let #ident = {
57                    #[allow(unused)]
58                    #[inline(always)]
59                    fn check_inputs<A, B, Item>(
60                        a: A, b: B
61                    ) -> impl #root::dfir_pipes::pull::Pull<
62                        Item = Item,
63                        Meta = A::Meta,
64                        CanPend = <A::CanPend as #root::dfir_pipes::Toggle>::Or<B::CanPend>,
65                        CanEnd = B::CanEnd,
66                    >
67                    where
68                        A: #root::dfir_pipes::pull::Pull<Item = Item, CanEnd = #root::dfir_pipes::Yes>,
69                        B: #root::dfir_pipes::pull::Pull<Item = Item, Meta = A::Meta>,
70                    {
71                        // NOTE(mingwei): `merge` (allowing async interleaving) may make more sense, but also
72                        // might inline worse. Merge may have correctness implications.
73                        #root::dfir_pipes::pull::Pull::chain(
74                            #root::dfir_pipes::pull::Pull::fuse(a),
75                            b,
76                        )
77                    }
78                    #chains
79                };
80            }
81        } else {
82            assert_eq!(1, outputs.len());
83            let output = &outputs[0];
84            quote_spanned! {op_span=>
85                let #ident = #output;
86            }
87        };
88        Ok(OperatorWriteOutput {
89            write_iterator,
90            ..Default::default()
91        })
92    },
93};
