10/// > 2 input streams of the same type T, 1 output stream of type T
11///
12/// Forms the multiset difference of the items in the input
13/// streams, returning items in the `pos` input that are not found in the
14/// `neg` input.
15///
16/// `difference` can be provided with one or two generic lifetime persistence arguments
17/// in the same way as [`join`](#join), see [`join`'s documentation](#join) for more info.
18///
19/// Note multiset semantics here: each (possibly duplicated) item in the `pos` input
20/// that has no match in `neg` is sent to the output.
21///
22/// ```dfir
23/// source_iter(vec!["cat", "cat", "elephant", "elephant"]) -> [pos]diff;
24/// source_iter(vec!["cat", "gorilla"]) -> [neg]diff;
25/// diff = difference_multiset() -> assert_eq(["elephant", "elephant"]);
26/// ```
27pub const DIFFERENCE_MULTISET: OperatorConstraints = OperatorConstraints {
28    name: "difference_multiset",
29    categories: &[OperatorCategory::MultiIn],
30    hard_range_inn: &(2..=2),
31    soft_range_inn: &(2..=2),
32    hard_range_out: RANGE_1,
33    soft_range_out: RANGE_1,
34    num_args: 0,
35    persistence_args: &(0..=2),
36    type_args: &(0..=1),
37    is_external_input: false,
38    has_singleton_output: false,
39    flo_type: None,
40    ports_inn: Some(|| super::PortListSpec::Fixed(parse_quote! { pos, neg })),
41    ports_out: None,
42    input_delaytype_fn: |idx| match idx {
43        PortIndexValue::Path(path) if "neg" == path.to_token_stream().to_string() => {
44            Some(DelayType::Stratum)
45        }
46        _else => None,
47    },
48    write_fn: |wc @ &WriteContextArgs {
49                   op_span,
50                   ident,
51                   inputs,
52                   op_inst: OperatorInstance { .. },
53                   ..
54               },
55               diagnostics| {
56        // Convert the type args to be `<K, ()>` where `K` is the input item type, defaulting to `_` if not provided.
57        let wc_with_types = WriteContextArgs {
58            op_inst: &OperatorInstance {
59                generics: OpInstGenerics {
60                    type_args: vec![
61                        wc.op_inst
62                            .generics
63                            .type_args
64                            .first()
65                            .cloned()
66                            .unwrap_or_else(|| parse_quote_spanned!(op_span=> _)),
67                        parse_quote_spanned!(op_span=> ()),
68                    ],
69                    ..wc.op_inst.generics.clone()
70                },
71                ..wc.op_inst.clone()
72            },
73            ..wc.clone()
74        };
75
76        let OperatorWriteOutput {
77            write_prologue,
78            write_prologue_after,
79            write_iterator,
80            write_iterator_after,
81        } = (super::anti_join_multiset::ANTI_JOIN_MULTISET.write_fn)(&wc_with_types, diagnostics)?;
82
83        let pos = &inputs[1];
84        let write_iterator = quote_spanned! {op_span=>
85            let #pos = #pos.map(|k| (k, ()));
86            #write_iterator
87            let #ident = #ident.map(|(k, ())| k);
88        };
89
90        Ok(OperatorWriteOutput {
91            write_prologue,
92            write_prologue_after,
93            write_iterator,
94            write_iterator_after,
95        })
96    },
97};
