7/// > 2 input streams of the same type, 1 output stream of the same type
8///
9/// Chains together a pair of streams, with all the elements of the first emitted before the second,
10/// emitting up to `N` elements where `N` is passed as an argument.
11///
12/// Since `chain_first_n` has multiple input streams, it needs to be assigned to
13/// a variable to reference its multiple input ports across statements.
14///
15/// ```dfir
16/// source_iter(vec!["hello", "world"]) -> [0]my_chain;
17/// source_iter(vec!["stay", "gold"]) -> [1]my_chain;
18/// my_chain = chain_first_n(3)
19///     -> map(|x| x.to_uppercase())
20///     -> assert_eq(["HELLO", "WORLD", "STAY"]);
21/// ```
22pub const CHAIN_FIRST_N: OperatorConstraints = OperatorConstraints {
23    name: "chain_first_n",
24    categories: &[OperatorCategory::MultiIn],
25    persistence_args: RANGE_0,
26    type_args: RANGE_0,
27    hard_range_inn: &(2..=2),
28    soft_range_inn: &(2..=2),
29    hard_range_out: RANGE_1,
30    soft_range_out: RANGE_1,
31    num_args: 1,
32    is_external_input: false,
33    has_singleton_output: false,
34    flo_type: None,
35    ports_inn: None,
36    ports_out: None,
37    input_delaytype_fn: |idx| match idx {
38        PortIndexValue::Int(idx) if idx.value == 0 => {
39            // will no longer be needed once subgraphs are always DAGs (only run once per tick)
40            Some(DelayType::Stratum)
41        }
42        _else => None,
43    },
44    write_fn: super::union::UNION.write_fn,
45};
