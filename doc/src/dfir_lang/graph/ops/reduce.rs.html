8/// > 1 input stream, 1 output stream
9///
10/// > Arguments: a closure which itself takes two arguments:
11/// > an `&mut Accum` accumulator mutable reference, and an `Item`. The closure should merge the item
12/// > into the accumulator.
13///
14/// Akin to Rust's built-in [`reduce`](https://doc.rust-lang.org/std/iter/trait.Iterator.html#method.reduce)
15/// operator, except that it takes the accumulator by `&mut` instead of by value. Reduces every
16/// item into an accumulator by applying a closure, returning the final result.
17///
18/// > Note: The closure has access to the [`context` object](surface_flows.mdx#the-context-object).
19///
20/// `reduce` can also be provided with one generic lifetime persistence argument, either
21/// `'tick` or `'static`, to specify how data persists. With `'tick`, values will only be collected
22/// within the same tick. With `'static`, the accumulated value will be remembered across ticks and
23/// items are aggregated with items arriving in later ticks. When not explicitly specified
24/// persistence defaults to `'tick`.
25///
26/// ```dfir
27/// source_iter([1,2,3,4,5])
28///     -> reduce::<'tick>(|accum: &mut _, elem| {
29///         *accum *= elem;
30///     })
31///     -> assert_eq([120]);
32/// ```
33pub const REDUCE: OperatorConstraints = OperatorConstraints {
34    name: "reduce",
35    categories: &[OperatorCategory::Fold],
36    hard_range_inn: RANGE_1,
37    soft_range_inn: RANGE_1,
38    hard_range_out: &(0..=1),
39    soft_range_out: &(0..=1),
40    num_args: 1,
41    persistence_args: &(0..=1),
42    type_args: RANGE_0,
43    is_external_input: false,
44    has_singleton_output: true,
45    flo_type: None,
46    ports_inn: None,
47    ports_out: None,
48    input_delaytype_fn: |_| Some(DelayType::Stratum),
49    write_fn: |wc @ &WriteContextArgs {
50                   root,
51                   context,
52                   df_ident,
53                   op_span,
54                   work_fn,
55                   work_fn_async,
56                   ident,
57                   inputs,
58                   is_pull,
59                   singleton_output_ident,
60                   arguments,
61                   ..
62               },
63               diagnostics| {
64        let [persistence] = wc.persistence_args_disallow_mutable(diagnostics);
65
66        let write_prologue = quote_spanned! {op_span=>
67            let #singleton_output_ident = #df_ident.add_state(::std::cell::RefCell::new(::std::option::Option::None));
68        };
69        let write_prologue_after = wc
70            .persistence_as_state_lifespan(persistence)
71            .map(|lifespan| quote_spanned! {op_span=>
72                #df_ident.set_state_lifespan_hook(#singleton_output_ident, #lifespan, move |rcell| { rcell.replace(::std::option::Option::None); });
73            }).unwrap_or_default();
74
75        let func = &arguments[0];
76        let accumulator_ident = wc.make_ident("accumulator");
77        let item_ident = wc.make_ident("item");
78
79        let foreach_body = quote_spanned! {op_span=>
80            #[inline(always)]
81            fn call_comb_type<Item>(
82                accum: &mut ::std::option::Option<Item>,
83                item: Item,
84                mut func: impl ::std::ops::FnMut(&mut Item, Item),
85            ) {
86                match accum {
87                    accum @ ::std::option::Option::None => *accum = ::std::option::Option::Some(item),
88                    ::std::option::Option::Some(accum) => (func)(accum, item),
89                }
90            }
91            #[allow(clippy::redundant_closure_call)]
92            call_comb_type(&mut *#accumulator_ident, #item_ident, #func);
93        };
94
95        let assign_accum_ident = quote_spanned! {op_span=>
96            #[allow(unused_mut)]
97            let mut #accumulator_ident = unsafe {
98                // SAFETY: handle from `#df_ident.add_state(..)`.
99                #context.state_ref_unchecked(#singleton_output_ident)
100            }.borrow_mut();
101        };
102
103        let write_iterator = if is_pull {
104            let input = &inputs[0];
105            quote_spanned! {op_span=>
106                #assign_accum_ident
107
108                // Eagerly consume input to ensure updated state.
109                {
110                    let __fut = #root::compiled::pull::ForEach::new(#input, |#item_ident| {
111                        #foreach_body
112                    });
113                    let () = #work_fn_async(__fut).await;
114                }
115
116                let #ident = #work_fn(
117                    || #root::futures::stream::iter(
118                        // 1 or 0 items (`Some` or `None`).
119                        ::std::clone::Clone::clone(&*#accumulator_ident)
120                    )
121                );
122            }
123        } else {
124            // Is only push when used as a singleton, so no need to push to `outputs[0]`.
125            quote_spanned! {op_span=>
126                let #ident = #root::sinktools::for_each::ForEach::new(|#item_ident| {
127                    #assign_accum_ident
128
129                    #foreach_body
130                });
131            }
132        };
133
134        let write_iterator_after = if Persistence::Static == persistence {
135            quote_spanned! {op_span=>
136                #context.schedule_subgraph(#context.current_subgraph(), false);
137            }
138        } else {
139            Default::default()
140        };
141
142        Ok(OperatorWriteOutput {
143            write_prologue,
144            write_prologue_after,
145            write_iterator,
146            write_iterator_after,
147        })
148    },
149};
