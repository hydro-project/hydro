9/// > 2 input streams of type `(K, V1)` and `(K, V2)`, 1 output stream of type `(K, (V1', V2'))` where `V1`, `V2`, `V1'`, `V2'` are lattice types
10///
11/// Performs a [`fold_keyed`](#fold_keyed) with lattice-merge aggregate function on each input and then forms the
12/// equijoin of the resulting key/value pairs in the input streams by their first (key) attribute.
13/// Unlike [`join`](#join), the result is not a stream of tuples, it's a stream of MapUnionSingletonMap
14/// lattices. You can (non-monotonically) "reveal" these as tuples if desired via [`map`](#map); see the examples below.
15///
16/// You must specify the the accumulating lattice types, they cannot be inferred. The first type argument corresponds to the `[0]` input of the join, and the second to the `[1]` input.
17/// Type arguments are specified in dfir using the rust turbofish syntax `::<>`, for example `_lattice_join_fused_join::<Min<_>, Max<_>>()`
18/// The accumulating lattice type is not necessarily the same type as the input, see the below example involving SetUnion for such a case.
19///
20/// Like [`join`](#join), `_lattice_join_fused_join` can also be provided with one or two generic lifetime persistence arguments, either
21/// `'tick` or `'static`, to specify how join data persists. With `'tick`, pairs will only be
22/// joined with corresponding pairs within the same tick. With `'static`, pairs will be remembered
23/// across ticks and will be joined with pairs arriving in later ticks. When not explicitly
24/// specified persistence defaults to `tick.
25///
26/// Like [`join`](#join), when two persistence arguments are supplied the first maps to port `0` and the second maps to
27/// port `1`.
28/// When a single persistence argument is supplied, it is applied to both input ports.
29/// When no persistence arguments are applied it defaults to `'tick` for both.
30/// It is important to specify all persistence arguments before any type arguments, otherwise the persistence arguments will be ignored.
31///
32/// The syntax is as follows:
33/// ```dfir,ignore
34/// _lattice_join_fused_join::<MaxRepr<usize>, MaxRepr<usize>>(); // Or
35/// _lattice_join_fused_join::<'static, MaxRepr<usize>, MaxRepr<usize>>();
36///
37/// _lattice_join_fused_join::<'tick, MaxRepr<usize>, MaxRepr<usize>>();
38///
39/// _lattice_join_fused_join::<'static, 'tick, MaxRepr<usize>, MaxRepr<usize>>();
40///
41/// _lattice_join_fused_join::<'tick, 'static, MaxRepr<usize>, MaxRepr<usize>>();
42/// // etc.
43/// ```
44///
45/// ### Examples
46///
47/// ```dfir
48/// use dfir_rs::lattices::Min;
49/// use dfir_rs::lattices::Max;
50///
51/// source_iter([("key", Min::new(1)), ("key", Min::new(2))]) -> [0]my_join;
52/// source_iter([("key", Max::new(1)), ("key", Max::new(2))]) -> [1]my_join;
53///
54/// my_join = _lattice_join_fused_join::<'tick, Min<usize>, Max<usize>>()
55///     -> map(|singleton_map| {
56///         let lattices::collections::SingletonMap(k, v) = singleton_map.into_reveal();
57///         (k, (v.into_reveal()))
58///     })
59///     -> assert_eq([("key", (Min::new(1), Max::new(2)))]);
60/// ```
61///
62/// ```dfir
63/// use dfir_rs::lattices::set_union::SetUnionSingletonSet;
64/// use dfir_rs::lattices::set_union::SetUnionHashSet;
65///
66/// source_iter([("key", SetUnionSingletonSet::new_from(0)), ("key", SetUnionSingletonSet::new_from(1))]) -> [0]my_join;
67/// source_iter([("key", SetUnionHashSet::new_from([0])), ("key", SetUnionHashSet::new_from([1]))]) -> [1]my_join;
68///
69/// my_join = _lattice_join_fused_join::<'tick, SetUnionHashSet<usize>, SetUnionHashSet<usize>>()
70///     -> map(|singleton_map| {
71///         let lattices::collections::SingletonMap(k, v) = singleton_map.into_reveal();
72///         (k, (v.into_reveal()))
73///     })
74///     -> assert_eq([("key", (SetUnionHashSet::new_from([0, 1]), SetUnionHashSet::new_from([0, 1])))]);
75/// ```
76pub const _LATTICE_JOIN_FUSED_JOIN: OperatorConstraints = OperatorConstraints {
77    name: "_lattice_join_fused_join",
78    categories: &[OperatorCategory::CompilerFusionOperator],
79    hard_range_inn: &(2..=2),
80    soft_range_inn: &(2..=2),
81    hard_range_out: RANGE_1,
82    soft_range_out: RANGE_1,
83    num_args: 0,
84    persistence_args: &(0..=2),
85    type_args: &(2..=2),
86    is_external_input: false,
87    has_singleton_output: false,
88    flo_type: None,
89    ports_inn: Some(|| super::PortListSpec::Fixed(parse_quote! { 0, 1 })),
90    ports_out: None,
91    input_delaytype_fn: |_| Some(DelayType::MonotoneAccum),
92    write_fn: |wc @ &WriteContextArgs {
93                   root,
94                   op_span,
95                   ident,
96                   op_inst:
97                       OperatorInstance {
98                           generics: OpInstGenerics { type_args, .. },
99                           ..
100                       },
101                   ..
102               },
103               diagnostics| {
104        let lhs_type = &type_args[0];
105        let rhs_type = &type_args[1];
106
107        let wc = WriteContextArgs {
108            arguments: &parse_quote_spanned! {op_span=>
109                #root::util::accumulator::FoldFrom::new(
110                    <#lhs_type as #root::lattices::LatticeFrom::<_>>::lattice_from,
111                    |state, delta| { #root::lattices::Merge::merge(state, delta); },
112                ),
113                #root::util::accumulator::FoldFrom::new(
114                    <#rhs_type as #root::lattices::LatticeFrom::<_>>::lattice_from,
115                    |state, delta| { #root::lattices::Merge::merge(state, delta); },
116                ),
117            },
118            ..wc.clone()
119        };
120
121        // Use `join_fused`'s codegen.
122        let OperatorWriteOutput {
123            write_prologue,
124            write_prologue_after,
125            write_iterator,
126            write_iterator_after,
127        } = (super::join_fused::JOIN_FUSED.write_fn)(&wc, diagnostics).unwrap();
128
129        let write_iterator = quote_spanned! {op_span=>
130            #write_iterator
131
132            #[allow(suspicious_double_ref_op, clippy::clone_on_copy)]
133            let #ident = #root::futures::stream::StreamExt::map(#ident, |(k, (v1, v2))| {
134                #root::lattices::map_union::MapUnionSingletonMap::new_from(
135                    (
136                        k,
137                        #root::lattices::Pair::<#lhs_type, #rhs_type>::new_from(v1, v2),
138                    ),
139                )
140            });
141        };
142
143        Ok(OperatorWriteOutput {
144            write_prologue,
145            write_prologue_after,
146            write_iterator,
147            write_iterator_after,
148        })
149    },
150};
