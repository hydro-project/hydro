10/// > 0 input streams, 1 output stream
11///
12/// > Arguments: [`Stream`](https://docs.rs/futures/latest/futures/stream/trait.Stream.html)
13///
14/// Given a [`Stream`](https://docs.rs/futures/latest/futures/stream/trait.Stream.html)
15/// of `(serialized payload, addr)` pairs, deserializes the payload and emits each of the
16/// elements it receives downstream.
17///
18/// ```rustbook
19/// async fn serde_in() {
20///     let addr = dfir_rs::util::ipv4_resolve("localhost:9000".into()).unwrap();
21///     let (outbound, inbound, _) = dfir_rs::util::bind_udp_bytes(addr).await;
22///     let mut flow = dfir_rs::dfir_syntax! {
23///         source_stream_serde(inbound) -> map(Result::unwrap) -> map(|(x, a): (String, std::net::SocketAddr)| x.to_uppercase())
24///             -> for_each(|x| println!("{}", x));
25///     };
26///     flow.run_available();
27/// }
28/// ```
29pub const SOURCE_STREAM_SERDE: OperatorConstraints = OperatorConstraints {
30    name: "source_stream_serde",
31    categories: &[OperatorCategory::Source],
32    hard_range_inn: RANGE_0,
33    soft_range_inn: RANGE_0,
34    hard_range_out: RANGE_1,
35    soft_range_out: RANGE_1,
36    num_args: 1,
37    persistence_args: RANGE_0,
38    type_args: &(0..=1),
39    is_external_input: true,
40    has_singleton_output: false,
41    flo_type: Some(FloType::Source),
42    ports_inn: None,
43    ports_out: None,
44    input_delaytype_fn: |_| None,
45    write_fn: |wc @ &WriteContextArgs {
46                   root,
47                   context,
48                   op_span,
49                   ident,
50                   arguments,
51                   op_inst:
52                       OperatorInstance {
53                           generics: OpInstGenerics { type_args, .. },
54                           ..
55                       },
56                   ..
57               },
58               _| {
59        let generic_type = type_args
60            .first()
61            .map(quote::ToTokens::to_token_stream)
62            .unwrap_or(quote_spanned!(op_span=> _));
63
64        let receiver = &arguments[0];
65        let stream_ident = wc.make_ident("stream");
66        let write_prologue = quote_spanned! {op_span=>
67            // TODO(mingwei): use `::std::pin::pin!(..)`?
68            let mut #stream_ident = Box::pin(#receiver);
69        };
70        let write_iterator = quote_spanned! {op_span=>
71            let #ident = #root::futures::stream::poll_fn(|_tick_cx| {
72                // Using the `tick_cx` will cause the tick to "block" (yield) until the stream is exhausted, which is not what we want.
73                // We want only the ready items, and will awaken this subgraph on a later tick when more items are available.
74                match #root::futures::stream::Stream::poll_next(#stream_ident.as_mut(), &mut ::std::task::Context::from_waker(&#context.waker())) {
75                    ::std::task::Poll::Ready(::std::option::Option::Some(::std::result::Result::Ok((payload, addr)))) =>
76                        ::std::task::Poll::Ready(::std::option::Option::Some(
77                            #root::util::deserialize_from_bytes::<#generic_type>(payload).map(|payload| (payload, addr))
78                        )),
79                    ::std::task::Poll::Ready(::std::option::Option::Some(::std::result::Result::Err(_)))
80                        | ::std::task::Poll::Ready(::std::option::Option::None)
81                        | ::std::task::Poll::Pending => ::std::task::Poll::Ready(::std::option::Option::None),
82                }
83            });
84        };
85        Ok(OperatorWriteOutput {
86            write_prologue,
87            write_iterator,
88            ..Default::default()
89        })
90    },
91};
