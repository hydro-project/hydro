6/// Buffers all input items and releases them in the next tick.
7/// See the [book discussion on time](../concepts/life_and_times) for details on ticks.
8///
9/// `defer_tick` is sometimes needed to separate conflicting data across time,
10/// in order to preserve invariants. Consider the following example, which implements
11/// a flip-flop -- the invariant is that it emit one of true or false in a given tick
12/// (but never both!)
13///
14/// ```rustbook
15/// pub fn main() {
16///     let mut df = dfir_rs::dfir_syntax! {
17///         source_iter(vec!(true))
18///                 -> state;
19///         state = union()
20///                 -> assert(|x| if context.current_tick().0 % 2 == 0 { *x == true } else { *x == false })
21///                 -> map(|x| !x)
22///                 -> defer_tick()
23///                 -> state;
24///     };
25///     for i in 1..100 {
26///         println!("tick {}", i);
27///         df.run_tick();
28///     }
29/// }
30/// ```
31///
32/// `defer_tick` can also be handy for comparing stream content across ticks.
33/// In the example below `defer_tick()` is used alongside `difference()` to
34/// filter out any items that arrive from `inp` in the current tick which match
35/// an item from `inp` in the previous
36/// tick.
37/// ```rustbook
38/// // Outputs 1 2 3 4 5 6 (on separate lines).
39/// let (input_send, input_recv) = dfir_rs::util::unbounded_channel::<usize>();
40/// let mut flow = dfir_rs::dfir_syntax! {
41///     inp = source_stream(input_recv) -> tee();
42///     inp -> [pos]diff;
43///     inp -> defer_tick() -> [neg]diff;
44///     diff = difference() -> for_each(|x| println!("{}", x));
45/// };
46///
47/// for x in [1, 2, 3, 4] {
48///     input_send.send(x).unwrap();
49/// }
50/// flow.run_tick();
51///
52/// for x in [3, 4, 5, 6] {
53///     input_send.send(x).unwrap();
54/// }
55/// flow.run_tick();
56/// ```
57///
58/// You can also supply a type parameter `defer_tick::<MyType>()` to specify what items flow
59/// through the the pipeline. This can be useful for helping the compiler infer types.
