7/// > 2 input streams of type S and T, 1 output stream of type (S, T)
8///
9/// Forms the multiset cross-join (Cartesian product) of the (possibly duplicated) items in the input streams, returning all
10/// tupled pairs regardless of duplicates.
11///
12/// ```dfir
13/// source_iter(vec!["happy", "happy", "sad"]) -> [0]my_join;
14/// source_iter(vec!["dog", "cat", "cat"]) -> [1]my_join;
15/// my_join = cross_join_multiset() -> sort() -> assert_eq([
16///     ("happy", "cat"),
17///     ("happy", "cat"),
18///     ("happy", "cat"),
19///     ("happy", "cat"),
20///     ("happy", "dog"),
21///     ("happy", "dog"),
22///     ("sad", "cat"),
23///     ("sad", "cat"),
24///     ("sad", "dog"), ]);
25/// ```
26pub const CROSS_JOIN_MULTISET: OperatorConstraints = OperatorConstraints {
27    name: "cross_join_multiset",
28    categories: &[OperatorCategory::MultiIn],
29    hard_range_inn: &(2..=2),
30    soft_range_inn: &(2..=2),
31    hard_range_out: RANGE_1,
32    soft_range_out: RANGE_1,
33    num_args: 0,
34    persistence_args: &(0..=2),
35    type_args: &(0..=1),
36    is_external_input: false,
37    flo_type: None,
38    ports_inn: Some(|| super::PortListSpec::Fixed(parse_quote! { 0, 1 })),
39    ports_out: None,
40    input_delaytype_fn: |_| None,
41    write_fn: |wc @ &WriteContextArgs {
42                   root,
43                   op_span,
44                   work_fn_async,
45                   ident,
46                   inputs,
47                   ..
48               },
49               diagnostics| {
50        let lhs = &inputs[0];
51        let rhs = &inputs[1];
52
53        let [lhs_persistence, rhs_persistence] = wc.persistence_args_disallow_mutable(diagnostics);
54
55        let lhs_state = wc.make_ident("lhs_state");
56        let rhs_state = wc.make_ident("rhs_state");
57        let write_prologue = quote_spanned! {op_span=>
58            let mut #lhs_state: ::std::vec::Vec<_> = ::std::vec::Vec::new();
59            let mut #rhs_state: ::std::vec::Vec<_> = ::std::vec::Vec::new();
60        };
61
62        let lhs_write_tick_end = match lhs_persistence {
63            Persistence::None | Persistence::Tick => quote_spanned! {op_span=>
64                #lhs_state.clear();
65            },
66            _ => Default::default(),
67        };
68        let rhs_write_tick_end = match rhs_persistence {
69            Persistence::None | Persistence::Tick => quote_spanned! {op_span=>
70                #rhs_state.clear();
71            },
72            _ => Default::default(),
73        };
74
75        let write_iterator = quote_spanned! {op_span=>
76            #work_fn_async(#root::dfir_pipes::pull::Pull::for_each(#lhs, |x| #lhs_state.push(x))).await;
77            #work_fn_async(#root::dfir_pipes::pull::Pull::for_each(#rhs, |x| #rhs_state.push(x))).await;
78
79            let #ident = #root::dfir_pipes::pull::iter(
80                #lhs_state
81                    .iter()
82                    .flat_map(|lhs| {
83                #rhs_state
84                    .iter()
85                    .map(move |rhs| (::std::clone::Clone::clone(lhs), ::std::clone::Clone::clone(rhs)))
86                    })
87            );
88        };
89
90        let replay_code = matches!(
91            (lhs_persistence, rhs_persistence),
92            (Persistence::Static, Persistence::Static)
93        )
94        .then(|| {
95            quote_spanned! {op_span=>
96                // Reschedule the subgraph lazily to ensure replay on later ticks.
97                
98            }
99        });
100        let write_iterator_after = quote_spanned! {op_span=>
101            #replay_code
102        };
103
104        // Ok(output)
105        Ok(OperatorWriteOutput {
106            write_prologue,
107            write_iterator,
108            write_iterator_after,
109            write_tick_end: quote_spanned! {op_span=>
110                #lhs_write_tick_end
111                #rhs_write_tick_end
112            },
113        })
114    },
115};
