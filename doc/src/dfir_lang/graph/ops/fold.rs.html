8/// > 1 input stream, 1 output stream
9///
10/// > Arguments: two arguments, both closures. The first closure is used to create the initial
11/// > value for the accumulator, and the second is used to combine new items with the existing
12/// > accumulator value. The second closure takes two two arguments: an `&mut Accum` accumulated
13/// > value, and an `Item`.
14///
15/// Akin to Rust's built-in [`fold`](https://doc.rust-lang.org/std/iter/trait.Iterator.html#method.fold)
16/// operator, except that it takes the accumulator by `&mut` instead of by value. Folds every item
17/// into an accumulator by applying a closure, returning the final result.
18///
19/// > Note: The closures have access to the [`context` object](surface_flows.mdx#the-context-object).
20///
21/// `fold` can also be provided with one generic lifetime persistence argument, either
22/// `'tick` or `'static`, to specify how data persists. With `'tick`, Items will only be collected
23/// within the same tick. With `'static`, the accumulated value will be remembered across ticks and
24/// will be aggregated with items arriving in later ticks. When not explicitly specified
25/// persistence defaults to `'tick`.
26///
27/// ```dfir
28/// // should print `Reassembled vector [1,2,3,4,5]`
29/// source_iter([1,2,3,4,5])
30///     -> fold::<'tick>(Vec::new, |accum: &mut Vec<_>, elem| {
31///         accum.push(elem);
32///     })
33///     -> assert_eq([vec![1, 2, 3, 4, 5]]);
34/// ```
35pub const FOLD: OperatorConstraints = OperatorConstraints {
36    name: "fold",
37    categories: &[OperatorCategory::Fold],
38    hard_range_inn: RANGE_1,
39    soft_range_inn: RANGE_1,
40    hard_range_out: &(0..=1),
41    soft_range_out: &(0..=1),
42    num_args: 2,
43    persistence_args: &(0..=1),
44    type_args: RANGE_0,
45    is_external_input: false,
46    has_singleton_output: true,
47    flo_type: None,
48    ports_inn: None,
49    ports_out: None,
50    input_delaytype_fn: |_| Some(DelayType::Stratum),
51    write_fn: |wc @ &WriteContextArgs {
52                   root,
53                   context,
54                   df_ident,
55                   op_span,
56                   ident,
57                   is_pull,
58                   inputs,
59                   singleton_output_ident,
60                   work_fn,
61                   arguments,
62                   ..
63               },
64               diagnostics| {
65        let init_fn = &arguments[0];
66        let func = &arguments[1];
67
68        let initializer_func_ident = wc.make_ident("initializer_func");
69        let init = quote_spanned! {op_span=>
70            (#initializer_func_ident)()
71        };
72
73        let [persistence] = wc.persistence_args_disallow_mutable(diagnostics);
74
75        let input = &inputs[0];
76        let accumulator_ident = wc.make_ident("accumulator");
77        let iterator_item_ident = wc.make_ident("iterator_item");
78
79        let write_prologue = quote_spanned! {op_span=>
80            #[allow(unused_mut, reason = "for if `Fn` instead of `FnMut`.")]
81            let mut #initializer_func_ident = #init_fn;
82
83            #[allow(clippy::redundant_closure_call)]
84            let #singleton_output_ident = #df_ident.add_state(::std::cell::RefCell::new(#init));
85        };
86        let write_prologue_after = wc
87            .persistence_as_state_lifespan(persistence)
88            .map(|lifespan| quote_spanned! {op_span=>
89                #[allow(clippy::redundant_closure_call)]
90                #df_ident.set_state_lifespan_hook(
91                    #singleton_output_ident, #lifespan, move |rcell| { rcell.replace(#init); },
92                );
93            }).unwrap_or_default();
94
95        let assign_accum_ident = quote_spanned! {op_span=>
96            #[allow(unused_mut)]
97            let mut #accumulator_ident = unsafe {
98                // SAFETY: handle from `#df_ident.add_state(..)`.
99                #context.state_ref_unchecked(#singleton_output_ident)
100            }.borrow_mut();
101        };
102        let iterator_foreach = quote_spanned! {op_span=>
103            #[inline(always)]
104            fn call_comb_type<Accum, Item>(
105                accum: &mut Accum,
106                item: Item,
107                func: impl Fn(&mut Accum, Item),
108            ) {
109                (func)(accum, item);
110            }
111            #[allow(clippy::redundant_closure_call)]
112            call_comb_type(&mut *#accumulator_ident, #iterator_item_ident, #func);
113        };
114
115        let write_iterator = if is_pull {
116            quote_spanned! {op_span=>
117                let #ident = {
118                    #assign_accum_ident
119
120                    #work_fn(|| #input.for_each(|#iterator_item_ident| {
121                        #iterator_foreach
122                    }));
123
124                    #[allow(clippy::clone_on_copy)]
125                    {
126                        ::std::iter::once(#work_fn(|| ::std::clone::Clone::clone(&*#accumulator_ident)))
127                    }
128                };
129            }
130        } else {
131            quote_spanned! {op_span=>
132                let #ident = #root::pusherator::for_each::ForEach::new(|#iterator_item_ident| {
133                    #assign_accum_ident
134
135                    #iterator_foreach
136                });
137            }
138        };
139
140        let write_iterator_after = if let Persistence::Static | Persistence::Tick = persistence {
141            quote_spanned! {op_span=>
142                #context.schedule_subgraph(#context.current_subgraph(), false);
143            }
144        } else {
145            Default::default()
146        };
147
148        Ok(OperatorWriteOutput {
149            write_prologue,
150            write_prologue_after,
151            write_iterator,
152            write_iterator_after,
153        })
154    },
155};
