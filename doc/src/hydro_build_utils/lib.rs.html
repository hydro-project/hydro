7#[macro_export]
8macro_rules! emit_nightly_configuration {
9    () => {
10        println!("cargo:rerun-if-env-changed=RUSTC_BOOTSTRAP");
11        println!("cargo::rustc-check-cfg=cfg(nightly)");
12        if matches!(
13            $crate::rustc_version::version_meta().map(|meta| meta.channel),
14            Ok($crate::rustc_version::Channel::Nightly)
15        ) || option_env!("RUSTC_BOOTSTRAP") == Some("1")
16        {
17            println!("cargo:rustc-cfg=nightly");
18        }
19    };
20}

22#[cfg(feature = "insta")]
23#[macro_export]
24macro_rules! nightly_wrapper {
25    ($statement:stmt) => {
26        $crate::insta::with_settings!({
27            prepend_module_to_snapshot => option_env!("CARGO_TARGET_TMPDIR").is_some(), // Only for integration tests.
28            snapshot_path => if cfg!(nightly) { "snapshots-nightly" } else { "snapshots" },
29        }, {
30            $statement;
31        });
32    }
33}

51#[cfg(feature = "trybuild")]
52#[macro_export]
53macro_rules! trybuild_compile_fail {
54    ($glob:expr) => {{
55        let source_dir = std::path::Path::new("tests/compile-fail");
56        let stderr_dir = source_dir.join(if cfg!(nightly) { "nightly" } else { "stable" });
57
58        // Remove any existing .rs files/symlinks in the channel dir first.
59        // Stale broken symlinks (e.g. from rebasing) break trybuild.
60        for entry in std::fs::read_dir(&stderr_dir).unwrap().flatten() {
61            if entry.path().extension().and_then(|e| e.to_str()) == Some("rs") {
62                let _ = std::fs::remove_file(entry.path());
63            }
64        }
65        // Symlink all .rs files from the source directory into the channel-specific
66        // stderr directory so trybuild can find them next to the .stderr files.
67        for entry in std::fs::read_dir(source_dir).unwrap().flatten() {
68            let path = entry.path();
69            if path.extension().and_then(|e| e.to_str()) == Some("rs") {
70                let dest = stderr_dir.join(entry.file_name());
71                let original = std::path::Path::new("..").join(entry.file_name());
72                #[cfg(unix)]
73                std::os::unix::fs::symlink(&original, &dest).unwrap();
74                #[cfg(windows)]
75                std::os::windows::fs::symlink_file(&original, &dest).unwrap();
76            }
77        }
78
79        let pattern = stderr_dir.join($glob);
80        let t = $crate::trybuild::TestCases::new();
81        t.compile_fail(pattern.to_str().unwrap());
82    }};
83}
